60 void NRemoteFSBrowser::reply_read_dir ( common::SignalArgs & args )
61 {
62  SignalOptions options( args );
63  QStringList completion_list;
64  int i;
65 
66  std::vector<std::string> dirs;
67  std::vector<std::string> files;

69  std::vector<std::string> file_dates;

71 
72  std::vector<std::string>::const_iterator it_dirs;
73  std::vector<std::string>::const_iterator it_files;
74 
75  m_current_path = options.value<std::string>("dirPath").c_str();
76 
77  // add an ending '/' if the string does not have any
78  if( !m_current_path.endsWith("/") )
79  m_current_path += "/";
80 
81 // if(!m_updatingCompletion)
82 // m_editPath->setText(m_current_path);
83 // else
84 // m_updatingCompletion = false;
85 
86  dirs = options.value<std::vector<std::string> >("dirs");
87  files = options.value<std::vector<std::string> >("files");
88  dir_dates = options.value<std::vector<std::string> >("dirDates");
89  file_dates = options.value< std::vector<std::string> >("fileDates");
90  file_sizes = options.value< std::vector<Uint> >("fileSizes");
91 
92  // notice the view(s) that the model is about to be completely changed
93  emit layoutAboutToBeChanged();
94 
95  //clear();
96  m_completion_model->setStringList( QStringList() );
97 
98  // delete old data
99  while( !m_data.empty() )
100  delete m_data.takeFirst();
101 
102  // add the directories
103  for( i = 0, it_dirs = dirs.begin() ; it_dirs != dirs.end() ; ++it_dirs, ++i )
104  {
105  FileInfo * fileInfo = new FileInfo();
106  QString name = it_dirs->c_str();
107 
108  fileInfo->name = name;
109  fileInfo->type = DIRECTORY;
110 
111  fileInfo->date_modified = ""; //QString::fromStdString(dir_dates[i]);
112 
113  if(!m_current_path.isEmpty() && name != "..")
114  name.prepend(m_current_path + (m_current_path.endsWith("/") ? "" : "/"));
115 
116  m_data.append(fileInfo);
117  completion_list << name;
118  }
119 
120  m_completion_model->setStringList( completion_list );
121 
122  // add the files
123  for( i = 0, it_files = files.begin() ; it_files != files.end() ; ++it_files, ++i )
124  {
125  FileInfo * fileInfo = new FileInfo();
126 
127  fileInfo->name = it_files->c_str();
128  fileInfo->type = FILE;
129  fileInfo->date_modified = file_dates[i].c_str();
130  fileInfo->file_size = file_sizes[i];
131 
132  m_data.append(fileInfo);
133  }
134 
135  // notice the view(s) that the model has finished changing
136  emit layoutChanged();
137  emit current_path_changed(m_current_path);
138 }

304 QVariant NRemoteFSBrowser::data ( const QModelIndex & index, int role ) const
305 {
306  QVariant data;
307 
308  if( index.isValid() )
309  {
310  FileInfo * item = m_data.at(index.row());
311 
312  if( role == Qt::DisplayRole )
313  {
314  switch( index.column() )
315  {
316  case 0: // item name
317  data = item->name;
318  break;
319 
320  case 1: // size (in case it's a file)
321  data = is_file(index) ? size_to_string(item->file_size) : QString();
322  break;
323 
324  case 2: // date modified
325  data = item->date_modified;
326  break;
327  }
328  }
329  else if( role == Qt::TextAlignmentRole && index.column() == 1)
330  return Qt::AlignRight;
331  else if( role == Qt::ToolTipRole )
332  {
333  if(item->type == FILE)
334  return QString("Size: %1\nDate modified: %2")
335  .arg(size_to_string(item->file_size)).arg(item->date_modified);
336  else
337  return QString("Date modified: %1").arg(item->date_modified);
338  }
339  }
340 
341  return data;
342 }

459 QString NRemoteFSBrowser::size_to_string( Uint size ) const
460 {
461  Uint division_step = 1024;
462 
463  int iter = 0;
464  QStringList units;
465  Real divided_size = size;
466 
467  units << "bytes" << "KB" << "MB" << "GB" << "TB";

471  divided_size /= division_step;
472  iter++;
473  }
474
