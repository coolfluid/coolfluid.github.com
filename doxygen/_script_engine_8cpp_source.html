83 void python_execute_function(){
84  boost::unique_lock<boost::mutex> locker(*python_code_queue_mutex,boost::defer_lock_t());
85  locker.lock();
86  boost::posix_time::millisec wait_init(500);//to wait that the component tree is created
87  boost::this_thread::sleep(wait_init);
88  while(python_should_run){
89  if (python_code_queue.size()){
90  std::pair<boost::python::handle<>,int> python_current_fragment=python_code_queue.front();
91  python_code_queue.pop();
92  locker.unlock();
93  current_execution_fragment=python_current_fragment.second;
94  try{
95  {
96  boost::lock_guard<boost::mutex> lock(*python_scope_mutex);
97  boost::python::handle<> dum(boost::python::allow_null(PyEval_EvalCode((PyCodeObject *)python_current_fragment.first.get(), global_scope.get(), local_scope.get())));
98  }
99  if (python_current_fragment.second){//we don't check errors on internal request (fragment code = 0)
100  PyObject *exc,*val,*trb,*obj;
101  char* error;
102  PyErr_Fetch(&exc, &val, &trb);
103  if (NULL != val){
104  if (PyArg_ParseTuple (val, "sO", &error, &obj)){
105  CFinfo << ":" << python_current_fragment.second << ":" << error << CFendl;
106  }else{
107  const char* val_str=PyString_AsString(PyObject_Str(val));
108  CFinfo << ":" << python_current_fragment.second << ":" << val_str << CFendl;
109  }
110  }
111  }
112  } catch(...) {
113  CFinfo << "Error while executing python code." << CFendl;
114  }
115  PyErr_Clear();
116  script_engine->check_python_change();
117  locker.lock();
118  }else{
119  script_engine->no_more_instruction();
120  python_code_queue_condition->wait(locker);
121  }
122  }
123  locker.unlock();
124 }

205 int ScriptEngine::execute_script(std::string script,int code_fragment){
206  boost::lock_guard<boost::mutex> lock(*python_code_queue_mutex);
207  // we compile the received fragment
208  std::stringstream ss;
209  if (code_fragment > 0)
210  code_fragment=fragment_generator++;
211  ss << code_fragment;
212  boost::python::handle<> src;
213  if (code_fragment == -1){//entire script 'flag'
214  src=boost::python::handle<>(boost::python::allow_null(Py_CompileString(script.c_str(), ss.str().c_str(), Py_file_input)));
215  if (NULL != src.get()){
216  boost::lock_guard<boost::mutex> lock(*python_scope_mutex);
217  boost::python::handle<> dum(boost::python::allow_null(PyEval_EvalCode((PyCodeObject *)src.get(), global_scope.get(), local_scope.get())));
218  }
219  }else{//single input compilation
220  src=boost::python::handle<>(boost::python::allow_null(Py_CompileString(script.c_str(), ss.str().c_str(), Py_single_input)));
221  if (NULL != src.get()){
222  python_code_queue.push(std::pair<boost::python::handle<>,int>(src,code_fragment));
223  python_code_queue_condition->notify_one();
224  }
225  if (code_fragment){//we don't check errors on internal request (fragment code = 0)
226  PyObject *exc,*val,*trb,*obj;
227  char* error;
228  PyErr_Fetch(&exc, &val, &trb);
229  if (NULL != val){
230  if (PyArg_ParseTuple (val, "sO", &error, &obj)){
231  CFinfo << ":" << code_fragment << ":" << error << CFendl;
232  }else{
233  const char* val_str=PyString_AsString(PyObject_Str(val));
234  CFinfo << ":" << code_fragment << ":" << val_str << CFendl;
235  }
236  }
237  }
238  }
239  PyErr_Clear();
240  return code_fragment;
241 }

245 void ScriptEngine::check_scope_difference(PythonDictEntry &entry,std::string name,std::vector<std::string> *add,std::vector<std::string> *sub, int rec,bool child){
246  static char value_limiter[105];//100 character limit (+ ...\0)
247  PyObject *py_key, *py_value;
248  char* key_str;
249  bool fetch_values=interpreter_mode==STOP || interpreter_mode==LINE_BY_LINE_EXECUTION;
250 #if PY_VERSION_HEX >= 0x02050000
251  ssize_t pos = 0;
252 #else
253  int pos = 0;
254 #endif
255  ssize_t dir_size;
256  std::vector<bool> reverse_found;
257  std::string c_name;
258  std::string value_string;
259  if (entry.name.size()){
260  c_name=(name.size()>0?name+".":"")+entry.name;
261  }
262  if (c_name == "sys.stdin" || c_name =="sys.stdout" || c_name == "sys.stderr")
263  return;
264  reverse_found.resize(entry.entry_list.size(),false);
265  bool loop=true;
266  if (entry.py_ref != NULL){
267  boost::python::handle<> dir_list;
268  if (rec > 0){
269  dir_list=boost::python::handle<>(boost::python::allow_null(PyObject_Dir(entry.py_ref)));
270  if (dir_list.get() != NULL)
271  dir_size=PyList_Size(dir_list.get());
272  else
273  loop=false;
274  }
275  while (loop) {
276  boost::python::handle<> key;
277  boost::python::handle<> value;
278 
279  if (rec==0){//scope fetch
280  loop=PyDict_Next(entry.py_ref, &pos, &py_key, &py_value);
281  key=boost::python::handle<>(boost::python::borrowed(boost::python::allow_null(py_key)));
282  value=boost::python::handle<>(boost::python::borrowed(boost::python::allow_null(py_value)));
283  }else{
284  key=boost::python::handle<>(boost::python::borrowed(boost::python::allow_null(PyList_GetItem(dir_list.get(),pos++))));
285  if (NULL != key.get())
286  value=boost::python::handle<>(boost::python::allow_null(PyObject_GetAttr(entry.py_ref,key.get())));
287  if (pos >= dir_size)
288  loop=false;
289  }
290 
291  if (NULL != key.get() && NULL != value.get()){
292  key_str=PyString_AsString(key.get());
293  if (key_str[0] != '_'){
294  bool found=false;
295  if (fetch_values)
296  value_string=PyString_AsString(PyObject_Str(value.get()));
297  bool callable=PyCallable_Check(value.get());
298  int i=0;
299  for(;i<entry.entry_list.size();i++){
300  if (entry.entry_list[i].py_ref==value.get() && entry.entry_list[i].name == key_str){
301  if (fetch_values && !callable){
302  if (entry.entry_list[i].value==value_string){
303  found=true;
304  reverse_found[i]=true;
305  break;
306  }
307  }else{
308  found=true;
309  reverse_found[i]=true;
310  break;
311  }
312  }
313  }
314  if (!found){
315  PythonDictEntry n_entry;
316  n_entry.py_ref=value.get();
317  n_entry.name=key_str;
318  if (PyCallable_Check(value.get())){
319  add->push_back((c_name.size()>0&&!child?c_name+".":"")+n_entry.name+"(:");
320  }else{
321  if (fetch_values){
322  char* value_str=PyString_AsString(PyObject_Str(value.get()));
323  strncpy(value_limiter,value_str,100);
324  if (strlen(value_limiter) > 99){
325  strcpy(value_limiter+99,"...");
326  }
327  n_entry.value=value_limiter;
328  }
329  add->push_back((c_name.size()>0&&!child?c_name+".":"")+n_entry.name+":"+n_entry.value+"{");
330  if (rec < 2)
331  check_scope_difference(n_entry,c_name,add,sub,rec+1,true);
332  add->push_back("}");
333  }
334  entry.entry_list.push_back(n_entry);
335  reverse_found.push_back(true);
336  }else{
337  if (!PyCallable_Check(value.get())){
338  PythonDictEntry n_entry=entry.entry_list[i];
339  if (rec < 2)
340  check_scope_difference(n_entry,c_name,add,sub,rec+1,false);
341  entry.entry_list[i]=n_entry;
342  }
343  }
344  }
345  }
346  }
347  }
348  for (int i=0;i<reverse_found.size();i++){
349  if (!reverse_found[i]){
350  reverse_found[i]=reverse_found[reverse_found.size()-1];
351  reverse_found.pop_back();
352  sub->push_back((c_name.size()>0?c_name+".":"")+entry.entry_list[i].name);
353  entry.entry_list[i]=entry.entry_list[entry.entry_list.size()-1];
354  entry.entry_list.pop_back();
355  i--;
356  }
357  }
358 }

480 int ScriptEngine::new_line_reached(int code_fragment,int line_number){
481  if (interpreter_mode==NORMAL_EXECUTION)
482  for (int i=0;i<break_points.size();i++)
483  if (break_points[i].first == code_fragment && break_points[i].second == line_number-1)
484  interpreter_mode=STOP;
485  switch (interpreter_mode){
486  case STOP:
487  case LINE_BY_LINE_EXECUTION:
488  python_should_wake_up=false;
489  python_scope_mutex->unlock();
490  emit_debug_trace(code_fragment,line_number);
491  check_python_change();
492  {
493  boost::unique_lock<boost::mutex> lock(*python_break_mutex);
494  while (!python_should_wake_up)
495  python_break_condition->wait(lock);
496  }
497  if (interpreter_mode==STOP)
498  interpreter_mode=NORMAL_EXECUTION;
499  if (python_should_break){
500  python_should_break=false;
501  return -1;
502  }
503  python_scope_mutex->lock();
504  break;
505  default:
506  break;
507  }
508  return 0;
509 }

513 void ScriptEngine::signal_change_debug_state(SignalArgs& node){
514  SignalOptions options( node );
515  int commande=options.option("command").value<int>();
516  std::pair<int,int> fragment;
517  bool push;
518  switch (commande){
519  case STOP:
520  if (interpreter_mode==NORMAL_EXECUTION)
521  interpreter_mode=STOP;
522  break;
523  case NORMAL_EXECUTION:
524  case LINE_BY_LINE_EXECUTION:
525  interpreter_mode=(debug_command)commande;
526  break;
527  case BREAK:
528  python_should_break=true;
529  case CONTINUE:
530  {
531  boost::lock_guard<boost::mutex> lock(*python_break_mutex);
532  python_should_wake_up=true;
533  }
534  python_break_condition->notify_one();
535  break;
536  case TOGGLE_BREAK_POINT:
537  fragment=std::pair<int,int>(options.option("fragment").value<int>(),options.option("line").value<int>());
538  push=true;
539  for (int i=0;i<break_points.size();i++){
540  if (break_points[i].first == fragment.first && break_points[i].second == fragment.second){
541  break_points[i]=break_points[break_points.size()-1];
542  break_points.pop_back();
543  push=false;
544  break;
545  }
546  }
547  if (push)
548  break_points.push_back(fragment);
549  default:
550  break;
551  }
