199 template <typename VectorT>
200 Uint MeshPartitioner::nb_connected_objects_in_part(const Uint part, VectorT& nb_connections_per_obj) const
201 {
202  // declaration for boost::tie
203  Handle< common::Component > comp;
204  Uint loc_idx;
205  Uint size = 0;
206  Uint idx = 0;
207  foreach_container((const Uint glb_obj)(const Uint loc_obj),*m_global_to_local)
208  {
209  if (part_of_obj(glb_obj) == part)
210  {
211  boost::tie(comp,loc_idx) = m_lookup->location(loc_obj);
212 
213  if (Handle< Dictionary > nodes = Handle<Dictionary>(comp))
214  {
215  if(!m_periodic_links[loc_idx].first)
216  {
217  const common::DynTable<Uint>& node_to_glb_elm = nodes->glb_elem_connectivity();
218  nb_connections_per_obj[idx] = node_to_glb_elm.row_size(loc_idx);
219  BOOST_FOREACH(const Uint linked_loc_idx, m_inverse_periodic_links[loc_idx])
220  {
221  nb_connections_per_obj[idx] += node_to_glb_elm.row_size(linked_loc_idx);
222  }
223  size += nb_connections_per_obj[idx++];
224  }
225  }
226  else if (Handle< Elements > elements = Handle<Elements>(comp))
227  {
228  const Connectivity& connectivity_table = elements->geometry_space().connectivity();
229  nb_connections_per_obj[idx] = 0;
230  boost_foreach (const Uint loc_node , connectivity_table[loc_idx])
231  {
232  ++nb_connections_per_obj[idx];
233  }
234  size += nb_connections_per_obj[idx++];
235  }
236  }
237  }
238  cf3_assert_desc(common::to_str(idx)+"!="+common::to_str(nb_objects_owned_by_part(part)), idx == nb_objects_owned_by_part(part));
239  return size;
240 }
