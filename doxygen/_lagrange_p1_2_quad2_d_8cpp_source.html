74 bool Quad2D::is_coord_in_element(const CoordsT& coord, const NodesT& nodes)
75 {
76  // Description found in http://hal.archives-ouvertes.fr/docs/00/12/27/30/PDF/exact_interpolation.pdf
77 
78  static const Real tolerance = 1e-6;
79 
80  if (coord[XX] > nodes.col(XX).maxCoeff()+tolerance)
81  return false;
82  if (coord[XX] < nodes.col(XX).minCoeff()-tolerance)
83  return false;
84  if (coord[YY] > nodes.col(YY).maxCoeff()+tolerance)
85  return false;
86  if (coord[YY] < nodes.col(YY).minCoeff()-tolerance)
87  return false;
88 
89 
90  m_D <<
91  nodes.col(XX).maxCoeff()-nodes.col(XX).minCoeff(),
92  nodes.col(YY).maxCoeff()-nodes.col(YY).minCoeff();
93  m_scale = 1./m_D.minCoeff();
94 
95  if (scp(nodes.row(0),nodes.row(Quad2D::nb_nodes-1),coord,m_scale) * scp(nodes.row(0),coord,nodes.row(1),m_scale) < -tolerance)
96  return false;
97  for (Uint i=1; i<Quad2D::nb_nodes-1; ++i)
98  {
99  if (scp(nodes.row(i),nodes.row(i-1),coord,m_scale) * scp(nodes.row(i),coord,nodes.row(i+1),m_scale) < -tolerance)
100  return false;
101  }
102  if (scp(nodes.row(Quad2D::nb_nodes-1),nodes.row(Quad2D::nb_nodes-2),coord,m_scale) * scp(nodes.row(Quad2D::nb_nodes-1),coord,nodes.row(0),m_scale) < -tolerance)
103  return false;
104 
105  return true;
106 }

109 void Quad2D::compute_mapped_coordinate(const CoordsT& coord, const NodesT& nodes, MappedCoordsT& mapped_coord)
110 {
111 
112  // Description found in http://hal.archives-ouvertes.fr/docs/00/12/27/30/PDF/exact_interpolation.pdf
113 
114  m_D <<
115  nodes.col(XX).maxCoeff()-nodes.col(XX).minCoeff(),
116  nodes.col(YY).maxCoeff()-nodes.col(YY).minCoeff();
117  m_scale = 1./m_D.minCoeff();
118 
119  const Real x = coord[XX] * m_scale;
120  const Real y = coord[YY] * m_scale;
121 
122  const Real xn1 = nodes(0, XX) * m_scale ;
123  const Real yn1 = nodes(0, YY) * m_scale ;
124  const Real xn2 = nodes(1, XX) * m_scale ;
125  const Real yn2 = nodes(1, YY) * m_scale ;
126  const Real xn3 = nodes(2, XX) * m_scale ;
127  const Real yn3 = nodes(2, YY) * m_scale ;
128  const Real xn4 = nodes(3, XX) * m_scale ;
129  const Real yn4 = nodes(3, YY) * m_scale ;
130 
131  const Real a0 = 0.25*( (xn1+xn2) + (xn3+xn4) );
132  const Real a1 = 0.25*( (xn2-xn1) + (xn3-xn4) );
133  const Real a2 = 0.25*( (xn3+xn4) - (xn1+xn2) );
134  const Real a3 = 0.25*( (xn1-xn2) + (xn3-xn4) );
135 
136  const Real b0 = 0.25*( (yn1+yn2) + (yn3+yn4) );
137  const Real b1 = 0.25*( (yn2-yn1) + (yn3-yn4) );
138  const Real b2 = 0.25*( (yn3+yn4) - (yn1+yn2) );
139  const Real b3 = 0.25*( (yn1-yn2) + (yn3-yn4) );
140 
141 
142  // x = a0 + a1*xi + a2*eta + a3*xi*eta
143  // y = b0 + b1*xi + b2*eta + b3*xi*eta
144 
145  // We have to invert these:
146  //
147  // xi = (y0*a3+a1*b2)/(a3*b1-a1*b3);
148  // A*eta^2 + B*eta + C = 0
149  //
150  // with
151  const Real x0 = x-a0;
152  const Real y0 = y-b0;
153  const Real A = a3*b2-a2*b3;
154  const Real B = (x0*b3+a1*b2) - (y0*a3+a2*b1);
155  const Real C = x0*b1-y0*a1;
156 
157  // 1) Solve for eta
158  if (std::abs(A)<1e-10)
159  {
160  mapped_coord[ETA] = -C/B;
161  }
162  else
163  {
164  const Real B2 = B*B;
165  Real q = -0.5*(B + math::Functions::sign(B)*std::sqrt(B2-4*A*C));
166  Real eta1 = q/A;
167  Real eta2 = C/q;
168  mapped_coord[ETA] = (std::abs(eta1)<=std::abs(eta2)) ? eta1 : eta2;
169  }
170 
171  // if (a1+a3*eta == 0) then we will divide by zero in to calculate xi.
172  if ( std::abs(a1+a3*mapped_coord[ETA]) < 1e-8 )
173  {
174  Real tolerance=1e-8;
175 
176  if (std::abs(a3)>tolerance)
177  {
178  mapped_coord[ETA] = -a1/a3;
179  mapped_coord[KSI] = (y0*a3+a1*b2)/(a3*b1-a1*b3);
180  }
181  else // if (std::abs(a3)<=tolerance)
182  {
183  mapped_coord[ETA] = x0/a2;
184  mapped_coord[KSI] = (y0*a2-x0*b2)/(a2*b1+x0*b3);
185  }
186  }
187  else // we won't divide by zero so we're good.
188  {
189  mapped_coord[KSI] = (x0-a2*mapped_coord[ETA])/(a1+a3*mapped_coord[ETA]);
190  }
191 
192  // clip to make it more robust
193  mapped_coord[KSI] = math::Functions::sign(mapped_coord[KSI])*std::min(1.,std::abs(mapped_coord[KSI]));
194  mapped_coord[ETA] = math::Functions::sign(mapped_coord[ETA])*std::min(1.,std::abs(mapped_coord[ETA]));
195 }

305 void Quad2D::compute_plane_jacobian_normal(const MappedCoordsT& mapped_coord, const NodesT& nodes, const CoordRef orientation, CoordsT& result)
306 {
307  const Real x0 = nodes(0,XX);
308  const Real y0 = nodes(0,YY);
309 
310  const Real x1 = nodes(1,XX);
311  const Real y1 = nodes(1,YY);
312 
313  const Real x2 = nodes(2,XX);
314  const Real y2 = nodes(2,YY);
315 
316  const Real x3 = nodes(3,XX);
317  const Real y3 = nodes(3,YY);
318 
319  const Real xi = mapped_coord[KSI];
320  const Real eta = mapped_coord[ETA];
321 
322  switch (orientation)
323  {
324  case KSI:
325  {
326  const Real dN0deta = -(1. - xi);
327  const Real dN1deta = -(1. + xi);
328  const Real dN2deta = (1. + xi);
329  const Real dN3deta = (1. - xi);
330  result[XX] = +0.25 * (y0*dN0deta + y1*dN1deta + y2*dN2deta + y3*dN3deta);
331  result[YY] = -0.25 * (x0*dN0deta + x1*dN1deta + x2*dN2deta + x3*dN3deta);
332  return;
333  }
334  case ETA:
335  {
336  const Real dN0dxi = -(1. - eta);
337  const Real dN1dxi = (1. - eta);
338  const Real dN2dxi = (1. + eta);
339  const Real dN3dxi = -(1. + eta);
340  result[XX] = -0.25 * (y0*dN0dxi + y1*dN1dxi + y2*dN2dxi + y3*dN3dxi);
341  result[YY] = +0.25 * (x0*dN0dxi + x1*dN1dxi + x2*dN2dxi + x3*dN3dxi);
342  return;
343  }
344  case ZTA:
345  throw common::ShouldNotBeHere(FromHere(),"");
346  }
347  throw common::ShouldNotBeHere(FromHere(),"orientation not defined");
348 }
