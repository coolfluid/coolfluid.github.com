182 physics::PhysModel& Model::create_physics( const std::string& builder )
183 {
184  std::string pm_name = Builder::extract_reduced_name(builder);
185 
186  boost::shared_ptr< physics::PhysModel > pm = boost::algorithm::contains( builder, "." ) ?
187  build_component_abstract_type< physics::PhysModel >( builder, pm_name ) :
188  build_component_abstract_type_reduced< physics::PhysModel >( builder, pm_name );

192  m_implementation->m_physics->mark_basic();
193 
194  configure_option_recursively(Tags::physical_model(), m_implementation->m_physics);
195

355 void Model::on_mesh_loaded_event(SignalArgs& args)
356 {
357  // If we have no domain, the event can't be for us
358  if(is_null(m_implementation->m_domain))
359  return;
360 
361  SignalOptions options(args);
362 
363  URI mesh_uri = options.value<URI>("mesh_uri");
364 
365  if(mesh_uri.base_path() == domain().uri()) // Only handle events coming from our own domain
366  {
367  // Get a reference to the mesh that changed
368  Handle<Mesh> mesh(access_component(mesh_uri));
369 
370  // Inform the solvers of the change
371  boost_foreach(Solver& solver, find_components<Solver>(*this))
372  {
373  solver.mesh_loaded(*mesh);
374  }
375  }
376 }
