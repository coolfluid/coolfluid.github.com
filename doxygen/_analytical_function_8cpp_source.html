100 {
101  clear();
102  m_function = function;
103 
104  m_parser = boost::shared_ptr<FunctionParser>( new FunctionParser() );
105  m_parser->AddConstant("pi", Consts::pi());
106 
107  // CFinfo << "Parsing Function: \'" << m_functions[i] << "\' Vars: \'" << m_vars << "\'\n" << CFendl;
108  std::stringstream ss;
109  for (Uint i=0; i<m_vars.size(); ++i)
110  {
111  if (i!=0) ss << ",";
112  ss << m_vars[i];
113  }
114  const int r = m_parser->Parse(m_function,ss.str());
115 
116  if ( m_parser->GetParseErrorType() != FunctionParser::FP_NO_ERROR )
117  {
118  std::string msg("ParseError in AnalyticalFunction::parse():\n");
119  msg += " Function [" + m_function + "]\n";

121  msg += " Vars [" + ss.str() + "]\n";
122  throw common::ParsingFailed (FromHere(),msg);
123  }
124  m_is_parsed = true;
