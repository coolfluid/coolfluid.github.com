122 Eigen::Matrix<Real,2,3> Rotate::compute_rotation_matrix_2d(const RealVector2& rotation_centre, const Real& theta)
123 {
124  enum { XX=0, YY=1, TT=2 };
125  Eigen::Matrix<Real,2,3> m;
126 
127  // Rotation centre
128  const Real& a = rotation_centre[XX];
129  const Real& b = rotation_centre[YY];
130 
131  // temp values
132  const Real cosT = std::cos(theta);
133  const Real oneMinusCosT = 1-cosT;
134  const Real sinT = std::sin(theta);
135 
136  // rotation matrix assembly
137  m(XX,XX) = cosT;
138  m(XX,YY) = -sinT;
139  m(XX,TT) = a*oneMinusCosT + b*sinT;
140 
141  m(YY,XX) = sinT;
142  m(YY,YY) = cosT;
143  m(YY,TT) = b*oneMinusCosT - a*sinT;
144  return m;
145 }

149 Eigen::Matrix<Real,3,4> Rotate::compute_rotation_matrix_3d(const RealVector3& point_on_axis, const RealVector3& axis_direction, const Real& theta)
150 {
151  enum { XX=0, YY=1, ZZ=2, TT=3 };
152  Eigen::Matrix<Real,3,4> m;
153 
154  // normalize axis of rotation
155  const Real l = axis_direction.norm();
156  const Real u = axis_direction[XX]/l;
157  const Real v = axis_direction[YY]/l;
158  const Real w = axis_direction[ZZ]/l;
159 
160  // point on axis
161  const Real& a = point_on_axis[XX];
162  const Real& b = point_on_axis[YY];
163  const Real& c = point_on_axis[ZZ];
164 
165  // temp values
166  const Real u2 = u*u;
167  const Real v2 = v*v;
168  const Real w2 = w*w;
169  const Real cosT = std::cos(theta);
170  const Real oneMinusCosT = 1-cosT;
171  const Real sinT = std::sin(theta);
172 
173  // rotation matrix assembly
174  m(XX,XX) = u2 + (v2 + w2) * cosT;
175  m(XX,YY) = u*v * oneMinusCosT - w*sinT;
176  m(XX,ZZ) = u*w * oneMinusCosT + v*sinT;
177  m(XX,TT) = (a*(v2 + w2) - u*(b*v + c*w))*oneMinusCosT + (b*w - c*v)*sinT;
178 
179  m(YY,XX) = u*v * oneMinusCosT + w*sinT;
180  m(YY,YY) = v2 + (u2 + w2) * cosT;
181  m(YY,ZZ) = v*w * oneMinusCosT - u*sinT;
182  m(YY,TT) = (b*(u2 + w2) - v*(a*u + c*w))*oneMinusCosT + (c*u - a*w)*sinT;
183 
184  m(ZZ,XX) = u*w * oneMinusCosT - v*sinT;
185  m(ZZ,YY) = v*w * oneMinusCosT + u*sinT;
186  m(ZZ,ZZ) = w2 + (u2 + v2) * cosT;
187  m(ZZ,TT) = (c*(u2 + v2) - w*(a*u + b*v))*oneMinusCosT + (a*v - b*u)*sinT;
188  return m;
189 }
