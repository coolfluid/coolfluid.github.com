215 void Component::rename ( const std::string& name )
216 {
217  if(name.empty())
218  throw BadValue(FromHere(), "Empty new name given for " + uri().string());
219 
220  if(!URI::is_valid_element(name))
221  throw BadValue(FromHere(), "Invalid new name '"+name+"' given for " + uri().string());
222

224  return;

229  raise_tree_updated_event();

233  if(is_not_null(m_parent->get_child(name)))
234  throw ValueExists(FromHere(), std::string("A component with name ") + this->name() + " already exists in " + uri().string());

237  CompLookupT::iterator lookup = m_parent->m_component_lookup.find(m_name);
238  const Uint idx = lookup->second;
239  m_parent->m_component_lookup.erase(lookup);
240  m_parent->m_component_lookup[name] = idx;
241  }
242 
243  m_name = name;
244 }

357 boost::shared_ptr<Component> Component::remove_component ( const std::string& name )
358 {
359  // find the component exists
360  Component::CompLookupT::iterator itr = m_component_lookup.find(name);
361 
362  if ( itr != m_component_lookup.end() ) // if exists
363  {
364  const Uint comp_idx = itr->second;
365  boost::shared_ptr<Component> comp = m_components[comp_idx]; // get the component
366  if(comp->has_tag(Tags::static_component()))
367  throw BadValue(FromHere(), "Error removing component " + comp->uri().string() + ", it is static!");
368 
369  m_component_lookup.erase(itr); // remove it from the lookup
370 
371  comp->change_parent( Handle<Component>() ); // set parent to invalid
372 
373  // Create new storage to eliminate the removed component
374  CompStorageT new_storage; new_storage.reserve(m_components.size() - 1);
375  // Insert all components that were before the one that got removed
376  new_storage.insert(new_storage.end(), m_components.begin(), m_components.begin() + (comp_idx));
377  //Copy the ones after, adjusting their lookup index in the process
378  const Uint after_deleted_begin = comp_idx + 1;
379  const Uint after_deleted_end = m_components.size();
380  for(Uint i = after_deleted_begin; i != after_deleted_end; ++i)
381  {
382  --m_component_lookup[m_components[i]->name()];
383  new_storage.push_back(m_components[i]);
384  }
385  m_components = new_storage;
386 
387  raise_tree_updated_event();
388 
389  return comp; // return it to client
390  }
391  else // if does not exist
392  {
393  throw ValueNotFound(FromHere(), "Dynamic component with name '"
394  + name + "' does not exist in component '"
395  + this->name() + "' with path ["
396  + uri().path() + "]");
397  }
398 }

487 Handle<Component> Component::access_component(const URI& path) const
488 {
489  // Return self for trivial path or at end of recursion.
490  if(path.path() == "." || path.empty())
491  return const_cast<Component*>(this)->handle<Component>();
492 
493  // If the path is absolute, make it relative and pass it to the root
494  if(path.is_absolute())
495  {
496  // String without protocol
497  std::string new_path = path.path();
498 
499  // Remove any leading /
500  boost::algorithm::trim_left_if(new_path, boost::algorithm::is_any_of("/"));
501 
502  if(new_path.empty())
503  {
504  return const_cast<Component*>(root().get())->handle<Component>();
505  }
506 
507  // Pass the rest to root
508  return root()->access_component(URI(new_path, cf3::common::URI::Scheme::CPATH));
509  }
510 
511  // Relative path
512  std::string path_str = path.path();
513  // Remove trailing /
514  boost::algorithm::trim_right_if(path_str, boost::algorithm::is_any_of("/"));
515 
516  // Find the first separator
517  const std::size_t first_sep = path_str.find("/");
518  const bool has_no_separator = (first_sep == std::string::npos);
519 
520  // Current part of the path to parse
521  const std::string current_part = has_no_separator ? path_str : path_str.substr(0, first_sep);
522 
523  // Remainder of the path to parse, set to "." if there were no more parts
524  const std::string next_part = has_no_separator ? "." : path_str.substr(first_sep+1, path_str.size());
525 
526  // Dispatch to self
527  if(current_part == "." || current_part.empty())
528  return access_component(next_part);
529 
530  // Dispatch to parent
531  if(current_part == "..")
532  return m_parent ? m_parent->access_component(next_part) : Handle<Component>();
533 
534  // Dispatch to child
535  Handle<Component const> child = get_child(current_part);
536  if(is_not_null(child))
537  return child->access_component(next_part);
538 
539  // Return null if not found
540  return Handle<Component>();
541 }
