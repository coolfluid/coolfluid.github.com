83 void Tetra3D::compute_mapped_coordinate(const CoordsT& coord, const NodesT& nodes, MappedCoordsT& mapped_coord)
84 {
85  RealMatrix3 M;
86  M.col(0) = nodes.row(1) - nodes.row(0);
87  M.col(1) = nodes.row(2) - nodes.row(0);
88  M.col(2) = nodes.row(3) - nodes.row(0);

189 void Tetra3D::compute_jacobian_adjoint(const MappedCoordsT& mapped_coord, const NodesT& nodes, JacobianT& result)
190 {
191  JacobianT J = jacobian(mapped_coord, nodes);
192  result(0, 0) = (J(1, 1)*J(2, 2) - J(1, 2)*J(2, 1));
193  result(0, 1) = -(J(0, 1)*J(2, 2) - J(0, 2)*J(2, 1));
194  result(0, 2) = (J(0, 1)*J(1, 2) - J(1, 1)*J(0, 2));
195  result(1, 0) = -(J(1, 0)*J(2, 2) - J(1, 2)*J(2, 0));
196  result(1, 1) = (J(0, 0)*J(2, 2) - J(0, 2)*J(2, 0));
197  result(1, 2) = -(J(0, 0)*J(1, 2) - J(0, 2)*J(1, 0));
198  result(2, 0) = (J(1, 0)*J(2, 1) - J(1, 1)*J(2, 0));
199  result(2, 1) = -(J(0, 0)*J(2, 1) - J(0, 1)*J(2, 0));
