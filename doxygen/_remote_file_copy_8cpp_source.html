82 void RemoteFileCopy::to_local_copy_requested(QModelIndex item, QModelIndexList copy_items){
83  std::vector<std::string> params;
84  std::string local_signature=core::SSHTunnel::get_local_signature();
85  local_signature.append(":");
86  if (item.isValid()){
87  if (local_files->isDir(item))
88  local_signature.append(local_files->filePath(item).toStdString());
89  else
90  local_signature.append(local_files->filePath(local_files->parent(item)).toStdString());
91  }else{
92  item=local_list_widget->rootIndex();
93  }
94  foreach (const QModelIndex & to_copy, copy_items){
95  QModelIndex to_copy_not_const = filter_remote_model->mapToSource(to_copy);
96  if (remote_files->is_directory(to_copy_not_const))
97  params.push_back("-r");
98  params.push_back("-p");
99  params.push_back(remote_files->retrieve_full_path(to_copy_not_const).toStdString());
100  params.push_back(local_signature);
101  params.push_back("#");
102  }
103  remote_files->copy_request(params);
104 }

122  params.push_back(dest);
123  params.push_back("#");
