22 LogStream::LogStream(const std::string & streamName, LogLevel level)
23 : m_buffer(),
24 m_streamName(streamName),
25 m_filter_level(level),
26 m_flushed(true)
27 {
28  iostreams::filtering_ostream * stream;
29  LogLevelFilter levelFilter(level);
30 
31  // SCREEN
32  stream = new iostreams::filtering_ostream();
33  stream->push(levelFilter);
34  stream->push(LogStampFilter(streamName));
35  stream->push(std::cout);
36  m_destinations[SCREEN] = stream;
37 
38  // FILE
39  m_destinations[FILE] = NULL;
40 
41  // STRING
42  stream = new iostreams::filtering_ostream();
43  stream->push(levelFilter);
44  stream->push(LogStampFilter(streamName));
45  stream->push(back_inserter(m_buffer));
46  m_destinations[STRING] = stream;
47 
48  // SYNC_SCREEN
49  stream = new iostreams::filtering_ostream();
50  stream->push(levelFilter);
51  stream->push(LogStampFilter(streamName));
52  stream->push(std::cout);
53  m_destinations[SYNC_SCREEN] = stream;
54 
55 
56  // by default, we use all destinations except SYNC_SCREEN
57  m_usedDests[SCREEN] = true;
58  m_usedDests[FILE] = true;
59  m_usedDests[STRING] = true;
60  m_usedDests[SYNC_SCREEN] = false;
61 
62  // by default, only the first processor outputs
63  m_filterRankZero[SCREEN] = true;
64  m_filterRankZero[FILE] = true;
65  m_filterRankZero[STRING] = true;
66  m_filterRankZero[SYNC_SCREEN] = true;

130 void LogStream::flush()
131 {
132  std::map<LogDestination, iostreams::filtering_ostream *>::iterator it;
133 
134  for(it = m_destinations.begin() ; it != m_destinations.end() ; it++)
135  {
136  if(this->isDestinationUsed(it->first))
137  {
138  it->second->strict_sync();
139  it->second->clear();
140  }
141  }
142 
143  this->getLevelFilter(SCREEN).resetToDefaultLevel();
144 
145  if(this->isFileOpen())
146  this->getLevelFilter(FILE).resetToDefaultLevel();
147 
148  this->getLevelFilter(STRING).resetToDefaultLevel();
149  this->getLevelFilter(SYNC_SCREEN).resetToDefaultLevel();
150 
151  this->getStampFilter(SCREEN).endMessage();
152 
153  if(this->isFileOpen())
154  this->getStampFilter(FILE).endMessage();
155 
156  this->getStampFilter(STRING).endMessage();
157  this->getStampFilter(SYNC_SCREEN).endMessage();
158 
159  if(!m_buffer.empty())
160  {
161  std::list<LogStringForwarder *>::iterator it = m_stringForwarders.begin();
162 
163  while(it != m_stringForwarders.end())
164  {
165  (*it)->message(m_buffer);
166  it++;
167  }
168  m_buffer.clear();
169  }
170 
171  m_flushed = true;
172 }
