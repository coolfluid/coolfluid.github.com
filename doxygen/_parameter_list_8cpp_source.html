167 void ParameterList::set_parameter_list(Teuchos::ParameterList& parameters)
168 {
169  m_parameters = Teuchos::rcpFromRef(parameters);
170 
171  for(Teuchos::ParameterList::ConstIterator it = parameters.begin(); it != parameters.end(); ++it)
172  {
173  if(it->second.isList())
174  {
175  Handle<ParameterList> new_parameters = create_component<ParameterList>(detail::param_list_name_to_comp(it->first));
176  new_parameters->mark_basic();
177  new_parameters->set_parameter_list(Teuchos::getValue<Teuchos::ParameterList>(it->second));
178  }
179  else
180  {
181  bool found = false;
182  boost::mpl::for_each<ParameterTypesT>(detail::AddParameter(*this, it->second, it->first, found));
183  if(!found)
184  {
185  CFdebug << "ParameterList: skipping parameter " << it->first << ": unsupported type " << it->second.getAny().typeName() << CFendl;
186  }
187  }
188  }
