114 {
115  m_mesh = mesh.handle<Mesh>();
116 
117  // Data to indicate if a node has a periodic link
118  Handle< common::List<Uint> > periodic_links_nodes(mesh.geometry_fields().get_child("periodic_links_nodes"));
119  Handle< common::List<bool> > periodic_links_active(mesh.geometry_fields().get_child("periodic_links_active"));
120 
121  const Uint nb_nodes = mesh.geometry_fields().size();
122  m_periodic_links.assign(nb_nodes, std::make_pair(false, 0));
123  m_inverse_periodic_links.assign(nb_nodes, std::vector<Uint>());
124 
125  if(is_not_null(periodic_links_active))
126  {
127  cf3_assert(periodic_links_active->size() == nb_nodes);
128  const common::List<Uint>& links_nodes = *periodic_links_nodes;
129  const common::List<bool>& links_active = *periodic_links_active;
130  for(Uint i = 0; i != nb_nodes; ++i)
131  {
132  if(links_active[i])
133  {
134  m_periodic_links[i] = std::make_pair(true, links_nodes[i]);
135  Uint final_target_node = links_nodes[i];
136  while(links_active[final_target_node])
137  {
138  final_target_node = links_nodes[final_target_node];
139  }
140  m_inverse_periodic_links[final_target_node].push_back(i);
141  }
142  }
143  }
144 
145  Dictionary& nodes = mesh.geometry_fields();
146  Uint tot_nb_owned_nodes(0);
147  for (Uint i=0; i<nodes.size(); ++i)
148  {
149  if (nodes.is_ghost(i) == false)
150  ++tot_nb_owned_nodes;
151  }
152 
153  Uint tot_nb_owned_elems(0);
154  boost_foreach( Entities& elements, find_components_recursively<Entities>(mesh) )
155  {
156  tot_nb_owned_elems += elements.size();
157  }
158 
159  Uint tot_nb_owned_obj = tot_nb_owned_nodes + tot_nb_owned_elems;
160

162  std::vector<Uint> nb_elems_per_proc(PE::Comm::instance().size());
163  std::vector<Uint> nb_obj_per_proc(PE::Comm::instance().size());
164  PE::Comm::instance().all_gather(tot_nb_owned_nodes, nb_nodes_per_proc);
165  PE::Comm::instance().all_gather(tot_nb_owned_elems, nb_elems_per_proc);
166  PE::Comm::instance().all_gather(tot_nb_owned_obj, nb_obj_per_proc);
167  m_start_id_per_part.resize(PE::Comm::instance().size());
168  m_start_node_per_part.resize(PE::Comm::instance().size());
169  m_start_elem_per_part.resize(PE::Comm::instance().size());
170  m_end_id_per_part.resize(PE::Comm::instance().size());
171  m_end_node_per_part.resize(PE::Comm::instance().size());
172  m_end_elem_per_part.resize(PE::Comm::instance().size());
173 
174  Uint start_id(0);
175  for (Uint p=0; p<PE::Comm::instance().size(); ++p)
176  {
177  m_start_id_per_part[p] = start_id;
178  m_end_id_per_part[p] = start_id + nb_obj_per_proc[p];
179  m_start_node_per_part[p] = start_id;
180  m_end_node_per_part[p] = start_id + nb_nodes_per_proc[p];
181  m_start_elem_per_part[p] = start_id + nb_nodes_per_proc[p];
182  m_end_elem_per_part[p] = start_id + nb_nodes_per_proc[p] + nb_elems_per_proc[p];
183 
184  start_id += nb_obj_per_proc[p];
185  }

189

204 void MeshPartitioner::build_global_to_local_index(Mesh& mesh)
205 {
206  Dictionary& nodes = mesh.geometry_fields();
207 
208  m_lookup->add(nodes);
209  boost_foreach ( const Handle<Entities>& elements, mesh.elements() )
210  m_lookup->add(*elements);
211 
212  m_nb_owned_obj = 0;
213  common::List<Uint>& node_glb_idx = nodes.glb_idx();
214  const Uint nb_nodes = nodes.size();
215  for (Uint i=0; i<nb_nodes; ++i)
216  {
217  if (!nodes.is_ghost(i) && !m_periodic_links[i].first)
218  {
219  //std::cout << PE::Comm::instance().rank() << " -- owning node " << node_glb_idx[i] << std::endl;
220  ++m_nb_owned_obj;
221  }
222  }
223 
224  boost_foreach ( const Handle<Entities>& elements, mesh.elements() )
225  {
226  m_nb_owned_obj += elements->size();
227  }
228 
229  const Uint tot_nb_obj = m_lookup->size();
230  m_global_to_local->reserve(tot_nb_obj);
231  Uint loc_idx=0;
232  //CFinfo << "adding nodes to map " << CFendl;
233  boost_foreach (const Uint glb_idx, node_glb_idx.array())
234  {
235  //CFinfo << " adding node with glb " << glb_idx << CFendl;
236  if (nodes.is_ghost(loc_idx) == false)
237  {
238  cf3_assert(glb_idx >= m_start_id_per_part[PE::Comm::instance().rank()]);
239  cf3_assert(glb_idx >= m_start_node_per_part[PE::Comm::instance().rank()]);
240  cf3_assert_desc(to_str(glb_idx)+">="+to_str(m_end_id_per_part[PE::Comm::instance().rank()]),glb_idx < m_end_id_per_part[PE::Comm::instance().rank()]);
241  cf3_assert_desc(to_str(glb_idx)+">="+to_str(m_end_node_per_part[PE::Comm::instance().rank()]),glb_idx < m_end_node_per_part[PE::Comm::instance().rank()]);
242  }
243  else
244  {
245  cf3_assert(glb_idx > m_start_id_per_part[PE::Comm::instance().rank()] ||
246  glb_idx <= m_end_id_per_part[PE::Comm::instance().rank()]);
247 
248  cf3_assert(glb_idx > m_start_node_per_part[PE::Comm::instance().rank()] ||
249  glb_idx <= m_end_node_per_part[PE::Comm::instance().rank()]);
250  }
251 
252  m_global_to_local->push_back(glb_idx,loc_idx++);
253  }
254 
255  //CFinfo << "adding elements " << CFendl;
256  boost_foreach ( const Handle<Entities>& elements, mesh.elements() )
257  {
258  boost_foreach (Uint glb_idx, elements->glb_idx().array())
259  {
260  cf3_assert_desc(to_str(glb_idx)+"<"+to_str(m_start_elem_per_part[PE::Comm::instance().rank()]),glb_idx >= m_start_elem_per_part[PE::Comm::instance().rank()]);
261  cf3_assert_desc(to_str(glb_idx)+">="+to_str(m_end_elem_per_part[PE::Comm::instance().rank()]),glb_idx < m_end_elem_per_part[PE::Comm::instance().rank()]);
262  cf3_assert_desc(to_str(glb_idx)+">="+to_str(m_end_id_per_part[PE::Comm::instance().rank()]),glb_idx < m_end_id_per_part[PE::Comm::instance().rank()]);
263  m_global_to_local->push_back(glb_idx,loc_idx++);
264  //CFinfo << " adding element with glb " << glb_idx << CFendl;
265  }
266  }
267 
268  m_global_to_local->sort_keys();
269 }

334 void flex_all_to_all(const std::vector<PE::Buffer>& send, PE::Buffer& recv)
335 {
336  std::vector<int> send_strides(send.size());
337  std::vector<int> send_displs(send.size());
338  for (Uint i=0; i<send.size(); ++i)
339  send_strides[i] = send[i].size();
340 
341  if (send.size()) send_displs[0] = 0;
342  for (Uint i=1; i<send.size(); ++i)
343  send_displs[i] = send_displs[i-1] + send_strides[i-1];
344 
345  PE::Buffer send_linear;
346 
347  send_linear.reserve(send_displs.back()+send_strides.back());
348  for (Uint i=0; i<send.size(); ++i)
349  send_linear.pack(send[i].buffer(),send[i].size());
350 
351  std::vector<int> recv_strides(PE::Comm::instance().size());
352  std::vector<int> recv_displs(PE::Comm::instance().size());
353  PE::Comm::instance().all_to_all(send_strides,recv_strides);
354  if (recv_displs.size()) recv_displs[0] = 0;
355  for (Uint i=1; i<PE::Comm::instance().size(); ++i)
356  recv_displs[i] = recv_displs[i-1] + recv_strides[i-1];
357  recv.reset();
358  recv.resize(recv_displs.back()+recv_strides.back());
359  MPI_CHECK_RESULT(MPI_Alltoallv, ((void*)send_linear.begin(), &send_strides[0], &send_displs[0], MPI_PACKED, (void*)recv.begin(), &recv_strides[0], &recv_displs[0], MPI_PACKED, PE::Comm::instance().communicator()));
360 }
