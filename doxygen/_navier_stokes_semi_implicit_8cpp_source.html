95  .pretty_name("Time")

102  m_apply_aup = create_component<solver::ActionDirector>("ApplyAup");

110  void execute()
111  {
112  typedef std::pair<Uint,Uint> BlockrowIdxT;
113 
114  a->reset(0.);
115  delta_p_sum->reset(0.);
116  u->assign(*u_lss->solution());
117  p->assign(*p_lss->solution());
118  for(Uint i = 0; i != nb_iterations; ++i)
119  {
120  u_lss->rhs()->reset(0.);
121  // Velocity system: compute delta_a_star
122  m_u_rhs_assembly->execute();
123  if(i == 0) // Apply velocity BC the first inner iteration
124  {
125  u_lss->rhs()->scale(m_time->dt());
126  velocity_bc->execute();
127  // The velocity BC deals with velocity, so we need to write this in terms of acceleration
128  u_lss->rhs()->scale(1./m_time->dt());
129  }
130  else // Zero boundary condition after first pass
131  {
132  BOOST_FOREACH(const BlockrowIdxT& diri_idx, Handle<math::LSS::TrilinosCrsMatrix>(u_lss->matrix())->get_dirichlet_nodes())
133  {
134  u_lss->rhs()->set_value(diri_idx.first, diri_idx.second, 0.);
135  }
136  }
137  u_lss->solution()->reset(0.);
138  CFdebug << "Solving velocity LSS..." << CFendl;
139  solve_u_lss->execute();
140  u_lss->solution()->sync();
141 
142  // Pressure system: compute delta_p
143  p_lss->rhs()->reset(0.);
144  m_p_rhs_assembly->execute();
145  p_lss->solution()->reset(0.);
146  // Apply BC if the first iteration, set RHS to 0 otherwise
147  if(i ==0)
148  {
149  pressure_bc->execute();
150  }
151  else
152  {
153  BOOST_FOREACH(const BlockrowIdxT& diri_idx, Handle<math::LSS::TrilinosCrsMatrix>(p_lss->matrix())->get_dirichlet_nodes())
154  {
155  p_lss->rhs()->set_value(diri_idx.first, diri_idx.second, 0.);
156  }
157  }
158  CFdebug << "Solving pressure LSS..." << CFendl;
159  if(i == 0 && is_not_null(m_p_strategy_first))
160  {
161  p_lss->options().set("solution_strategy_component", m_p_strategy_first);
162  }
163  else if (i > 0 && is_not_null(m_p_strategy_second))
164  {
165  p_lss->options().set("solution_strategy_component", m_p_strategy_second);
166  }
167  solve_p_lss->execute();
168  p_lss->solution()->sync();
169 
170  // Compute delta_a
171  u_lss->rhs()->reset(0.);
172  m_apply_aup->execute(); // Compute Aup*delta_p (stored in u_lss RHS)
173  // delta_a is delta_a_star for the dirichlet nodes
174  BOOST_FOREACH(const BlockrowIdxT& diri_idx, Handle<math::LSS::TrilinosCrsMatrix>(u_lss->matrix())->get_dirichlet_nodes())
175  {
176  u_lss->rhs()->set_value(diri_idx.first, diri_idx.second, 0.);
177  }
178  Thyra::apply(*lumped_m_op, Thyra::NOTRANS, *aup_delta_p, delta_a.ptr(), -1., 1.); // delta_a = delta_a_star - Ml_inv*Aup*delta_p
179  u_lss->solution()->sync(); // delta_a is a link to u_lss->solution(), so it needs a sync after matrix apply
180 
181  const math::LSS::Vector& da = *u_lss->solution();
182  const math::LSS::Vector& dp = *p_lss->solution();
183 
184  a->update(da);
185  u->update(da, m_time->dt());
186  p->update(dp);
187  delta_p_sum->update(dp);
188  }
189 
190  u_lss->solution()->assign(*u);
191  p_lss->solution()->assign(*p);
192  }

241  void execute()
242  {
243  // We use the velocity RHS to constrct the lumped mass matrix during assembly
244  inner_loop->lumped_m_diag->assign(*inner_loop->u_lss->rhs());
245  Handle<math::LSS::ThyraVector> lumped_diag(inner_loop->lumped_m_diag);
246  Teuchos::RCP< Thyra::VectorBase<Real> > lumped_inv_diag = lumped_diag->thyra_vector();
247  Thyra::reciprocal(*lumped_inv_diag, lumped_inv_diag.ptr());
248  // Construct the diagonal op
249  inner_loop->lumped_m_op = Thyra::diagonal(lumped_inv_diag);
250  inner_loop->lumped_m_diag->sync();
251 
252  // Set up pressure RCG, if needed
253  if(is_not_null(inner_loop->m_p_strategy_first) && is_not_null(inner_loop->m_p_strategy_second))
254  {
255  inner_loop->m_p_strategy_first->set_matrix(inner_loop->p_lss->matrix());
256  inner_loop->m_p_strategy_first->set_solution(inner_loop->p_lss->solution());
257  inner_loop->m_p_strategy_first->set_rhs(inner_loop->p_lss->rhs());
258  inner_loop->m_p_strategy_first->mark_basic();
259 
260  inner_loop->m_p_strategy_second->set_matrix(inner_loop->p_lss->matrix());
261  inner_loop->m_p_strategy_second->set_solution(inner_loop->p_lss->solution());
262  inner_loop->m_p_strategy_second->set_rhs(inner_loop->p_lss->rhs());
263  inner_loop->m_p_strategy_second->mark_basic();
264  }
265  }
266 
267  Handle<InnerLoop> inner_loop;
