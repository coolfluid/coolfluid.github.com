110 void BulkVelocity::execute()
111 {
112  m_integral_value = 0.;
113  m_area = 0.;
114 
115  if(m_loop_regions.empty())
116  {
117  CFwarn << "BulkVelocity has no regions" << CFendl;
118  return;
119  }
120 
121  mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(*m_loop_regions.front());
122 
123  const std::string elems_name = "cf3.mesh.LagrangeP0";
124  Handle<mesh::Dictionary> elems_P0_handle(mesh.get_child(elems_name));
125  mesh::Dictionary& elems_P0 = is_null(elems_P0_handle) ? mesh.create_discontinuous_space(elems_name,"cf3.mesh.LagrangeP0") : *elems_P0_handle;
126  if(!elems_P0.has_tag("ufem_dict"))
127  elems_P0.add_tag("ufem_dict");
128 
129  Handle<mesh::Field> is_local_element(elems_P0.get_child("surface_integrator"));
130  if(is_null(is_local_element))
131  {
132  is_local_element = elems_P0.create_field("surface_integrator", "is_local_element").handle<mesh::Field>();
133  is_local_element->add_tag("surface_integrator");
134  }
135  if(!is_local_element->has_tag("local_element_initialized"))
136  {
137  const Uint nb_elems = is_local_element->size();
138  for(Uint i = 0; i != nb_elems; ++i)
139  {
140  is_local_element->array()[i][0] = !elems_P0.is_ghost(i);
141  }
142  }
143 
144  solver::actions::Proto::ProtoAction::execute();
145 
146  //TODO: Stop this from counting overlapping faces twice
147  Real global_integral = m_integral_value;
148  Real global_area = m_area;
149 
150  if(common::PE::Comm::instance().is_active())
151  {
152  common::PE::Comm::instance().all_reduce(common::PE::plus(), &m_integral_value, 1, &global_integral);
153  common::PE::Comm::instance().all_reduce(common::PE::plus(), &m_area, 1, &global_area);
154  }
155  m_result = global_integral / global_area;
156 
157  m_changing_result = true;
158  options().set("result", m_result);
159  m_changing_result = false;
160 
161  if(is_not_null(m_history))
162  {
163  m_history->set("bulk_velocity", m_result);
164  m_history->save_entry();
165  }
166 }
