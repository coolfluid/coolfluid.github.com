104 void TwoPointCorrelation::execute()
105 {
106  setup();
107 
108  const Uint nb_used_nodes = m_used_node_lids.size();
109 
110  common::PE::Comm& comm = common::PE::Comm::instance();
111  if(comm.is_active() && comm.rank() == m_root)
112  {
113  const Uint nb_x_gids = m_x_positions.size();
114  const Uint nb_y_gids = m_y_positions.size();
115  for(Uint i = 0; i != nb_used_nodes; ++i)
116  {
117  const Uint gid = m_used_node_y_gids[i]*nb_x_gids+m_used_node_x_gids[i];
118  m_sampled_values[gid] = m_field->array()[m_used_node_lids[i]];
119  }
120  }
121  else
122  {
123  for(Uint i = 0; i != nb_used_nodes; ++i)
124  {
125  m_sampled_values[i] = m_field->array()[m_used_node_lids[i]];
126  }
127  }
128 
129  if(comm.is_active())
130  m_comm_pattern->synchronize("samples");
131 
132  if(!comm.is_active() || comm.rank() == m_root)
133  {
134  const Uint nb_x_gids = m_x_positions.size();
135  const Uint nb_y_gids = m_y_positions.size();
136 
137  const RealMatrix old_x_sum = m_x_corr*static_cast<Real>(m_count);
138  const RealMatrix old_y_sum = m_y_corr*static_cast<Real>(m_count);
139  m_x_corr.setZero();
140  m_y_corr.setZero();
141 
142  const Uint dim = m_field->row_size();
143 
144  for(Uint i = 0; i != nb_x_gids; ++i)
145  {
146  const RealRowVector y_ref = Eigen::Map<RealRowVector const>(&m_sampled_values[i][0], dim);
147  for(Uint j = 0; j != nb_y_gids; ++j)
148  {
149  const Uint x_ref_gid = nb_x_gids*j;
150  Eigen::Map<RealRowVector const> mapped_x_ref(&m_sampled_values[x_ref_gid][0], dim);
151  Eigen::Map<RealRowVector const> mapped_val(&m_sampled_values[x_ref_gid + i][0], dim);
152  m_x_corr.row(i).array() += mapped_x_ref.array() * mapped_val.array();
153  m_y_corr.row(j).array() += y_ref.array() * mapped_val.array();
154  }
155  }
156 
157  ++m_count;
158 
159  m_x_corr = (m_x_corr/static_cast<Real>(nb_y_gids) + old_x_sum) / static_cast<Real>(m_count);
160  m_y_corr = (m_y_corr/static_cast<Real>(nb_x_gids) + old_y_sum) / static_cast<Real>(m_count);
161 
162  if(m_count % m_interval == 0)
163  {
164  const Uint normal = options().value<Uint>("normal");
165  const Uint x_direction = (normal+1) % 3;
166  const Uint y_direction = (normal+2) % 3;
167  const Real coord = options().value<Real>("coordinate");
168 
169  const common::URI original_uri = options().value<common::URI>("file");
170  std::string rewritten_path = original_uri.path();
171  boost::algorithm::replace_all(rewritten_path, "{iteration}", common::to_str(m_count));
172 
173  boost::filesystem::fstream file(rewritten_path, std::ios::out);
174  file << "# Autocorrelation at level " << coord << " in direction " << x_direction << " for field " << m_field->descriptor().description() << "\n";
175  for(Uint i = 0; i != nb_x_gids; ++i)
176  {
177  file << m_x_positions[i];
178  for(Uint j = 0; j != dim; ++j)
179  file << "," << common::to_str(m_x_corr(i,j));
180  file << "\n";
181  }
182  file << "# Autocorrelation at level " << coord << " in direction " << y_direction << " for field " << m_field->descriptor().description() << "\n";
183  for(Uint i = 0; i != nb_y_gids; ++i)
184  {
185  file << m_y_positions[i];
186  for(Uint j = 0; j != dim; ++j)
187  file << "," << common::to_str(m_y_corr(i,j));
188  file << "\n";
189  }
190  file.close();
191  }
192  }

201 void TwoPointCorrelation::setup()
202 {
203  if(is_not_null(m_field))
204  return;
205 
206  if(is_not_null(m_comm_pattern))
207  remove_component(*m_comm_pattern);
208 
209  m_field = options().value< Handle<mesh::Field> >("field");
210  if(is_null(m_field))
211  throw common::SetupError(FromHere(), "No field configured for " + uri().path());
212 
213  const mesh::Dictionary& dict = m_field->dict();
214  const Uint nb_nodes = dict.size();
215  const mesh::Field& coords = dict.coordinates();
216 
217  if(coords.row_size() != 3)
218  {
219  throw common::SetupError(FromHere(), "TwoPointCorreleation must be used on a 3D problem");
220  }
221 
222  const Uint normal = options().value<Uint>("normal");
223  if(normal >= coords.row_size())
224  throw common::SetupError(FromHere(), "normal " + common::to_str(normal) + " is not allowed for mesh of dimension " + common::to_str(coords.row_size()));
225 
226  const Uint x_direction = (normal+1) % 3;
227  const Uint y_direction = (normal+2) % 3;
228 
229  const Real coordinate = options().value<Real>("coordinate");
230 
231  const Real threshold = options().value<Real>("threshold");
232 
233  std::set<Real, detail_twopoint::threshold_compare> unique_x_coords((detail_twopoint::threshold_compare(threshold)));
234  std::set<Real, detail_twopoint::threshold_compare> unique_y_coords((detail_twopoint::threshold_compare(threshold)));
235  Uint node_counter = 0;
236  for(Uint node_idx = 0; node_idx != nb_nodes; ++node_idx )
237  {
238  if(!dict.is_ghost( node_idx ) && ::fabs(coords[node_idx][normal] - coordinate) < threshold)
239  {
240  unique_x_coords.insert(coords[node_idx][x_direction]);
241  unique_y_coords.insert(coords[node_idx][y_direction]);
242  ++node_counter;
243  }
244  }
245  const Uint nb_used_nodes = node_counter;
246 
247  common::PE::Comm& comm = common::PE::Comm::instance();
248 
249  if(comm.is_active())
250  {
251  std::vector<Real> my_unique_x_coords(unique_x_coords.begin(), unique_x_coords.end());
252  std::vector<Real> my_unique_y_coords(unique_y_coords.begin(), unique_y_coords.end());
253  std::vector< std::vector<Real> > gathered_x_coords, gathered_y_coords;
254  comm.all_gather(my_unique_x_coords, gathered_x_coords);
255  comm.all_gather(my_unique_y_coords, gathered_y_coords);
256  BOOST_FOREACH(const std::vector<Real>& vec, gathered_x_coords)
257  {
258  unique_x_coords.insert(vec.begin(), vec.end());
259  }
260  BOOST_FOREACH(const std::vector<Real>& vec, gathered_y_coords)
261  {
262  unique_y_coords.insert(vec.begin(), vec.end());
263  }
264  }
265 
266  std::map<Real, Uint, detail_twopoint::threshold_compare> x_coords_map((detail_twopoint::threshold_compare(threshold)));
267  std::map<Real, Uint, detail_twopoint::threshold_compare> y_coords_map((detail_twopoint::threshold_compare(threshold)));
268  Uint coord_x_gid = 0;
269  Uint coord_y_gid = 0;
270  BOOST_FOREACH(const Real coord, unique_x_coords)
271  {
272  x_coords_map[coord] = coord_x_gid++;
273  }
274  BOOST_FOREACH(const Real coord, unique_y_coords)
275  {
276  y_coords_map[coord] = coord_y_gid++;
277  }
278 
279  m_x_positions.assign(unique_x_coords.begin(), unique_x_coords.end());
280  m_y_positions.assign(unique_y_coords.begin(), unique_y_coords.end());
281 
282  CFinfo << "Found " << m_x_positions.size() << "x" << m_y_positions.size() << " unique coordinates in direction normal to " << normal << CFendl;
283 
284  m_used_node_lids.clear();
285  m_used_node_lids.reserve(nb_used_nodes);
286  m_used_node_x_gids.resize(nb_used_nodes); m_used_node_y_gids.resize(nb_used_nodes);
287  for(Uint node_idx = 0; node_idx != nb_nodes; ++node_idx )
288  {
289  if(!dict.is_ghost( node_idx ) && ::fabs(coords[node_idx][normal] - coordinate) < threshold)
290  {
291  cf3_assert(x_coords_map.find(coords[node_idx][x_direction]) != x_coords_map.end());
292  cf3_assert(y_coords_map.find(coords[node_idx][y_direction]) != y_coords_map.end());
293 
294  m_used_node_x_gids[m_used_node_lids.size()] = x_coords_map[coords[node_idx][x_direction]];
295  m_used_node_y_gids[m_used_node_lids.size()] = y_coords_map[coords[node_idx][y_direction]];
296  m_used_node_lids.push_back(node_idx);
297  }
298  }
299 
300  const Uint nb_x_gids = m_x_positions.size();
301  const Uint nb_y_gids = m_y_positions.size();
302 
303 
304 
305  if(comm.is_active())
306  {
307  m_root = 0;
308  std::vector< std::vector<Uint> > send_gids, recv_gids;
309  send_gids.resize(comm.size());
310  send_gids[m_root].reserve(nb_used_nodes);
311  for(Uint i = 0; i != nb_used_nodes; ++i)
312  {
313  send_gids[m_root].push_back(m_used_node_y_gids[i]*nb_x_gids+m_used_node_x_gids[i]);
314  }
315 
316  comm.all_to_all(send_gids, recv_gids);
317 
318  if(comm.rank() == m_root)
319  {
320  m_ranks.resize(nb_x_gids*nb_y_gids);
321  m_gids.resize(nb_x_gids*nb_y_gids);
322  for(Uint rank = 0; rank != comm.size(); ++rank)
323  {
324  BOOST_FOREACH(const Uint gid, recv_gids[rank])
325  {
326  m_gids[gid] = gid;
327  m_ranks[gid] = rank;
328  }
329  }
330 
331  m_sampled_values.resize(boost::extents[nb_x_gids*nb_y_gids][m_field->row_size()]);
332  m_x_corr.resize(nb_x_gids, m_field->row_size()); m_x_corr.setZero();
333  m_y_corr.resize(nb_y_gids, m_field->row_size()); m_y_corr.setZero();
334  }
335  else
336  {
337  m_gids = send_gids[m_root];
338  m_ranks.assign(nb_used_nodes, comm.rank());
339  m_sampled_values.resize(boost::extents[nb_used_nodes][m_field->row_size()]);
340  }
341 
342  m_comm_pattern = create_component<common::PE::CommPattern>("CommPattern");
343  m_comm_pattern->insert("gid", m_gids, 1, false);
344  m_comm_pattern->setup(Handle<common::PE::CommWrapper>(m_comm_pattern->get_child("gid")), m_ranks);
345  m_comm_pattern->insert("samples", m_sampled_values);
346  }
347  else
348  {
349  m_x_corr.resize(nb_x_gids, m_field->row_size()); m_x_corr.setZero();
350  m_y_corr.resize(nb_y_gids, m_field->row_size()); m_y_corr.setZero();
351  m_sampled_values.resize(boost::extents[nb_x_gids*nb_y_gids][m_field->row_size()]);
352  }
353 }
