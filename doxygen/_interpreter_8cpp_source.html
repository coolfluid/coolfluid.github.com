94 void Interpreter::interpret_alias(std::vector<std::string>& unrecognized_commands)
95 {
96  Uint cnt(0);
97  boost_foreach(std::string& command, unrecognized_commands)
98  {
99  std::string alias_str = command;
100  std::string arg_str = "";
101  alias_str.erase( boost::algorithm::find_first(alias_str," ").begin(),alias_str.end());
102  if (m_alias.find(alias_str) != m_alias.end())
103  {
104  if (alias_str != command)

106  unrecognized_commands.erase(unrecognized_commands.begin()+cnt);

127 std::vector<std::string> Interpreter::split_command_line(const std::string& input)
128 {
129  std::vector<std::string> result;
130 
131  std::string::const_iterator i = input.begin(), e = input.end();
132  for(;i != e; ++i)
133  if (!isspace((unsigned char)*i))
134  break;
135 
136  if (i != e)
137  {
138  std::string current;
139  bool inside_quoted = false;
140  int backslash_count = 0;
141 
142  for(; i != e; ++i)
143  {
144  if (*i == '"')
145  {
146  // '"' preceded by even number (n) of backslashes generates
147  // n/2 backslashes and is a quoted block delimiter
148  if (backslash_count % 2 == 0)
149  {
150  current.append(backslash_count / 2, '\\');
151  inside_quoted = !inside_quoted;
152  // '"' preceded by odd number (n) of backslashes generates
153  // (n-1)/2 backslashes and is literal quote.
154  }
155  else
156  {
157  current.append(backslash_count / 2, '\\');
158  current += '"';
159  }
160  backslash_count = 0;
161  }
162  else if (*i == '\\')
163  {
164  ++backslash_count;
165  } else
166  {
167  // Not quote or backslash. All accumulated backslashes should be
168  // added
169  if (backslash_count)
170  {
171  current.append(backslash_count, '\\');
172  backslash_count = 0;
173  }
174  if (isspace((unsigned char)*i) && !inside_quoted)
175  {
176  // Space outside quoted section terminate the current argument
177  result.push_back(current);
178  current.resize(0);
179  for(;i != e && isspace((unsigned char)*i); ++i)
180  ;
181  --i;
182  }
183  else
184  {
185  current += *i;
186  }
187  }
188  }
189 
190  // If we have trailing backslashes, add them
191  if (backslash_count)
192  current.append(backslash_count, '\\');
193 
194  // If we have non-empty 'current' or we're still in quoted
195  // section (even if 'current' is empty), add the last token.
196  if (!current.empty() || inside_quoted)
197  result.push_back(current);
198  }
199  return result;
200 }

204 void Interpreter::handle_read_line(std::string line)
205 {
206 
207  line = BasicCommands::filter_env_vars(line);
208 
209  std::vector<std::string> args;
210 
211  args = split_command_line(std::string("--") + line);
212 
213  try
214  {
215  parsed_options parsed = command_line_parser(args).options(m_desc).allow_unregistered().run();
216  variables_map vm;
217  store( parsed , vm);
218  notify(vm);
219  std::vector<std::string> unrecognized_commands_no_args = collect_unrecognized( parsed.options , include_positional);
220  std::vector<std::string> unrecognized_commands;
221  boost_foreach(std::string& command, unrecognized_commands_no_args)
222  {
223  if ( std::string(command.begin(),command.begin()+2) == "--")
224  unrecognized_commands.push_back(std::string(command.begin()+2,command.end()));
225  else if ( std::string(command.begin(),command.begin()+1) == "-")
226  throw boost::program_options::unknown_option(" argument cannot start with '-'");
227  else
228  unrecognized_commands.back() = unrecognized_commands.back() + " " + command;
229  }
230  boost_foreach(const unrecognized_commands_handler_t& handle, m_handle_unrecognized_commands )
231  {
232  if (unrecognized_commands.size())
233  {
234  handle(unrecognized_commands);
235  }
236  }
237  if (unrecognized_commands.size())
238  throw boost::program_options::unknown_option("");
239  m_history.push_back(line);
240  }
241  catch (boost::program_options::unknown_option &e)
242  {
243  CFerror << "error: " << e.what() << CFendl;
244  }
245  catch (boost::program_options::invalid_command_line_syntax &e)
246  {
247  CFerror << "error: " << e.what() << CFendl;

413 void Interpreter::interpret(std::istream &input_stream)
414 {
415  std::string command;
416 
417  write_prompt();
418 
419  while (std::getline(input_stream, command, '\n'))
420  {
421  bool multi_line = boost::algorithm::iends_with(command, "\\");
422  while ( multi_line )
423  {
424  boost::algorithm::erase_last(command,"\\");
425  std::string more;
426  std::getline(input_stream, more, '\n');
427  command += more;
428  multi_line = boost::algorithm::iends_with(command, "\\");
429  }
430 
431  handle_read_line(command);
432  write_prompt();
433  }
