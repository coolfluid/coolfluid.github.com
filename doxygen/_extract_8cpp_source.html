87 void Extract::execute()
88 {
89 
90  Mesh& mesh = *m_mesh;
91 
92 
93  // Storage of regions to keep
94  std::set<std::string> keep_region_paths;
95 
96  std::vector<std::string> args = options().value< std::vector<std::string> >("regions");
97 
98  // special cases "volumes" and "surfaces" as arg
99  BOOST_FOREACH(const std::string region_name, args)
100  {
101  if (boost::regex_match(region_name,boost::regex("[Ss]urface(s)?"))) // Surface, Surfaces, surface, surfaces
102  {
103  BOOST_FOREACH( const Elements& elements, find_components_recursively_with_filter<Elements>(mesh,IsElementsSurface()))
104  {
105  keep_region_paths.insert(elements.parent()->uri().path());
106  }
107  }
108  else if (boost::regex_match(region_name,boost::regex("[Vv]olume(s)?"))) // Volume, Volumes, volume, volumes
109  {
110  BOOST_FOREACH( const Elements& elements, find_components_recursively_with_filter<Elements>(mesh,IsElementsVolume()))
111  {
112  keep_region_paths.insert(elements.parent()->uri().path());
113  }
114  }
115  }
116 
117  // For every region, see if its path matches the regex, and store it in a list
118  BOOST_FOREACH( Region& region, find_components_recursively<Region>(mesh))
119  {
120  // see if this region has to be deleted.
121  bool found = false;
122  BOOST_FOREACH(const std::string& expression, args)
123  {
124  if (boost::regex_match(region.uri().path(),boost::regex(".*"+expression+".*")))
125  {
126  found = true;
127  keep_region_paths.insert(region.uri().path());
128  Handle<Component> parent = region.parent();
129  while ( parent != mesh.handle() )
130  {
131  keep_region_paths.insert(parent->uri().path());
132  parent = parent->parent();
133  }
134  break;
135  }
136  }
137  }
138 
139  CFdebug << "regions to keep:" << CFendl;
140  BOOST_FOREACH(const URI& region_path, keep_region_paths)
141  {
142  CFdebug << " - " << region_path << CFendl;
143  }
144 
145  // Remove regions whose name doesn't appear in the parsed list "keep_region"
146  BOOST_FOREACH( Region& region, find_components_recursively<Region>(mesh))
147  {
148  bool found = (std::find(keep_region_paths.begin(),keep_region_paths.end(),region.uri().path()) != keep_region_paths.end());
149  if (!found)
150  {
151  if (region.parent())
152  region.parent()->remove_component(region.name());
153  }
154  }
