37 PoissonProto::PoissonProto ( const std::string& name ) : LSSAction ( name )
38 {
39  // This determines the name of the field that will be used to store the solution
40  set_solution_tag("poisson_solution");
41 
42  // Create action components that wil be executed in the order they are created here:
43  // 1. Set the linear system matrix and vectors to zero
44  create_component<math::LSS::ZeroLSS>("ZeroLSS");
45 
46  // 2. Assemble the system matrix and RHS using a proto expression
47  Handle<ProtoAction> assembly = create_component<ProtoAction>("Assembly");
48 
49  // The unknown function. The first template argument is a constant to distinguish each variable at compile time
50  FieldVariable<0, ScalarField> u("u", solution_tag());
51 
52  // The source term, to be set at runtime using an initial condition
53  FieldVariable<1, ScalarField> f("f", "source_term");
54 
55  // The expression itself
56  assembly->set_expression(elements_expression
57  (
58  mesh::LagrangeP1::CellTypes(), // All P1 volume cell types are supported
59  group
60  (
61  _A = _0, _a = _0,// The element matrix and RHS vector, initialized to 0
62  element_quadrature // Integration over the element
63  (
64  _A(u) += transpose(nabla(u)) * nabla(u),
65  _a[u] += transpose(N(u))*f
66  ),
67  system_matrix += _A, // Assemble into the global linear system
68  system_rhs += _a
69  )
70  ));
71
