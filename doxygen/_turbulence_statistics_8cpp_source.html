42  {

108 {
109  setup();

120  if(m_dim == 2)
121  {
122  for(Uint i = 0; i != nb_nodes; ++i)
123  {
124  const mesh::Field::ConstRow velocity = velocity_array[used_nodes_list[i]];
125  mesh::Field::Row means = means_array[used_nodes_list[i]];
126  const Real u = velocity[XX+m_velocity_field_offset]; const Real v = velocity[YY+m_velocity_field_offset];
127  const Real p = pressure_array[used_nodes_list[i]][m_pressure_field_offset];
128 
129  detail::update_mean(means[0], u, m_count);

131  detail::update_mean(means[2], u*u, m_count);
132  detail::update_mean(means[3], v*v, m_count);
133  detail::update_mean(means[4], u*v, m_count);

136  }

139  {
140  const Uint probe_begin = my_probe_idx*stride;
141  const mesh::Field::ConstRow velocity = velocity_array[m_probe_nodes[my_probe_idx]];
142  const Real u = velocity[XX+m_velocity_field_offset]; const Real v = velocity[YY+m_velocity_field_offset];
143 
144  m_means[probe_begin ](u);
145  m_means[probe_begin+1](v);
146  m_means[probe_begin+2](u*u);
147  m_means[probe_begin+3](v*v);
148  m_means[probe_begin+4](u*v);
149 
150  m_rolling_means[probe_begin ](u);
151  m_rolling_means[probe_begin+1](v);
152  m_rolling_means[probe_begin+2](u*u);
153  m_rolling_means[probe_begin+3](v*v);
154  m_rolling_means[probe_begin+4](u*v);
155  }
156  }
157  else if(m_dim == 3)
158  {
159  for(Uint i = 0; i != nb_nodes; ++i)
160  {
161  const mesh::Field::ConstRow velocity = velocity_array[used_nodes_list[i]];
162  mesh::Field::Row means = means_array[used_nodes_list[i]];
163  const Real u = velocity[XX+m_velocity_field_offset]; const Real v = velocity[YY+m_velocity_field_offset]; const Real w = velocity[ZZ+m_velocity_field_offset];
164  const Real p = pressure_array[used_nodes_list[i]][m_pressure_field_offset];
165 
166  detail::update_mean(means[0], u, m_count);
167  detail::update_mean(means[1], v, m_count);
168  detail::update_mean(means[2], w, m_count);
169  detail::update_mean(means[3], u*u, m_count);
170  detail::update_mean(means[4], v*v, m_count);
171  detail::update_mean(means[5], w*w, m_count);
172  detail::update_mean(means[6], u*v, m_count);
173  detail::update_mean(means[7], u*w, m_count);
174  detail::update_mean(means[8], v*w, m_count);
175  detail::update_mean(means[9], p, m_count);
176  detail::update_mean(means[10], p*p, m_count);
177  }
178

181  const Uint probe_begin = my_probe_idx*stride;

189  m_means[probe_begin+4](v*v);
190  m_means[probe_begin+5](w*w);
191  m_means[probe_begin+6](u*v);
192  m_means[probe_begin+7](u*w);
193  m_means[probe_begin+8](v*w);

196  m_rolling_means[probe_begin+1](v);

199  m_rolling_means[probe_begin+4](v*v);
200  m_rolling_means[probe_begin+5](w*w);
201  m_rolling_means[probe_begin+6](u*v);
202  m_rolling_means[probe_begin+7](u*w);
203  m_rolling_means[probe_begin+8](v*w);
204  }
205  }
206 
207  // Write the probe data
208  for(Uint my_probe_idx = 0; my_probe_idx != nb_my_probes; ++my_probe_idx)
209  {
210  const Uint probe_begin = my_probe_idx*stride;
211  const Uint probe_end = probe_begin + stride;
212  boost::filesystem::fstream& file = *m_probe_files[my_probe_idx];

219

225  file << "\n";
226  }

267 void TurbulenceStatistics::setup()
268 {
269  if(!m_options_changed)
270  return;
271  m_options_changed = false;
272 
273  m_used_nodes.reset();
274 
275  Handle<mesh::Region> region = options().value< Handle<mesh::Region> >("region");
276  if(is_null(region))
277  {
278  return;
279  }
280 
281  const std::string velocity_var_name = options().value<std::string>("velocity_variable_name");
282  const std::string pressure_var_name = options().value<std::string>("pressure_variable_name");
283  m_velocity_field.reset();
284  m_pressure_field.reset();
285 
286  const mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(*region);
287  Handle<mesh::Dictionary> dictionary;
288  BOOST_FOREACH(const Handle<mesh::Dictionary>& dict, mesh.dictionaries())
289  {
290  BOOST_FOREACH(const Handle<mesh::Field>& field, dict->fields())
291  {
292  if(field->has_variable( velocity_var_name ))
293  {
294  if(is_not_null(m_velocity_field))
295  throw common::SetupError(FromHere(), "There are two fields that contain the variable " + velocity_var_name );
296 
297  m_velocity_field = field;
298  dictionary = dict;
299  }
300  }
301  }
302 
303  if(is_null(m_velocity_field))
304  throw common::SetupError(FromHere(), "There is no field with the variable " + velocity_var_name );
305 
306  m_dim = mesh.dimension();
307  if(m_dim == 1)
308  {
309  throw common::SetupError(FromHere(), "TurbulenceStatistics are not supported in 1D");
310  }
311  cf3_assert(m_velocity_field->var_length(velocity_var_name) == m_dim);
312  m_velocity_field_offset = m_velocity_field->var_offset( velocity_var_name );
313 
314  BOOST_FOREACH(const Handle<mesh::Field>& field, dictionary->fields())
315  {
316  if(field->has_variable( pressure_var_name ))
317  {
318  m_pressure_field = field;
319  }
320  }
321 
322  if(is_null(m_pressure_field))
323  throw common::SetupError(FromHere(), "There is no field with the variable " + pressure_var_name );
324 
325  m_pressure_field_offset = m_pressure_field->var_offset( pressure_var_name );
326 
327  const mesh::Field& coords = dictionary->coordinates();
328 
329  common::PE::Comm& comm = common::PE::Comm::instance();
330 
331  m_used_nodes = mesh::build_used_nodes_list(*region, *dictionary, true, false);
332  const int nb_probes = m_probe_locations.size();
333  std::vector<int> my_probes_found(nb_probes, 0);
334  m_probe_nodes.clear();
335  m_probe_indices.clear();
336  const common::List<Uint>::ListT& used_nodes_list = m_used_nodes->array();
337  BOOST_FOREACH(const Uint node_idx, used_nodes_list)
338  {
339  if(dictionary->is_ghost(node_idx))
340  continue;
341  for(int probe_idx = 0; probe_idx != nb_probes; ++probe_idx)
342  {
343  if(m_probe_locations[probe_idx].size() != m_dim)
344  {
345  throw common::SetupError(FromHere(), "Probe coordinates of dimension " + common::to_str(m_probe_locations[probe_idx].size()) + " do not match dimension " + common::to_str(m_dim));
346  }
347  if(((RealVector::Map(&coords[node_idx][0], m_dim) - m_probe_locations[probe_idx]).array().abs() < 1e-10).all())
348  {
349  m_probe_nodes.push_back(node_idx);
350  m_probe_indices.push_back(probe_idx);
351  my_probes_found[probe_idx] = 1;
352  }
353  }
354  }
355 
356  std::vector<int> global_probes_found(nb_probes);
357  if(comm.is_active() && nb_probes > 0)
358  {
359  comm.all_reduce(common::PE::plus(), my_probes_found, global_probes_found);
360  }
361  else
362  {
363  global_probes_found = my_probes_found;
364  }
365  for(int probe_idx = 0; probe_idx != nb_probes; ++probe_idx)
366  {
367  if(global_probes_found[probe_idx] == 0)
368  throw common::SetupError(FromHere(), "Probe " + common::to_str(probe_idx) + " has no matching node");
369  if(global_probes_found[probe_idx] > 1)
370  throw common::SetupError(FromHere(), "Probe " + common::to_str(probe_idx) + " was found on " + common::to_str(global_probes_found[probe_idx]) + " CPUs");
371  }
372 
373  const common::URI original_uri = options().value<common::URI>("file");
374 
375  // Init probe files
376  m_probe_files.clear();
377  const Uint nb_my_probes = m_probe_nodes.size();
378  for(Uint my_idx = 0; my_idx != nb_my_probes; ++my_idx)
379  {
380  const Uint probe_idx = m_probe_indices[my_idx];
381  std::string probe_path = (original_uri.base_path() / (original_uri.base_name() + "-probe-" + common::to_str(probe_idx) + original_uri.extension())).path();
382 
383  m_probe_files.push_back(boost::make_shared<boost::filesystem::fstream>(probe_path, std::ios_base::out));
384  boost::filesystem::fstream& file = *m_probe_files.back();
385  if(!file)
386  throw common::FileSystemError(FromHere(), "Failed to open file " + probe_path);
387 
388  file << "# Probe data for probe " << probe_idx << " at point " << m_probe_locations[probe_idx].transpose() << "\n";
389  if(m_dim == 2)
390  {
391  file << "# U, V, uu, vv, uv, U_rolling, V_rolling, uu_rolling, vv_rolling, uv_rolling\n";
392  }
393  else if(m_dim == 3)
394  {
395  file << "# U, V, W, uu, vv, ww, uv, uw, vw, U_rolling, V_rolling, W_rolling, uu_rolling, vv_rolling, ww_rolling, uv_rolling, uw_rolling, vw_rolling\n";
396  }
397  }
398 
399  // Create a field for the statistics data
400  m_statistics_field = Handle<mesh::Field>(dictionary->get_child("turbulence_statistics"));
401  if(is_null(m_statistics_field))
402  {
403  if(m_dim == 2)
404  {
405  m_statistics_field = dictionary->create_field("turbulence_statistics", "V[vector],uu,vv,uv,p,pp").handle<mesh::Field>();
406  }
407  else if(m_dim == 3)
408  {
409  m_statistics_field = dictionary->create_field("turbulence_statistics", "V[vector],uu,vv,ww,uv,uw,vw,p,pp").handle<mesh::Field>();
410  }
411  m_statistics_field->add_tag("turbulence_statistics");
412  }
413 
414  // Reset statistics without changing m_count
415  const Uint nb_accs = (2.*m_dim + m_dim-1 + m_dim-2)*m_probe_nodes.size();
416  m_means.assign(nb_accs, MeanAccT());
417  m_rolling_means.assign(nb_accs, RollingAccT(boost::accumulators::tag::rolling_window::window_size = options().value<Uint>("rolling_window")));
418 }
