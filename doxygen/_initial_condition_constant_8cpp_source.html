67 void InitialConditionConstant::trigger_tag()
68 {
69  const std::string field_tag = options().value<std::string>("field_tag");
70  if(field_tag.empty())
71  return;
72 
73  // Wipe the old options
74  BOOST_FOREACH(const std::string& opt_name, m_variable_options)
75  {
76  options().erase(opt_name);
77  }
78 
79  m_variable_options.clear();
80 
81  VariablesDescriptor& descriptor = find_component_with_tag<VariablesDescriptor>(physical_model().variable_manager(), field_tag);
82  const Uint nb_vars = descriptor.nb_vars();
83  for(Uint i = 0; i != nb_vars; ++i)
84  {
85  const std::string& var_name = descriptor.user_variable_name(i);
86  m_variable_options.push_back(var_name);
87  if(descriptor.var_length(i) == 1)
88  {
89  options().add(var_name, 0.).description(std::string("Initial condition for variable " + var_name)).mark_basic();
90  }
91  else
92  {
93  options().add(var_name, std::vector<Real>()).description(std::string("Initial condition for variable " + var_name)).mark_basic();
94  }
95  }

98 void InitialConditionConstant::execute()
99 {
100  if(m_loop_regions.empty())
101  CFwarn << "No regions to loop over for action " << uri().string() << CFendl;
102 
103  const std::string field_tag = options().value<std::string>("field_tag");
104  VariablesDescriptor& descriptor = find_component_with_tag<VariablesDescriptor>(physical_model().variable_manager(), field_tag);
105 
106  // Construct a vector with the values to use
107  std::vector<Real> values; values.reserve(descriptor.size());
108  const Uint nb_vars = m_variable_options.size();
109  CFdebug << "Using descriptor " << descriptor.description() << " for InitialConditionConstant" << CFendl;
110  cf3_assert(nb_vars == descriptor.nb_vars());
111 
112  for(Uint i = 0; i != nb_vars; ++i)
113  {
114  const Option& opt = options().option(m_variable_options[i]);
115  if(descriptor.var_length(i) == 1) // scalar
116  {
117  values.push_back(opt.value<Real>());
118  }
119  else
120  {
121  std::vector<Real> val = opt.value< std::vector<Real> >();
122  // By default, use zero
123  if(val.empty())
124  val.resize(descriptor.var_length(i), 0.);
125  cf3_assert(val.size() == descriptor.var_length(i));
126  values.insert(values.end(), val.begin(), val.end());
127  }
128  }
129  BOOST_FOREACH(const Handle<Region>& region, m_loop_regions)
130  {
131  CFdebug << " Action " << name() << ": running over region " << region->uri().path() << CFendl;
132 
133  // Build a list of used entities
134  mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(*region);
135  std::vector< Handle<mesh::Entities const> > used_entities;
136  BOOST_FOREACH(const mesh::Entities& entities, common::find_components_recursively<mesh::Entities>(*region))
137  {
138  used_entities.push_back(entities.handle<mesh::Entities>());
139  }
140  boost::shared_ptr< common::List<Uint> > used_nodes_ptr = mesh::build_used_nodes_list(used_entities, mesh.geometry_fields(), true);
141 
142  common::List<Uint>& nodes = *used_nodes_ptr;
143 
144  Field& field = find_field(*region, field_tag);
145  BOOST_FOREACH(const Uint node, nodes.array())
146  {
147  field.set_row(node, values);
148  }
149  }
150 }
