168 
169  Eigen::Matrix<Real, 1, nb_nodes> adv;
170  Eigen::Matrix<Real, nb_nodes, nb_nodes> App;
171  const UVecT u_plus_dt_da = u_vec + dt*delta_a;
172  const UVecT a_plus_da = a + delta_a;
173 
174  App.setZero();
175  result.setZero();
176 
177  typedef mesh::Integrators::GaussMappedCoords<IntegralOrder<ElementT>::value, ElementT::shape> GaussT;
178 
179  for(Uint gauss_idx = 0; gauss_idx != GaussT::nb_points; ++gauss_idx)
180  {
181  // This precomputes the required matrix operators
182  u.support().compute_shape_functions(GaussT::instance().coords.col(gauss_idx));
183  u.support().compute_jacobian(GaussT::instance().coords.col(gauss_idx));
184  u.compute_values(GaussT::instance().coords.col(gauss_idx));
185 
186  const Real w = GaussT::instance().weights[gauss_idx] * u.support().jacobian_determinant();
187  const Real tau_w = w*tau_ps;
188 
189  adv = tau_ps*u.eval() * u.nabla(); // advection operator
190  App += tau_w * u.nabla().transpose() * u.nabla();
191  for(Uint i = 0; i != dim; ++i)
192  {
193  const Real a = w*(u.nabla().row(i)*u_plus_dt_da.row(i).transpose())[0];
194  const Real b = w*(adv*u_plus_dt_da.row(i).transpose())[0];
195  const Real c = tau_w*(u.shape_function()*a_plus_da.row(i).transpose())[0];
196 
197  result -= a*(u.shape_function() + adv*0.5).transpose() + (b+c)*u.nabla().row(i).transpose();
198  }
199  }
200 
201  result -= App*p_vec;
202 
203  return result;
204  }

228  const StorageT& operator()(StorageT& result, const UT& u, const NUT& nu_eff, const UVecT1& a_vec, const UVecT2& dt_a_min_u_in, const PVecT& p_plus_dp_in, const Real& tau_su, const Real& tau_bulk) const
229  {
230  typedef typename UT::EtypeT ElementT;
231 
232  static const Uint nb_nodes = ElementT::nb_nodes;
233  static const Uint dim = ElementT::dimension;
234 
235  Eigen::Matrix<Real, 1, nb_nodes> adv;
236  Eigen::Matrix<Real, 1, nb_nodes> N_plus_adv;
237  const Eigen::Matrix<Real, nb_nodes, 1> p_plus_dp = p_plus_dp_in;
238  const UVecT1 dt_a_min_u = dt_a_min_u_in;
239 
240  result.setZero();
241 
242  static const int ideal_order = IntegralOrder<ElementT>::value;
243 
244  if(ideal_order != 2)
245  {
247  typedef mesh::Integrators::GaussMappedCoords<2, ElementT::shape> Gauss2T;
248  for(Uint gauss_idx = 0; gauss_idx != Gauss2T::nb_points; ++gauss_idx)
249  {
250  // This precomputes the required matrix operators
251  u.support().compute_shape_functions(Gauss2T::instance().coords.col(gauss_idx));
252  u.support().compute_jacobian(Gauss2T::instance().coords.col(gauss_idx));
253  u.compute_values(Gauss2T::instance().coords.col(gauss_idx));
254 
255  const Real w = Gauss2T::instance().weights[gauss_idx] * u.support().jacobian_determinant();
256 
257  adv = u.eval() * u.nabla(); // advection operator
258  N_plus_adv = tau_su*adv + u.shape_function();
259 
260  // Compose the bulk and skew symmetric term
261  Real f = 0.;
262  for(Uint i = 0; i != dim; ++i)
263  {
264  f += u.nabla().row(i)*dt_a_min_u.row(i).transpose();
265  }
266  f *= w;
267  const Real f_skew = 0.5*f;
268 
269  for(Uint i = 0; i != dim; ++i)
270  {
271  const Real c = -w*(u.shape_function()*(a_vec.row(i)).transpose())[0];
272  const Real e = w*adv*dt_a_min_u.row(i).transpose();
273 
274  result.template segment<nb_nodes>(i*nb_nodes) += N_plus_adv.transpose() * (c + e + (f_skew * u.eval()[i]));
275  }
276  }
277  }
278 
279  typedef mesh::Integrators::GaussMappedCoords<ideal_order, ElementT::shape> GaussT;
280 
281  for(Uint gauss_idx = 0; gauss_idx != GaussT::nb_points; ++gauss_idx)
282  {
283  // This precomputes the required matrix operators
284  u.support().compute_shape_functions(GaussT::instance().coords.col(gauss_idx));
285  u.support().compute_jacobian(GaussT::instance().coords.col(gauss_idx));
286  u.compute_values(GaussT::instance().coords.col(gauss_idx));
287  nu_eff.compute_values(GaussT::instance().coords.col(gauss_idx));
288 
289  const Real w = GaussT::instance().weights[gauss_idx] * u.support().jacobian_determinant();
290  const Real w_visc = w * nu_eff.eval();
291  const Real tau_su_w = w * tau_su;
292 
293  adv = u.eval() * u.nabla(); // advection operator
294  const Real b = -w*(u.shape_function()*p_plus_dp)[0];
295 
296  // Compose the bulk and skew symmetric term
297  Real f = 0.;
298  for(Uint i = 0; i != dim; ++i)
299  {
300  f += u.nabla().row(i)*dt_a_min_u.row(i).transpose();
301  }
302  f *= w;
303  const Real f_bulk = tau_bulk*f;
304  if(ideal_order == 2)
305  {
306  N_plus_adv = tau_su*adv + u.shape_function();
307  const Real f_skew = 0.5*f;
308  for(Uint i = 0; i != dim; ++i)
309  {
310  const Real a = tau_su_w*(u.nabla().row(i)*p_plus_dp)[0];
311  const Real c = -w*(u.shape_function()*(a_vec.row(i)).transpose())[0];
312  const Eigen::Matrix<Real, dim, 1> d = w_visc*(u.nabla()*dt_a_min_u.row(i).transpose());
313  const Real e = w*adv*dt_a_min_u.row(i).transpose();
314 
315 
316  result.template segment<nb_nodes>(i*nb_nodes) += adv.transpose() * a
317  + u.nabla().row(i).transpose() * (b + f_bulk)
318  + N_plus_adv.transpose() * (c + e + (f_skew * u.eval()[i]))
319  + u.nabla().transpose() * d;
320  }
321  }
322  else
323  {
324  for(Uint i = 0; i != dim; ++i)
325  {
326  const Real a = tau_su_w*(u.nabla().row(i)*p_plus_dp)[0];
327  const Eigen::Matrix<Real, dim, 1> d = w_visc*(u.nabla()*dt_a_min_u.row(i).transpose());
328 
329  result.template segment<nb_nodes>(i*nb_nodes) += adv.transpose() * a
330  + u.nabla().row(i).transpose() * (b + f_bulk)
331  + u.nabla().transpose() * d;
332  }
333  }
334  }
335 
336  return result;
337  }
