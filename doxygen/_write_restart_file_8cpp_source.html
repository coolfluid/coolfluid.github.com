68  throw common::SetupError(FromHere(), "No fields configured");

70  Handle<Time> time = options().value< Handle<Time> >(solver::Tags::time());

74  // Enssure each field is from the same mesh

79  if(parent_mesh != mesh && is_not_null(mesh))
80  throw common::SetupError(FromHere(), "Fields do not belong to the same mesh");
81  mesh = parent_mesh;
82  }
83  cf3_assert(is_not_null(mesh));

88  data_writer->options().set("file", binfile);
