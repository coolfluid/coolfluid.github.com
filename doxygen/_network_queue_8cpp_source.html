127 void NetworkQueue::insert_transaction( const QString & uuid,
128  NetworkQueue::Priority priority )
129 {
130  if ( !m_new_transactions.contains(uuid) )
131  throw ValueNotFound( FromHere(), "No transaction found with UuiD [" +
132  uuid.toStdString() + "].");
133  else
134  {
135  Transaction * transaction = m_new_transactions[uuid];
136 
137  switch ( priority )
138  {
139  case HIGH:
140  m_transactions.prepend( transaction );
141  m_current_index++;
142  break;
143 
144  case MEDIUM:
145 
146  if( m_current_index == -1 )
147  m_transactions.append( transaction );
148  else
149  {
150  m_transactions.insert( m_current_index, transaction );
151  m_current_index++;
152  }
153  break;
154 
155  case LOW:
156  m_transactions.append( transaction );
157  break;
158 
159  case IMMEDIATE:
160  throw BadValue( FromHere(), "Immediate priority cannot be applied to a transaction." );
161  }
162 
163  start(); // run the transaction (ignored if another one is already being executed)
164 
165  SignalFrame args = transaction->actions[0];
166 
167  m_new_transactions.remove( uuid );
168  }
169 }
