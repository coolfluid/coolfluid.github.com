91 void MeshReader::signal_read( SignalArgs& node )
92 {
93  SignalOptions options( node );
94 
95  // get the domain
96  Handle< Component > domain = options.value< Handle<Component> >("domain");
97 
98  std::vector<URI> files = options.value< std::vector<URI> >("files");
99  if (files.empty() && options.check("file") )
100  files.push_back( options.value<URI>("file"));
101 
102  // create a mesh in the domain
103  if( !files.empty() )
104  {
105  Handle<Mesh> mesh = domain->create_component<Mesh>("Mesh");
106 
107  // Get the file paths
108  boost_foreach(const URI& file, files)
109  {
110  // Call the concrete implementation
111  mesh->block_mesh_changed(true);
112  do_read_mesh_into(file, *mesh);
113  mesh->block_mesh_changed(false);
114  }
115  // Raise an event to indicate that a mesh was loaded happened
116  mesh->raise_mesh_loaded();
117 
118  SignalFrame reply = node.create_reply(uri());
119  SignalOptions reply_options(reply);
120  reply_options.add("created_component", mesh->uri());
121  }
122  else
123  {
124  throw BadValue( FromHere(), "No mesh was read because no files were selected." );
125  }
126 }

199 void MeshReader::remove_empty_element_regions(Region& parent_region)
200 {
201  boost_foreach(Elements& region, find_components_recursively<Elements>(parent_region))
202  {
203  // find the empty regions
204  Uint empty_on_this_rank = region.geometry_space().connectivity().array().empty();
205  Uint empty_on_all_ranks = empty_on_this_rank;
206 
208  if (PE::Comm::instance().is_active())
209  PE::Comm::instance().instance().all_reduce( PE::logical_and(), &empty_on_this_rank, 1, &empty_on_all_ranks);
210 
211  if ( empty_on_all_ranks )
212  {
213  boost::shared_ptr< Elements > removed = boost::dynamic_pointer_cast<Elements>(region.parent()->remove_component(region.name()));
214  removed.reset();
215  }
216  }
217 
218  // loop over regions
219  boost_foreach(Region& region, find_components_recursively<Region>(parent_region))
220  {
221  // find the empty regions
222  if ( find_components<Region>(region).empty() && find_components<Elements>(region).empty() )
223  {
224  boost::shared_ptr< Region > removed = boost::dynamic_pointer_cast<Region>(region.parent()->remove_component(region.name()));
225  removed.reset();
226  }
