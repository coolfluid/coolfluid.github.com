63  if(is_null(source_pressures))

65

67  if(is_null(destination_field))

69 
70  if(source_velocities->size() != destination_field->size())
71  throw common::SetupError(FromHere(), "Source field size " + common::to_str(source_velocities->size()) + " does not match target size " + common::to_str(destination_field->size()));
72 
73  const Uint nb_rows = source_velocities->size();
74  const Uint dim = source_velocities->row_size();
75 
76  if(destination_field->row_size() != dim+1)

80  const mesh::Field::ArrayT& source_pressures_array = source_pressures->array();
