72  // ------------------------------

74  // (a) Create bounding box of all coordinates, to define a space
75  // This bounding box is not the one from the mesh, as it doesn't have
76  // to cover the mesh, or can exist in more than one mesh.
77  math::BoundingBox bounding_box;
78  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
79  {
80  Entities& entities = *entities_handle;
81  const ShapeFunction& shape_function = space(entities).shape_function();
82  RealVector node_coord(entities.element_type().dimension());
83  for (Uint elem=0; elem<entities.size(); ++elem)
84  {
85  elem_coordinates = entities.geometry_space().get_coordinates(elem);
86  for (Uint node=0; node<elem_coordinates.rows(); ++node)
87  {
88  node_coord = elem_coordinates.row(node);
89  bounding_box.extend(node_coord);
90  }
91  }
92  }
93  bounding_box.make_global();
94 
95  Uint point_idx = 0;
96  // (b) Create unique indices for every coordinate
97  math::Hilbert compute_glb_idx(bounding_box, 20); // functor
98  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
99  {
100  Entities& entities = *entities_handle;
101  const ShapeFunction& shape_function = space(entities).shape_function();
102  for (Uint elem=0; elem<entities.size(); ++elem)
103  {
104  elem_coordinates = entities.geometry_space().get_coordinates(elem);
105  for (Uint node=0; node<shape_function.nb_nodes(); ++node)
106  {
107  RealVector space_coordinates = entities.element_type().shape_function().value(shape_function.local_coordinates().row(node)) * elem_coordinates ;
108  boost::uint64_t hash = compute_glb_idx(space_coordinates);
109  if(points.insert(std::make_pair(hash, point_idx)).second)
110  ++point_idx;
111  }
112  }
113  }
114 
115  // (c) Create the coordinates field
116  Field& coordinates = create_field(mesh::Tags::coordinates(),"coords[vector]");
117 
118  // step 3: resize
119  // --------------
120  resize(points.size());
121  m_coordinates = coordinates.handle<Field>();
122  for (Uint i=0; i<size(); ++i)
123  rank()[i] = UNKNOWN;
124 
125  // step 2: collect nodes in a set
126  // ------------------------------
127  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
128  {
129  Entities& entities = *entities_handle;
130  Dictionary& geometry = entities.geometry_fields();
131  Connectivity& geometry_node_connectivity = entities.geometry_space().connectivity();
132  common::List<Uint>& geometry_rank = entities.geometry_fields().rank();
133  const ShapeFunction& shape_function = space(entities).shape_function();
134  Connectivity& connectivity = const_cast<Space&>(space(entities)).connectivity();
135  connectivity.resize(entities.size());
136  for (Uint elem=0; elem<entities.size(); ++elem)
137  {
138  elem_coordinates = entities.geometry_space().get_coordinates(elem);
139  for (Uint node=0; node<shape_function.nb_nodes(); ++node)
140  {
141  RealVector space_coordinates = entities.element_type().shape_function().value(shape_function.local_coordinates().row(node)) * elem_coordinates ;
142  boost::uint64_t hash = compute_glb_idx(space_coordinates);
143  const Uint idx = points[hash];
144  connectivity[elem][node] = idx;
145  coordinates.set_row(idx, space_coordinates);
146  rank()[idx] = UNKNOWN;
147  glb_idx()[idx] = UNKNOWN;
148  }
149  }
150  }
151 
152  // step 5: fix unknown ranks
153  // -------------------------
154  cf3_assert(size() == m_rank->size());
155  cf3_assert(size() == m_glb_idx->size());
156  cf3_assert(size() == m_coordinates->size());
157 
158  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
159  {
160  Entities& entities = *entities_handle;
161  const Connectivity& space_connectivity = space(entities).connectivity();
162  for (Uint e=0; e<entities.size(); ++e)
163  {
164  boost_foreach(const Uint node, space_connectivity[e])
165  rank()[node] = std::min(rank()[node] , entities.rank()[e] );
166  }
167  }
168 
169  std::map<boost::uint64_t,Uint> hash_to_idx;
170  std::map<boost::uint64_t,Uint>::iterator hash_to_idx_iter;
171  std::map<boost::uint64_t,Uint>::iterator hash_not_found = hash_to_idx.end();
172 
173  std::vector<boost::uint64_t> coord_hash(size());
174  RealVector dummy(coordinates.row_size());
175  Uint c(0);
176  for (Uint i=0; i<size(); ++i)
177  {
178  math::copy(coordinates[i],dummy);
179  boost::uint64_t hash = compute_glb_idx(dummy);
180  hash_to_idx[hash] = i;
181  coord_hash[c++]=hash;
182  }
183 
184  // - Communicate unknown ranks to all processes
185  std::vector< std::vector<boost::uint64_t> > recv_hash(Comm::instance().size());
186  if (Comm::instance().is_active())
187  Comm::instance().all_gather(coord_hash,recv_hash);
188  else
189  recv_hash[0] = coord_hash;
190 
191  // - Search this process contains the missing ranks of other processes
192  std::vector< std::vector<Uint> > send_found_on_rank(Comm::instance().size());
193  for (Uint p=0; p<Comm::instance().size(); ++p)
194  {
195  send_found_on_rank[p].resize(recv_hash[p].size(),UNKNOWN);
196  {
197  for (Uint h=0; h<recv_hash[p].size(); ++h)
198  {
199  hash_to_idx_iter = hash_to_idx.find(recv_hash[p][h]);
200  if ( hash_to_idx_iter != hash_not_found )
201  {
202  send_found_on_rank[p][h] = rank()[ hash_to_idx_iter->second ];
203  }
204  }
205  }
206 
207  }
208 
209  // - Communicate which processes found the missing ranks
210  std::vector< std::vector<Uint> > recv_found_on_rank(Comm::instance().size());
211  if (Comm::instance().is_active())
212  Comm::instance().all_to_all(send_found_on_rank,recv_found_on_rank);
213  else
214  recv_found_on_rank[0] = send_found_on_rank[0];
215 
216 
217  // - Set the missing rank to the lowest process number that found it
218  for (Uint n=0; n<size(); ++n)
219  {
220  Uint rank_that_owns = UNKNOWN;
221 
222  for (Uint p=0; p<Comm::instance().size(); ++p)
223  {

225  }
226  rank()[n] = rank_that_owns;
227 
228  cf3_assert(rank()[n] != UNKNOWN);
229  }
230 
231  // step 5: fix unknown glb_idx
232  // ---------------------------
233  Uint nb_owned = 0;
234  std::vector<Uint> ghosts; ghosts.reserve(size());
235  for (Uint i=0; i<size(); ++i)
236  {
237  if (is_ghost(i))
238  ghosts.push_back(i);
239  else
240  ++nb_owned;
241  }
242  std::vector<boost::uint64_t> ghosts_hashed(ghosts.size());
243  for (Uint g=0; g<ghosts.size(); ++g)
244  {
245  math::copy(coordinates[ghosts[g]], dummy);
246  ghosts_hashed[g] = compute_glb_idx(dummy);
247  }
248  std::vector<Uint> nb_owned_per_proc(Comm::instance().size(),nb_owned);
249  if( Comm::instance().is_active() )
250  Comm::instance().all_gather(nb_owned, nb_owned_per_proc);
251 
252  std::vector<Uint> start_id_per_proc(Comm::instance().size());
253 
254  Uint start_id=0;
255  for (Uint p=0; p<Comm::instance().size(); ++p)
256  {
257  start_id_per_proc[p] = start_id;
258  start_id += nb_owned_per_proc[p];
259  }
260  start_id = start_id_per_proc[Comm::instance().rank()];
261  for (Uint i=0; i<size(); ++i)
262  {
263  if (! is_ghost(i))
264  glb_idx()[i] = start_id++;
265  else
266  glb_idx()[i] = UNKNOWN;
267  }
268 
269  std::vector< std::vector<boost::uint64_t> > recv_ghosts_hashed(Comm::instance().size());
270  if (Comm::instance().is_active())
271  Comm::instance().all_gather(ghosts_hashed,recv_ghosts_hashed);
272  else
273  recv_ghosts_hashed[0] = ghosts_hashed;
274 
275  // - Search this process contains the missing ranks of other processes
276  std::vector< std::vector<Uint> > send_glb_idx_on_rank(Comm::instance().size());
277  for (Uint p=0; p<Comm::instance().size(); ++p)
278  {
279  send_glb_idx_on_rank[p].resize(recv_ghosts_hashed[p].size(),UNKNOWN);
280  if (p!=Comm::instance().rank())
281  {
282  for (Uint h=0; h<recv_ghosts_hashed[p].size(); ++h)
283  {
284  hash_to_idx_iter = hash_to_idx.find(recv_ghosts_hashed[p][h]);
285  if ( hash_to_idx_iter != hash_not_found )
286  {
287  send_glb_idx_on_rank[p][h] = glb_idx()[hash_to_idx_iter->second];
288  }
289  }
290  }
291  }
292 
293  // - Communicate which processes found the missing ghosts
294  std::vector< std::vector<Uint> > recv_glb_idx_on_rank(Comm::instance().size());
295  if (Comm::instance().is_active())
296  Comm::instance().all_to_all(send_glb_idx_on_rank,recv_glb_idx_on_rank);
297  else
298  recv_glb_idx_on_rank[0] = send_glb_idx_on_rank[0];
299 
300  // - Set the missing rank to the lowest process number that found it
301  for (Uint g=0; g<ghosts.size(); ++g)
302  {
303  cf3_assert(rank()[ghosts[g]] < Comm::instance().size());
304  glb_idx()[ghosts[g]] = recv_glb_idx_on_rank[rank()[ghosts[g]]][g];
305  }

311 {
312  // Reserve memory in m_connectivity->array()
313  m_connectivity->array().resize(size());

315  std::vector<Uint> connectivity_sizes(size());

317  {

319  {
320  boost_foreach (const Uint node_idx, space->connectivity()[elem_idx])
321  {
322  cf3_assert_desc(to_str(node_idx)+"<"+to_str(size())+" --> something wrong with the element-node connectivity table from space "+space->uri().path(),node_idx<size());

325  }
326  }
327  for (Uint i=0; i<size(); ++i)
328  {
329  m_connectivity->array()[i].clear();
