49  boost::python::object operator()(boost::python::tuple args, boost::python::dict kwargs)
50  {
51  // Get the signature
52  common::SignalArgs node;
53  ( * m_signal->signature() ) ( node );
54  common::XML::SignalOptions options(node);
55 
56  // We support keywordless calling only for signals taking a single mandatory argument
57  // This is because the options are in alphabetical order and thus different from C++, making it counter-intuitive to learn
58  // two different orders. This policy can be reconsidered when the options store is no longer alphabetic.
59  if(len(args) == 2)
60  {
61  // We have only one (possibly optional) argument, so just set it
62  if(options.store.size() == 1)
63  {
64  options.begin()->second->change_value(python_to_any(args[1]));
65  }
66  else
67  {
68  // Find the first non-optional argument and set it
69  bool found = false;
70  for(common::OptionList::iterator opt_it = options.begin(); opt_it != options.end(); ++opt_it)
71  {
72  if(opt_it->second->has_tag("basic"))
73  {
74  if(found)
75  throw common::IllegalCall(FromHere(), "Method " + m_signal->name() + " can not be called using unnamed arguments since it has more than one basic option");
76  opt_it->second->change_value(python_to_any(args[1]));
77  found = true;
78  }
79  }
80  }
81  }
82  else // All other cases should use keywords
83  {
84  const boost::python::list keys = kwargs.keys();
85  const Uint nb_kwargs = len(keys);
86  for(Uint i = 0; i != nb_kwargs; ++i)
87  {
88  boost::python::extract<std::string> extracted_key(keys[i]);
89  const std::string key = extracted_key();
90  if(options.check(key))
91  {
92  common::Option& option = options[key];
93  option.change_value(python_to_any(kwargs[key]));
94  }
95  else
96  {
97  const boost::python::object& kw_val = kwargs[key];
98  if(kw_val.ptr()->ob_type == &PyList_Type)
99  {
100  const boost::python::list& kw_list = static_cast<const boost::python::list&>(kw_val);
101  const Uint nb_vals = boost::python::len(kw_list);
102  std::vector<std::string> values_str(nb_vals);
103  for(Uint i = 0; i != nb_vals; ++i)
104  {
105  values_str[i] = boost::python::extract<std::string>(boost::python::str(kw_list[i]))();
106  }
107  options.add(common::OptionFactory::instance().create_option(key, type_name(kw_val), values_str));
108  }
109  else
110  {
111  std::string tp_name = type_name(kw_val);
113  if(boost::starts_with(tp_name, "handle["))
114  tp_name = "handle[cf3.common.Component]";
115  options.add(common::OptionFactory::instance().create_option(key, tp_name, boost::python::extract<std::string>(boost::python::str(kw_val))()));
116  }
117  }
118  }
119 
120  }
121 
122  options.flush();
123 
124  CFdebug << "Calling signal " << m_signal->name() << " with arguments: ";
125  for(SignalOptions::const_iterator opt = options.begin(); opt != options.end(); ++opt)
126  CFdebug << opt->first << " = " << opt->second->value_str() << ", ";
127  CFdebug << CFendl;
128 
129  (*m_signal->signal())(node);
130  // Process reply
131  SignalFrame reply = node.get_reply();
132  if(reply.node.is_valid())
133  {
134  SignalOptions reply_options(reply);
135  if(reply_options.check("created_component"))
136  return wrap_component(m_component->access_component(reply_options["created_component"].value< common::URI >()));
137  if(reply_options.check("return_value"))
138  return any_to_python(reply_options["return_value"].value());
139  }
140 
141  return boost::python::object();
142  }
143 
144  std::string documentation()
145  {
146  std::stringstream doc_str;
147  doc_str << m_signal->description() << "\n";
148  doc_str << "Valid keywords are:\n";
149

151  common::SignalArgs node;
152  ( * m_signal->signature() ) ( node );

263 {
264  common::Component& comp = boost::python::extract<ComponentWrapper&>(self)().component();
265 
266  // Try setting an option, if there is a basic option with the attribute name

268  if(comp.options().check(opt_name) && comp.options().option(opt_name).has_tag("basic"))
269  {

271  return;

273

275  if(is_not_null(child_comp) && child_comp->has_tag("basic"))

278  boost::python::throw_error_already_set();
279  }
280 
281  PyErr_SetString(PyExc_AttributeError, ("Component " + comp.uri().path() + " has no basic option " + attr).c_str());
