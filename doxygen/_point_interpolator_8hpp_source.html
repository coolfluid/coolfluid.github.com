78 template <typename VectorT>
79 bool APointInterpolator::interpolate(const Field& field, const RealVector& coordinate, VectorT& interpolated_value)
80 {
81  // 1) Compute all useful information for interpolation
82  const bool element_found = compute_storage(coordinate,m_element,m_stencil,m_source_field_points,m_source_field_weights);
83  if (!element_found)
84  {
85  // Not interpolated because coordinate not found with element finder
86  return false;
87  }
88 
89  // 2) Interpolate
90 
91  for (Uint v=0; v<interpolated_value.size(); ++v)
92  {
93  interpolated_value[v]=0.;
94  for (Uint i=0; i<m_source_field_points.size(); ++i)
95  {
96  interpolated_value[v] += field[m_source_field_points[i]][v] * m_source_field_weights[i];
97  }
98  }
99  return true;
100 }
