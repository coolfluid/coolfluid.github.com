116 void NScriptEngine::execute_line( const QString & line , int fragment_number, QVector<int> break_points){
117  if (connected){
118  const common::URI script_engine_path("//Tools/Python/ScriptEngine", common::URI::Scheme::CPATH);
119  SignalOptions options;
120  QString repl=QString(line);
121  repl.replace(QString("\t"),QString(";"));
122  repl.replace(QString("\n"),QString("?"));
123  options.add("script", repl.toStdString());
124  options.add("fragment",fragment_number);
125  options.add("breakpoints",break_points.toStdVector());
126  SignalFrame frame = options.create_frame("execute_script", uri(), script_engine_path);
127  NetworkQueue::global()->send( frame, NetworkQueue::IMMEDIATE );
128  }
129 }

133 void NScriptEngine::emit_debug_command(debug_command command, int fragment, int line){
