266 inline void Buffer::reserve(const Uint size)
267 {
268  if(m_size+static_cast<int>(size) > m_capacity)
269  {
270  m_capacity = std::max(2*m_capacity,m_size+static_cast<int>(size));

310 template <typename T>
311 inline void Buffer::pack(const T* data, const Uint data_size)
312 {
313  if (data_size)
314  {
315  // get size of the package
316  int size;
317  MPI_Pack_size(data_size, get_mpi_datatype<T>() , Comm::instance().communicator() , &size);
318 
319  // reserve buffer to fit the package
320  reserve(size);
321 
322  // pack the package in the buffer, and modify the packed_size
323  int index = static_cast<int>(m_size);
324  MPI_Pack((void*)data, data_size , get_mpi_datatype<T>(), m_buffer, m_capacity, &index, Comm::instance().communicator());
325  m_size = index;
326  cf3_assert(m_size <= m_capacity);
327  if (m_strides.size())
328  m_strides.back()=m_size-m_displs.back();
329  }
330 }

464 inline void Buffer::broadcast(const Uint root)
465 {
466  // broadcast buffer size
467  int p = m_size;
468  MPI_Bcast( &p, 1, get_mpi_datatype(p), root, PE::Comm::instance().communicator() );
469 
470  // resize the buffer on receiving ranks
471  if (Comm::instance().rank()!=root)
472  {
473  reserve(p);
474  m_size = p;
475  }
476 
477  // broadcast buffer as MPI_PACKED
478  MPI_Bcast( m_buffer, m_size, MPI_PACKED, root, PE::Comm::instance().communicator() );
479 }

483 inline void Buffer::all_gather(Buffer& recv)
484 {
485  std::vector<int> strides;
486  Comm::instance().all_gather((int)size(),strides);
487  std::vector<int> displs(strides.size());
488  if (strides.size())
489  {
490  int sum_strides = strides[0];
491  displs[0] = 0;
492  for (Uint i=1; i<strides.size(); ++i)
493  {
494  displs[i] = displs[i-1] + strides[i-1];
495  sum_strides += strides[i];
496  }
497  recv.reset();
498  recv.resize(displs.back()+strides.back());
499  MPI_CHECK_RESULT(MPI_Allgatherv, (begin(), size(), MPI_PACKED, recv.begin(), &strides[0], &displs[0], MPI_PACKED, Comm::instance().communicator()));
500  }
501  else
502  {
503  recv.reset();
504  }
505 }

509 inline void Buffer::all_to_all(PE::Buffer& recv)
510 {
511  cf3_assert(strides().size() == PE::Comm::instance().size());
512  recv.reset();
513  Comm::instance().all_to_all(strides(),recv.strides());
514  cf3_assert(recv.strides().size() == PE::Comm::instance().size());
515  recv.displs().resize(PE::Comm::instance().size());
516  recv.displs()[0]=0;
517  for (Uint pid=1; pid<Comm::instance().size(); ++pid)
518  recv.displs()[pid] = recv.displs()[pid-1] + recv.strides()[pid-1];
519  recv.resize(recv.displs().back()+recv.strides().back());
520  MPI_CHECK_RESULT(MPI_Alltoallv, ((void*)begin(), &strides()[0], &displs()[0], MPI_PACKED, (void*)recv.begin(), &recv.strides()[0], &recv.displs()[0], MPI_PACKED, Comm::instance().communicator()));
521 }
