63 void NodeElementConnectivity::build_connectivity()
64 {
65  cf3_assert(elements().size());

67  cf3_assert(elements().components()[0]);

69  cf3_assert(m_nodes);

72 
73  // Reserve memory in m_connectivity->array()
74  std::vector<Uint> connectivity_sizes(nodes.size());
75  boost_foreach(Handle<Component> elements_comp, m_elements->components() )
76  {
77  Entities& elements = dynamic_cast<Entities&>(*elements_comp);
78  boost_foreach (Connectivity::ConstRow elem_nodes, elements.geometry_space().connectivity().array() )
79  {
80  boost_foreach (const Uint node_idx, elem_nodes)
81  {
82  cf3_assert(node_idx<nodes.size());
83  ++connectivity_sizes[node_idx];
84  }
85  }
86  }
87  Uint i(0);
88  boost_foreach (DynTable<Uint>::Row row, m_connectivity->array() )
89  {
90  row.reserve(connectivity_sizes[i++]);
91  }
92 
93  // fill m_connectivity->array()
94  Uint glb_elem_idx = 0;
95  boost_foreach(Handle<Component> elements_comp, m_elements->components() )
96  {

99  {
