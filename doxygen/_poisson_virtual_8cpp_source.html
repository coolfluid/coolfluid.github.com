50  virtual void execute()
51  {
52  typedef mesh::Integrators::GaussMappedCoords<2, GeoShape::TRIAG> GaussT;
53 
54 
55  if(is_null(m_lss))
56  throw common::SetupError(FromHere(), "LSS not set for " + uri().path());
57 
58  math::LSS::Matrix& lss_mat = *m_lss->matrix();
59  math::LSS::Vector& lss_rhs = *m_lss->rhs();
60 
61  BOOST_FOREACH(const Handle<mesh::Region>& region, m_loop_regions)
62  {
63  BOOST_FOREACH(const mesh::Elements& elements, find_components_recursively_with_filter<Elements>(*region, IsElementType<mesh::LagrangeP1::Triag2D>()))
64  {
65  const Uint nb_elems = elements.size();
66  const mesh::Connectivity& connectivity = elements.geometry_space().connectivity();
67  const mesh::ShapeFunction& sf = elements.geometry_space().shape_function();
68  const mesh::ElementType& et = elements.element_type();
69  const mesh::Field& coordinates = elements.geometry_fields().coordinates();
70  math::LSS::BlockAccumulator acc;
71  const Uint nb_nodes = sf.nb_nodes();
72  acc.resize(nb_nodes, 1);
73  RealMatrix jacobian_adj;
74  jacobian_adj.resize(sf.dimensionality(), sf.dimensionality());
75  RealRowVector N;
76  N.resize(sf.nb_nodes());
77  RealColVector f;
78  f.resize(sf.nb_nodes());
79  RealMatrix nodes;
80  nodes.resize(nb_nodes, sf.dimensionality());
81  RealMatrix nabla;
82  nabla.resize(sf.dimensionality(), nb_nodes);
83 
84  const mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(elements);
85  const mesh::Field& source_term = common::find_component_recursively_with_tag<mesh::Field>(mesh, "source_term");
86 
87  std::vector<RealMatrix> gauss_gradients(GaussT::nb_points, RealMatrix(sf.dimensionality(), nb_nodes));
88  for(Uint gauss_idx = 0; gauss_idx != GaussT::nb_points; ++gauss_idx)
89  {
90  sf.compute_gradient(GaussT::instance().coords.col(gauss_idx), gauss_gradients[gauss_idx]);
91  }
92 
93  for(Uint i = 0; i != nb_elems; ++i)
94  {
95  acc.neighbour_indices(connectivity[i]);
96  mesh::fill(nodes, coordinates, connectivity[i]);
97  for(Uint j = 0; j != nb_nodes; ++j)
98  f[j] = source_term[acc.indices[j]][0];
99 
100  acc.reset();
101 
102  for(Uint gauss_idx = 0; gauss_idx != GaussT::nb_points; ++gauss_idx)
103  {
104  et.compute_jacobian_adjoint(GaussT::instance().coords.col(gauss_idx), nodes, jacobian_adj);
105  const Real w = GaussT::instance().weights[gauss_idx];
106  const Real det_jac = et.jacobian_determinant(GaussT::instance().coords.col(gauss_idx), nodes);
107  sf.compute_value(GaussT::instance().coords.col(gauss_idx), N);
108  nabla.noalias() = jacobian_adj*gauss_gradients[gauss_idx];
109  acc.mat.noalias() += w * nabla.transpose()*nabla / det_jac;
110  acc.rhs.noalias() += w*det_jac*N.transpose()*(N*f);
111  }
112 
113  lss_mat.add_values(acc);
114  lss_rhs.add_rhs_values(acc);
115  }
116  }
117  }
118  }

125 PoissonVirtual::PoissonVirtual ( const std::string& name ) : LSSAction ( name )
126 {
127  // This determines the name of the field that will be used to store the solution
128  set_solution_tag("poisson_solution");
129 
130  // Create action components that wil be executed in the order they are created here:
131  // 1. Set the linear system matrix and vectors to zero
132  create_component<math::LSS::ZeroLSS>("ZeroLSS");
133 
134  // 2. Assemble the system matrix and RHS using an action component
135  Handle<PoissonVirtualAssembly> assembly = create_component<PoissonVirtualAssembly>("Assembly");
136  options().option("lss").link_to(&assembly->m_lss);
137 
138  // 3. Apply bondary conditions
139  Handle<BoundaryConditions> bc = create_component<BoundaryConditions>("BoundaryConditions");
140  bc->mark_basic();
141  bc->set_solution_tag(solution_tag());
142 
143  // 4. Solve the linear system
144  create_component<math::LSS::SolveLSS>("SolveLSS");
145 
146  // 5. Update the solution
147  // The unknown function. The first template argument is a constant to distinguish each variable at compile time
148  FieldVariable<0, ScalarField> u("u", solution_tag());
149  // The source term, to be set at runtime using an initial condition
150  FieldVariable<1, ScalarField> f("f", "source_term");
151  create_component<ProtoAction>("Update")->set_expression(nodes_expression(u = solution(u)));
152  // Dummy action to have f created automatically
153  create_component<ProtoAction>("DummyF")->set_expression(nodes_expression(f));
154 }
