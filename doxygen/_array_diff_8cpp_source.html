49  template<typename NumberT>
50  void operator()(const NumberT)
51  {
52  if(m_left->derived_type_name() != m_right->derived_type_name())
53  throw common::SetupError(FromHere(), "Array types do not match");
54 
55  Handle< List<NumberT> const > left_list(m_left);
56  Handle< List<NumberT> const > right_list(m_right);
57  if(is_not_null(left_list) && is_not_null(right_list))
58  {
59  m_found_type = true;
60  if(left_list->size() != right_list->size())
61  {
62  CFdebug << m_prefix << "Array sizes don't match: left: " << left_list->size() << ", right: " << right_list->size() << CFendl;
63  m_arrays_equal = false;
64  return;
65  }
66  compare_lists(boost::is_floating_point<NumberT>(), *left_list, *right_list);
67  }
68  else
69  {
70  Handle< Table<NumberT> const > left_table(m_left);
71  Handle< Table<NumberT> const > right_table(m_right);
72  if(is_not_null(left_table) && is_not_null(right_table))
73  {
74  m_found_type = true;
75  if(left_table->size() != right_table->size() || left_table->row_size() != right_table->row_size())
76  {
77  CFdebug << m_prefix << "Array sizes don't match: left: " << left_table->size() << "x" << left_table->row_size() << ", right: " << right_table->size() << "x" << right_table->row_size() << CFendl;
78  m_arrays_equal = false;
79  return;
80  }
81  compare_tables(boost::is_floating_point<NumberT>(), *left_table, *right_table);

165  // compare floating point numbers
166  template<typename T>
167  Real compare_numbers(const T left, const T right)
168  {
169  // Don't try to compare zero using ULPs
170  if(fabs(left) < m_zero_threshold && fabs(right) < m_zero_threshold)
171  {
172  return 0.;
173  }
174 
175  return fabs(boost::math::float_distance(left, right));
176  }

221 void ArrayDiff::execute()
222 {
223  typedef boost::mpl::vector5<int, Uint, float, double, bool> allowed_types;
224  const Handle<Component const> left = options().value< Handle<Component const> >("left");
225  const Handle<Component const> right = options().value< Handle<Component const> >("right");
226 
227  if(is_null(left))
228  throw common::SetupError(FromHere(), "Left array is not set in " + uri().path());
229 
230  if(is_null(right))
231  throw common::SetupError(FromHere(), "Right array is not set in " + uri().path());
232 
233  const Uint max_ulps = options().value<Uint>("max_ulps");
234  const Real zero_threshold = options().value<Real>("zero_threshold");
235  bool found_type = false;
236  bool arrays_equal = false;
237  boost::mpl::for_each<allowed_types>(detail::DoArrayDiff(left, right, max_ulps, zero_threshold, found_type, arrays_equal));
238  if(!found_type)
239  throw common::NotImplemented(FromHere(), "ArrayDiff does not support diff for type " + left->derived_type_name());
240 
241  CFdebug << "rank " << PE::Comm::instance().rank() << ": arrays " << left->uri().path() << " and " << right->uri().path();
242  if(arrays_equal)
243  {
244  CFdebug << " are equal";
245  }
246  else
247  {
248  CFdebug << " differ";
249  }
250  CFdebug << CFendl;
251 
252  properties().set("arrays_equal", arrays_equal);
253 }
