83 std::vector<std::string> CreateField::split_function_list(const std::string& str)
84 {
85  std::string trimmed_str(str);
86  boost::algorithm::trim(trimmed_str);
87  boost::algorithm::trim_left_if (trimmed_str,boost::algorithm::is_any_of("["));
88  boost::algorithm::trim_right_if(trimmed_str,boost::algorithm::is_any_of("]"));
89  std::vector<std::string> split;
90  Uint scope_level = 0;
91  Uint start_pos = 0;
92  Uint pos = 0;
93  boost_foreach(const char& c, trimmed_str)
94  {
95  if (c == '(') ++scope_level;
96  if (c == ')') --scope_level;
97  if (c == ',' && scope_level == 0)
98  {
99  split.push_back( std::string(trimmed_str.begin()+start_pos,trimmed_str.begin()+pos) );
100  start_pos = pos + 1;
101  }
102  ++pos;
103  }
104  split.push_back( std::string(trimmed_str.begin()+start_pos,trimmed_str.end()) );
105  return split;
106 }

125 Handle<Field> CreateField::create_field(const std::string& name, Dictionary& dict, const std::vector<std::string>& _functions)
126 {
127  std::vector<std::string> functions_str=_functions;
128  std::vector<std::string> variables_str(functions_str.size());
129 
130  for (Uint f=0; f<functions_str.size(); ++f)
131  {
132  std::string str = functions_str[f];
133  std::vector<std::string> split;
134  boost::split(split, str, boost::is_any_of("="));
135 
136  variables_str[f] = split[0];
137  functions_str[f] = split[1];
138  }
139 
140  std::string var_description;
141  for(Uint i=0; i<variables_str.size(); ++i)
142  {
143  if (i!=0) var_description += ",";
144  var_description += variables_str[i];
145  }
146 
147  Field& new_field = dict.create_field(name, var_description);
148 
149  std::vector<std::string> functions;
150  boost_foreach(const std::string& function_list, functions_str)
151  {
152  std::vector<std::string> split_functions = split_function_list(function_list);
153  boost_foreach(const std::string& f_str, split_functions)
154  {
155  functions.push_back(f_str);
156  }
157  }
158 
159  if (new_field.row_size()!=functions.size())
160  {
161  std::stringstream ss;
162  ss << "Number of variables and functions doesn't match:\n";
163  ss << "variables = " << var_description << "\n";
164  ss << "functions = \n";
165  boost_foreach(const std::string& function, functions)
166  ss << " " << function << "\n";
167  throw SetupError(FromHere(),ss.str());
168  }
169 
170  std::vector< std::string > var_names;
171  std::vector< std::string > mod_var_names;
172  std::vector< Table<Real>::ArrayT const* > var_arrays;
173  std::vector< Uint > var_array_idx;
174 
175  std::string var_name;
176  std::string mod_var_name;
177 
178  boost_foreach ( Field& field, find_components_recursively<Field>(dict))
179  {
180  for (Uint v=0; v<field.row_size(); ++v)
181  {
182  var_name =field.name()+"["+to_str(v)+"]";
183  mod_var_name=field.name()+"_"+to_str(v)+"_";
184 
185  for (Uint s=0; s<functions.size(); ++s)
186  boost::algorithm::replace_all(functions[s],var_name,mod_var_name);
187  var_names.push_back(mod_var_name);
188  var_arrays.push_back(&field.array());
189  var_array_idx.push_back(v);
190  }
191  }
192 
193  // Add the x, y, z coordinates as aliases
194  const Field& coordinates = dict.coordinates();
195  std::vector<std::string> xyz(3);
196  xyz[0] = "x";
197  xyz[1] = "y";
198  xyz[2] = "z";
199  for (Uint d=0; d<coordinates.row_size(); ++d)
200  {
201  var_names.push_back(xyz[d]);
202  var_arrays.push_back(&coordinates.array());
203  var_array_idx.push_back(d);
204  }
205 
206 
207  // parse function
208 
209  VectorialFunction vectorial_function;
210  vectorial_function.functions(functions);
211  vectorial_function.variables(var_names);
212  vectorial_function.parse();
213 
214 
215  // Evaluate function
216 
217  std::vector<Real> params(var_names.size());
218  for (Uint pt=0; pt<new_field.size(); ++pt)
219  {
220  for (Uint v=0; v<params.size(); ++v)
221  {
222  params[v] = (*var_arrays[v])[pt][var_array_idx[v]];
223  }
224  Table<Real>::ArrayT::reference ref = new_field.array()[pt];
225  vectorial_function.evaluate(params,ref);
226  }
227 
228  return new_field.handle<Field>();
229 }
