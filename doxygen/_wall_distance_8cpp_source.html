50 struct WallProjection
51 {
52  WallProjection(const Field& coordinates, const NodeConnectivity& node_connectivity) :
53  m_coords(coordinates),
54  m_node_connectivity(node_connectivity)
55  {
56  }
57 
58  // Get the wall distance for an inner node, looking at the elements that are adjacent to the given surface node
59  Real operator()(const Uint inner_node_idx, const Uint surface_node_idx)
60  {
61  RealMatrix elem_coords;
62  const Uint dim = m_coords.row_size();
63  const RealVector inner_coord = to_vector(m_coords[inner_node_idx]);
64  std::vector<Uint> neighbor_nodes; // Collect neighboring nodes, so we can project onto a sharp corner in 3D if needed (i.e. near a step)
65  // Loop over all surface elements around the given node
66  BOOST_FOREACH(const NodeConnectivity::ElementReferenceT& element_ref, m_node_connectivity.node_element_range(surface_node_idx))
67  {
68  // Get the element coordinates
69  const Entities& elem_entities = *m_node_connectivity.entities()[element_ref.first];
70  const ElementType& etype = elem_entities.element_type();
71  const Uint element_nb_nodes = etype.nb_nodes();
72  elem_coords.resize(element_nb_nodes, dim);
73  const Connectivity::ConstRow conn_row = elem_entities.geometry_space().connectivity()[element_ref.second];
74  fill(elem_coords, m_coords, conn_row);
75 
76  // We consider lines, triangles and quads as viable surface elements
77  if(element_nb_nodes < 2 || element_nb_nodes > 4 || etype.order() != 1)
78  {
79  throw common::SetupError(FromHere(), "Unsupported surface element of type " + etype.name() + " in surface region " + elem_entities.uri().path());
80  }
81 
82  bool in_element = false;
83 
84  if(element_nb_nodes == 2) // line segment
85  {
86  cf3_assert(dim == 2);
87  cf3_assert(etype.shape() == GeoShape::LINE);
88  RealVector e1 = elem_coords.row(1) - elem_coords.row(0); // line segment vector
89  Real e1_len = e1.norm();
90  e1 /= e1_len;
91  const Real projection = e1.dot(inner_coord - elem_coords.row(0).transpose());
92  // If the projection of the node along the normal fits inside the element, we can take the normal distance
93  in_element = projection > 0 && projection < e1_len;
94  }
95  if(element_nb_nodes == 3)
96  {
97  cf3_assert(dim == 3);
98  cf3_assert(etype.shape() == GeoShape::TRIAG);
99  RealVector3 e1 = (elem_coords.row(1) - elem_coords.row(0)).normalized();
100  RealVector3 en = elem_coords.row(2) - elem_coords.row(0);
101  RealVector3 e2 = (e1.cross(en)).cross(e1).normalized();
102  RealVector3 p = inner_coord - elem_coords.row(0).transpose();
103 
104  // Construct 2D coordinates for the boundary element
105  Eigen::Matrix<Real, 3, 2> triag_coords_2d;
106  triag_coords_2d.row(0).setZero();
107  triag_coords_2d(1,0) = e1.dot(elem_coords.row(1) - elem_coords.row(0));
108  triag_coords_2d(1,1) = 0.;
109  triag_coords_2d(2,0) = e1.dot(elem_coords.row(2) - elem_coords.row(0));
110  triag_coords_2d(2,1) = e2.dot(elem_coords.row(2) - elem_coords.row(0));
111 
112  RealVector2 p_proj(2);
113  p_proj[0] = p.dot(e1);
114  p_proj[1] = p.dot(e2);
115 
116  in_element = LagrangeP1::Triag2D::is_coord_in_element(p_proj, triag_coords_2d);
117  const Uint origin_corner = std::find(conn_row.begin(), conn_row.end(), surface_node_idx) - conn_row.begin();
118  if(origin_corner == 0)
119  {
120  neighbor_nodes.push_back(conn_row[1]);
121  neighbor_nodes.push_back(conn_row[2]);
122  }
123  else if(origin_corner == 1)
124  {
125  neighbor_nodes.push_back(conn_row[0]);
126  neighbor_nodes.push_back(conn_row[2]);
127  }
128  else
129  {
130  neighbor_nodes.push_back(conn_row[0]);
131  neighbor_nodes.push_back(conn_row[1]);
132  }
133  }
134  if(element_nb_nodes == 4)
135  {
136  cf3_assert(dim == 3);
137  cf3_assert(etype.shape() == GeoShape::QUAD);
138  RealVector3 e1 = (elem_coords.row(1) - elem_coords.row(0)).normalized();
139  RealVector3 en = elem_coords.row(3) - elem_coords.row(0);
140  RealVector3 e2 = (e1.cross(en)).cross(e1).normalized();
141  RealVector3 p = inner_coord - elem_coords.row(0).transpose();
142 
143  // Construct 2D coordinates for the boundary element
144  Eigen::Matrix<Real, 4, 2> quad_coords_2d;
145  quad_coords_2d.row(0).setZero();
146  for(int i = 1; i != 4; ++i)
147  {
148  quad_coords_2d(i, 0) = e1.dot(elem_coords.row(i) - elem_coords.row(0));
149  quad_coords_2d(i, 1) = e2.dot(elem_coords.row(i) - elem_coords.row(0));
150  }
151 
152  RealVector2 p_proj(2);
153  p_proj[0] = p.dot(e1);
154  p_proj[1] = p.dot(e2);
155 
156  in_element = LagrangeP1::Quad2D::is_coord_in_element(p_proj, quad_coords_2d);
157  const Uint origin_corner = std::find(conn_row.begin(), conn_row.end(), surface_node_idx) - conn_row.begin();
158  if(origin_corner == 0 || origin_corner == 2)
159  {
160  neighbor_nodes.push_back(conn_row[1]);
161  neighbor_nodes.push_back(conn_row[3]);
162  }
163  else
164  {
165  neighbor_nodes.push_back(conn_row[0]);
166  neighbor_nodes.push_back(conn_row[2]);
167  }
168  }
169 
170  // If the projection was in an element, we can just proceed to compute the normal distance
171  if(in_element)
172  {
173  RealVector n(dim); // normal vector
174  etype.compute_normal(elem_coords, n);
175  return fabs((n/n.norm()).dot(inner_coord - elem_coords.row(0).transpose()));
176  }
177  }
178  // If we got here, no projections on the elements gave a result
179  // First, verify the 3D case where we need to project on "step" edges
180  BOOST_FOREACH(const Uint neighbor_node, neighbor_nodes)
181  {
182  const RealVector surface_coord = to_vector(m_coords[surface_node_idx]);
183  const RealVector neighbor_coord = to_vector(m_coords[neighbor_node]);
184  RealVector e1 = neighbor_coord - surface_coord;
185  Real e1_len = e1.norm();
186  e1 /= e1_len;
187  const Real projection = e1.dot(inner_coord - surface_coord);
188  if(projection > 0 && projection < e1_len)
189  {
190  return (inner_coord - (surface_coord + e1*projection)).norm();
191  }
192  }
193  return (inner_coord - to_vector(m_coords[surface_node_idx])).norm();
194  }
195 
196  const Field& m_coords;
197  const NodeConnectivity& m_node_connectivity;
198 };

210 void WallDistance::execute()
211 {
212  Mesh& mesh = *m_mesh;
213 
214  Field& d = mesh.geometry_fields().create_field("WallDistance", "wall_distance");
215  d.add_tag("wall_distance");
216  const Field& coords = mesh.geometry_fields().coordinates();
217  const Uint nb_nodes = coords.size();
218 
219  boost::shared_ptr<NodeConnectivity> node_connectivity = common::allocate_component<NodeConnectivity>("NodeConnectivity");
220  std::vector< Handle<Entities const> > surface_entities;
221  BOOST_FOREACH(const Handle<Region const>& region, m_regions)
222  {
223  BOOST_FOREACH(const mesh::Elements& elements, common::find_components_recursively_with_filter<mesh::Elements>(*region, IsElementsSurface()))
224  {
225  surface_entities.push_back(elements.handle<Entities const>());
226  }
227  }
228 
229  node_connectivity->initialize(nb_nodes, surface_entities);
230 
231  boost::shared_ptr< common::List< Uint > > surface_nodes_ptr = build_used_nodes_list(surface_entities, mesh.geometry_fields(), true);
232  const common::List<Uint>& surface_nodes = *surface_nodes_ptr;
233  const Uint nb_surface_nodes = surface_nodes.size();
234 
235  detail::WallProjection normal_distance(coords, *node_connectivity);
236 
237  for(Uint inner_node_idx = 0; inner_node_idx != nb_nodes; ++inner_node_idx)
238  {
239  bool is_surface_node = false;
240  Real shortest_distance = 1e20;
241  Uint closest_surface_node = 0;
242  for(Uint j = 0; j != nb_surface_nodes; ++j)
243  {
244  const Uint surface_node_idx = surface_nodes[j];
245  if(surface_node_idx == inner_node_idx)
246  {
247  is_surface_node = true;
248  break;
249  }
250  const Real d2 = (to_vector(coords[inner_node_idx]) - to_vector(coords[surface_node_idx])).squaredNorm();
251  if(d2 < shortest_distance)
252  {
253  shortest_distance = d2;
254  closest_surface_node = surface_node_idx;
255  }
256  }
257 
258  d[inner_node_idx][0] = is_surface_node ? 0. : normal_distance(inner_node_idx, closest_surface_node);
259  }
260 
261 }
