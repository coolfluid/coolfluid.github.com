42 struct CorrectionMatrix
43 {
44  typedef void result_type;
45 
46  template<typename UT, typename MatrixT>
47  void operator()(UT& u, MatrixT& A)
48  {
49  static const Uint nb_nodes = UT::EtypeT::nb_nodes;
50  Eigen::Matrix<Real, nb_nodes, nb_nodes> mass_matrix;
51  mass_matrix.setZero();
52 
53  // 4th order quadrature
54  typedef mesh::Integrators::GaussMappedCoords<4, UT::EtypeT::shape> GaussT;
55  for(Uint gauss_idx = 0; gauss_idx != GaussT::nb_points; ++gauss_idx)
56  {
57  u.shape_function(GaussT::instance().coords.col(gauss_idx));
58  const Real w = GaussT::instance().weights[gauss_idx] * u.support().jacobian_determinant(GaussT::instance().coords.col(gauss_idx));
59  mass_matrix.noalias() += u.shape_function().transpose() * (u.shape_function()*w);
60  }
61 
62  for(Uint i = 0; i != UT::EtypeT::dimension; ++i)
63  {
64  A.template block<nb_nodes, nb_nodes>(i*nb_nodes, i*nb_nodes).noalias() = mass_matrix;
65  }
66  }
67 };

71 NavierStokesChorin::NavierStokesChorin ( const std::string& name ) : solver::Action( name )
72 {
73  using boost::proto::lit;
74 
75  PhysicsConstant nu("kinematic_viscosity");
76 
77  // Manage the auxiliary velocity
78  Handle<LSSActionUnsteady> auxiliary_lss = create_component<LSSActionUnsteady>("AuxiliaryLSS");
79  auxiliary_lss->options().set("matrix_builder", std::string("cf3.math.LSS.TrilinosCrsMatrix"));
80  auxiliary_lss->options().set("use_geometry_space", false);
81  auxiliary_lss->set_solution_tag("navier_stokes_u_solution");
82 
83  // The auxiliary velocity field, of P2 order
84  FieldVariable<0, VectorField> u("Velocity", auxiliary_lss->solution_tag(), mesh::LagrangeP2::LibLagrangeP2::library_namespace());
85 
86  // Zero the LSS
87  auxiliary_lss->create_component<math::LSS::ZeroLSS>("ZeroLSS");
88 
89  // Matrix assembly
90  Handle<ProtoAction> auxiliary_mat_assembly = auxiliary_lss->create_component<ProtoAction>("MatrixAssembly");
91  auxiliary_mat_assembly->set_expression(elements_expression(LagrangeP1P2(),
92  group
93  (
94  _A( u ) = _0, _T( u ) = _0,
95  element_quadrature
96  (
97  _A( u[_i], u[_i]) += transpose(nabla( u ))*nabla( u ),
98  _T( u[_i], u[_i]) += transpose(N( u ))*N( u )
99  ),
100  auxiliary_lss->system_matrix += auxiliary_lss->invdt()*_T + nu*_A
101  )));
102 
103  // RHS assembly
104  Handle<ProtoAction> auxiliary_rhs_assembly = auxiliary_lss->create_component<ProtoAction>("RHSAssembly");
105  auxiliary_rhs_assembly->set_expression(elements_expression(LagrangeP1P2(),
106  group
107  (
108  _a[u] = _0, // Set the element vector for the u_star variable to 0 and at the same time indicate that u_star is part of the linear system
109  element_quadrature
110  (
111  _a[u[_i]] += auxiliary_lss->invdt() * transpose(N(u))*u[_i] - transpose(N(u))*(u*nabla(u))*_col(nodal_values(u), _i)
112  ),
113  auxiliary_lss->system_rhs += _a
114  )));
115 
116  Handle<LSSActionUnsteady> pressure_lss = create_component<LSSActionUnsteady>("PressureLSS");
117  pressure_lss->options().set("matrix_builder", std::string("cf3.math.LSS.TrilinosCrsMatrix"));
118  pressure_lss->set_solution_tag("navier_stokes_p_solution");
119  pressure_lss->create_component<math::LSS::ZeroLSS>("ZeroLSS");
120 
121  // The pressure field. LagrangeP1 by default
122  FieldVariable<1, ScalarField> p("Pressure", pressure_lss->solution_tag());
123 
124  Handle<ProtoAction> pressure_mat_assembly = pressure_lss->create_component<ProtoAction>("MatrixAssembly");
125  pressure_mat_assembly->set_expression(elements_expression(LagrangeP1(),
126  group
127  (
128  _A(p) = _0,
129  element_quadrature(_A(p) += transpose(nabla(p))*nabla(p)),
130  pressure_lss->system_matrix += _A
131  )));
132 
133  Handle<ProtoAction> pressure_rhs_assembly = pressure_lss->create_component<ProtoAction>("RHSAssembly");
134  pressure_rhs_assembly->set_expression(elements_expression(LagrangeP1P2(),
135  group
136  (
137  _a[p] = _0,
138  element_quadrature(_a[p] += transpose(N(p))*divergence(u)),
139  pressure_lss->system_rhs += -lit(pressure_lss->invdt())*_a
140  )));
141 
142  // Velocity correction
143  Handle<LSSActionUnsteady> correction_lss = create_component<LSSActionUnsteady>("CorrectionLSS");
144  correction_lss->options().set("matrix_builder", std::string("cf3.math.LSS.TrilinosCrsMatrix"));
145  correction_lss->options().set("use_geometry_space", false);
146  correction_lss->set_solution_tag(auxiliary_lss->solution_tag());
147  correction_lss->create_component<math::LSS::ZeroLSS>("ZeroLSS");
148 
149  Handle<ProtoAction> correction_matrix_assembly = correction_lss->create_component<ProtoAction>("MatrixAssembly");
150  correction_matrix_assembly->set_expression(elements_expression(LagrangeP1P2(),
151  group
152  (
153  _A(u) = _0,
154  element_quadrature(_A(u[_i], u[_i]) += transpose(N(u))*N(u)),
155  //correction_matrix(u, _A),
156  correction_lss->system_matrix += _A
157  )));
158 
159  Handle<ProtoAction> correction_rhs_assembly = correction_lss->create_component<ProtoAction>("RHSAssembly");
160  correction_rhs_assembly->set_expression(elements_expression(LagrangeP1P2(),
161  group
162  (
163  _a[u] = _0,
164  element_quadrature(_a[u[_i]] += transpose(N(u))*(u[_i] - lit(correction_lss->dt()) * gradient(p)[_i])),
165  correction_lss->system_rhs += _a
166  )));
167 
168  // Velocity boundary conditions
169  Handle<BoundaryConditions> u_bc = create_component<BoundaryConditions>("VelocityBC");
170  u_bc->mark_basic();
171  u_bc->set_solution_tag(auxiliary_lss->solution_tag());
172  auxiliary_lss->create_component<common::Link>("BC")->link_to(*u_bc);
173  correction_lss->create_component<common::Link>("BC")->link_to(*u_bc);
174 
175  // Pressure boundary conditions
176  Handle<BoundaryConditions> p_bc = create_component<BoundaryConditions>("PressureBC");
177  p_bc->mark_basic();
178  p_bc->set_solution_tag(pressure_lss->solution_tag());
179  pressure_lss->create_component<common::Link>("BC")->link_to(*p_bc);
180 
181  // Solution of the systems
182  auxiliary_lss->create_component<math::LSS::SolveLSS>("SolveLSS");
183  pressure_lss->create_component<math::LSS::SolveLSS>("SolveLSS");
184  correction_lss->create_component<math::LSS::SolveLSS>("SolveLSS");
185 
186  // Update the fields
187  auxiliary_lss->create_component<ProtoAction>("Update")->set_expression(nodes_expression(u = auxiliary_lss->solution(u)));
188  pressure_lss->create_component<ProtoAction>("Update")->set_expression(nodes_expression(p = pressure_lss->solution(p)));
189  correction_lss->create_component<ProtoAction>("Update")->set_expression(nodes_expression(u = correction_lss->solution(u)));
190 }

198

202  correction_lss->execute();
203

207  pressure_lss->options().set("disabled_actions", disabled);
208  correction_lss->options().set("disabled_actions", disabled);
209 
210  // Don't zero the matrix anymore after the first run
211  auxiliary_lss->get_child("ZeroLSS")->options().set("reset_matrix", false);
