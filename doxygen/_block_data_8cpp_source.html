64  void create_mapped_coords(const Uint segments, const Real* gradings, common::Table<Real>::ArrayT& mapped_coords, const Uint nb_edges)
65  {
66  const Real eps = 1500*std::numeric_limits<Real>::epsilon();
67  mapped_coords.resize(boost::extents[segments+1][nb_edges]);
68  for(Uint edge = 0; edge != nb_edges; ++edge)
69  {
70  Real grading = gradings[edge];
71  if(fabs(grading-1.) > 1.e-6)
72  {
73  const Real r = pow(grading, 1. / static_cast<Real>(segments - 1)); // expansion ratio
74  for(Uint i = 0; i <= segments; ++i)
75  {
76  const Real result = 2. * (1. - pow(r, (int)i)) / (1. - grading*r) - 1.;
77  mapped_coords[i][edge] = result;
78  cf3_assert(fabs(result) < (1. + eps));
79  }
80  }
81  else
82  {
83  const Real step = 2. / static_cast<Real>(segments);
84  for(Uint i = 0; i <= segments; ++i)
85  {
86  mapped_coords[i][edge] = i*step - 1.;
87  cf3_assert(fabs(mapped_coords[i][edge]) < 1. + eps);
88  }
89  }
90  const Real start = mapped_coords[0][edge];
91  cf3_assert(fabs(start+1.) < eps);
92  mapped_coords[0][edge] = -1.;
93  const Real end = mapped_coords[segments][edge];
94  cf3_assert(fabs(end-1.) < eps);
95  mapped_coords[segments][edge] = 1.;
96  }
97  }

342  positive_faces[1] = 2;
343 
344  negative_faces[0] = 3;

348  block_list.assign(nb_blocks, Block(dimensions));
349  patch_map.clear();
350  const Table<Uint>& block_subdivs = *block_subdivisions;
351  Uint block_start = 0;
352  for(Uint block_idx = 0; block_idx != nb_blocks; ++block_idx)
353  {
354  Block& block = block_list[block_idx];
355  block.is_local = block_idx >= partition_begin && block_idx < partition_end;
356  block.start_index = block_start;

359  const Table<Uint>::ConstRow subdiv_row = block_subdivs[block_idx];

361  Uint stride = 1;
362  Uint nb_points = 1;
363  block.nb_elems = 1;
364  for(Uint i = 0; i != dimensions; ++i)
365  {
366  // Add the block
367  if(!face_conn.has_adjacent_element(block_idx, positive_faces[i]) || !face_conn.has_adjacent_element(block_idx, negative_faces[i]))
368  {
369  std::stringstream error_str;
370  error_str << "Block " << block_idx << " has no adjacent element for patch [ ";

372  const Uint bad_face_idx = !face_conn.has_adjacent_element(block_idx, positive_faces[i]) ? positive_faces[i] : negative_faces[i];
373  BOOST_FOREACH(const Uint i, faces.nodes_range(bad_face_idx))
374  {
375  error_str << row[i] << " ";
376  }
377  error_str << "]. Did you flip the ordering of patch nodes?";

379  }

381  block.strides[i] = stride;
382  block.bounded[i] = face_conn.node_connectivity().entities()[adj_elem.first]->element_type().dimensionality() != dimensions;
383  block.nb_points[i] = subdiv_row[i] + (block.bounded[i] ? 1 : 0);
384  block.segments[i] = subdiv_row[i];
385  block.nb_elems *= subdiv_row[i];
386  stride *= block.nb_points[i];
387  nb_points *= block.nb_points[i];
388  block.neighbors[i] = block.bounded[i] ? nullptr : &block_list[adj_elem.second];
389  }

393  for(Uint i = 0; i != dimensions; ++i)
394  {
395  FaceConnectivity::ElementReferenceT adj_elems[2];
396  adj_elems[0] = face_conn.adjacent_element(block_idx, negative_faces[i]);
397  adj_elems[1] = face_conn.adjacent_element(block_idx, positive_faces[i]);
398  // check for a patch both in the positive and negative direction
399  for(Uint dir = 0; dir != 2; ++dir)
400  {
401  const Uint patch_orientation = dir == 0 ? negative_faces[i] : positive_faces[i];
402  if(face_conn.node_connectivity().entities()[adj_elems[dir].first]->element_type().dimensionality() == (dimensions-1))
403  {
404  patch_map[face_conn.node_connectivity().entities()[adj_elems[dir].first]->parent()->name()].push_back(new Patch(block, i, dir * (block.nb_points[i]-1), patch_orientation));
405  }
406  else
407  {
408  cf3_assert(face_conn.node_connectivity().entities()[adj_elems[dir].first]->element_type().dimensionality() == dimensions);
409  cf3_assert(adj_elems[dir].second < block_regions.size());
410  const std::string& other_region = block_regions[adj_elems[dir].second];
411  if(other_region != my_region)
412  {
413  const std::string internal_patch_name = std::string("region_bnd_") + my_region + "_" + other_region;
414  patch_map[internal_patch_name].push_back(new Patch(block, i, dir * (block.nb_points[i]), patch_orientation));
415  }
416  }
417  }
418  }
419  block_start += nb_points;
420  }

556  mapped_coords[ETA] = (w[0][ETA]*eta[j][0] + w[1][ETA]*eta[j][1] + w[2][ETA]*eta[j][2] + w[3][ETA]*eta[j][3]) / w_mag[ETA];
557  mapped_coords[ZTA] = (w[0][ZTA]*zta[k][0] + w[1][ZTA]*zta[k][1] + w[2][ZTA]*zta[k][2] + w[3][ZTA]*zta[k][3]) / w_mag[ZTA];
558 
559  typename ET::SF::ValueT sf;
560  ET::SF::compute_value(mapped_coords, sf);
561 
562  // Transform to real coordinates
563  typename ET::CoordsT coords = sf * block_nodes;
564 
565  // Store the result
566  const Uint node_idx = to_local(block_list[block_idx][i][j][k].global_idx());
567  cf3_assert(node_idx < mesh_coords.size());
568  mesh_coords[node_idx][XX] = coords[XX];
569  mesh_coords[node_idx][YY] = coords[YY];
570  mesh_coords[node_idx][ZZ] = coords[ZZ];
571  }

577  template<typename ET>
578  void fill_block_coordinates_2d(Table<Real>& mesh_coords, const Uint block_idx)
579  {
580  typename ET::NodesT block_nodes;
581  fill(block_nodes, *points, (*blocks)[block_idx]);
582  const Table<Uint>::ConstRow& segments = (*block_subdivisions)[block_idx];
583  const Table<Real>::ConstRow& gradings = (*block_gradings)[block_idx];
584 
585  common::Table<Real>::ArrayT ksi, eta; // Mapped coordinates along each edge
586  detail::create_mapped_coords(segments[XX], &gradings[0], ksi, 2);
587  detail::create_mapped_coords(segments[YY], &gradings[2], eta, 2);
588 
589  Real w[2][2]; // weights for each edge
590  Real w_mag[2]; // Magnitudes of the weights
591  for(Uint j = 0; j <= segments[YY]; ++j)
592  {
593  for(Uint i = 0; i <= segments[XX]; ++i)
594  {
595  // Weights are calculating according to the BlockMesh algorithm
596  w[0][KSI] = (1. - ksi[i][0])*(1. - eta[j][0]) + (1. + ksi[i][0])*(1. - eta[j][1]);
597  w[1][KSI] = (1. - ksi[i][1])*(1. + eta[j][0]) + (1. + ksi[i][1])*(1. + eta[j][1]);
598  w_mag[KSI] = (w[0][KSI] + w[1][KSI]);
599 
600  w[0][ETA] = (1. - eta[j][0])*(1. - ksi[i][0]) + (1. + eta[j][0])*(1. - ksi[i][1]);
601  w[1][ETA] = (1. - eta[j][1])*(1. + ksi[i][0]) + (1. + eta[j][1])*(1. + ksi[i][1]);
602  w_mag[ETA] = (w[0][ETA] + w[1][ETA]);
603 
604  // Get the mapped coordinates of the node to add
605  typename ET::MappedCoordsT mapped_coords;
606  mapped_coords[KSI] = (w[0][KSI]*ksi[i][0] + w[1][KSI]*ksi[i][1]) / w_mag[KSI];
607  mapped_coords[ETA] = (w[0][ETA]*eta[j][0] + w[1][ETA]*eta[j][1]) / w_mag[ETA];
608 
609  typename ET::SF::ValueT sf;
610  ET::SF::compute_value(mapped_coords, sf);
611 
612  // Transform to real coordinates
613  typename ET::CoordsT coords = sf * block_nodes;
614 
615  // Store the result
616  const Uint node_idx = to_local(block_list[block_idx][i][j].global_idx());
617  cf3_assert(node_idx < mesh_coords.size());
618  mesh_coords[node_idx][XX] = coords[XX];
619  mesh_coords[node_idx][YY] = coords[YY];
620  }
621  }

806  bool build_block_layer(const Uint block_idx, const Uint direction, const std::vector<Uint>& transverse_directions, const std::vector<Uint>& output_block_layer, const std::vector<Uint>& added_blocks, std::set<Uint>& recursed_blocks)
807  {
808  if(!recursed_blocks.insert(block_idx).second)
809  return true;
810 
811  // Block was already found or is not part of the search
812  if(std::count(output_block_layer.begin(), output_block_layer.end(), block_idx) != 0 ||
813  std::count(added_blocks.begin(), added_blocks.end(), block_idx) != 0)
814  return true;
815 
816  cf3_assert(face_connectivity->has_adjacent_element(block_idx, direction));
817  const FaceConnectivity::ElementReferenceT adj_elem = face_connectivity->adjacent_element(block_idx, direction);
818  // If we have a volume element as neighbor in the main negative direction
819  // and it is not in the previous layer, then we are not in the current layer.
820  if(face_connectivity->node_connectivity().entities()[adj_elem.first]->element_type().dimensionality() == face_connectivity->node_connectivity().entities()[adj_elem.first]->element_type().dimension() &&
821  std::count(added_blocks.begin(), added_blocks.end(), adj_elem.second) == 0)
822  return false;
823 
824  // Recurse through the transverse directions
825  BOOST_FOREACH(const Uint transverse_direction, transverse_directions)
826  {
827  const FaceConnectivity::ElementReferenceT transverse_elem = face_connectivity->adjacent_element(block_idx, transverse_direction);
828  if(face_connectivity->node_connectivity().entities()[transverse_elem.first]->element_type().dimensionality() < face_connectivity->node_connectivity().entities()[transverse_elem.first]->element_type().dimension()) // skip faces
829  continue;
830  if(!build_block_layer(transverse_elem.second, direction, transverse_directions, output_block_layer, added_blocks, recursed_blocks))
831  return false;
832  }
833 
834  return true;
835  }

838  {
839  layer.added_blocks.insert(layer.added_blocks.end(), layer.global_layer.begin(), layer.global_layer.end());
840  layer.global_layer.clear();
841  const Uint nb_blocks = blocks->size();
842  for(Uint block_idx = 0; block_idx != nb_blocks; ++block_idx)
843  {
844  if(std::count(layer.added_blocks.begin(), layer.added_blocks.end(), block_idx) != 0)
845  continue;
846 
847  std::set<Uint> recursed_blocks;
848  if(build_block_layer(block_idx, start_direction, transverse_directions, layer.global_layer, layer.added_blocks, recursed_blocks))
849  layer.global_layer.push_back(block_idx);
850  }
851 
852  const Uint partition_begin = block_distribution[partition];
853  const Uint partition_end = block_distribution[partition+1];
854  layer.local_layer.clear();
855  layer.segment_size = 0;
856  layer.nb_segments = 0;
857  Table<Uint>& subdivs = *block_subdivisions;
858  const Uint dimensions = points->row_size();
859  BOOST_FOREACH(const Uint block_idx, layer.global_layer)
860  {
861  if(layer.nb_segments == 0)

867  cf3_assert(layer.nb_segments == subdivs[block_idx][direction]);
868  }

877  if(direction == 0) { i = 1; j = 2; }
878  if(direction == 1) { i = 0; j = 2; }
879  if(direction == 2) { i = 0; j = 1; }
880  layer.segment_size += subdivs[block_idx][i]*subdivs[block_idx][j];
881  }
882 
883  if(block_idx >= partition_begin && block_idx < partition_end)
884  layer.local_layer.push_back(block_idx);
885  }
886  layer.remaining_nb_segments = layer.nb_segments;

919  {
920  start_face_nodes.push_back(face_node_idx);
921  }
922

924  BlocksPartitioning::IndicesT end_face_nodes;

932  BlocksPartitioning::IndicesT transverse_axes;

934  {

936  transverse_axes.assign(1, direction == XX ? YY : XX);

939  {
940  if(direction == XX)
941  {
942  transverse_directions = boost::assign::list_of(Hexa::ETA_NEG)(Hexa::ETA_POS)(Hexa::ZTA_NEG)(Hexa::ZTA_POS);
943  transverse_axes = boost::assign::list_of(YY)(ZZ);

948  transverse_axes = boost::assign::list_of(XX)(ZZ);

951  {
952  transverse_directions = boost::assign::list_of(Hexa::ETA_NEG)(Hexa::ETA_POS)(Hexa::KSI_NEG)(Hexa::KSI_POS);
953  transverse_axes = boost::assign::list_of(YY)(XX);

973  {
974  for(Uint i = 0; i != (nb_partitions+1); ++i)
975  {
976  node_mapping[i][node_idx] = node_idx;
977  }
978  node_is_mapped[0][node_idx] = true;
979  node_is_mapped[nb_partitions][node_idx] = true;
980  }
981 
982  // total number of elements
983  Uint global_nb_elements = 0;
984  for(Uint block = 0; block != nb_blocks; ++block)
985  global_nb_elements += block_list[block].nb_elems;
986 
987  // All the blocks at the start of the direction to partition in
988  const Uint nb_existing_partitions = block_distribution.size()-1;
989  for(Uint existing_partition = 0; existing_partition != nb_existing_partitions; ++existing_partition)
990  {
991  const common::Table<Real>::ArrayT& block_coordinates = points->array();
992  BlocksPartitioning blocks_to_partition = blocks_in;
993  BlockLayer layer;
994  build_block_layer(direction, start_direction, transverse_directions, existing_partition, layer);
995 
996  print_vector(CFdebug << "Examining block layer: ", layer.local_layer); CFdebug << CFendl;
997 
998  // Size of one partition
999  const Uint partition_size = static_cast<Uint>( ceil( static_cast<Real>(global_nb_elements) / static_cast<Real>(nb_partitions) ) );
1000 
1001  Uint nb_partitioned = 0;
1002  Uint block_layer_offset = 0;
1003  for(Uint partition = 0; partition != nb_partitions; ++partition)
1004  {
1005  // Update block distribution
1006  blocks_out.block_distribution.push_back(blocks_out.block_points.size());
1007 
1008  Uint partition_remaining_size = partition == (nb_partitions-1) ? global_nb_elements-nb_partitioned : partition_size;
1009  while(partition_remaining_size)
1010  {
1011  // Get the total size of a slice of elements in the current direction
1012  cf3_assert(layer.segment_size);

1019  partition_remaining_size = global_nb_elements - nb_partitioned;

1023  bool is_last_layer = false;
1024

1026  is_first_layer = true;

1030  block_layer_offset += partition_nb_slices;
1031  BOOST_FOREACH(const Uint block_idx, layer.local_layer)
1032  {
1033  common::Table<Real>::ArrayT mapped_coords;

1035  &blocks_in.block_gradings[block_idx][mapped_stride*direction], mapped_coords, mapped_stride);

1046  {
