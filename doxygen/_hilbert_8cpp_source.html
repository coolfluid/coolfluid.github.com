143 void Hilbert::recursive_algorithm_1d(const Real& p)
144 {
145  m_min_distance=Consts::real_max();
146  for (m_idx=0; m_idx<2; ++m_idx)
147  {
148  m_distance = std::abs(p-m_box1[m_idx]);
149  if (m_distance < m_min_distance)
150  {
151  m_quadrant=m_idx;
152  m_min_distance = m_distance;
153  }
154  }
155  if(m_quadrant == A)
156  {
157  m_box1[B]=0.5*(m_box1[A]+m_box1[B]);
158  }
159  else
160  {
161  m_box1[A]=0.5*(m_box1[A]+m_box1[B]);
162  }
163 
164  m_key += (boost::uint64_t) std::ldexp((long double) 0.5*m_quadrant,(m_max_level-m_level));
165 
166  ++m_level;
167  if (m_level < m_max_level)
168  recursive_algorithm_1d(p);
169 
170 }

173 void Hilbert::recursive_algorithm_2d(const RealVector2& p)
174 {
175  m_min_distance=Consts::real_max();
176  for (m_idx=0; m_idx<4; ++m_idx)
177  {
178  m_distance = (p-m_box2[m_idx]).squaredNorm();
179  if (m_distance < m_min_distance)
180  {
181  m_quadrant=m_idx;
182  m_min_distance = m_distance;
183  }
184  }
185 
186  switch (m_quadrant)
187  {
188  case A:
189  m_tmp_box2[A] = m_box2[A];
190  m_tmp_box2[B] = 0.5*(m_box2[A]+m_box2[D]);
191  m_tmp_box2[C] = 0.5*(m_box2[A]+m_box2[C]);
192  m_tmp_box2[D] = 0.5*(m_box2[A]+m_box2[B]);
193  break;
194  case B:
195  m_tmp_box2[A] = 0.5*(m_box2[A]+m_box2[B]);
196  m_tmp_box2[B] = m_box2[B];
197  m_tmp_box2[C] = 0.5*(m_box2[B]+m_box2[C]);
198  m_tmp_box2[D] = 0.5*(m_box2[B]+m_box2[D]);
199  break;
200  case C:
201  m_tmp_box2[A] = 0.5*(m_box2[A]+m_box2[C]);
202  m_tmp_box2[B] = 0.5*(m_box2[B]+m_box2[C]);
203  m_tmp_box2[C] = m_box2[C];
204  m_tmp_box2[D] = 0.5*(m_box2[C]+m_box2[D]);
205  break;
206  case D:
207  m_tmp_box2[A] = 0.5*(m_box2[C]+m_box2[D]);
208  m_tmp_box2[B] = 0.5*(m_box2[B]+m_box2[D]);
209  m_tmp_box2[C] = 0.5*(m_box2[A]+m_box2[D]);
210  m_tmp_box2[D] = m_box2[D];
211  break;
212  }
213 
214  m_box2 = m_tmp_box2;
215 
216  m_key += (boost::uint64_t) std::ldexp((long double) 0.25*m_quadrant,2*(m_max_level-m_level));
