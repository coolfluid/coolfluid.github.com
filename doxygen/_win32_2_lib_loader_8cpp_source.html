44 void LibLoader::load_library(const std::string& lib)
45 {
46  CFLog( VERBOSE , "LibLoader: Attempting to load '" << lib << "'\n" );
47  using boost::filesystem::path;
48  std::string libname = lib + ".dll";
49 
50  // attempt to load a module
51  HINSTANCE hdl = nullptr;
52  std::vector<path>::const_iterator itr = m_search_paths.begin();
53  for (; itr != m_search_paths.end() ; ++itr)
54  {
55  // set the current search directory
56  CFLog ( VERBOSE, "searching in dir: '" << (*itr).string() << "'\n" );
57  SetDllDirectory((*itr).string().c_str());
58 
59  // try to load the library
60  hdl = LoadLibrary(TEXT(libname.c_str()));
61  if( hdl != nullptr ) break;
62  CFLog ( VERBOSE, "didnt find '" << lib << "' in dir '" << (*itr).string() << "'\n" );
63  }
64 
65  // searhc in the global path
66  hdl = LoadLibrary(TEXT(libname.c_str()));
67  if( hdl == nullptr )
68  {
69  char * pPath;
70  pPath = getenv ("PATH");
71  if (pPath != nullptr)
72  CFLog ( VERBOSE, "didnt find '" << lib << "' in global path variable PATH='" << pPath << "'\n" );
73  }
74 
75  // check for success
76  if(hdl != NULL)
77  {
78  CFLog( VERBOSE, "LibLoader: Loaded '" << libname << "'\n" );
79  }
80  else
81  {
82  CFLog( VERBOSE, "LoadLibrary() failed to load module : '" << libname << "'\n" );
83  throw LibLoadingError ("Module failed to load '" + libname + "'" );
84  }
85 }
