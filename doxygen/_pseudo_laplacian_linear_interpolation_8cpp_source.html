69 void PseudoLaplacianLinearInterpolation::pseudo_laplacian_weighted_linear_interpolation(const RealVector& t_point, const std::vector<RealVector>& s_points, std::vector<Real>& weights)
70 {
71  const Uint dim = s_points[0].size();
72  cf3_assert(t_point.size() == s_points[0].size());
73  switch (dim)
74  {
75  case DIM_3D:
76  {
77  Real Ixx(0),Ixy(0),Ixz(0),Iyy(0),Iyz(0),Izz(0), Rx(0),Ry(0),Rz(0), Lx,Ly,Lz, dx,dy,dz;
78  RealVector Dx(s_points.size());
79  RealVector Dy(s_points.size());
80  RealVector Dz(s_points.size());
81  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
82  {
83  dx = s_points[s_pt_idx][XX] - t_point[XX];
84  dy = s_points[s_pt_idx][YY] - t_point[YY];
85  dz = s_points[s_pt_idx][ZZ] - t_point[ZZ];
86 
87  Ixx += dx*dx;
88  Ixy += dx*dy;
89  Ixz += dx*dz;
90  Iyy += dy*dy;
91  Iyz += dy*dz;
92  Izz += dz*dz;
93 
94  Rx += dx;
95  Ry += dy;
96  Rz += dz;
97 
98  Dx[s_pt_idx]=dx;
99  Dy[s_pt_idx]=dy;
100  Dz[s_pt_idx]=dz;
101  }
102  Lx = (-(Iyz*Iyz*Rx) + Iyy*Izz*Rx + Ixz*Iyz*Ry - Ixy*Izz*Ry - Ixz*Iyy*Rz + Ixy*Iyz*Rz)/
103  (Ixz*Ixz*Iyy - 2.*Ixy*Ixz*Iyz + Ixy*Ixy*Izz + Ixx*(Iyz*Iyz - Iyy*Izz));
104  Ly = (Ixz*Iyz*Rx - Ixy*Izz*Rx - Ixz*Ixz*Ry + Ixx*Izz*Ry + Ixy*Ixz*Rz - Ixx*Iyz*Rz)/
105  (Ixz*Ixz*Iyy - 2.*Ixy*Ixz*Iyz + Ixx*Iyz*Iyz + Ixy*Ixy*Izz - Ixx*Iyy*Izz);
106  Lz = (-(Ixz*Iyy*Rx) + Ixy*Iyz*Rx + Ixy*Ixz*Ry - Ixx*Iyz*Ry - Ixy*Ixy*Rz + Ixx*Iyy*Rz)/
107  (Ixz*Ixz*Iyy - 2.*Ixy*Ixz*Iyz + Ixy*Ixy*Izz + Ixx*(Iyz*Iyz - Iyy*Izz));
108 
109  Real S(0);
110  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
111  {
112  weights[s_pt_idx] = 1.0 + Lx*Dx[s_pt_idx] + Ly*Dy[s_pt_idx] + Lz*Dz[s_pt_idx];
113  S += weights[s_pt_idx];
114  }
115 
116  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
117  weights[s_pt_idx] /= S;
118  return;
119  }
120  case DIM_2D:
121  {
122  Real Ixx(0),Ixy(0),Iyy(0), Rx(0),Ry(0), Lx,Ly, dx,dy;
123  RealVector Dx(s_points.size());
124  RealVector Dy(s_points.size());
125  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
126  {
127  dx = s_points[s_pt_idx][XX] - t_point[XX];
128  dy = s_points[s_pt_idx][YY] - t_point[YY];
129 
130  Ixx += dx*dx;
131  Ixy += dx*dy;
132  Iyy += dy*dy;
133 
134  Rx += dx;
135  Ry += dy;
136 
137  Dx[s_pt_idx]=dx;
138  Dy[s_pt_idx]=dy;
139  }
140  Lx = (Ixy*Ry - Iyy*Rx)/(Ixx*Iyy-Ixy*Ixy);
141  Ly = (Ixy*Rx - Ixx*Ry)/(Ixx*Iyy-Ixy*Ixy);
142 
143  Real S(0);
144  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
145  {
146  weights[s_pt_idx] = 1.0 + Lx*Dx[s_pt_idx] + Ly*Dy[s_pt_idx] ;
147  S += weights[s_pt_idx];
148  }
149  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
150  weights[s_pt_idx] /= S;
151  return;
152  }
153  case DIM_1D:
154  {
155  Real Ixx(0), Rx(0), Lx, dx;
156  RealVector Dx(s_points.size());
157  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
158  {
159  dx = s_points[s_pt_idx][XX] - t_point[XX];
160 
161  Ixx += dx*dx;
162

164

166  }

168

171  {
172  weights[s_pt_idx] = 1.0 + Lx*Dx[s_pt_idx];
173  S += weights[s_pt_idx];
174  }
175  for (Uint s_pt_idx=0; s_pt_idx<s_points.size(); ++s_pt_idx)
176  weights[s_pt_idx] /= S;
