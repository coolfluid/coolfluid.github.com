117 template<Uint dim>
118 struct DNSCollisionKernel
119 {
120  typedef Eigen::Matrix<Real, dim, 1> VectorT;
121  typedef Eigen::Matrix<Real, dim, dim> MatrixT;
122 
123  DNSCollisionKernel(const mesh::Field& particle_velocity_field, const std::vector<mesh::Field*>& concentration_fields, const std::vector<mesh::Field*>& weighted_volume_fields, const std::vector<mesh::Field*>& gradient_fields, const Real reference_volume, const physics::PhysModel& physical_model) :
124  m_particle_velocity_field(particle_velocity_field),
125  m_concentration_fields(concentration_fields),
126  m_weighted_volume_fields(weighted_volume_fields),
127  m_gradient_fields(gradient_fields),
128  m_reference_volume(reference_volume)
129  {
130  }
131 
132  Real apply(const Uint node_idx, const Uint alpha, const Uint gamma)
133  {
134  const Real vol_alpha = (*m_weighted_volume_fields[alpha])[node_idx][0] / (*m_concentration_fields[alpha])[node_idx][0] * m_reference_volume;
135  const Real vol_gamma = (*m_weighted_volume_fields[gamma])[node_idx][0] / (*m_concentration_fields[gamma])[node_idx][0] * m_reference_volume;
136  const Eigen::Map<VectorT const> v_alpha(&m_particle_velocity_field[node_idx][alpha*dim]);
137  const Eigen::Map<VectorT const> v_gamma(&m_particle_velocity_field[node_idx][gamma*dim]);
138  const Real r_alpha = ::pow(3./(4.*pi())*vol_alpha, 1./3.);
139  const Real r_gamma = ::pow(3./(4.*pi())*vol_gamma,1./3.);
140  const Real r_col_3 = pow_int(r_alpha + r_gamma, 3); // Collision radius ^3
141 
142  // Effect of different particle size:
143  const Real beta1 = (v_alpha - v_gamma).norm() * pi() * pow_int(r_alpha + r_gamma, 2);
144 
145  // Velocity gradient tensor (transposed, but not important here)
146  const Eigen::Map<MatrixT> g(&((*m_gradient_fields[gamma])[node_idx][0]));
147  // Rate of strain tensor
148  const MatrixT s = (g + g.transpose())/2.;
149  std::vector<Real> ev(3, 0.); // vector so we can use std::sort
150  Eigen::Map<VectorT>(ev.data()) = s.template selfadjointView<Eigen::Upper>().eigenvalues();
151  std::sort(ev.begin(), ev.end(), std::greater<double>()); // Sort from largest to smallest
152  Real beta2 = 0;
153  const Real approx_zero = 1e-8;
154  if(ev[2] >= -approx_zero) // All >= 0
155  {
156  beta2 = 0.;
157  }
158  else if(ev[0] < approx_zero) // All < 0
159  {
160  beta2 = -4./3.*pi()*r_col_3*(ev[0]+ev[1]+ev[2]);
161  }
162  // The above conditions ensure that always ev[0] > 0 and ev[2] < 0
163  else if(::fabs(ev[1]) < approx_zero)
164  {
165  cf3_assert(ev[2]<0);
166  const Real k1 = -ev[0]/ev[2];
167  const Real sqrt_k1 = ::sqrt(k1);
168  beta2 = 8.*(-ev[2])*r_col_3*( 1./3*( sqrt_k1 + (k1-1)*::atan(sqrt_k1) ) - pi()/6.*(k1-1.) );
169  }
170  else if(ev[1] >= 0.)
171  {
172  const Real k1 = -ev[0]/ev[2];
173  const Real k2 = -ev[1]/ev[2];
174  beta2 = 8.*(-ev[2])*r_col_3*(collision_integral(k1, k2) - pi()/6.*(k1+k2-1.));
175  }
176  else if(ev[1] < 0.)
177  {
178  cf3_assert(ev[0] > 0);
179  const Real k1 = -ev[1]/ev[0];
180  const Real k2 = -ev[2]/ev[0];
181  beta2 = 8.*ev[0]*r_col_3*(collision_integral(k1, k2));
182  }
183  else
184  {
185  std::stringstream msg;
186  msg << "Unknown combination of eigenvalues: " << ev[0] << ", " << ev[1] << ", " << ev[2];
187  CFdebug << msg.str() << CFendl;
188  beta2 = 0.;
189  }
190 
191  return beta1 + beta2;
192  }
193 
194  // Numerical integration needed for the general case
195  Real collision_integral(const Real k1, const Real k2)
196  {
197  // The gauss points and weights
198  static const Real gauss[6][2] = {
199  {0.053038319518181845,0.1345579416596007},
200  {0.26608552564569055,0.28334147689638234},
201  {0.5979870928963115,0.36749874484146516},
202  {0.9728092338985851,0.36749874484146516},
203  {1.304710801149206,0.28334147689638234},
204  {1.5177580072767147,0.1345579416596007}
205  };
206 
207  Real result = 0;
208  for(int i = 0; i != 6; ++i)
209  {
210  const Real phi = gauss[i][0];
211  const Real cos_sin = k1*pow_int(::cos(phi),2) + k2*pow_int(::sin(phi),2);
212  result += gauss[i][1] * 2./3. * ::pow(cos_sin, 1.5)/::sqrt(1 + cos_sin);
213  }
214  return result;
215  }

245 struct MomentSourceFunctor : FunctionBase
246 {
247  typedef void result_type;
248 
249  MomentSourceFunctor(mesh::Region& region, const std::vector<std::string>& concentration_tags, const std::vector<std::string>& weighted_volume_tags, const std::vector<std::string>& gradient_tags, const Real reference_volume, const physics::PhysModel& physical_model) :
250  m_nb_phases(concentration_tags.size()),
251  m_mat(2*m_nb_phases, 2*m_nb_phases),
252  m_rhs(2*m_nb_phases)
253  {
254  mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(region);
255  mesh::Dictionary& dict = mesh.geometry_fields(); // TODO: Generalize to other dicts
256  m_source_field = &(dict.field("moment_source_terms"));
257 
258  m_concentration_fields.reserve(m_nb_phases);
259  m_weighted_volume_fields.reserve(m_nb_phases);
260  for(Uint i = 0; i != m_nb_phases; ++i)
261  {
262  m_concentration_fields.push_back(&(dict.field(concentration_tags[i])));
263  m_weighted_volume_fields.push_back(&(dict.field(weighted_volume_tags[i])));
264  m_gradient_fields.push_back(&(dict.field(gradient_tags[i])));
265  }
266  m_beta.reset(new CollisionKernelT(dict.field("ufem_particle_velocity"), m_concentration_fields, m_weighted_volume_fields, m_gradient_fields, reference_volume, physical_model));
267  m_collision_rate_field = Handle<mesh::Field>(dict.get_child("collision_rate")).get();
268  }
269 
270  ~MomentSourceFunctor()
271  {
272  m_source_field->synchronize();
273  }
274 
275  void operator()(const Uint node_idx)
276  {
277  const Uint nb_moments = 2*m_nb_phases;
278  m_rhs.setZero();
279  for(Uint alpha = 0; alpha != m_nb_phases; ++alpha) // For all phases
280  {
281  const Real n_alpha = (*m_concentration_fields[alpha])[node_idx][0];
282  const Real vol_alpha = (*m_weighted_volume_fields[alpha])[node_idx][0] / n_alpha;
283  for(int ki = 0; ki != nb_moments; ++ki) // For all moments
284  {
285  const Real k = static_cast<Real>(ki);
286  m_mat(ki, alpha) = (1.-k) * pow_int(vol_alpha, ki);
287  m_mat(ki, alpha+m_nb_phases) = k * pow_int(vol_alpha, ki-1);
288  }
289  for(Uint gamma = 0; gamma != m_nb_phases; ++gamma)
290  {
291  const Real n_gamma = (*m_concentration_fields[gamma])[node_idx][0];
292  const Real vol_gamma = (*m_weighted_volume_fields[gamma])[node_idx][0] / n_gamma;
293  const Real beta = m_beta->apply(node_idx, alpha, gamma);
294 // std::cout << "computing for vol_alpha " << vol_alpha << ", vol_gamma: " << vol_gamma << ", n_alpha " << n_alpha << ", n_gamma " << n_gamma << ", beta " << beta << std::endl;
295 // std::cout << "collision value: " << (beta*n_alpha*n_gamma) << std::endl;
296  for(int k = 0; k != nb_moments; ++k ) // For all moments
297  {
298  m_rhs[k] += (0.5*pow_int(vol_alpha + vol_gamma, k ) - pow_int(vol_alpha, k )) * (beta*n_alpha*n_gamma);
299  }
300  if(is_not_null(m_collision_rate_field))
301  {
302  const Uint var_idx = alpha*m_nb_phases + gamma;
303  cf3_assert(var_idx < m_collision_rate_field->row_size());
304  (*m_collision_rate_field)[node_idx][var_idx] = n_alpha*n_gamma*beta;
305  }
306  }
307  }
308 
309  Eigen::Map<RealVector> x(&(*m_source_field)[node_idx][0], nb_moments);
310 
311  // Check for NaN
312  for(Uint k = 0; k != nb_moments; ++k)
313  {
314  if(!std::isfinite(m_rhs[k]))
315  {
316  x.setZero();
317  return;
318  }
319  }
320 
321  Eigen::FullPivLU<RealMatrix> lu(m_mat);
322  if(!lu.isInvertible())
323  {
324  Eigen::JacobiSVD<RealMatrix> svd(m_mat, Eigen::ComputeThinU | Eigen::ComputeThinV);
325  x = svd.solve(m_rhs);
326  }
327  else
328  {
329  x = lu.solve(m_rhs);
330  }
331 
332 // std::cout << "matrix:\n" << m_mat << std::endl;
333 // std::cout << "rhs: " << m_rhs.transpose() << std::endl;
334 // std::cout << "x: " << x.transpose() << std::endl;
335  }
336 
337  mesh::Field* m_source_field;
338  std::vector<mesh::Field*> m_concentration_fields;
339  std::vector<mesh::Field*> m_weighted_volume_fields;
340  std::vector<mesh::Field*> m_gradient_fields;
341 
342  const Uint m_nb_phases;
343  RealMatrix m_mat;
344  RealVector m_rhs;
345 
346  boost::scoped_ptr<CollisionKernelT> m_beta;
347  mesh::Field* m_collision_rate_field;
348 };

608  {
