45 void TermComputer::compute_term(mesh::Field& term, mesh::Field& wave_speed)
46 {
47  term = 0.;
48  boost_foreach( const Handle<mesh::Entities const>& cells, term.entities_range() )
49  {
50  if (loop_cells(cells))
51  {
52  const mesh::Space& space = term.space(*cells);
53  const Uint nb_elems = space.size();
54  const Uint nb_nodes_per_elem = space.shape_function().nb_nodes();
55  for (Uint e=0; e<nb_elems; ++e)
56  {
57  compute_term(e,m_tmp_term,m_tmp_ws);
58  for (Uint s=0; s<nb_nodes_per_elem; ++s)
59  {
60  const Uint p=space.connectivity()[e][s];
61  for (Uint eq=0; eq<m_tmp_term[s].size(); ++eq)
62  {
63  term[p][eq] += m_tmp_term[s][eq];
64  }
65  wave_speed[p][0] = m_tmp_ws[s];
66  }
67  }
68  }
69  }
70 }
