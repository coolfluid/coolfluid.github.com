118 void PythonCodeContainer::repaint_border_area(QPaintEvent *event){
119  if (BorderArea::debug_arrow==NULL){

121  BorderArea::debug_arrow=new QPixmap(":/Icons/debug_arrow.png");
122  }
123  QPainter painter(border_area);
124  painter.fillRect(event->rect(), Qt::lightGray);
125  QTextBlock block = firstVisibleBlock();
126  int block_number = block.blockNumber();
127  int vertical_displace=tool_bar->height();
128  int top = (int) blockBoundingGeometry(block).translated(contentOffset()).top()+vertical_displace;
129  int bottom = top + (int) blockBoundingRect(block).height();
130  int debug_arrow_block=python_console->get_debug_arrow_block();

133  if (block.isVisible() && bottom >= event->rect().top()) {
134  switch (block.userState()){
135  case PROMPT_1://display prompt 1
136  painter.drawText(0, top, border_width-4, fontMetrics().height(),
137  Qt::AlignRight, ">>>");
138  break;
139  case PROMPT_2://display prompt 2
140  painter.drawText(0, top, border_width-4, fontMetrics().height(),
141  Qt::AlignRight, "...");
142  break;
143  case LINE_NUMBER://display bloc number
144  painter.drawText(0, top, border_width-4, fontMetrics().height(),
145  Qt::AlignRight, QString::number(block_number + 1));
146  break;
147  default:
148  painter.drawText(0, top, border_width-4, fontMetrics().height(),
149  Qt::AlignRight, QString::number(block.userState()));
150  }
151  int break_point_index=break_points.indexOf(block_number);
152  if (break_point_index!=-1)
153  painter.drawPixmap(0,top,16,16,*BorderArea::break_point);
154  if (block_number == debug_arrow_block)
155  painter.drawPixmap(2,top,16,16,*BorderArea::debug_arrow);
156  }
157  block = block.next();
158  top = bottom;
159  bottom = top + (int) blockBoundingRect(block).height();

161  }

192 void PythonCodeContainer::keyPressEvent(QKeyEvent *e){
193  static QRegExp indent_case("^[^#:]*:[ ]*(#[^$]*)?$");
194  QTextCursor c=textCursor();
195  QTextCursor temp;
196  if (editable_zone(c)){
197  if (completer != NULL && completer->popup() != NULL && completer->popup()->isVisible()){
198  switch(e->key()){
199  case Qt::Key_Backspace:
200  if (editable_zone(c)){
201  temp=c;temp.movePosition(QTextCursor::Left);
202  if (editable_zone(temp)){
203  QPlainTextEdit::keyPressEvent(e);
204  break;
205  }
206  }
207  case Qt::Key_Enter:
208  case Qt::Key_Escape:
209  case Qt::Key_Tab:
210  case Qt::Key_Backtab:
211  case Qt::Key_Return:
212  e->ignore();
213  return;
214  default:
215  QPlainTextEdit::keyPressEvent(e);
216  }
217  QString word_under_cursor=get_word_under_cursor(c);
218  QAbstractItemView* popup=completer->popup();
219  if (e->text().size()>=0){//text entered
220  //completer->setModel(new QStringListModel(generate_dictionary(word_under_cursor)));
221  completer->setCompletionPrefix(word_under_cursor);
222  popup->setCurrentIndex(completer->completionModel()->index(0,0));
223  QRect completer_rect=cursorRect(c);
224  completer_rect.moveTopLeft(completer_rect.topLeft()+offset_border);
225  completer_rect.setWidth(popup->sizeHintForColumn(0)
226  +popup->verticalScrollBar()->sizeHint().width());
227  completer->complete(completer_rect);
228  }
229  }else{
230  if (e->key()==Qt::Key_Return){//auto indentation
231  QTextCursor c=textCursor();
232  QString current_line=document()->findBlockByNumber(c.blockNumber()).text();
233  int tab_number=0;
234  for (int i=0;i<current_line.size();i++){
235  if (current_line[i]=='\t')
236  tab_number++;
237  else
238  break;
239  }
240  if (indent_case.indexIn(current_line) > -1)
241  tab_number++;
242  new_line(tab_number);
243  }else if (e->key()==Qt::Key_Escape){
244  QString word_under_cursor=get_word_under_cursor(c);
245  if (word_under_cursor.length() >= 0){
246  setTextCursor(c);
247  if (static_cast<PythonCodeContainer*>(completer->widget()) != NULL)
248  disconnect(completer,SIGNAL(activated(QString)),static_cast<PythonCodeContainer*>(completer->widget()),SLOT(insert_completion(QString)));
249  completer->setWidget(this);
250  //completer->setModel(new QStringListModel(generate_dictionary(word_under_cursor)));
251  completer->setCompletionPrefix(word_under_cursor);
252  QAbstractItemView* popup=completer->popup();
253  popup->setCurrentIndex(completer->completionModel()->index(0,0));
254  QRect completer_rect=cursorRect();
255  completer_rect.moveTopLeft(completer_rect.topLeft()+offset_border);
256  completer_rect.setWidth(popup->sizeHintForColumn(0)
257  +popup->verticalScrollBar()->sizeHint().width());
258  completer->complete(completer_rect);
259  connect(completer,SIGNAL(activated(QString)),this,SLOT(insert_completion(QString)));
260  }
261  }else{
262  key_press_event(e);
263  }
264  }
265  }else{//allow copy in uneditable zones
266  if (e->modifiers() == Qt::ControlModifier && e->key() == Qt::Key_C)
267  QPlainTextEdit::keyPressEvent(e);
268  else
269  key_press_event(e);
270  }
271  ensureCursorVisible();
272 }

363 void PythonCodeContainer::add_to_dictionary(int &i,const QStringList &add,CustomStandardItem *item){
364  while(i<add.size()){
365  QString s=add[i];
366  QChar c=s[s.size()-1];
367  if (c=='{' || c==':'){
368  s.chop(1);
369  int two_point=s.indexOf(':');
370  QString item_path=s.mid(0,two_point);
371  CustomStandardItem *n_value;
372  if (two_point != -1)
373  n_value=new CustomStandardItem(s.mid(two_point+1));
374  else
375  n_value=new CustomStandardItem("");
376  CustomStandardItem *current_item=item;
377  QStringList path_list=item_path.split('.');
378  for (int j=0;j<path_list.size()-1;j++){
379  for (int k=0;k<current_item->rowCount();k++){
380  CustomStandardItem *n_item=(CustomStandardItem*)current_item->child(k);
381  if (n_item->text()==path_list[j]){
382  current_item=n_item;
383  }
384  }
385  }
386  CustomStandardItem *n_item=new CustomStandardItem(path_list.last());
387  current_item->appendRow(n_item);
388  current_item->setChild(n_item->row(),1,n_value);
389  i++;
390  if (c!=':')
391  add_to_dictionary(i,add,n_item);
392  }else if (c=='}'){
393  i++;
394  return;
395  }else{
396  CustomStandardItem *n_item=new CustomStandardItem(s);
397  item->appendRow(n_item);
398  i++;
399  }
400  }
401 }

469 QString PythonCodeContainer::get_word_under_cursor(QTextCursor &c){
470  QString block=c.block().text();
471  static QRegExp complete_word("[\\w\\.]+");
472  int position_in_block=c.position() - c.block().position();
473  int index = complete_word.indexIn(block);
474  while (index >= 0) {
475  int length = complete_word.matchedLength();
476  if (index < position_in_block && index+length >= position_in_block){
477  c.setPosition(c.block().position()+index+length);
478  return block.mid(index,length);
479  }
480  index = complete_word.indexIn(block, index + length);
481  }
482  return "";
483 }
