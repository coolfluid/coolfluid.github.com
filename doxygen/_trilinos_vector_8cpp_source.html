71 void TrilinosVector::create_blocked(common::PE::CommPattern& cp, const VariablesDescriptor& vars, const std::vector<Uint>& periodic_links_nodes, const std::vector<bool>& periodic_links_active)
72 {
73  // if built
74  if (m_is_created) destroy();
75 
76  // prepare intermediate data
77  int nmyglobalelements=0;
78  std::vector<int> myglobalelements(0);
79 
80  std::vector<Uint> my_ranks;
81  create_map_data(cp, vars, m_p2m, myglobalelements, my_ranks, nmyglobalelements, periodic_links_nodes, periodic_links_active);
82 
83  m_data.resize(myglobalelements.size());
84 
85  std::vector<Uint> gids(myglobalelements.begin(), myglobalelements.end()); // need Uint data for GIDs
86 
87  if(is_not_null(get_child("CommPattern")))
88  remove_component("CommPattern");
89  m_comm_pattern = common::allocate_component<common::PE::CommPattern>("CommPattern");
90  m_comm_pattern->insert("gid",gids,1,false);
91  m_comm_pattern->setup(Handle<common::PE::CommWrapper>(m_comm_pattern->get_child("gid")),my_ranks);
92 
93  m_comm_pattern->insert(name(), m_data, true);
94 
95  // map (its actually blockmap insteady of rowmap, to involve ghosts)
96  m_map = Teuchos::rcp(new Epetra_Map(-1,nmyglobalelements,&myglobalelements[0],0,m_comm));
97  // create vector
98  m_vec=Teuchos::rcp(new Epetra_Vector(View, *m_map, &m_data[0]));
99 
100  m_neq=vars.size();
101  m_blockrow_size=cp.isUpdatable().size();
102  m_is_created=true;
103 }
