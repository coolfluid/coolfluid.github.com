282 void History::write_file(boost::filesystem::fstream& file)
283 {
284  // Write header, containing the variables
285  // format: # var1 var2 vector[0] vector[1]
286  file << file_header();
287 
288  flush();

292  {
293  const Uint var_begin = m_variables->offset(var_idx);

295  for (Uint i=0; i<var_length; ++i)
296  file << "\t" <<std::scientific << std::setw(16) << (*m_table)[row][var_begin+i];
297  }

304 void History::read_file(boost::filesystem::fstream& file)
305 {
306  bool logging = m_logging;
