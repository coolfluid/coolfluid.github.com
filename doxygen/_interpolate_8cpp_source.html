75 void Interpolate::execute()
76 {
77  if (is_null(m_source))
78  throw SetupError(FromHere(),uri().string()+": Source field not configured");
79 
80  if (is_null(m_target))
81  throw SetupError(FromHere(),uri().string()+": Target field not configured");
82 
83  const Field& source = *m_source;
84  Field& target = *m_target;
85 
86  if (source.row_size() != target.row_size())
87  throw BadValue(FromHere(),"Field "+source.uri().string()+" has "+to_str(source.row_size())+" variables.\n"
88  "Field "+target.uri().string()+" has "+to_str(target.row_size())+" variables.");
89  const Uint nb_vars(source.row_size());
90 
91  if(find_parent_component_ptr<Mesh>(source) == find_parent_component_ptr<Mesh>(target))
92  {
94  boost_foreach(const Handle<Entities>& elements_handle, target.dict().entities_range())
95  {
96  Entities& elements = *elements_handle;
97  if (source.dict().defined_for_entities(elements_handle) == false)
98  continue;
99  // throw BadValue(FromHere(),"Source field "+source.uri().string()+" is not defined in elements "+elements.uri().string());
100 
101  const Space& s_space = source.space(elements);
102  const Space& t_space = target.space(elements);
103  const ShapeFunction& s_sf = s_space.shape_function();
104  const ShapeFunction& t_sf = t_space.shape_function();
105 
107  RealMatrix interpolate(t_sf.nb_nodes(),s_sf.nb_nodes());
108  for (Uint t_pt = 0; t_pt<t_sf.nb_nodes(); ++t_pt)
109  interpolate.row(t_pt) = s_sf.value(t_sf.local_coordinates().row(t_pt));
110 
112  for (Uint e=0; e<elements.size(); ++e)
113  {
114  Connectivity::ConstRow s_field_indexes = s_space.connectivity()[e];
115  Connectivity::ConstRow t_field_indexes = t_space.connectivity()[e];
116 
119  for (Uint t_pt=0; t_pt<t_sf.nb_nodes(); ++t_pt)
120  {
121  for (Uint var=0; var<nb_vars; ++var)
122  target[t_field_indexes[t_pt]][var] = 0.;
123 
124  for (Uint s_pt=0; s_pt<s_sf.nb_nodes(); ++s_pt)
125  {
126  for (Uint var=0; var<nb_vars; ++var)
127  target[t_field_indexes[t_pt]][var] += interpolate(t_pt,s_pt) * source[s_field_indexes[s_pt]][var];
128  }
129  }
130  }
131  }
132  }
133  else // fields are on different meshes
134  {
135  interpolate(source,target.coordinates(),target);
136  }
137 }

169  }

171  const Uint dimension = source_mesh.dimension();
172  const Uint nb_vars = source.row_size();
173  m_source = Handle<Field const>(source.handle<Component>());
174 
175  Entity element;
176  std::deque<Uint> missing_cells;
177 
178  RealVector coord(dimension); coord.setZero();
179  const Uint target_dim = coordinates.row_size();
180 
181  for(Uint i=0; i<coordinates.size(); ++i)
182  {
183  for (Uint d=0; d<target_dim; ++d)
184  coord[d] = coordinates[i][d];
185  if( m_octtree->find_element(coord,element) )
186  {
187  interpolate_coordinate( coord, *element.comp, element.idx, target[i] );

189  }
190  else
191  {
192 // std::cout << PERank<< "coord " << coord.transpose() << " not found " << std::endl;
193  for (Uint v=0; v<nb_vars; ++v)
194  target[i][v] = math::Consts::real_max();

196  }
197  }
198

200  std::vector<Real> recv_coords;
201 
202  Uint c(0);
203  boost_foreach(const Uint i, missing_cells)
204  {
205  for(Uint d=0; d<target_dim; ++d)
206  send_coords[c++]=coordinates[i][d];
207  }
208 
209  for (Uint root=0; root<PE::Comm::instance().size(); ++root)
210  {
211 
212 // if (root == PE::Comm::instance().rank())
213 // {
214 // std::cout << PERank << "send coords = ";
215 // for(Uint i=0; i<send_coords.size(); ++i)
216 // std::cout << send_coords[i] << " ";

223 // {

226 // std::cout << recv_coords[i] << " ";
227 // std::cout << std::endl;
228 // }
229 
230 
231  // size is only because it doesn't get resized for this rank
232  std::vector<Real> send_target_rows(nb_vars*missing_cells.size(),math::Consts::uint_max());
233 
234  if (root!=Comm::instance().rank())
235  {
236  std::vector<RealVector> recv_coordinates(recv_coords.size()/target_dim) ;
237  boost_foreach(RealVector& realvec, recv_coordinates)
238  realvec.resize(target_dim);
239 
240  c=0;
241  for (Uint i=0; i<recv_coordinates.size(); ++i)
242  {
243  for(Uint d=0; d<target_dim; ++d)
244  recv_coordinates[i][d]=recv_coords[c++];
245  }
246 
247  send_target_rows.resize(recv_coordinates.size()*nb_vars);
248  for (Uint i=0; i<recv_coordinates.size(); ++i)
249  {
250  for (Uint d=0; d<target_dim; ++d)
251  coord[d] = recv_coordinates[i][d];
252 
253  if( m_octtree->find_element(coord,element) )
254  {
255 // std::cout<< PERank << " send to " << root << ": interpolate for coord (" << coord.transpose() << ") in " << element_component->uri().path() << "["<<element_idx<<"]" << std::endl;
256  boost::multi_array<Real,2> target_row(boost::extents[1][nb_vars]);
257  interpolate_coordinate( coord, *element.comp, element.idx, target_row[0] );
258  for (Uint v=0; v<nb_vars; ++v)
259  send_target_rows[i*nb_vars+v] = target_row[0][v];
260  }
261  else
262  {
263 // std::cout << PERank<< " send to " << root << ": coord " << coord.transpose() << " not found " << std::endl;
264  for (Uint v=0; v<nb_vars; ++v)
265  send_target_rows[i*nb_vars+v] = math::Consts::real_max();
266  }
267  }
268  }
269 
270  std::vector<Real> recv_target_rows(nb_vars*missing_cells.size()*Comm::instance().size());
271  PE::Comm::instance().gather(send_target_rows,recv_target_rows,root);
272 
273  if( root==Comm::instance().rank())
274  {
275  const Uint nb_cells = missing_cells.size();
276  for (Uint i=0; i<missing_cells.size(); ++i)
277  {
278  for(Uint p=0; p<Comm::instance().size(); ++p)
279  {
280  for(Uint v=0; v<nb_vars; ++v)
281  target[missing_cells[i]][v] = std::min(recv_target_rows[v+nb_vars*(i+p*nb_cells)] , target[missing_cells[i]][v]);
282  }
283  }
284  } // endif root==rank
285  } // end foreach root
286 
287 
288  missing_cells.clear();
289  for (Uint i=0; i<target.size(); ++i)
290  {
291  bool found = true;
292  for (Uint v=0; v<target.row_size(); ++v)
293  {
294  if (target[i][v] == math::Consts::uint_max())
295  {
296  target[i][v] = 0.;
297  found &= false;
298  }
299  }
300  if (!found)
301  missing_cells.push_back(i);
302  }
303  if(missing_cells.size())
304  {
305  std::cout << PERank << "could not interpolate " << missing_cells.size() << " coordinates because point was not found on source mesh. Values set to zero for coordinates : ";
306  for(Uint i=0; i<missing_cells.size(); ++i)
307  {
308  std::cout << "(";

310  {
311  if (d!=0) std::cout << " ";
312  std::cout << coordinates[missing_cells[i]][d];
313  }
314  std::cout << ") ";
315  }
316  std::cout << std::endl;
317  }

323 void Interpolate::interpolate_coordinate(const RealVector& target_coord, const Entities& element_component, const Uint element_idx, Field::Row target_row)
324 {
325  cf3_assert(is_null(m_source) == false);
326  const Field& source = *m_source;
327  const Space& source_space = source.space(element_component);
328  const ShapeFunction& sf = source_space.shape_function();
329 
330  RealMatrix source_geom_nodes(element_component.element_type().nb_nodes(),element_component.element_type().dimension());
331  element_component.geometry_space().put_coordinates(source_geom_nodes,element_idx);
332  RealVector local_coord(sf.dimensionality());
333  element_component.element_type().compute_mapped_coordinate(target_coord,source_geom_nodes,local_coord);
334  RealRowVector sf_value(sf.nb_nodes());
335  sf.compute_value(local_coord,sf_value);
336 
337  Connectivity::ConstRow source_indexes = source_space.connectivity()[element_idx];
338  for(Uint v=0; v<target_row.size(); ++v)
339  {
340  target_row[v]=0.;
341  for(Uint i=0; i<source_indexes.size(); ++i)
342  {
343  target_row[v] += source[source_indexes[i]][v] * sf_value[i];
344  }
345  }
346 }
