69  int setup_solver()
70  {
71  if(is_null(m_matrix))
72  throw common::SetupError(FromHere(), "Null matrix for " + m_self.uri().path());
73 
74  if(is_null(m_rhs))
75  throw common::SetupError(FromHere(), "Null RHS for " + m_self.uri().path());
76 
77  if(is_null(m_solution))
78  throw common::SetupError(FromHere(), "Null solution vector for " + m_self.uri().path());
79 
80  if(!m_factory.Query(m_solver_type))
81  throw common::SetupError(FromHere(), "Solver " + m_solver_type + " is not available.");
82 
83  m_problem.SetOperator(m_matrix->epetra_matrix().get());
84  m_problem.SetLHS(m_solution->epetra_vector().get());
85  m_problem.SetRHS(m_rhs->epetra_vector().get());
86 
87  // Create the solver
88  m_solver.reset(m_factory.Create(m_solver_type, m_problem));
89 
90  // Factorize the matrix
91  CFdebug << "Directstrategy: factoring matrix" << CFendl;
92  m_solver->SymbolicFactorization();
93  m_solver->NumericFactorization();
94 
95  return 0;
96  }
