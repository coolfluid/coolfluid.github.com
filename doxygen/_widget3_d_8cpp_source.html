311 void Widget3D::connect_to_server(QString given_host,QString port)
312 {
313 
314  //Be sure we are not connected yet and all widget are in a correct state
315  disconnect_from_server();
316 
317  //pqServerResource configuration
318  QString host = "cs://"; //cs:// =>client-server rc:// => remote server connection (not implemented yet by paraview)
319  host += given_host;
320  host += ":";
321  host += port;
322 
323  //create new remote server
324  m_server = m_object_builder->createServer(pqServerResource(host));
325 
326  //if the server is remote then :
327  if(m_server && m_server->isRemote()){
328  //show user info
329  NLog::global()->add_message("Connected to paraview server");
330 
331  //show the "load file" button
332  m_action_load_file->setEnabled(true);
333 
334  //change connect button to disconnect button
335  m_action_connect->setEnabled(false);
336  m_action_disconnect->setEnabled(true);
337 
338  //create server view
339  create_view();
340  }else{
341  NLog::global()->add_error("Error while connecting to paraview server");
342  //Set Server to a stable state
343  m_server = m_object_builder->createServer(pqServerResource("builtin:"));
344  create_view();
345  }

763 void Widget3D::set_solid_color(){
764 
765  pqPipelineRepresentation* repr = qobject_cast<pqPipelineRepresentation*>
766  (m_source_list.at(m_actor_list->row(m_actor_list->selectedItems().at(0)))
767  ->getRepresentation(m_RenderView));
768  if (!repr)
769  {
770 // qCritical() << "No active representation.";
771  return;
772  }
773 
774  if (repr->getColorField() == pqPipelineRepresentation::solidColor())
775  {
776  // Get the color property.
777  vtkSMProxy * proxy = repr->getProxy();
778  vtkSMProperty * diffuse = proxy->GetProperty("DiffuseColor");
779  vtkSMProperty * ambient = proxy->GetProperty("AmbientColor");
780  int reprType = repr->getRepresentationType();
781  bool use_ambient = (reprType == vtkSMPVRepresentationProxy::WIREFRAME ||
782  reprType == vtkSMPVRepresentationProxy::POINTS ||
783  reprType == vtkSMPVRepresentationProxy::OUTLINE);
784  if (diffuse && ambient)
785  {
786  // Get the current color from the property.
787  QList<QVariant> rgb =
788  pqSMAdaptor::getMultipleElementProperty(diffuse);
789  QColor color(Qt::white);
790  if(rgb.size() >= 3)
791  {
792  color = QColor::fromRgbF(rgb[0].toDouble(), rgb[1].toDouble(),
793  rgb[2].toDouble());
794  }
795 
796  // Let the user pick a new color.
797  color = QColorDialog::getColor(color, this);
798  if(color.isValid())
799  {
800  // Set the properties to the new color.
801  rgb.clear();
802  rgb.append(color.redF());
803  rgb.append(color.greenF());
804  rgb.append(color.blueF());
805  pqSMAdaptor::setMultipleElementProperty(
806  use_ambient? ambient : diffuse, rgb);
807  proxy->UpdateVTKObjects();
808  // need to break any global-property link that might have existed
809  // with this property.
810  pqStandardColorLinkAdaptor::breakLink(proxy,
811  use_ambient? "AmbientColor" : "DiffuseColor");
812  }
813  }
814  }
