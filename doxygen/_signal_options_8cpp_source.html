99 boost::shared_ptr<Option> SignalOptions::xml_to_option( const XmlNode & node )
100 {
101  cf3_assert( node.is_valid() );
102 
103  bool advanced; // indicates whether the option should be advanced or not
104  std::string opt_type( Map::get_value_type(node) ); // option type
105 
106  // get some attributes
107  rapidxml::xml_attribute<>* key_attr = node.content->first_attribute( Protocol::Tags::attr_key() );
108  rapidxml::xml_attribute<>* descr_attr = node.content->first_attribute( Protocol::Tags::attr_descr() );
109  rapidxml::xml_attribute<>* p_name_attr = node.content->first_attribute( Protocol::Tags::attr_pretty_name() );
110 
111  rapidxml::xml_attribute<>* mode_attr = node.content->first_attribute( "mode" );
112 
113  // check the key is present and not empty
114  if( is_null(key_attr) || key_attr->value()[0] == '\0' )
115  throw ProtocolError(FromHere(), "The option key is missing or empty.");
116 
117  std::string key_str( key_attr->value() ); // option name
118 
119  // get the description and the pretty name (if present)
120  std::string descr_str( is_not_null(descr_attr) ? descr_attr->value() : "" );
121  std::string pretty_name( is_not_null(p_name_attr) ? p_name_attr->value() : "" );
122 
123  advanced = is_null(mode_attr) || std::strcmp(mode_attr->value(), "adv") == 0;
124 
125  boost::shared_ptr<Option> option;
126 
127  // if it is a single value
128  if( Map::is_single_value(node) )
129  {
130  XmlNode type_node( node.content->first_node(opt_type.c_str()) );
131  option = make_option(key_str, type_node);
132  boost::shared_ptr<OptionURI> option_uri = boost::dynamic_pointer_cast<OptionURI>(option);
133 
134  // Add the schemes if we have a URI
135  if( is_not_null(option_uri) )
136  {
137  // in the case of a URI value, we have to create an OptionURI and
138  // check for supported schemes
139 
140  rapidxml::xml_attribute<>* schemes_attr = nullptr;
141  std::vector<std::string> schemes;
142  std::vector<std::string>::iterator it;
143 
144  schemes_attr = node.content->first_attribute( Protocol::Tags::attr_uri_schemes() );
145 
146  // add the supported schems (if present)
147  if( is_not_null(schemes_attr) && schemes_attr->value_size() != 0 )
148  {
149  std::string schemes_str(schemes_attr->value());
150 
151  boost::algorithm::split(schemes, schemes_str, boost::algorithm::is_any_of(","));
152 
153  for(it = schemes.begin() ; it != schemes.end() ; it++)
154  {
155  URI::Scheme::Type scheme = URI::Scheme::Convert::instance().to_enum(*it);
156 
157  if( scheme == URI::Scheme::INVALID )
158  throw CastingFailed(FromHere(), "[" + *it + "] is not a valid scheme.");
159 
160  option_uri->supported_protocol(scheme);
161  }
162  }
163  }
164  }
165  else if( Map::is_array_value(node) ) // it is an array value
166  {
167  option = make_option_array(key_str, node);
168  }
169  else
170  throw ProtocolError(FromHere(), "Node [" + std::string(node.content->name()) +"] does not "
171  "represent either a single value nor an array value.");
172 
173  // if the option is not advanced, mark it as basic
174  if( !advanced && option.get() != nullptr )
175  option->mark_basic();
176 
177  if(is_not_null(option))
178  set_option_attributes(*option, pretty_name, descr_str, node);
179 
180  return option;
181 }

190 void add_opt_to_xml( Map& opt_map, boost::shared_ptr<Option> opt, bool is_array)
191 {
192  cf3_assert( opt_map.content.is_valid() );
193  cf3_assert( is_not_null( opt.get() ) );
194 
195  XmlNode value_node;
196  bool basic = opt->has_tag("basic");
197  std::string desc = opt->description();
198 
199 
200  if( !is_array )
201  {
202  value_node = opt_map.set_value( opt->name(), opt->type(), opt->value_str(), desc );
203  }
204  else
205  {
206  value_node = opt_map.set_array( opt->name(), opt->element_type(), opt->value_str(), opt->separator(), desc );
207  }
208 
209  value_node.set_attribute( Protocol::Tags::attr_pretty_name(), opt->pretty_name() );
210  value_node.set_attribute( "is_option", "true" );
211  value_node.set_attribute( "mode", (basic ? "basic" : "adv") );
212 
213  if( opt->has_restricted_list() )
214  {
215  Map value_map(value_node);
216  value_map.set_array( Protocol::Tags::key_restricted_values(), opt->element_type(), opt->restricted_list_str(), opt->separator() );
217  }
218 
219  if( opt->type() == common::class_name<URI>() )
220  {
221  boost::shared_ptr<OptionURI> uri_opt = boost::dynamic_pointer_cast<OptionURI>(opt);
222  if(is_null(uri_opt))
223  throw ShouldNotBeHere(FromHere(), "Option " + opt->name() + " has type " + opt->type() + " but can't be cast to OptionURI. TypeID is " + typeid(opt.get()).name());
224  std::vector<URI::Scheme::Type> prots = uri_opt->supported_protocols();
225  std::vector<URI::Scheme::Type>::iterator it = prots.begin();
226 
227  for( ; it != prots.end() ; it++)
228  value_node.set_attribute( Protocol::Tags::attr_uri_schemes(),
229  URI::Scheme::Convert::instance().to_str(*it));
230  }
231 }

251 SignalOptions::SignalOptions( SignalFrame & frame, const std::string & name )
252  : OptionList()
253 {
254  cf3_assert( frame.node.is_valid() );
255 
256  if( name.empty() )
257  main_map = frame.map( Protocol::Tags::key_options() ).main_map;
258  else
259  main_map = frame.map( name ).main_map;
260 
261  rapidxml::xml_node<> * value_node = main_map.content.content->first_node();
262 
263  for( ; is_not_null(value_node) ; value_node = value_node->next_sibling() )
264  {
265  boost::shared_ptr<Option> option = xml_to_option( XmlNode(value_node) );
266 
267  if( check(option->name()) )
268  throw ValueExists(FromHere(), "Option [" + option->name() + "] already exists.");
269 
270  store[ option->name() ] = option;
271  }
272 }
