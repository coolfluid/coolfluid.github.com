53  template <class T>
54  void bracket(const Real a, const Real b, T& func)
55  {
56  const Real GLIMIT=100.0,TINY=1.0e-20;
57  const Real GOLD=1.618034;
58  ax=a; bx=b;
59  Real fu;
60  fa=func(ax);
61  fb=func(bx);
62  if (fb > fa) {
63  swap(ax,bx);
64  swap(fb,fa);
65  }
66  cx=bx+std::min(GOLD*(bx-ax),m_maxStep);
67  fc=func(cx);
68  while (fb > fc) {
69  Real r=(bx-ax)*(fb-fc);
70  Real q=(bx-cx)*(fb-fa);
71  Real u=bx-((bx-cx)*q-(bx-ax)*r)/
72  (2.0*math::Functions::change_sign(std::max(std::abs(q-r),TINY),q-r));
73  Real ulim=bx+GLIMIT*(cx-bx);
74  if ((bx-u)*(u-cx) > 0.0) {
75  fu=func(u);
76  if (fu < fc) {
77  ax=bx;
78  bx=u;
79  fa=fb;
80  fb=fu;
81  return;
82  } else if (fu > fb) {
83  cx=u;
84  fc=fu;
85  return;
86  }
87  u=cx+std::min(GOLD*(cx-bx),m_maxStep);
88  fu=func(u);
89  } else if ((cx-u)*(u-ulim) > 0.0) {
90  fu=func(u);
91  if (fu < fc) {
92  shft3(bx,cx,u,u+std::min(GOLD*(u-cx),m_maxStep));
93  shft3(fb,fc,fu,func(u));
94  }
95  } else if ((u-ulim)*(ulim-cx) >= 0.0) {
96  u=ulim;
97  fu=func(u);
98  } else {
99  u=cx+std::min(GOLD*(cx-bx),m_maxStep);
100  fu=func(u);
101  }
102  shft3(ax,bx,cx,u);
103  shft3(fa,fb,fc,fu);
104  }
105  }

174  template <class T>
175  Real minimize(T &func)
176  {
177  const Real R=0.61803399,C=1.0-R;
178  Real x1,x2;
179  Real x0=ax;
180  Real x3=cx;
181  if (std::abs(cx-bx) > std::abs(bx-ax)) {
182  x1=bx;
183  x2=bx+C*(cx-bx);
184  } else {
185  x2=bx;
186  x1=bx-C*(bx-ax);
187  }
188  Real f1=func(x1);
189  Real f2=func(x2);
190  while (std::abs(x3-x0) > tol*(std::abs(x1)+std::abs(x2))) {
191  if (f2 < f1) {
192  shft3(x0,x1,x2,R*x2+C*x3);
193  shft2(f1,f2,func(x2));
194  } else {
195  shft3(x3,x2,x1,R*x1+C*x0);
196  shft2(f2,f1,func(x1));
197  }
198  }
199  if (f1 < f2) {
200  xmin=x1;
201  fmin=f1;
202  } else {
203  xmin=x2;
204  fmin=f2;
205  }
206  return xmin;
207  }

242  template <class T>
243  Real minimize(T &func)
244  {
245  const Uint ITMAX=100;
246  const Real CGOLD=0.3819660;
247  const Real ZEPS=math::Consts::eps()*1.0e-3;
248  Real a,b,d=0.0,etemp,fu,fv,fw,fx;
249  Real p,q,r,tol1,tol2,u,v,w,x,xm;
250  Real e=0.0;
251 
252  a=(ax < cx ? ax : cx);
253  b=(ax > cx ? ax : cx);
254  x=w=v=bx;
255  fw=fv=fx=func(x);
256  for (Uint iter=0;iter<ITMAX;iter++) {
257  xm=0.5*(a+b);
258  tol2=2.0*(tol1=tol*std::abs(x)+ZEPS);
259  if (std::abs(x-xm) <= (tol2-0.5*(b-a))) {
260  fmin=fx;
261  return xmin=x;
262  }
263  if (std::abs(e) > tol1) {
264  r=(x-w)*(fx-fv);
265  q=(x-v)*(fx-fw);
266  p=(x-v)*q-(x-w)*r;
267  q=2.0*(q-r);
268  if (q > 0.0) p = -p;
269  q=std::abs(q);
270  etemp=e;
271  e=d;
272  if (std::abs(p) >= std::abs(0.5*q*etemp) || p <= q*(a-x)
273  || p >= q*(b-x))
274  d=CGOLD*(e=(x >= xm ? a-x : b-x));
275  else {
276  d=p/q;
277  u=x+d;
278  if (u-a < tol2 || b-u < tol2)
279  d=math::Functions::change_sign(tol1,xm-x);
280  }
281  } else {
282  d=CGOLD*(e=(x >= xm ? a-x : b-x));
283  }
284  u=(std::abs(d) >= tol1 ? x+d : x+math::Functions::change_sign(tol1,d));
285  fu=func(u);
286  if (fu <= fx) {
287  if (u >= x) a=x; else b=x;
288  shft3(v,w,x,u);
289  shft3(fv,fw,fx,fu);
290  } else {
291  if (u < x) a=u; else b=u;
292  if (fu <= fw || w == x) {
293  v=w;
294  w=u;
295  fv=fw;
296  fw=fu;
297  } else if (fu <= fv || v == x || v == w) {
298  v=u;
299  fv=fu;
300  }
301  }
302  }
303  throw("Too many iterations in brent");
304  }

357  xm=0.5*(a+b);

361  fmin=fx;

363  }
364  if (std::abs(e) > tol1) {
365  d1=2.0*(b-a);

368  if (dv != dx) d2=(v-x)*dx/(dx-dv);

370  u2=x+d2;

372  ok2 = (a-u2)*(u2-b) > 0.0 && dx*d2 <= 0.0;
373  olde=e;
374  e=d;
