60 std::string OSystemLayer::dump_back_trace ()
61 {
62 #define CF_BUFFER_SIZE 256
63 
64 static int i = 0;
65 ++i;
66 
67 
68 std::ostringstream oss;
69 int j, nptrs;
70 void *buffer[CF_BUFFER_SIZE];
71 char **strings;
72 
73 //printf ("dumping %d backtrace ...\n", i);
74 oss << "dumping " << i << " backtrace ...\n";
75 nptrs = backtrace(buffer, CF_BUFFER_SIZE);
76 oss << "\nbacktrace() returned " << nptrs << " addresses\n";
77 
78 strings = backtrace_symbols(buffer, nptrs);
79 if (strings == NULL)
80  oss << "\nno backtrace_symbols found\n";
81 
82 #ifdef CF3_HAVE_CXXABI_H
83 
84 boost::regex e("(.+)\\((_.+)(\\+.+)\\)");
85 boost::match_results<std::string::const_iterator> what;
86 
87 // iterate over the returned symbol lines. skip the first, it is the
88 // address of this function.
89 for (j = 1; j < nptrs; j++)
90 {
91  std::string trace(strings[j]);
92 
93  // what[0] -> orginal string
94  // what[1] -> library name
95  // what[2] -> function name
96  // what[3] -> memory offset
97  if (boost::regex_search(trace,what,e))
98  {
99  trace = std::string(what[2].first,what[2].second);
100  size_t maxName = 1024;
101  int demangleStatus; // will be assigned in abi::__cxa_demangle
102  char* demangledName; // will be allocated in abi::__cxa_demangle
103  if ((demangledName = abi::__cxa_demangle(trace.c_str(), NULL, &maxName,
104  &demangleStatus)) && demangleStatus == 0)
105  {
106  trace = demangledName; // the demangled name is now in our trace string
107  }
108 
109  delete_ptr_array(demangledName);
110  }
111  oss << trace << "\n";
112 }
113 #else
114 for (j = 0; j < nptrs; j++)
115  oss << strings[j] << "\n";
116 #endif
117 
118 free(strings);
119 #undef CF_BUFFER_SIZE
120 
121 oss << "\nexit dumping backtrace ...";
122 
123 return oss.str();
124 
125 }
