135  {
136  name=what[1];
137  type=what[3];
138  subtype=what[5];
139  value=what[6];

141 
142  if (type == "bool")
143  store[name]=from_str<bool>(value);
144  else if (type == "unsigned")
145  store[name]=from_str<Uint>(value);
146  else if (type == "integer")
147  store[name]=from_str<int>(value);
148  else if (type == "real")
149  store[name]=from_str<Real>(value);
150  else if (type == "string")
151  store[name]=value;
152  else if (type == "uri")
153  store[name]=from_str<URI>(value);
154  else if (type == "array")
155  {
156  std::vector<std::string> array;
157 
158  // the strings could have comma's inside, brackets, etc...
159  Uint in_brackets(0);
160  std::string::iterator first = value.begin();

162  for ( ; it!=value.end(); ++it)

164  if (*it == '(') // opening bracket

166  else if (*it == ')') // closing bracket
167  --in_brackets;
168  else if (*it == ',' && in_brackets == 0)
169  {
170  array.push_back(std::string(first,it));
171  boost::algorithm::trim(array.back());
172  first = it+1;
173  }
174  }

176  boost::algorithm::trim(array.back());

178  if (subtype == "bool")
