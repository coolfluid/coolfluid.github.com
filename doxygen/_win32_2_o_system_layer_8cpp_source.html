17 // The arraysize(arr) macro returns the # of elements in an array arr.
18 // The expression is a compile-time constant, and therefore can be
19 // used in defining new arrays, for example. If you use arraysize on
20 // a pointer by mistake, you will get a compile-time error.
21 //
22 // One caveat is that arraysize() doesn't accept any array of an
23 // anonymous type or a type defined inside a function. In these rare
24 // cases, you have to use the unsafe ARRAYSIZE_UNSAFE() macro below. This is
25 // due to a limitation in C++'s template system. The limitation might
26 // eventually be removed, but it hasn't happened yet.

99  // For the given trace, attempts to resolve the symbols, and output a trace
100  // to the ostream os. The format for each line of the backtrace is:
101  //
102  // <tab>SymbolName[0xAddress+Offset] (FileName:LineNo)
103  //
104  // This function should only be called if Init() has been called. We do not
105  // LOG(FATAL) here because this code is called might be triggered by a
106  // LOG(FATAL) itself.
107  void OutputTraceToStream(const void* const* trace,
108  int count,
109  std::ostream* os) {
110 
111  /* AutoLock lock(lock_); */ // from chromium
112 
113  for (int i = 0; (i < count) && os->good(); ++i) {
114  const int kMaxNameLength = 256;
115  DWORD_PTR frame = reinterpret_cast<DWORD_PTR>(trace[i]);
116 
117  // Code adapted from MSDN example:
118  // http://msdn.microsoft.com/en-us/library/ms680578(VS.85).aspx
119  ULONG64 buffer[
120  (sizeof(SYMBOL_INFO) +
121  kMaxNameLength * sizeof(wchar_t) +
122  sizeof(ULONG64) - 1) /
123  sizeof(ULONG64)];
124 
125  // Initialize symbol information retrieval structures.
126  DWORD64 sym_displacement = 0;
127  PSYMBOL_INFO symbol = reinterpret_cast<PSYMBOL_INFO>(&buffer[0]);
128  symbol->SizeOfStruct = sizeof(SYMBOL_INFO);
129  symbol->MaxNameLen = kMaxNameLength;
130  BOOL has_symbol = SymFromAddr(GetCurrentProcess(), frame,
131  &sym_displacement, symbol);
132 
133  // Attempt to retrieve line number information.
134  DWORD line_displacement = 0;
135  IMAGEHLP_LINE64 line = {};
136  line.SizeOfStruct = sizeof(IMAGEHLP_LINE64);
137  BOOL has_line = SymGetLineFromAddr64(GetCurrentProcess(), frame,
138  &line_displacement, &line);
139 
140  // Output the backtrace line.
141  (*os) << "\t";
142  if (has_symbol) {
143  (*os) << symbol->Name << " [0x" << trace[i] << "+"
144  << sym_displacement << "]";
145  } else {
146  // If there is no symbol informtion, add a spacer.
147  (*os) << "(No symbol) [0x" << trace[i] << "]";
148  }
149  if (has_line) {
150  (*os) << " (" << line.FileName << ":" << line.LineNumber << ")";
151  }
152  (*os) << "\n";
153  }
154  }
155 
156 private:
157 
158  SymbolContext() : init_error_(ERROR_SUCCESS) {
159  // Initializes the symbols for the process.
160  // Defer symbol load until they're needed, use undecorated names, and
161  // get line numbers.
162  SymSetOptions(SYMOPT_DEFERRED_LOADS |
163  SYMOPT_UNDNAME |
164  SYMOPT_LOAD_LINES);
165  if (SymInitialize(GetCurrentProcess(), NULL, TRUE)) {

169  // TODO(awong): Handle error: SymInitialize can fail with

173  // process).
174  std::cerr << "SymInitialize failed: " << init_error_ << std::endl;
175  }
176  }

188 StackTrace::StackTrace(EXCEPTION_POINTERS* exception_pointers) {
189  // When walking an exception stack, we need to use StackWalk64().
190  count_ = 0;
191  // Initialize stack walking.
192  STACKFRAME64 stack_frame;
193  memset(&stack_frame, 0, sizeof(stack_frame));
194 #if defined(_WIN64)
195  int machine_type = IMAGE_FILE_MACHINE_AMD64;
196  stack_frame.AddrPC.Offset = exception_pointers->ContextRecord->Rip;
197  stack_frame.AddrFrame.Offset = exception_pointers->ContextRecord->Rbp;
198  stack_frame.AddrStack.Offset = exception_pointers->ContextRecord->Rsp;
199 #else
200  int machine_type = IMAGE_FILE_MACHINE_I386;
201  stack_frame.AddrPC.Offset = exception_pointers->ContextRecord->Eip;
202  stack_frame.AddrFrame.Offset = exception_pointers->ContextRecord->Ebp;
203  stack_frame.AddrStack.Offset = exception_pointers->ContextRecord->Esp;
204 #endif
205  stack_frame.AddrPC.Mode = AddrModeFlat;
206  stack_frame.AddrFrame.Mode = AddrModeFlat;
207  stack_frame.AddrStack.Mode = AddrModeFlat;
208  while (StackWalk64(machine_type,
209  GetCurrentProcess(),
210  GetCurrentThread(),
211  &stack_frame,
212  exception_pointers->ContextRecord,
213  NULL,
214  &SymFunctionTableAccess64,
215  &SymGetModuleBase64,
216  NULL) &&
217  count_ < arraysize(trace_)) {
218  trace_[count_++] = reinterpret_cast<void*>(stack_frame.AddrPC.Offset);
219  }

226 void StackTrace::OutputToStream(std::ostream* os) {
227  SymbolContext* context = SymbolContext::Get();
228  DWORD error = context->init_error();
229  if (error != ERROR_SUCCESS) {
230  (*os) << "Error initializing symbols (" << error
231  << "). Dumping unresolved backtrace:\n";
232  for (int i = 0; (i < count_) && os->good(); ++i) {
233  (*os) << "\t" << trace_[i] << "\n";
234  }
235  } else {
236  (*os) << "Backtrace:\n";
237  context->OutputTraceToStream(trace_, count_, os);
238  }
