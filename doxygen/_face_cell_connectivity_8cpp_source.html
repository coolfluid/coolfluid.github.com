111 void FaceCellConnectivity::build_connectivity()
112 {
113 
114  if (used().size() == 0 )
115  {
116  CFwarn << "No elements are given to build faces of" << CFendl;
117  return;
118  }
119 
120  // declartions
121  m_connectivity->resize(0);
122  common::Table<Entity>::Buffer f2c = m_connectivity->create_buffer();
123  common::Table<Uint>::Buffer face_number = m_face_nb_in_elem->create_buffer();
124  common::List<bool>::Buffer is_bdry_face = m_is_bdry_face->create_buffer();
125  common::Table<Uint>::Buffer cell_rotation = m_cell_rotation->create_buffer();
126  common::Table<bool>::Buffer cell_orientation = m_cell_orientation->create_buffer();
127 
128  Dictionary& geometry_fields = find_parent_component<Mesh>(*used()[0]).geometry_fields();
129  Uint tot_nb_nodes = geometry_fields.size();
130  std::vector < std::vector<Uint> > mapNodeFace(tot_nb_nodes);
131  std::vector<Uint> face_nodes; face_nodes.reserve(100);
132  std::vector<Entity> dummy_element_row(2);
133  std::vector<Uint> tmp_row(2);
134  Uint max_nb_faces(0);
135 
136  // calculate max_nb_faces
137  boost_foreach ( Handle< Component > elements_comp, used() )
138  {
139  Handle<Elements> elements(elements_comp);
140  if (elements->element_type().dimensionality() != elements->element_type().dimension() )
141  continue;
142  const Uint nb_faces = elements->element_type().nb_faces();
143  max_nb_faces += nb_faces * elements->size() ;
144  }
145 
146  if (m_face_building_algorithm)
147  {
148  // allocate storage if doesn't exist that says if the element is at the boundary of a region
149  // ( = not the same as the mesh boundary)
150  boost_foreach (Handle< Component > elements_comp, used())
151  {
152  Elements& elements = dynamic_cast<Elements&>(*elements_comp);
153  Handle< Component > comp = elements.get_child("is_bdry");
154  if ( is_null( comp ) || is_null(Handle< common::List<bool> >(comp)) )
155  {
156  common::List<bool>& is_bdry_elem = * elements.create_component< common::List<bool> >("is_bdry");
157 
158  const Uint nb_elem = elements.size();
159  is_bdry_elem.resize(nb_elem);
160 
161  for (Uint e=0; e<nb_elem; ++e)
162  is_bdry_elem[e] = true;
163  }
164  cf3_assert( Handle< common::List<bool> >(elements.get_child("is_bdry")) );
165  }
166  }
167 
168  // Declarations to save frequent allocations in the loop algorithm
169  Uint nb_inner_faces = 0;
170  Uint nb_matched_nodes = 1;
171  Uint face_node_idx;
172  Uint face_node;
173  Uint face;
174  Uint connected_face;
175  Uint node;
176  Uint nb_nodes;
177  bool found_face = false;
178  Handle< Component > elem_location_comp;
179  Uint elem_location_idx;
180 
181  // loop over the element types
182  m_nb_faces=0;
183  boost_foreach (Handle< Component > elements_comp, used() )
184  {
185  Elements& elements = dynamic_cast<Elements&>(*elements_comp);
186  const Uint nb_faces_in_elem = elements.element_type().nb_faces();
187 
188  Handle< common::List<bool> > is_bdry_elem;
189 
190  if (m_face_building_algorithm)
191  is_bdry_elem = Handle< common::List<bool> >(elements.get_child("is_bdry"));
192 
193  // loop over the elements of this type
194  Uint loc_elem_idx=0;
195  boost_foreach(Connectivity::ConstRow elem_nodes, elements.geometry_space().connectivity().array() )
196  {
197  if ( is_not_null(is_bdry_elem) )
198  if ( (*is_bdry_elem)[loc_elem_idx] == false )
199  continue;
200 
201  Entity element(elements,loc_elem_idx);
202 
203  // loop over the faces in the current element
204  for (Uint face_idx = 0; face_idx != nb_faces_in_elem; ++face_idx)
205  {
206  // construct sets of nodes that make the corresponding face in this element
207  nb_nodes = elements.element_type().face_type(face_idx).nb_nodes();
208  face_nodes.resize(nb_nodes);
209 
210  Uint i(0);
211  boost_foreach(const Uint face_node_idx, elements.element_type().faces().nodes_range(face_idx))
212  face_nodes[i++] = elem_nodes[face_node_idx];
213 
214 
215  // consider the first node belonging to the current face
216  // check if you find a face ID shared between all the other
217  // nodes building a face
218  node = face_nodes[0];
219  cf3_assert(node<tot_nb_nodes);
220 
221  found_face = false;
222  // search for matching face if they are registered to the node yet
223  boost_foreach( face, mapNodeFace[node] )
224  {
225  nb_matched_nodes = 1;
226  if (nb_nodes > 1)
227  {
228  for (face_node_idx=1; face_node_idx!=nb_nodes; ++face_node_idx)
229  {
230  boost_foreach (connected_face, mapNodeFace[face_nodes[face_node_idx]])
231  {
232  if (connected_face == face)
233  {
234  ++nb_matched_nodes;
235  break;
236  }
237  }
238 
239 
240  if (nb_matched_nodes == nb_nodes) // face matches the nodes
241  {
242  // the corresponding face already exists, meaning
243  // that the face is an internal one, shared by two elements
244  // here you set the second element (==state) neighbor of the face
245  found_face = true;
246  f2c.get_row(face)[1]=element;
247  face_number.get_row(face)[1]=face_idx;
248  // since it has two neighbor cells,
249  // this face is surely NOT a boundary face
250  is_bdry_face.get_row(face)=false;
251 
252  // First node in first face element:
253  Uint first_node_loc_idx = f2c.get_row(face)[0].get_nodes()[
254  f2c.get_row(face)[0].element_type().faces().nodes_range(
255  face_number.get_row(face)[0])[0]
256  ];
257 
258  // Find orientation ( or find match between first face-nodes of both neighbouring elements )
259  Uint rotation;
260  for (rotation=0; rotation<=nb_nodes; ++rotation)
261  {
262  if (face_nodes[rotation] == first_node_loc_idx)
263  {
264  cell_rotation.get_row(face)[1]=rotation;
265  break;
266  }
267  }
268  // Following assertion fails, it means the correct orientation was not found! This should never happen!
269  cf3_always_assert(rotation != nb_nodes);
270 
271  // increment number of inner faces (they always have 2 states)
272  ++nb_inner_faces;
273  break;
274  }
275  }
276  }
277  else // this only applies to the 1D case
278  {
279  // the corresponding faceID already exists, meaning
280  // that the face is an internal one, shared by two elements
281  // here you set the second element (==state) neighbor of the face
282  found_face = true;
283  //(*elem_to_face)[elemID][iFace] = currFaceID;
284  f2c.get_row(face)[1]=element;
285  face_number.get_row(face)[1]=face_idx;
286  is_bdry_face.get_row(face)=false;
287 
288  // increment number of inner faces (they always have 2 states)
289  ++nb_inner_faces;
290  break;
291  }
292  }
293  if (found_face == false)
294  {
295  // a new face has been found
296  // add the ID of the new face in the corresponding nodes
297  // referencing it
298  boost_foreach (face_node, face_nodes)
299  {
300  mapNodeFace[face_node].push_back(m_nb_faces);
301  }
302 
303  // increment the number of faces
304  dummy_element_row[0]=element;
305  f2c.add_row(dummy_element_row);
306 
307  tmp_row[0]=face_idx;
308  face_number.add_row(tmp_row);
309  tmp_row[0] = MATCHED;
310  tmp_row[1] = INVERTED;
311  cell_orientation.add_row(tmp_row);
312  tmp_row[0] = 0;
313  tmp_row[1] = 0;
314  cell_rotation.add_row(tmp_row);
315  is_bdry_face.add_row(true);
316  ++m_nb_faces;
317  }
318  }
319  ++loc_elem_idx;
320  } // end foreach element
321  } // end foreach elements component
322 
323  f2c.flush();
324  face_number.flush();
325  is_bdry_face.flush();
326  cell_rotation.flush();
327  cell_orientation.flush();
328 
329  // CFinfo << "Total nb faces [" << m_nb_faces << "]" << CFendl;
330  // CFinfo << "Inner nb faces [" << nb_inner_faces << "]" << CFendl;
331 
332  // total number of boundary + partition boundary faces
333  //const Uint nb_bdry_plus_partition_faces = m_nb_faces - nb_inner_faces;
334  //CFinfo << "Boundary and Partition faces [" << nb_bdry_plus_partition_faces << "]" << CFendl;
335 
336  cf3_assert(m_nb_faces <= max_nb_faces);
337  cf3_assert(nb_inner_faces <= max_nb_faces);
338 
339 
340  cf3_assert(m_nb_faces == m_connectivity->size());
341 
342  if (m_face_building_algorithm)
343  {
344  for (Uint f=0; f<m_connectivity->size(); ++f)
345  {
346  ElementConnectivity::Row elem_row = (*m_connectivity)[f];
347  boost_foreach (Entity& elem, elem_row)
348  {
349  if ( is_not_null(elem.comp) )
350  {
351  common::List<bool>& is_bdry_elem = *Handle< common::List<bool> >(elem.comp->get_child("is_bdry"));
352  is_bdry_elem[elem.idx] = is_bdry_elem[elem.idx] || is_bdry_face.get_row(f) ;
353  }
354  }
355  }
356  }
357 }
