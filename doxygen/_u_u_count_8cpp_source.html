44 UUCount::UUCount(const std::string& str)
45 {
46  if(str.empty())
47  {
48  m_uuid = boost::uuids::nil_uuid();
49  m_count = 0;
50  }
51  else
52  {
53  std::vector<std::string> split_parts;
54  boost::algorithm::split(split_parts, str, boost::algorithm::is_any_of(":"));
55  if(split_parts.size() != 2)
56  throw ParsingFailed(FromHere(), "Did not find exactly one ':' separating UUID and count when parsing UUID string " + str);
57 
58  try
59  {
60  m_uuid = boost::uuids::string_generator()(split_parts.front());
61  }
62  catch(std::runtime_error&)
63  {
64  throw ParsingFailed(FromHere(), "Invalid UUID string: " + split_parts.front());
65  }
66 
67  try
68  {
69  m_count = boost::lexical_cast<Uint>(split_parts.back());
70  }
