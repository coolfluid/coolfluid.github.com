61 void create_line(Mesh& mesh, const Real x_len, const Uint x_segments)
62 {
63  Region& region = mesh.topology().create_region("fluid");
64  Dictionary& nodes = mesh.geometry_fields();
65  mesh.initialize_nodes(x_segments+1,DIM_1D);
66  const Real x_step = x_len / static_cast<Real>(x_segments);
67  for(Uint i = 0; i <= x_segments; ++i)
68  {
69  nodes.coordinates()[i][XX] = static_cast<Real>(i) * x_step;
70  }
71

73  cells->initialize("cf3.mesh.LagrangeP1.Line1D",nodes);
74  cells->resize(x_segments);
75  Table<Uint>& connectivity = cells->geometry_space().connectivity();
76  for(Uint i = 0; i < x_segments; ++i)
77  {
78  Table<Uint>::Row nodes = connectivity[i];
79  nodes[0] = i;
80  nodes[1] = i+1;

86  xneg->resize(1);
