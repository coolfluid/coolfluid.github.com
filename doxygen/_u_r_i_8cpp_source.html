199 URI URI::base_path () const
200 {
201  using namespace boost::algorithm;
202 
203  if(m_path == "/")
204  return *this;
205 
206  if(!contains(m_path, CF3_URI_SEPARATOR))
207  return URI("./", m_scheme);
208  else
209  {
210  std::string rpath = m_path;
211  rpath.erase ( find_last(rpath,CF3_URI_SEPARATOR).begin(), rpath.end() );
212  if(rpath.empty())
213  {
214  cf3_assert(is_absolute()); // this case should only happen on first-level absolute paths such as /Model
215  rpath = "/";
216  }
217  return URI(rpath, m_scheme);
218  }

267 void URI::split_path(const std::string & path, URI::Scheme::Type & protocol,
268  std::string & real_path)
269 {
270  // by default the protocol is CPATH
271  protocol = URI::Scheme::CPATH;
272  real_path = path;
273 
274  size_t colon_pos = path.find_first_of(':');
275 
276  // if the colon has been found
277  if(colon_pos != std::string::npos)
278  {
279  // extract the procotol
280  std::string protocol_str = path.substr(0, colon_pos);
281 
282  // extract the path
283  real_path = real_path.substr(colon_pos + 1, path.length() - colon_pos - 1);
284 
285  // check that the protocol is valid
286  protocol = URI::Scheme::Convert::instance().to_enum(protocol_str);
287 
288  if(protocol == URI::Scheme::INVALID)
289  throw ProtocolError(FromHere(), "\'" + protocol_str + "\' is not a supported protocol");
290  }

306 void URI::cleanup()
307 {
308  // Regex crashes here on GCC 4.1, probably due to static variable cleanup inside boost. So we avoid using it.
309  if(m_scheme == cf3::common::URI::Scheme::CPATH && !m_path.empty())
310  {
311  const Uint path_size = m_path.size();
312  std::string cleaned_path;
313  cleaned_path.reserve(path_size);
314  cleaned_path.push_back(m_path[0]);
315  for(Uint i = 1; i != path_size; ++i)
316  {
317  if(m_path[i] != '/' || m_path[i-1] != '/')
318  cleaned_path.push_back(m_path[i]);
319  }
320  m_path = cleaned_path;
321  }
322 }
