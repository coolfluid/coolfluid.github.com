591 void NTree::list_tree_reply(SignalArgs & args)
592 {
593  //QMutexLocker locker(m_mutex);
594  emit begin_update_tree();
595  beginResetModel();
596 
597  try
598  {
599  Handle< NRoot > tree_root = m_root_node->node()->castTo<NRoot>();
600  boost::shared_ptr< CNode > root_node = CNode::create_from_xml(args.main_map.content.content->first_node());
601  ComponentIterator<CNode> it = component_begin<CNode>(*root_node->root());
602  ComponentIterator<CNode> root_end = component_end<CNode>(*root_node->root());
603  URI currentIndexPath;
604 
605  if(m_current_index.isValid())
606  {
607  currentIndexPath = index_to_tree_node(m_current_index)->node()->uri();
608  }
609 
610  //
611  // rename the root
612  //
613  tree_root->rename(root_node->name());
614  tree_root->rename(root_node->name());
615 
616  //
617  // remove old nodes
618  //
619  ComponentIterator<CNode> itRem = component_begin<CNode>(*tree_root);
620  ComponentIterator<CNode> tree_root_end = component_end<CNode>(*tree_root);
621 
622  QList<std::string> list_to_remove;
623  QList<std::string>::iterator itList;
624 
625  for( ; itRem != tree_root_end ; itRem++)
626  {
627  if(!itRem->is_local_component() && !itRem->is_root() )
628  list_to_remove << itRem->name();
629  }
630 
631  itList = list_to_remove.begin();
632 
633  for( ; itList != list_to_remove.end() ; itList++)
634  {
635  tree_root->access_component_checked(*itList)->handle<CNode>()->about_to_be_removed();
636  tree_root->remove_component(*itList);
637  }
638 
639  //
640  // add the new nodes
641  //
642 
643  std::vector<std::string> names_to_add;
644  names_to_add.reserve(root_node->count_children());
645  for( ; it != root_end ; it++)
646  names_to_add.push_back(it.get()->name());
647  BOOST_FOREACH(const std::string& name, names_to_add)
648  tree_root->add_component( root_node->remove_component(name) );
649 
650  // child count may have changed, ask the root TreeNode to update its internal data
651  m_root_node->update_child_list();
652 
653  // retrieve the previous index, if it still exists
654  if(!currentIndexPath.path().empty())
655  m_current_index = this->index_from_path(currentIndexPath);
656 
657 
658  NLog::global()->add_message("Tree updated.");
659  }
660  catch(XmlError & xe)
661  {
662  NLog::global()->add_exception(xe.what());
663  }
664 
665  // tell the view to update the whole thing
666  endResetModel();
667 
668  emit end_update_tree();
669 
670  emit current_index_changed(m_current_index, QModelIndex());
671 
672 }
