147 void Mesh::update_structures()
148 {
149  Uint entities_idx=0;
150  Uint dict_idx=0;
151  m_elements.clear();
152  m_dictionaries.clear();
153  std::map<std::string,Handle<Dictionary> > dicts_map; // sorted by string
154  boost_foreach ( Entities& elements, find_components_recursively<Entities>(topology()) )
155  {
156  m_elements.push_back(elements.handle<Entities>());
157  boost_foreach(const Handle<Space>& space, elements.spaces())
158  {
159  dicts_map[space->dict().uri().string()]=space->dict().handle<Dictionary>();
160  }
161  }
162  m_dictionaries.reserve(dicts_map.size());
163  foreach_container( (const std::string& uri) (const Handle<Dictionary>& dict), dicts_map)
164  {
165  m_dictionaries.push_back(dict);
166  dict->update_structures();
167  }
168 
169  // Set private member m_entities_idx in each Entities
170  for (entities_idx=0; entities_idx<m_elements.size(); ++entities_idx)
171  {
172  m_elements[entities_idx]->m_entities_idx = entities_idx;
173  }
174 
175  bool found=false;
176  boost_foreach (const Handle<Dictionary> dict, m_dictionaries)
177  {
178  if (dict == m_geometry_fields)
179  {
180  found = true;
181  break;
182  }
183  }
184  if (!found)
185  {
186  if (m_geometry_fields)
187  m_dictionaries.push_back(m_geometry_fields);
188  }
189 
190  // Set private member m_dict_idx in each Space
191  for (dict_idx=0; dict_idx<m_dictionaries.size(); ++dict_idx)
192  {
193  boost_foreach(const Handle<Space>& space, m_dictionaries[dict_idx]->spaces())
194  {
195  space->m_dict_idx = dict_idx;
196  }

400 bool Mesh::check_sanity(std::vector<std::string>& messages) const
401 {
402  Uint nb_messages_init = messages.size();
403  if (dimension() == 0)
404  messages.push_back("mesh.dimension not configured");
405 
406  if (dimensionality() == 0)
407  messages.push_back("mesh.dimensionality not configured");
408 
409  if (dimensionality() > dimension())
410  messages.push_back("dimensionality ["+to_str(dimensionality())+"] > dimension ["+to_str(dimension())+"]");
411 
412  if(geometry_fields().coordinates().row_size() != dimension())
413  messages.push_back("coordinates dimension ["+to_str(geometry_fields().coordinates().row_size())+"] does not match mesh.dimension ["+to_str(dimension())+"]");
414 
415  boost_foreach(const Dictionary& dict, find_components_recursively<Dictionary>(*this))
416  {
417  dict.check_sanity(messages);
418  }
419 
420  if (Comm::instance().size()>1)
421  {
422  std::set<Uint> unique_node_gids;
423  boost_foreach(const Uint gid, geometry_fields().glb_idx().array())
424  {
425  std::pair<std::set<Uint>::iterator, bool > inserted = unique_node_gids.insert(gid);
426  if (inserted.second == false)
427  {
428  messages.push_back(geometry_fields().glb_idx().uri().string()+" has non-unique entries. (entry "+to_str(gid)+" exists more than once, no further checks)");
429  break;
430  }
431  }
432  }
433 
434  std::set<Uint> unique_elem_gids;
435  boost_foreach(const Entities& entities, find_components_recursively<Entities>(*this))
436  {
437  if (entities.rank().size() != entities.size())
438  messages.push_back(entities.uri().string()+": size() ["+to_str(entities.size())+"] != rank().size() ["+to_str(entities.rank().size())+"]");
439  if (entities.glb_idx().size() != entities.size())
440  messages.push_back(entities.uri().string()+": size() ["+to_str(entities.size())+"] != glb_idx().size() ["+to_str(entities.glb_idx().size())+"]");
441 
442  if (Comm::instance().size()>1)
443  {
444  boost_foreach(const Uint gid, entities.glb_idx().array())
445  {
446  std::pair<std::set<Uint>::iterator, bool > inserted = unique_elem_gids.insert(gid);
447  if (inserted.second == false)
448  {
449  messages.push_back(entities.glb_idx().uri().string()+" has non-unique entries. (entry "+to_str(gid)+" exists more than once, no further checks)");
450  break;
451  }
452  }
453  }
454  }
455 
456  return (messages.size()-nb_messages_init) == 0;
457 
458 }
