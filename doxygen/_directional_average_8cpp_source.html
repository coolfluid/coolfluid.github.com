89 void DirectionalAverage::execute()
90 {
91  setup();
92 
93  std::fill(m_averages.begin(), m_averages.end(), 0.);
94 
95  const mesh::Field::ArrayT& field_values = m_field->array();
96  const Uint nb_nodes = field_values.size();
97  const mesh::Dictionary& dict = m_field->dict();
98  const common::List<bool>* periodic_links_active = Handle<common::List<bool> const>(dict.get_child("periodic_links_active")).get();
99  const Uint stride = m_field->row_size()+1;
100  for(Uint node_idx = 0; node_idx != nb_nodes; ++node_idx)
101  {
102  if(!dict.is_ghost(node_idx) && !(is_not_null(periodic_links_active) && (*periodic_links_active)[node_idx]))
103  {
104  const Uint position_idx = m_node_position_indices[node_idx];
105  const Uint node_begin = stride*position_idx;
106  ++m_averages[node_begin];
107  const mesh::Field::ConstRow row = field_values[node_idx];
108  std::transform(row.begin(), row.end(), m_averages.begin()+node_begin+1, m_averages.begin()+node_begin+1, std::plus<Real>());
109  }
110  }
111 
112  common::PE::Comm& comm = common::PE::Comm::instance();
113  std::vector<Real> global_averages;
114  if(comm.is_active())
115  {
116  comm.reduce(common::PE::plus(), m_averages, global_averages, 0);
117  }
118  else
119  {
120  global_averages = m_averages;
121  }
122 
123  if(comm.rank() == 0)
124  {
125  boost::filesystem::fstream file(options().value<common::URI>("file").path(), std::ios::out);
126  const Uint nb_positions = m_positions.size();
127  file << "# Position, Count, Averages for " << m_field->descriptor().description() << "\n";
128  for(Uint pos_idx = 0; pos_idx != nb_positions; ++pos_idx)
129  {
130  const Uint avg_begin = pos_idx*stride;
131  const Uint avg_end = avg_begin+stride;
132  const Real count = global_averages[avg_begin];
133  file << common::to_str(m_positions[pos_idx]) << " " << count;
134 
135  for(Uint j = avg_begin+1; j != avg_end; ++j)
136  {
137  file << " " << common::to_str(global_averages[j]/count);
138  }
139  file << "\n";
140  }
141  file.close();
142  }

150 void DirectionalAverage::setup()
151 {
152  if(is_not_null(m_field))
153  return;
154 
155  m_field = options().value< Handle<mesh::Field> >("field");
156  if(is_null(m_field))
157  throw common::SetupError(FromHere(), "No field configured for " + uri().path());
158 
159  const mesh::Dictionary& dict = m_field->dict();
160  const Uint nb_nodes = dict.size();
161  const mesh::Field& coords = dict.coordinates();
162 
163  const Uint direction = options().value<Uint>("direction");
164  if(direction >= coords.row_size())
165  throw common::SetupError(FromHere(), "Direction " + common::to_str(direction) + " is not allowed for mesh of dimension " + common::to_str(coords.row_size()));
166 
167  std::set<Real, detail::threshold_compare> unique_coords(detail::threshold_compare(options().value<Real>("threshold")));
168 
169  for(Uint node_idx = 0; node_idx != nb_nodes; ++node_idx )
170  {
171  if(!dict.is_ghost( node_idx ))
172  unique_coords.insert(coords[node_idx][direction]);
173  }
174 
175  common::PE::Comm& comm = common::PE::Comm::instance();
176  if(comm.is_active())
177  {
178  std::vector<Real> my_unique_coords(unique_coords.begin(), unique_coords.end());
179  std::vector< std::vector<Real> > gathered_coords;
180  comm.all_gather(my_unique_coords, gathered_coords);
181  BOOST_FOREACH(const std::vector<Real>& vec, gathered_coords)
182  {
183  unique_coords.insert(vec.begin(), vec.end());
184  }
185  }
186 
187  std::map<Real, Uint, detail::threshold_compare> coords_map(detail::threshold_compare(options().value<Real>("threshold")));
188  Uint coord_gid = 0;
189  BOOST_FOREACH(const Real coord, unique_coords)
190  {
191  coords_map[coord] = coord_gid++;
192  }
193 
194  m_positions.assign(unique_coords.begin(), unique_coords.end());
195 
196  CFinfo << "Found " << m_positions.size() << " unique coordinates in direction " << direction << CFendl;
197 
198  m_node_position_indices.resize(nb_nodes);
199  for(Uint node_idx = 0; node_idx != nb_nodes; ++node_idx )
200  {
201  if(!dict.is_ghost( node_idx ))
202  {
203  cf3_assert(coords_map.find(coords[node_idx][direction]) != coords_map.end());
204  m_node_position_indices[node_idx] = coords_map[coords[node_idx][direction]];
205  }
206  }
207 
208  // For each position, we store the count and then the sum of all values in the field
209  m_averages.resize(m_positions.size() * (m_field->row_size()+1));
210 }
