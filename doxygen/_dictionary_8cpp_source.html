272 bool Dictionary::check_sanity(std::vector<std::string>& messages) const
273 {
274  Uint nb_messages_init = messages.size();
275  if (rank().size() != size())
276  messages.push_back(uri().string()+": size() ["+to_str(size())+"] != rank().size() ["+to_str(rank().size())+"]");
277 
278  if (glb_idx().size() != size())
279  messages.push_back(uri().string()+": size() ["+to_str(size())+"] != glb_idx().size() ["+to_str(glb_idx().size())+"]");
280 
281  std::set<Uint> unique_gids;
282  if (Comm::instance().size()>1)
283  {
284  for (Uint i=0; i<size(); ++i)
285  {
286  if (rank()[i] >= PE::Comm::instance().size())
287  {
288  messages.push_back(rank().uri().string()+"["+to_str(i)+"] has invalid entry. (entry = "+to_str(rank()[i])+" , no further checks)");
289  break;
290  }
291  }
292  for (Uint i=0; i<size(); ++i)
293  {
294  std::pair<std::set<Uint>::iterator, bool > inserted = unique_gids.insert(glb_idx()[i]);
295  if (inserted.second == false)
296  {
297  messages.push_back(glb_idx().uri().string()+"["+to_str(i)+"] has non-unique entries. (glb_idx "+to_str(glb_idx()[i])+" exists more than once, no further checks)");
298  break;
299  }
300  }
301  }
302 
303  const common::List<Uint>* periodic_links_nodes = Handle<common::List<Uint> const>(get_child("periodic_links_nodes")).get();
304  const common::List<bool>* periodic_links_active = Handle<common::List<bool> const>(get_child("periodic_links_active")).get();
305  if(is_not_null(periodic_links_active))
306  {
307  for (Uint i=0; i != size(); ++i)
308  {
309  if((*periodic_links_active)[i])
310  {
311  const Uint link_lid = (*periodic_links_nodes)[i];
312  if((*periodic_links_active)[link_lid])
313  {
314  messages.push_back(periodic_links_nodes->uri().string()+"["+to_str(i)+"] links to another periodic node at " + common::to_str(link_lid));
315  }
316  if(rank()[i] != rank()[link_lid])
317  {
318  messages.push_back(periodic_links_nodes->uri().string()+"["+to_str(i)+"] is at rank " + common::to_str(rank()[i]) + ", but the link " + common::to_str(link_lid) + " is at rank " + common::to_str(rank()[link_lid]));
319  }
320  }
321  }
322  }

490 Field& Dictionary::create_coordinates()
491 {
492  if (has_coordinates())
493  m_coordinates.reset();
494 
495  Field& coordinates = create_field("coordinates","coords[vector]");
496  CFdebug << " - Interpolating coordinates from geometry" << CFendl;
497  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
498  {
499  Entities& entities = *entities_handle;
500  Space& geometry_space = entities.geometry_space();
501  const ShapeFunction& geom_sf = geometry_space.shape_function();
502  RealMatrix geom_nodes;
503  entities.geometry_space().allocate_coordinates(geom_nodes);
504 
505  const Space& entities_space = space(entities);
506  const ShapeFunction& sf = entities_space.shape_function();
507  const RealMatrix& local_coords = sf.local_coordinates();
508 
509  RealMatrix interpolation(sf.nb_nodes(),geom_sf.nb_nodes());
510  for (Uint i=0; i<sf.nb_nodes(); ++i)
511  interpolation.row(i) = geom_sf.value(local_coords.row(i));
512 
513  RealMatrix coords(sf.nb_nodes(),geom_nodes.cols());
514 
515  for (Uint e=0; e<entities.size(); ++e)
516  {
517  entities.geometry_space().put_coordinates(geom_nodes,e);
518  coords.noalias() = interpolation*geom_nodes;
519 
520  for (Uint i=0; i<sf.nb_nodes(); ++i)
521  {
522  const Uint pt = entities_space.connectivity()[e][i];
523  for(Uint d=0; d<coords.cols(); ++d)
524  coordinates[pt][d] = coords(i,d);
525  }
526  }
527  }
528 
529  m_coordinates = coordinates.handle<Field>();
530  m_coordinates->add_tag(mesh::Tags::coordinates());
531 
532  return coordinates;
533 }
