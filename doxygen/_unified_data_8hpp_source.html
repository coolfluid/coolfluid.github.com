118 inline void UnifiedData::add(DATA& data)
119 {
120  Handle<DATA> actual_data = Handle<DATA>(data.template handle<Component>());
121 
122  // if it is not added yet, add
123  if (m_start_idx.find(actual_data.get()) == m_start_idx.end())
124  {
125  m_start_idx[actual_data.get()] = m_size;
126 
127  m_data_links->create_component<common::Link>("data_component_"+common::to_str(m_data_vector.size()))->link_to(*actual_data);
128 
129  m_data_vector.push_back(actual_data);
130  m_size += actual_data->size();
131 
132  common::List<Uint>::Buffer data_start_indices = m_data_indices->create_buffer();
133  data_start_indices.add_row(m_size);
134  data_start_indices.flush();
135  }
136 
137  cf3_assert(m_data_vector.size()==m_data_indices->size()-1);
