98 void BrowserDialog::path_edited( const QString & text )
99 {
100  QString path;
101  bool send = false;
102 
103  // if user just typed a '/', the path to explore is the current path in the field
104  if(text.endsWith("/"))
105  {
106  send = true;
107  path = text;
108  }
109 
110  // if user just deleted a '/' or lengths of texts differ of more than one
111  // character (this may happen if user pasted a path or delete more than
112  // one character at a time), the path to explore is the parent directory of
113  // the path in the field
114  else if(m_old_path.endsWith("/") || qAbs(m_old_path.length() - text.length()) > 1)
115  {
116  m_updating_completer = true;
117  send = true;
118  path = QFileInfo(text).path();
119  }
120 
121  if(send)
122  m_model->open_dir(path);
123 
124  m_old_path = text;
125 }

129 void BrowserDialog::keyPressEvent(QKeyEvent * event)
130 {
131  // key code for the pressed key
132  int pressed_key = event->key();
133  // modifiers keys pressed (ctrl, shift, alt, etc...)
134  Qt::KeyboardModifiers modifiers = event->modifiers();
135 
136  QDialog::keyPressEvent(event);
137 
138  // if the path line edit has the focus
139  if(m_edit_path->hasFocus())
140  {
141  // Qt::Key_Enter : enter key located on the keypad
142  // Qt::Key_Return : return key
143  if(pressed_key == Qt::Key_Enter || pressed_key == Qt::Key_Return)
144  m_edit_path->setText(m_edit_path->text());
145 
146  return;
147  }
148 
149  // if user pressed Enter key, it is similar to clicking on a button
150  // (if any has the focus). Note: if none has the focus the default one ("Ok") is
151  // taken (this is managed by QDialogButtonBox class)
152  // Qt::Key_Enter : enter key located on the keypad
153  // Qt::Key_Return : return key
154  if(pressed_key == Qt::Key_Enter || pressed_key == Qt::Key_Return)
155  {
156  if(m_buttons->button(QDialogButtonBox::Ok)->hasFocus())
157  accept();
158  else if(m_buttons->button(QDialogButtonBox::Cancel)->hasFocus())
159  reject();
160  }
161 
162  else if(pressed_key == Qt::Key_Backspace)
163  m_model->open_dir(m_model->current_path() + ".."); // back to the parent directory
164 
165  // if user pressed either no modifier key or Shift key *and* another key,
166  // the filter line edit takes the focus
167  else if(modifiers == Qt::NoModifier ||
168  (modifiers == Qt::ShiftModifier && !event->text().isEmpty()))
169  {
170  m_view->clearFocus();
171 
172  m_edit_filter->setText(m_edit_filter->text() + event->text());
173  path_edited(m_edit_filter->text());
174  m_edit_filter->setFocus(Qt::NoFocusReason);
175  }
176 }

197 bool BrowserDialog::show( bool multi_select, QVariant & selected )
198 {
199  if( multi_select )
200  m_view->setSelectionMode(QAbstractItemView::ExtendedSelection); // allow multi-selection
201  else
202  m_view->setSelectionMode(QAbstractItemView::SingleSelection); // mono-selection
203 
204  connect(NLog::global().get(), SIGNAL(new_message(QString, uiCommon::LogMessage::Type)),
205  this, SLOT(message(QString, uiCommon::LogMessage::Type)));
206 
207  bool ok_clicked = exec() == Accepted;
208  QString path = m_model->current_path();
209 
210  // if user clicked on "OK"
211  if(ok_clicked)
212  {
213  QModelIndexList selected_indexes = m_view->selectionModel()->selectedIndexes();
214 
215  if( !multi_select && selected_indexes.count() == 1 )
216  {
217  QModelIndex index_in_model = m_filter_model->mapToSource( selected_indexes.at(0) );
218  selected = path + m_model->data( index_in_model, Qt::DisplayRole ).toString();
219  }
220  else if( multi_select )
221  {
222  QModelIndexList::iterator it = selected_indexes.begin();
223  QStringList list;
224 
225  for( ; it != selected_indexes.end() ; ++it )
226  {
227  QModelIndex index_in_model = m_filter_model->mapToSource( *it );
228 
229  list << path + m_model->data( index_in_model, Qt::DisplayRole ).toString();
230  }
231 
232  selected = list;
233  }
234  }
235 
236  disconnect(NLog::global().get());
237 
238  return ok_clicked;
239 }
