63  QWidget * parent )
64 {
65  GraphicalValue * value = nullptr;
66 
67  if(option.get() == nullptr)
68  return value;
69 
70  std::string type(option->type());
71 
72  if(!boost::starts_with(type, "array"))
73  {
74  if(option->has_restricted_list())
75  value = new GraphicalRestrictedList(option, parent);
76  else
77  {
78 
79  if(type == common::class_name<bool>()) // bool option
80  value = new GraphicalBool(option->value<bool>(), parent);
81  else if(type == common::class_name<Real>()) // Real option
82  value = new GraphicalDouble(option->value<Real>(), parent);
83  else if(type == common::class_name<int>()) // int option
84  value = new GraphicalInt(false, option->value<int>(), parent);
85  else if(type == common::class_name<Uint>()) // Uint option
86  value = new GraphicalInt(true, option->value<Uint>(), parent);
87  else if(type == common::class_name<std::string>()) // string option
88  value = new GraphicalString(option->value<std::string>().c_str(), parent);
89  else if(type == common::class_name<URI>()) // URI option
90  value = new GraphicalUri(boost::dynamic_pointer_cast<OptionURI>(option), parent);
91  else
92  throw CastingFailed(FromHere(), type + ": Unknown type");
93  }
94  }
95  else
96  {
97  if(option->has_restricted_list())
98  value = new GraphicalArrayRestrictedList(option, parent);
99  else
100  {
101  std::string value_str( option->value_str() );
102  std::string type( option->element_type() );
103  QString sep( option->separator().c_str() );
104 
105  if(type == common::class_name<bool>()) // bool option
106  {
107  QRegExp regex("(true)|(false)|(1)|(0)|(on)|(off)");
108  value = new GraphicalArray(new QRegExpValidator(regex, parent), sep, parent);
109  }
110  else if(type == common::class_name<Real>()) // Real option
111  {
112  QDoubleValidator * val = new QDoubleValidator(nullptr);
113  val->setNotation(QDoubleValidator::ScientificNotation);
114  value = new GraphicalArray(val, sep, parent);
115  }
116  else if(type == common::class_name<int>()) // int option
117  value = new GraphicalArray(new QIntValidator(), sep, parent);
118  else if(type == common::class_name<Uint>()) // Uint option
119  {
120  QIntValidator * val = new QIntValidator();
121  val->setBottom(0);
122  value = new GraphicalArray(val, sep, parent);
123  }
124  else if(type == common::class_name<std::string>()) // string option
125  value = new GraphicalArray(nullptr,sep, parent);
126  else if(type == common::class_name<URI>()) // URI option
127  value = new GraphicalUriArray(sep, parent);
128  else
129  throw CastingFailed(FromHere(), type + ": Unknown type");
130

132  }
