50 template<typename TriangulationComplexT>
51 void cgal_to_coolfluid(const TriangulationComplexT& complex, Mesh& mesh)
52 {
53  // Map CGAL vertex handles to coolfluid point indices
54  typedef ::CGAL::Unique_hash_map<typename TriangulationComplexT::Vertex_handle,Uint,::CGAL::Handle_hash_function> VertexMapT;
55  VertexMapT vertex_map(0, complex.number_of_cells()); // estimate the number of vertices equal to the cell count
56 
57  const Uint nb_cells = complex.number_of_cells();
58 
59  std::vector<Point> points; points.reserve(nb_cells*2);
60  std::vector<Uint> cells; cells.reserve(nb_cells*4);
61 
62  for(typename TriangulationComplexT::Cell_iterator cell = complex.cells_begin(); cell != complex.cells_end(); ++cell) {
63  for(Uint i = 0; i != 4; ++i) {
64  typename TriangulationComplexT::Vertex_handle v = cell->vertex(i);
65  if(!vertex_map.is_defined(v)) { // store the point, if we didn't have it already
66  vertex_map[v] = points.size();
67  points.push_back(v->point());
68  }
69  cells.push_back(vertex_map[v]);
70  }
71  }
72 
73  // Coordinates
74  const Uint nb_points = points.size();
75  Region& region = mesh.topology().create_region("region");
76  Dictionary& nodes = mesh.geometry_fields();
77  mesh.initialize_nodes(nb_points,DIM_3D);
78  Field& coordinates = nodes.coordinates();
79  for(Uint i = 0 ; i != nb_points; ++i)
80  {
81  Field::Row coords_row = coordinates[i];
82  coords_row[XX] = ::CGAL::to_double(points[i].x());
83  coords_row[YY] = ::CGAL::to_double(points[i].y());
84  coords_row[ZZ] = ::CGAL::to_double(points[i].z());
85  }
86 
87  // Connectivity
88  Handle<Cells> tetras = region.create_component<Cells>("Tetras");
89  tetras->initialize("cf3.mesh.LagrangeP1.Tetra3D",nodes);
90  tetras->resize(cells.size());
91  Table<Uint>& connectivity = tetras->geometry_space().connectivity();
92  for(Uint i = 0; i != nb_cells; ++i)
93  {
94  Table<Uint>::Row row = connectivity[i];
95  for(Uint j = 0; j != 4; ++j)
96  {
97  row[j] = cells[i*4 + j];
98  }
99  }
100 
101  List<Uint>& gids = mesh.geometry_fields().glb_idx(); gids.resize(nb_points);
102  List<Uint>& ranks = mesh.geometry_fields().rank(); ranks.resize(nb_points);
103  for(Uint i = 0; i != nb_points; ++i)
104  {
105  ranks[i] = 0;
106  gids[i] = i;
107  }
108 
109  mesh.raise_mesh_loaded();
110 }
