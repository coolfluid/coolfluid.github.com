140  template<typename UT>
141  void compute_coefficients(const UT& u, const Real element_nu, const Real& dt, Real& tau_ps, Real& tau_su, Real& tau_bulk) const
142  {
143  typedef typename UT::EtypeT ElementT;
144  static const Uint dim = ElementT::dimension;
145  typedef mesh::Integrators::GaussMappedCoords<1, ElementT::shape> GaussT;
146 
147  u.compute_values(GaussT::instance().coords.col(0));
148 
149  if(supg_type == SUPGTypes::TEZDUYAR)
150  {
151  // Get the minimal edge length
152  Real h_rgn = 1e10;
153  for(Uint i = 0; i != ElementT::nb_nodes; ++i)
154  {
155  for(Uint j = 0; j != ElementT::nb_nodes; ++j)
156  {
157  if(i != j)
158  {
159  h_rgn = std::min(h_rgn, (u.support().nodes().row(i) - u.support().nodes().row(j)).squaredNorm());
160  }
161  }
162  }
163  h_rgn = sqrt(h_rgn);
164 
165  const Real umag = detail::norm(u.eval());
166  const Real h_ugn = h_rgn;//fabs(2.*umag / (u.eval()*u.nabla()).sum());
167 
168  const Real tau_adv_inv = (2.*umag)/h_ugn;
169  const Real tau_time_inv = 2./dt;
170  const Real tau_diff_inv = (4.*element_nu)/(h_rgn*h_rgn);
171 
172  tau_su = 1./(tau_adv_inv + tau_time_inv + tau_diff_inv);
173  tau_ps = tau_su;
174  tau_bulk = tau_su*umag*umag;
175  }
176  else if(supg_type == SUPGTypes::METRIC)
177  {
178  u.support().compute_jacobian(GaussT::instance().coords.col(0));
179 
180  const Eigen::Matrix<Real, ElementT::dimensionality, ElementT::dimensionality> gij = u.support().jacobian_inverse().transpose() * u.support().jacobian_inverse();
181  const Real tau_adv_sq = fabs((u.eval()*gij*detail::transpose(u.eval()))[0]); // Very close 0 but slightly negative sometimes
182  const Real tau_diff = element_nu*element_nu*gij.squaredNorm();
183 
184  tau_su = 1. / sqrt((4.*c1*c1/(dt*dt)) + tau_adv_sq + c2*tau_diff);
185  tau_ps = tau_su;
186 
187  // Use the standard SUPG factor to compute the bulk viscosity, or it goes up way too much
188  const Real tau_su_std = 1. / sqrt((4./(dt*dt)) + tau_adv_sq + 16.*tau_diff);
189  tau_bulk = (1./tau_su_std) / gij.trace();
190  }
191  else if(supg_type == SUPGTypes::CF2)
192  {
193  const Real he = UT::dimension == 2 ? sqrt(4./3.141592654*u.support().volume()) : ::pow(3./4./3.141592654*u.support().volume(),1./3.);
194  const Real ree=u_ref*he/(2.*element_nu);
195  cf3_assert(ree > 0.);
196  const Real xi = ree < 3. ? ree/3. : 1.;
197  tau_ps = he*xi/(2.*u_ref);
198  tau_bulk = he*u_ref/xi;
199 
200  tau_su = 0.;
201  const Real umag = detail::norm(u.eval());
202  if(umag > 1e-10)
203  {
204  const Real h = 2.*umag / (u.eval()*u.nabla()).sum();
205  const Real tau_adv = h/(2.*umag);
206  const Real tau_time = 0.5*dt;
207  const Real tau_diff = h*h/(4.*element_nu);
208  tau_su = 1./(1./tau_adv + 1./tau_time + 1./tau_diff);
209  }
210  }
211  else
212  {
213  throw common::ShouldNotBeHere(FromHere(), "Unsupported SUPG method chosen");
214  }
215 
216  tau_ps *= alpha_ps;
217  tau_su *= alpha_su;
218  tau_bulk *= alpha_bu;
219  }
