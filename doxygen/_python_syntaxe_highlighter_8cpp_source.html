103 void PythonSyntaxeHighlighter::highlightBlock(const QString &text){
104  static QRegExp two_points("^[^#:]*:[ ]*(#[^$]*)?$");
105  currentBlock().setUserData(NULL);//reset error if there where one
106  if (text.length()){
107  int a=currentBlock().userState();
108  if (a >= PythonCodeContainer::PROMPT_1){
109  int start=0,end=text.size();
110  int first_word=0;
111  if (text.contains(two_points)){
112  first_word=1;
113  for (;start<text.size();start++){
114  QChar c=text[start];
115  if (c != '#'){
116  if (first_word==1){
117  if (c != '\t' && c != ' ')
118  first_word=2;
119  }else if (first_word==2){
120  if (c == '\t' || c == ' ')
121  first_word=3;
122  }else{
123  if (c != '\t' && c != ' '){
124  first_word=4;
125  break;
126  }
127  }
128  }
129  }
130  }else{
131  for (;start<end && (text[start] == '\t' || text[start] == ' ');start++);
132  }
133  end=start;
134  for (;end<text.size() && text[end] != ':' && text[end] != '#';end++);
135  if ((first_word == 0 || first_word==4) && start != end){
136  const QString error=python_pre_compiler->try_compile(text.mid(start,end-start));
137  if (error.length()){
138  setFormat(start,end-start,*error_format);
139  currentBlock().setUserData(new TextBlockErrorData(error));
140  return;
141  }
142  }
143  foreach (const HighlightingRule &rule, *highlighting_rules) {
144  QRegExp expression(rule.pattern);
145  int index = expression.indexIn(text);
146  while (index >= 0) {
147  int length = expression.matchedLength();
148  setFormat(index, length, *rule.format);
149  index = expression.indexIn(text, index + length);
150  }
151  }
152  }
153  }
154 }
