104 inline void test(const Real& A, const Real& B, Accumulator& Result)
105 {
106  const Real abs_A = fabs(A);
107  const Real abs_B = fabs(B);
108  const Real threshold = 10*std::numeric_limits<Real>::epsilon();
109  if(abs_A < threshold && abs_B < threshold)
110  {
111  Result.ulps(ceil(fabs(abs_B - abs_A) / threshold));
112  }
113  else
114  {
115  Result.ulps(std::fabs(boost::math::float_distance(A, B)));
116  }
117 }
