553 void complete_elements(const mesh::Mesh& mesh, const GidMapT& periodic_links_gid_map)
554 {
555  common::PE::Comm& comm = common::PE::Comm::instance();
556 
557  boost::shared_ptr<mesh::NodeConnectivity> node_connectivity = common::allocate_component<mesh::NodeConnectivity>("NodeConnectivity");
558  node_connectivity->initialize(mesh.elements());
559 
560  const Uint nb_geom_nodes = mesh.geometry_fields().size();
561  common::List<Uint>& geom_node_ranks = mesh.geometry_fields().rank();
562 
563  std::vector<Uint> send_sizes(comm.size(), 0); // Number of Uint values to send to each CPU
564  std::vector< std::set<mesh::NodeConnectivity::ElementReferenceT> > element_sets(comm.size()); // Sets of elements to send
565 
566  // Packed elements to send. Packing is as follows:
567  // - total packed size
568  // - entities index into mesh.entities();
569  // - element GID
570  // then for each space:
571  // - Space index (in the entities.spaces() vector)
572  // - GIDs of the nodes
573  std::vector< std::vector<Uint> > send_elements(comm.size());
574 
575  // Find the elements to pack
576  for(Uint node_idx = 0; node_idx != nb_geom_nodes; ++node_idx)
577  {
578  const Uint destination_rank = geom_node_ranks[node_idx];
579 
580  if(destination_rank == comm.rank())
581  continue;
582 
583  BOOST_FOREACH(const mesh::NodeConnectivity::ElementReferenceT& element_ref, node_connectivity->node_element_range(node_idx))
584  {
585  if(element_sets[destination_rank].insert(element_ref).second)
586  {
587  const mesh::Entities& entities = *mesh.elements()[element_ref.first];
588  send_sizes[destination_rank] += 3;
589  const Uint nb_spaces = entities.spaces().size();
590  for(Uint space_idx = 0; space_idx != nb_spaces; ++space_idx)
591  {
592  send_sizes[destination_rank] += 1 + entities.spaces()[space_idx]->shape_function().nb_nodes();
593  }
594  }
595  }
596  }
597 
598  // Pack the elements
599  mesh::Dictionary& geom_dict = mesh.geometry_fields();
600  common::List<Uint>& geom_node_gids = geom_dict.glb_idx();
601  for(Uint rank = 0; rank != comm.size(); ++rank)
602  {
603  std::vector<Uint>& send_elements_for_rank = send_elements[rank];
604  send_elements_for_rank.reserve(send_sizes[rank]);
605  BOOST_FOREACH(const mesh::NodeConnectivity::ElementReferenceT& element_ref, element_sets[rank])
606  {
607  const Uint entities_idx = element_ref.first;
608  const Uint element_local_id = element_ref.second;
609  const mesh::Entities& entities = *mesh.elements()[entities_idx];
610 
611  // Build the "header"
612  send_elements_for_rank.push_back(3); // 3 numbers for packed size, entities idx and element GID
613  Uint& packed_size = send_elements_for_rank.back();
614  send_elements_for_rank.push_back(entities_idx);
615  send_elements_for_rank.push_back(entities.glb_idx()[element_local_id]);
616 
617  // Add the spaces
618  const Uint nb_spaces = entities.spaces().size();
619  for(Uint space_idx = 0; space_idx != nb_spaces; ++space_idx)
620  {
621  send_elements_for_rank.push_back(space_idx);
622  ++packed_size;
623  const mesh::Space& space = *entities.spaces()[space_idx];
624  BOOST_FOREACH(const Uint node_local_id, space.connectivity()[element_local_id])
625  {
626  send_elements_for_rank.push_back(space.dict().glb_idx()[node_local_id]);
627  ++packed_size;
628  }
629  }
630  }
631  }
632 
633  // Buffers for elements received from other ranks
634  std::vector< std::vector<Uint> > recv_elements(comm.size());
635 
636  // Do the communication
637  comm.all_to_all(send_elements, recv_elements);
638 
639  // Build an index of the owned elements so far
640  const Uint nb_entities = mesh.elements().size();
641  std::vector< std::set<Uint> > element_gid_sets(nb_entities); // Sets to check if an element GID is already added
642  for(Uint entities_idx = 0; entities_idx != nb_entities; ++entities_idx)
643  {
644  BOOST_FOREACH(const Uint element_gid, mesh.elements()[entities_idx]->glb_idx().array())
645  {
646  element_gid_sets[entities_idx].insert(element_gid);
647  }
648  }
649 
650  // Indicates if the received elements starting at a given index need to be unpacked
651  std::vector< std::vector<bool> > needs_unpack(comm.size());
652 
653  // Number of elements to unpack for each Entities in the mesh
654  std::vector<Uint> nb_new_elements(nb_entities, 0);
655 
656  // Count the number of elements to unpack
657  for(Uint rank = 0; rank != comm.size(); ++rank)
658  {
659  const std::vector<Uint>& recv_elements_for_rank = recv_elements[rank];
660  needs_unpack[rank].resize(recv_elements_for_rank.size(), true); // Unpack everything by default
661  Uint recv_idx = 0;
662  const Uint recv_size = recv_elements_for_rank.size();
663  while(recv_idx != recv_size)
664  {
665  const Uint entry_size = recv_elements_for_rank[recv_idx];
666  const Uint entities_idx = recv_elements_for_rank[recv_idx+1];
667  const Uint new_gid = recv_elements_for_rank[recv_idx+2];
668 
669  cf3_assert(entities_idx < element_gid_sets.size());
670 
671  // If we already owned the element, it doesn't need to be unpacked
672  if(!element_gid_sets[entities_idx].insert(new_gid).second)
673  {
674  needs_unpack[rank][recv_idx] = false;
675  }
676  else
677  {
678  ++nb_new_elements[entities_idx];
679  }
680 
681  recv_idx += entry_size;
682  cf3_assert(recv_idx <= recv_size);
683  }
684  }
