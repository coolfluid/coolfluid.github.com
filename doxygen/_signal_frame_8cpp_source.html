300 void SignalFrame::insert( std::vector<std::string>& input )
301 {
302  // extract: variable_name:type=value or variable_name:array[type]=value1,value2
303  boost::regex expression( "([[:word:]]+)(\\:([[:word:]]+)(\\[([[:word:]]+)\\])?=(.*))?" );
304  boost::match_results<std::string::const_iterator> what;
305 
306  boost_foreach (const std::string& arg, input)
307  {
308 
309  std::string name;
310  std::string type;
311  std::string subtype; // in case of array<type>
312  std::string value;
313 
314  if (regex_search(arg,what,expression))
315  {
316  name=what[1];
317  type=what[3];
318  subtype=what[5];
319  value=what[6];
320 
321  //CFinfo << name << ":" << type << (subtype.empty() ? std::string() : std::string("["+subtype+"]")) << "=" << value << CFendl;
322 
323  if(type == "array")
324  {
325  set_array(name, subtype, value, " ; ");
326  }
327  else
328  {
329  set_option(name, type, value);
330  }
331  }
332  else
333  throw ParsingFailed(FromHere(), "Could not parse [" + arg + "].\n"+
334  "Format should be:\n"
335  " - for simple types: variable_name:type=value\n"
336  " - for array types: variable_name:array[type]=value1,value2\n"
337  " with possible type: [bool,unsigned,integer,real,string,uri]");
338  }
339 }

392 std::string SignalFrame::to_script( int indentation ) const
393 {
394  std::string str;
395  std::string target = node.attribute_value("target");
396 
397  if( target == "configure" )
398  str = target + ' ' + node.attribute_value("receiver");
399  else
400  str = "call " + node.attribute_value("receiver") + "/" + target;
401 
402  std::string indent( indentation + str.length() + 1, ' ');
403  SignalFrame frame( *this );
404  SignalOptions opts( frame );
405 
406  OptionList::OptionStorage_t::const_iterator it = opts.store.begin();
407 
408  for (; it != opts.store.end() ; it++ )
409  {
410  boost::shared_ptr<Option const> option = it->second;
411 
412  std::string opt = option->name() + ':';
413  std::string tag = option->type();
414  tag += '=' + option->value_str();
415 
416  if( it == opts.begin() )
417  str += " " + opt + tag;
418  else
419  str += " \\\n" + indent + opt + tag;
420  }
421 
422 
423  return str;
424 }
