105 Handle<Library> Libraries::autoload_library_with_namespace( const std::string& libnamespace )
106 {
107  if( libnamespace.empty() )
108  throw BadValue( FromHere(), "Library namespace is empty - cannot guess library name" );
109 
110  const std::string lib_name = namespace_to_libname( libnamespace );
111 
112  try // to auto-load in case builder not there
113  {
114  CFdebug << "Auto-loading plugin " << lib_name << CFendl;
115  OSystem::instance().lib_loader()->load_library(lib_name);
116  Handle<Library> lib(get_child( libnamespace ));
117  cf3_assert( is_not_null(lib) );
118  return lib;
119  }
120  catch(const std::exception& e)
121  {
122  CFwarn << "Library " << lib_name << " failed to load with error " << e.what() << CFendl;
123  return Handle<Library>();
124  }
125 }
