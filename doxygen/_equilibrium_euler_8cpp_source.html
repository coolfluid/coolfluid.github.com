74  template<typename VectorT, typename TensorT>
75  void compute_velocity(const VectorT& u, const VectorT& u1, const VectorT& g, const Real tau, const TensorT& grad_u)
76  {
77  const TensorT invgrad = (TensorT::Identity() + tau*grad_u).inverse();
78  const VectorT du_dt = (u - u1)/dt;
79  v = u - tau*(1-beta)*invgrad*(du_dt + grad_u*u - g);
80  }

136 void EquilibriumEuler::on_regions_set()
137 {
138  if(options().value<bool>("compute_gradient"))
139  {
140  if(is_null(m_velocity_gradient))
141  {
142  m_velocity_gradient = Handle<common::Action>(create_component("VelocityGradient", "cf3.UFEM.VelocityGradient"));
143  m_velocity_gradient->configure_option_recursively("physical_model", options()["physical_model"].value());
144  m_velocity_gradient->options().set("initial_conditions", options()["initial_conditions"].value());
145  options().option("initial_conditions").link_option(m_velocity_gradient->options().option_ptr("initial_conditions"));
146  }
147  m_velocity_gradient->options().set("velocity_variable", options().option("velocity_variable").value());
148  m_velocity_gradient->options().set("velocity_tag", options().option("velocity_tag").value());
149  m_velocity_gradient->options().set("regions", options().option("regions").value());
150  }
151 
152  const Uint dim = physical_model().ndim();
153  const std::string grad_tag = "velocity_gradient";
154 
155  // Fluid velocity
156  FieldVariable<0, VectorField> u(options().value<std::string>("velocity_variable"), options().value<std::string>("velocity_tag"));
157  FieldVariable<1, VectorField> u1("AdvectionVelocity1", "linearized_velocity");
158  FieldVariable<2, VectorField> v(options().value<std::string>("particle_velocity_variable"), "ufem_particle_velocity");
159  FieldVariable<3, VectorField> g("Force", "body_force");
160  FieldVariable<4, ScalarField> tau(options().value<std::string>("tau_variable"), "ufem_particle_relaxation_time");
161 
162  m_functor->v.resize(dim);
163 
164  if(dim == 2)
165  {
166  FieldVariable<5, VectorField> grad_ux("grad_ux", grad_tag);
167  FieldVariable<6, VectorField> grad_uy("grad_uy", grad_tag);
168 
169  set_expression(nodes_expression_2d
170  (
171  group
172  (
173  lit(*m_functor)(u, u1, g, tau, grad_ux, grad_uy),
174  v[_i] = lit(m_functor->v)[_i]
175  )
176  ));
177  }
178  else if(dim == 3)
179  {
180  FieldVariable<5, VectorField> grad_ux("grad_ux", grad_tag);
181  FieldVariable<6, VectorField> grad_uy("grad_uy", grad_tag);
182  FieldVariable<7, VectorField> grad_uz("grad_uz", grad_tag);
183 
184  set_expression(nodes_expression_3d
185  (
186  group
187  (
188  lit(*m_functor)(u, u1, g, tau, grad_ux, grad_uy, grad_uz),
189  v[_i] = lit(m_functor->v)[_i]
190  )
191  ));
192  }
193  else
194  {
195  throw common::SetupError(FromHere(), "Unsupported dimension " + common::to_str(dim) + " for EquilibriumEuler");
196  }
197 }
