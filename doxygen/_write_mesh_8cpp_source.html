163 void WriteMesh::write_mesh( const Mesh& mesh, const URI& file, const std::vector<URI>& fields)
164 {
165  update_list_of_available_writers();
166 
170 
171  URI filepath = file;
172 
173  if( filepath.scheme() != URI::Scheme::FILE )
174  filepath.scheme( URI::Scheme::FILE );
175 
176 
177  const std::string extension = filepath.extension();
178 
179  if ( m_extensions_to_writers.count(extension) == 0 )
180  throw FileFormatError (FromHere(), "No meshwriter exists for files with extension " + extension);
181 
182  if (m_extensions_to_writers[extension].size()>1)
183  {
184  std::string msg;
185  msg = filepath.string() + " has ambiguous extension " + extension + "\n"
186  + "Possible writers for this extension are: \n";
187  boost_foreach(const Handle< MeshWriter > writer , m_extensions_to_writers[extension])
188  msg += " - " + writer->name() + "\n";
189  throw FileFormatError( FromHere(), msg);
190  }
191 
192  // substitute the regex wildcards in the file name
193 
194  const MeshMetadata& metadata = mesh.metadata();
195 
196  std::string file_str = filepath.path();
197  boost::regex re("\\$\\{(\\w+)\\}");
198  boost::sregex_iterator itr(file_str.begin(), file_str.end(), re);
199  boost::sregex_iterator end;
200  std::vector<std::pair<std::string,std::string> > matches;
201  for(; itr!=end; ++itr)
202  {
203  matches.push_back( std::make_pair((*itr)[0],(*itr)[1]) );
204  }
205 
206  for (Uint i=0; i<matches.size(); ++i)
207  {
208  if ( metadata.check(matches[i].second) )
209  {
210  std::string replace_str;
211  if (matches[i].second == "iter")
212  {
213  std::stringstream ss;
214  ss << std::setw( 4 ) << std::setfill( '0' ) << metadata.properties().value<Uint>(matches[i].second);
215  replace_str = ss.str();
216  }
217  else if (matches[i].second == "time")
218  {
219  std::stringstream ss;
220  ss << std::setprecision(4) << std::setiosflags(std::ios_base::scientific) << std::setw(10) << metadata.properties().value<Real>(matches[i].second);
221  replace_str = ss.str();
222  }
223  else
224  {
225  replace_str = metadata.properties().value_str(matches[i].second);
226  }
227  boost::algorithm::replace_first(file_str,matches[i].first,replace_str);
228  }
229  }
230 
231  // change the path in the filepath
232 
233  filepath.path( file_str );
234 
235  // get the correct writer based on the extension
236 
237  Handle< MeshWriter > writer = m_extensions_to_writers[extension][0];
238  writer->options().set("fields",fields);
239  writer->options().set("mesh",mesh.handle<Mesh>());
240  writer->options().set("file", filepath);
241 
242  writer->execute();
243 }
