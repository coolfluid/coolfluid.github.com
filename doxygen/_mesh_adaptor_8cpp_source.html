1055  std::vector<boost::uint64_t> glb_nodes(dict->size());
1056  for (Uint n=0; n<glb_nodes.size(); ++n)
1057  {
1058  // fill the nodes to send to other cpu's
1059  glb_nodes[n] = dict->glb_idx()[n];
1060

1062  cf3_assert(n<dict->rank().size());
1063  dict->rank()[n] = PE::Comm::instance().rank();

1073  PE::Comm::instance().broadcast(glb_nodes,glb_nodes_from_root,root);

1078
