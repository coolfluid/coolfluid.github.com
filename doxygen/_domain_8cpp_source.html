134 Mesh& Domain::load_mesh( const URI& file, const std::string& name )
135 {
136  Group& tools = Core::instance().tools();
137 
138  Handle<LoadMesh> mesh_loader;
139 
140  mesh_loader = find_component_ptr<LoadMesh>( tools );
141 
142  if ( is_null(mesh_loader) )
143  mesh_loader = find_component_ptr<LoadMesh>( *this );
144 
145  if ( is_null(mesh_loader) )
146  mesh_loader = create_component<LoadMesh>("mesh_loader");
147 
148  Handle<Mesh> mesh = create_component<Mesh>(name);
149 
150  mesh_loader->options().set("dimension",dimension());
151  mesh_loader->load_mesh_into(file, *mesh);
152 
153  CFdebug << "Loaded mesh " << file.string() << " into mesh " << name << CFendl;
154 
155  // rebalance the mesh if necessary and create global idx and ranks
156 
157  build_component_abstract_type<MeshTransformer>("cf3.mesh.actions.LoadBalance","load_balancer")
158  ->transform(mesh);
159 
160  // raise an event to indicate that a mesh was rebalanced (changed)
161  // in serial it is important still to raise the event
162  // considering it like a rebalance that had no effect
163 
164  SignalOptions options;
165  options.add("mesh_uri", mesh->uri());
166  options.add("mesh_rebalanced", true);
167  SignalArgs args = options.create_frame();
168  Core::instance().event_handler().raise_event( "mesh_changed", args);
169 
170  mesh->check_sanity();
171 
172  return *mesh;
173 }
