255 {
256  cf3_assert(m_is_created);
257  values.mat.setZero();
258  const Uint nb_nodes = values.indices.size();
259  const int num_entries = nb_nodes*m_neq;
260  int extracted_num_entries;
261  Real* extracted_values;
262  int* extracted_indices;
263  cf3_assert(values.mat.rows() == num_entries);
264  std::map<int, int> reverse_idx_map;
265  // Convert the index vector
266  for(Uint i = 0; i != nb_nodes; ++i)
267  {
268  const Uint local_start_idx = values.indices[i]*m_neq;
269  for(int j = 0; j != m_neq; ++j)
270  {
271  m_converted_indices[i*m_neq+j] = m_p2m[local_start_idx+j];
272  reverse_idx_map[m_p2m[local_start_idx+j]] = i*m_neq + j;
273  }
274  }
275  // get the values
276  for(Uint i = 0; i != nb_nodes; ++i)
277  {
278  for(int j = 0; j != m_neq; ++j)
279  {
280  if(m_converted_indices[i*m_neq+j] >= m_num_my_elements)
281  continue;
282  TRILINOS_THROW(m_mat->ExtractMyRowView(m_converted_indices[i*m_neq+j], extracted_num_entries, extracted_values, extracted_indices));
283  for(int k = 0; k != extracted_num_entries; ++k)
284  {
285  const std::map<int,int>::const_iterator it = reverse_idx_map.find(extracted_indices[k]);
286  if(it != reverse_idx_map.end())

325 void TrilinosCrsMatrix::symmetric_dirichlet(const Uint blockrow, const Uint ieq, const Real value, Vector& rhs)
326 {
327  // We assume that we have an epetra RHS with the same storage structure as the matrix!
328  Epetra_Vector& epetra_rhs = *dynamic_cast<TrilinosVector&>(rhs).epetra_vector();
329 
330  int num_entries;
331  Real* extracted_values;
332  int* extracted_indices;
333 
334  const int bc_col = m_p2m[blockrow*m_neq+ieq];
335 
336  m_dirichlet_nodes.push_back(std::make_pair(blockrow, ieq));
337 
338  const Uint nb_connected_nodes = m_starting_indices[blockrow+1] - m_starting_indices[blockrow];
339  std::vector<int> row_indices; row_indices.reserve(m_neq*nb_connected_nodes);
340  const Uint conn_start = m_starting_indices[blockrow];
341  const Uint conn_end = m_starting_indices[blockrow+1];
342  for(Uint i = conn_start; i != conn_end; ++i)
343  {
344  for(Uint j = 0; j != m_neq; ++j)
345  {
346  row_indices.push_back(m_p2m[m_node_connectivity[i]*m_neq+j]);
347  }
348  }
349 
350  DirichletEntryT& cached_col_values = m_symmetric_dirichlet_values[bc_col];
351 
352  if(cached_col_values.empty())
353  {
354  BOOST_FOREACH(const int other_row, row_indices)
355  {
356  if(other_row >= m_num_my_elements)
357  continue;
358 
359  TRILINOS_THROW(m_mat->ExtractMyRowView(other_row, num_entries, extracted_values, extracted_indices));
360  const int nb_entries_const = num_entries;
361  Uint i;
362  if(other_row != bc_col)
363  {
364  for(i = 0; i != nb_entries_const; )
365  {
366  if(extracted_indices[i] == bc_col)
367  {
368  cached_col_values[other_row] = extracted_values[i];
369  epetra_rhs[other_row] -= extracted_values[i] * value;
370  extracted_values[i] = 0;
371  break;
372  }
373  ++i;
374  }
375  cf3_assert(i != nb_entries_const);
376  }
377  else
378  {
379  for(i = 0; i != nb_entries_const; ++i)
380  {
381  if(extracted_indices[i] == bc_col)
382  {
383  extracted_values[i] = 1.;
384  }
385  else
386  {
387  extracted_values[i] = 0;
388  }
389  }
390  }
391  }
392  }
393  else // Reuse the cached values, if the matrix wasn't reset since the previous BC application
394  {
395  for(DirichletEntryT::const_iterator it = cached_col_values.begin(); it != cached_col_values.end(); ++it)
396  {
397  epetra_rhs[it->first] -= it->second * value;
398  }
399  }
400 
401  rhs.set_value(blockrow, ieq, value);
402 }

413 void TrilinosCrsMatrix::tie_blockrow_pairs (const Uint iblockrow_to, const Uint iblockrow_from)
414 {
415  cf3_assert(m_is_created);
416  int num_entries_from;
417  Real* extracted_values_from;
418  int* extracted_indices_from;
419  const int row_from_begin = iblockrow_from*m_neq;
420 
421  int num_entries_to;
422  Real* extracted_values_to;
423  int* extracted_indices_to;
424  const int row_to_begin = iblockrow_to*m_neq;
425 
426  if(m_p2m[row_from_begin] >= m_num_my_elements || m_p2m[row_to_begin] >= m_num_my_elements)
427  return;
428 
429  const int nb_eq = m_neq;
430  for(int i = 0; i != nb_eq; ++i)
431  {
432  TRILINOS_THROW(m_mat->ExtractMyRowView(m_p2m[row_from_begin+i], num_entries_from, extracted_values_from, extracted_indices_from));
433  TRILINOS_THROW(m_mat->ExtractMyRowView(m_p2m[row_to_begin+i], num_entries_to, extracted_values_to, extracted_indices_to));
434  if (num_entries_to != num_entries_from)
435  {
436  throw common::BadValue(FromHere(),"Number of entries do not match for the two block rows to be tied together.");
437  }
438  int diag, pair;
439  for(int j = 0; j != num_entries_from; ++j)
440  {
441  if(extracted_indices_from[j] != extracted_indices_to[j])
442  {
443  throw common::BadValue(FromHere(),"Indices of the entries do not match for the two block rows to be tied together.");
444  }
445 
446  if(extracted_indices_from[j] == m_p2m[row_from_begin+i])
447  diag = j;
448  if(extracted_indices_to[j] == m_p2m[row_to_begin+i])
449  pair = j;
450 
451  extracted_values_to[j] += extracted_values_from[j];
452  extracted_values_from[j] = 0.;
453  }
454  extracted_values_from[diag] = 1.;
455  extracted_values_from[pair] = -1.;
456  for(int k = 0; k != nb_eq; ++k)
457  {
458  extracted_values_to[pair-i+k] += extracted_values_to[diag-i+k];
459  extracted_values_to[diag-i+k] = 0.;
460  }
461  }
462 }
