54 void C3DViewBuilder::signal_create_3dview(SignalArgs &args)
55 {
56  SignalOptions options( args );
57 
58  std::string name = options.value<std::string>("3DView name");
59  URI parent = options.value<URI>("Parent");
60 
61  // some checks
62  if(name.empty())
63  throw BadValue(FromHere(), "The 3DView name is empty.");
64 
65  if(parent.empty())
66  throw BadValue(FromHere(), "The parent is empty.");
67 
68  if(parent.scheme() != URI::Scheme::CPATH)
69  throw InvalidURI(FromHere(), "The parent scheme is not CPATH");
70 
71  // create and add the component
72  Handle<Component> parent_comp = Core::instance().root().access_component(parent);
73  Handle<C3DView> view = parent_comp->create_component<C3DView>( name );
74  view->mark_basic();
75 }
