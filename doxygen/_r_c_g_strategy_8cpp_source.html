101  int setup_solver()
102  {
103  if(is_null(m_matrix))
104  throw common::SetupError(FromHere(), "Null matrix for " + m_self.uri().path());
105 
106  if(is_null(m_rhs))
107  throw common::SetupError(FromHere(), "Null RHS for " + m_self.uri().path());
108 
109  if(is_null(m_solution))
110  throw common::SetupError(FromHere(), "Null solution vector for " + m_self.uri().path());
111 
112  // Create the problem
113  m_problem = Teuchos::rcp( new Belos::LinearProblem<Real,MV,OP>(m_matrix->epetra_matrix(), m_solution->epetra_vector(), m_rhs->epetra_vector()) );
114 
115  if(is_not_null(m_xcoords))
116  {
117  m_ml_parameter_list->set("repartition: enable",1);
118  m_ml_parameter_list->set("repartition: max min ratio",1.3);
119  m_ml_parameter_list->set("repartition: min per proc",500);
120  m_ml_parameter_list->set("repartition: partitioner","Zoltan");
121  m_ml_parameter_list->set("repartition: Zoltan dimensions",m_dim);
122 
123  m_ml_parameter_list->set("x-coordinates", m_xcoords);
124  m_ml_parameter_list->set("y-coordinates", m_ycoords);
125  m_ml_parameter_list->set("z-coordinates", m_zcoords);
126  }
127 
128  // Set the preconditioner
129  m_ml_prec = Teuchos::rcp(new ML_Epetra::MultiLevelPreconditioner(*m_matrix->epetra_matrix(), *m_ml_parameter_list, true));
130  Teuchos::RCP<Belos::EpetraPrecOp> belos_prec = Teuchos::rcp(new Belos::EpetraPrecOp(m_ml_prec));
131  m_problem->setLeftPrec(belos_prec);
132 
133  m_solver = Teuchos::rcp(new Belos::RCGSolMgr<double,MV,OP>(m_problem, m_solver_parameter_list));
134 
135  return 0;
136  }
