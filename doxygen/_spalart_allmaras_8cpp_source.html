74 void SpalartAllmaras::trigger_set_expression()
75 {
76  // The code will only be active for these element types
77  boost::mpl::vector2<mesh::LagrangeP1::Quad2D, mesh::LagrangeP1::Triag2D> allowed_elements;
78 
79  FieldVariable<0, ScalarField> nu_sa("SAViscosity", solution_tag());
80  FieldVariable<1, VectorField> u("Velocity",options().value<std::string>("velocity_tag"));
81  FieldVariable<2, ScalarField> nu_eff("EffectiveViscosity", "navier_stokes_viscosity"); // This is the viscosity that needs to be modified to be visible in NavierStokes
82  FieldVariable<3, ScalarField> d("wall_distance", "wall_distance");
83 
84  PhysicsConstant nu_lam("kinematic_viscosity");
85 
86  Handle<ProtoAction>(get_child("Assembly"))->set_expression(
87  elements_expression
88  (
89  allowed_elements,
90  group
91  (
92  _A = _0, _T = _0,
93  compute_tau.apply( u, nu_eff, lit(dt()), lit(tau_su)),
94  comp_sa( u, nu_sa, d, nu_lam),
95  element_quadrature
96  (
97  _A(nu_sa) += transpose(N(nu_sa) + tau_su*u*nabla(nu_sa)) * u * nabla(nu_sa) // advection terms
98  - lit(m_sa_coeff.op.cb1) * lit(m_sa_coeff.op.Stilde) * transpose(N(nu_sa)) * N(nu_sa) // production
99  + lit(m_sa_coeff.op.fw) * lit(m_sa_coeff.op.cw1) * nu_sa / (d*d) * transpose(N(nu_sa)) * N(nu_sa) // destruction
100  + (lit(1.)/lit(m_sa_coeff.op.sigma) * (nu_sa + nu_lam) + diffusion_coeff(u,nu_sa)) * transpose(nabla(nu_sa)) * nabla(nu_sa) // diffusion
101  + lit(m_sa_coeff.op.cb2) / lit(m_sa_coeff.op.sigma) * transpose(N(nu_sa)) * transpose(gradient(nu_sa))*nabla(nu_sa), // diffusion added term
102  _T(nu_sa,nu_sa) += transpose(N(nu_sa) + tau_su*u*nabla(nu_sa)) * N(nu_sa) // Time, standard and SUPG
103  ),
104  system_matrix += invdt() * _T + 1.0 * _A,
105  system_rhs += -_A * _x
106  )
107  ));
108 
109  Handle<ProtoAction>(get_child("Update"))->set_expression(nodes_expression(
110  group
111  (
112  nu_sa += solution(nu_sa),
113  nu_eff = nu_lam + nu_sa * _fv1(nu_sa/nu_lam, lit(m_sa_coeff.op.cv1))
114  )));
115 }
