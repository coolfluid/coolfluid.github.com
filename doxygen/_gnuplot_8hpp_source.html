78 GnuplotPty::GnuplotPty(bool debug_messages) :
79  pty_fn(),
80  pty_fh(NULL),
81  master_fd(-1),
82  slave_fd(-1)
83 {
84 // adapted from http://www.gnuplot.info/files/gpReadMouseTest.c
85  if(0 > openpty(&master_fd, &slave_fd, NULL, NULL, NULL)) {
86  perror("openpty");
87  throw std::runtime_error("openpty failed");
88  }
89  char pty_fn_buf[1024];
90  if(ttyname_r(slave_fd, pty_fn_buf, 1024)) {
91  perror("ttyname_r");
92  throw std::runtime_error("ttyname failed");
93  }
94  pty_fn = std::string(pty_fn_buf);
95  if(debug_messages) {
96  std::cerr << "fn=" << pty_fn << std::endl;
97  }
98 
99  // disable echo
100  struct termios tios;
101  if(tcgetattr(slave_fd, &tios) < 0) {
102  perror("tcgetattr");
103  throw std::runtime_error("tcgetattr failed");
104  }
105  tios.c_lflag &= ~(ECHO | ECHONL);
106  if(tcsetattr(slave_fd, TCSAFLUSH, &tios) < 0) {
107  perror("tcsetattr");
108  throw std::runtime_error("tcsetattr failed");
109  }
110 
111  pty_fh = fdopen(master_fd, "r");
112  if(!pty_fh) {
113  throw std::runtime_error("fdopen failed");
114  }
115 }

176  template <class T>
177  Gnuplot &send(const boost::multi_array<T,2>& x, const boost::multi_array<T,2>& y)
178  {
179  if (x.shape()[1] != 1 || y.shape()[1] != 1)
180  throw std::runtime_error("multi-arrays have more than 1 column");
181  if (x.shape()[0] != y.shape()[0])
182  throw std::runtime_error("multi-arrays have different rows");
