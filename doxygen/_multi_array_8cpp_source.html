86 void get_multi_array( const Map & map, const std::string & name,
87  boost::multi_array<Real, 2> & array,
88  std::vector<std::string> & labels )
89 {
90  cf3_assert( map.content.is_valid() );
91  cf3_assert( !name.empty());
92 
93  // make the code a bit more readable
94  using namespace boost; // first_finder() and is_iequal()
95  using namespace boost::algorithm; // split_iterator and make_split_iterator()
96  typedef split_iterator<std::string::iterator> StringSplitIterator;
97  typedef boost::tokenizer<boost::char_separator<char> > Tokenizer;
98 
99  std::vector<Uint> sizes;
100  std::string delimiter;
101  rapidxml::xml_attribute<char> * attr = nullptr;
102 
103  //
104  // 1. Some information gathering and checks
105  //
106 
107  // 1a. check the array exists
108  XmlNode array_node = map.find_value(name, Protocol::Tags::node_array());
109 
110  if(!array_node.is_valid())
111  throw ValueNotFound(FromHere(), "Could not find a multi-array of name [" + name + "]." );
112 
113  // 1b. look for labels and data nodes
114  XmlNode labels_node( array_node.content->first_node( common::class_name<std::string>().c_str() ) );
115  XmlNode data_node( array_node.content->first_node( common::class_name<Real>().c_str() ) );
116 
117  if(!data_node.is_valid())
118  throw ValueNotFound(FromHere(), "Could not find data for multi-array [" + name + "]." );
119 
120  // 1c. get the delimiter
121  attr = data_node.content->first_attribute( Protocol::Tags::attr_array_delimiter() );
122 
123  if( is_null(attr) )
124  throw XmlError(FromHere(), "Could not find the delimiter for multi-array [" + name + "].");
125 
126  delimiter = attr->value();
127 
128  // 1d. get the multi-array size and resize the target multi-array
129  attr = data_node.content->first_attribute( Protocol::Tags::attr_array_size() );
130 
131  if( is_null(attr) )
132  throw XmlError(FromHere(), "Could not find the size of multi-array [" + name + "].");
133 
134  Map::split_string( attr->value(), ":", sizes, 2 );
135 
136  if( sizes.size() != 2 )
137  throw XmlError(FromHere(), "The multi-array size ["+ std::string(attr->value()) +"] is not valid.");
138 
139  array.resize(boost::extents[ sizes[0] ][ sizes[1] ] );
140 
141  // 1e. get the labels
142  if( labels_node.is_valid() )
143  Map::split_string( labels_node.content->value(), delimiter, labels);
144 
145  //
146  // 2. Fill the multi-array
147  //
148 
149  // the array is written in the XML as a 2D array, with a new line after each
150  // row. Thus we first need to tokenize the string on line breaks and then
151  // split the line depending on the delimiter and cast each element to Real.
152 
153  boost::char_separator<char> sep("\n");
154  std::string str( data_node.content->value() );
155  Tokenizer tokens( str, sep );
156  Tokenizer::iterator tok_iter = tokens.begin();
157 
158  for (int row = 0 ; tok_iter != tokens.end() && row < sizes[0]; ++tok_iter, ++row)
159  {
160  std::string value_str(*tok_iter);
161  StringSplitIterator it = make_split_iterator(value_str, first_finder(delimiter, is_iequal()));
162 
163  for(int col = 0 ; it != StringSplitIterator() && col < sizes[1] ; ++it, ++col )
164  {
165  // split_iterator works with ranges in the given string. Each range
166  // is a portion of the string between 2 delimiters.
167  // copy_range() copies the content of this range to a new string (in our case).
168  std::string val( boost::copy_range<std::string>(*it) );
169 
170  try
171  {
172  array[row][col] = from_str<Real>(val);
173  }
174  catch(boost::bad_lexical_cast e)
175  {
176  throw CastingFailed(FromHere(), "Enable to cast [" + val + "] to Real.");
177  }
178  }
179  }
180 }
