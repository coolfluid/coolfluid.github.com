135 bool PDESolver::stop_condition()
136 {
137  Uint nb_criteria = 0;
138  bool finish = false;
139  boost_foreach(Criterion& stop_criterion, find_components<Criterion>(*this))
140  {
141  finish |= stop_criterion();
142  ++nb_criteria;
143  }
144 
145  if ( m_pde->time() && m_time_step_computer->options().value<bool>("time_accurate") )
146  {
147  if (m_pde->time()->current_time() >= m_pde->time()->end_time())
148  return true;
149  }
150 
151  if (m_pde->time()->iter() >= options().value<Uint>("max_iteration"))
152  return true; // stop
153 
154  return finish;
155 }

159 void PDESolver::do_iteration()
160 {
161  if (m_pre_iteration) m_pre_iteration->execute();
162 
163  step();
164 
165  m_pde->time()->current_time() += m_pde->time()->dt();
166  ++m_pde->time()->iter();
167 
168  if (m_post_iteration) m_post_iteration->execute();
169 
170  iteration_summary();
171 
172  history()->save_entry();
173 
174 // CFinfo << " " << iteration_summary() << CFendl;
175  if ( options().value<bool>("print_iteration_summary") )
176  CFinfo << " " << history()->entry().summary() << CFendl;
177 
178  m_pde->time()->options().set( "current_time", m_pde->time()->current_time() );
179  m_pde->time()->options().set( "iteration", m_pde->time()->iter() );
180 }
