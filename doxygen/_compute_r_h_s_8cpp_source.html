77 void ComputeRHS::compute_rhs(const Uint elem_idx, std::vector<RealVector>& rhs, std::vector<Real>& wave_speed)
78 {
79  boost_foreach( RealVector& rhs_in_pt, rhs)
80  {
81  rhs_in_pt.setZero();
82  }
83  boost_foreach( Real& wave_speed_in_pt, wave_speed)
84  {
85  wave_speed_in_pt = 0.;
86  }
87 
88  for (Uint t=0; t<m_term_computers.size(); ++t)
89  {
90  if (m_loop_cells[t])
91  {
92  m_term_computers[t]->compute_term(elem_idx,m_tmp_term,m_tmp_ws);
93  for (Uint p=0; p<rhs.size(); ++p)
94  {
95  for (Uint eq=0; eq<rhs[p].size(); ++eq)
96  {
97  rhs[p][eq] += m_tmp_term[p][eq];
98  }
99  wave_speed[p] = std::max(wave_speed[p],m_tmp_ws[p]);
100  }
101  }
102  }

107 void ComputeRHS::compute_rhs(mesh::Field& rhs, mesh::Field& wave_speed)
108 {
109  const Uint nb_eqs = rhs.row_size();
110  mesh::Dictionary& dict = rhs.dict();
111  boost_foreach(const Handle<mesh::Entities>& cells, dict.entities_range() )
112  {
113  if ( loop_cells(cells) )
114  {
115  const Space& space = dict.space(*cells);
116 
117  // Element-loop
118  const Uint nb_elems = cells->size();
119  const Uint nb_sol_pts = space.shape_function().nb_nodes();
120 
121  std::vector<RealVector> elem_rhs(nb_sol_pts,RealVector(nb_eqs));
122  std::vector<Real> elem_wave_speed(nb_sol_pts);
123 
124  for (Uint elem_idx=0; elem_idx<nb_elems; ++elem_idx)
125  {
126  if (cells->is_ghost(elem_idx)==false)
127  {
128  compute_rhs(elem_idx,elem_rhs,elem_wave_speed);
129 
130  mesh::Connectivity::ConstRow nodes = space.connectivity()[elem_idx];
131  for (Uint sol_pt=0; sol_pt<nb_sol_pts; ++sol_pt)
132  {
133  for (Uint eq=0; eq<nb_eqs; ++eq)
134  {
135  rhs[nodes[sol_pt]][eq] = elem_rhs[sol_pt][eq];
136  }
137  wave_speed[nodes[sol_pt]][0] = elem_wave_speed[sol_pt];
138  }
139  }
140  }
141  }
142  }
