37 // Specialized code for triangles
38 struct PoissonTriagAssembly
39 {
40  typedef void result_type;
41 
42  // Functor that takes: source term f, Linear system lss, element matrix and vector acc
43  template<typename FT, typename LSST>
44  void operator()(const FT& f, LSST& lss, math::LSS::BlockAccumulator& acc) const
45  {
46  typedef mesh::LagrangeP1::Triag2D ElementT;
47  // Get the coordinates of the element nodes
48  const ElementT::NodesT& nodes = f.support().nodes();
49 
50  // Compute normals
51  ElementT::NodesT normals;
52  normals(0, XX) = nodes(1, YY) - nodes(2, YY); normals(0, YY) = nodes(2, XX) - nodes(1, XX);
53  normals(1, XX) = nodes(2, YY) - nodes(0, YY); normals(1, YY) = nodes(0, XX) - nodes(2, XX);
54  normals(2, XX) = nodes(0, YY) - nodes(1, YY); normals(2, YY) = nodes(1, XX) - nodes(0, XX);
55 
56  // Jacobian determinant
57  const Real det_jac = normals(2, YY)*normals(1, XX) - normals(1, YY)*normals(2, XX);
58  const Real c = 1. / (2.*det_jac);
59 
60  acc.neighbour_indices(f.support().element_connectivity());
61 
62  for(Uint i = 0; i != 3; ++i)
63  for(Uint j = 0; j != 3; ++j)
64  acc.mat(i, j) = c * (normals(i, XX)*normals(j, XX) + normals(i, YY)*normals(j, YY));
65 
66  // Get the values of the source term
67  const Real f0 = f.value()[0];
68  const Real f1 = f.value()[1];
69  const Real f2 = f.value()[2];
70 
71  acc.rhs[0] = (2*f0 + f1 + f2);
72  acc.rhs[1] = (f0 + 2*f1 + f2);
73  acc.rhs[2] = (f0 + f1 + 2*f2);
74  acc.rhs *= det_jac/24.;
75 
76  lss.matrix().add_values(acc);
77  lss.rhs().add_rhs_values(acc);
78  }
79 };
