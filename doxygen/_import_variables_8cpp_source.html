81 void ImportVariables::execute()
82 {
83  // Check correct configuration
84  Handle<Dictionary> dict ( m_mesh->access_component_checked( options().value<URI>("dict") ) );
85  if (is_null(dict))
86  throw SetupError(FromHere(), "Dictionary not found in "+m_mesh->uri().string());
87 
88  std::vector< std::string > var_names = options().value< std::vector<std::string> >("variable_names");
89  if (var_names.size() == 0)
90  var_names = m_import_var_names;
91  if ( var_names.size() != m_import_var_names.size() )
92  throw SetupError(FromHere(), "Variable names has different size than number of variables to import");
93 
94  if (options().value< std::vector<URI> >("files").size() == 0)
95  throw SetupError(FromHere(), "No files have been configured to load variables from");
96 
97  // Import mesh
98  boost::shared_ptr<Mesh> loaded_mesh = allocate_component<Mesh>("imported");
99  m_mesh_loader->load_multiple_files( options().value< std::vector<URI> >("files") , *loaded_mesh );
100 
101  m_interpolator->options().set("store",true);
102 
103  if (m_import_var_names.size())
104  {
105  // Find the variables
106  for (Uint v=0; v<var_names.size(); ++v)
107  {
108  std::string import_var_name = m_import_var_names[v];
109  std::string var_name = var_names[v];
110  boost_foreach (const Handle<Dictionary>& loaded_dict, loaded_mesh->dictionaries() )
111  {
112  boost_foreach (const Handle<Field>& loaded_field, loaded_dict->fields())
113  {
114  if (loaded_field->has_variable(import_var_name))
115  {
116  std::string var_dimensionality = math::VariablesDescriptor::Dimensionalities::Convert::instance().to_str( loaded_field->descriptor().dimensionality(import_var_name) );
117  Uint var_idx = loaded_field->var_offset(import_var_name);
118  Uint var_length = loaded_field->var_length(import_var_name);
119  std::vector<Uint> source_vars(var_length);
120  std::vector<Uint> target_vars(var_length);
121  for (Uint i=0; i<var_length; ++i)
122  {
123  source_vars[i] = var_idx+i;
124  target_vars[i] = i;
125  }
126  Field& imported_field = dict->create_field(var_name, var_name+"["+var_dimensionality+"]");
127  m_interpolator->interpolate_vars(*loaded_field,imported_field,source_vars,target_vars);
128  }
129  }
130  }
131  }
132  }
133  else
134  {

136  {
