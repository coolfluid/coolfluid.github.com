68 void Notifier::notify(const std::string& event, const std::string& description)
69 {
70  int response = prowl_push_msg(
71  const_cast<char*>(m_api_key.c_str()),
72  m_priority,
73  const_cast<char*>(m_application_name.c_str()),
74  const_cast<char*>(event.c_str()),
75  const_cast<char*>(description.c_str())
76  );
77 
78  if (response != SUCCESS)
79  {
80  switch (response)
81  {
82  case BAD_REQUEST:
83  CFerror << "Bad request, the parameters you provided did not validate" << CFendl;
84  CFerror << "Possible priorities: [-2, -1, 0 , 1, 2]" << CFendl;
85  break;
86  case NOT_AUTHORIZED:
87  CFerror << "Not authorized, the given API key ["<<m_api_key<<"] is not valid, and does not correspond to a user." << CFendl;
88  break;
89  case NOT_ACCEPTABLE:
90  CFerror << "Not acceptable, your IP address has exceeded the API limit." << CFendl;
91  break;
92  case NOT_APPROVED:
93  CFerror << "Not approved, the user has yet to approve your retrieve request." << CFendl;
94  break;
95  case INTERNAL_SERVER_ERROR:
96  CFerror << "Internal server error, something failed to execute properly on the Prowl side." << CFendl;
97  break;
98  case INVALID_RESPONSE:
99  CFerror << "Prowl sent an invalid response back" << CFendl;
100  break;
101  default:
102  CFerror << "Error code ["<<response<<"], unknown error" << CFendl;
103  break;
104  }
105  }
106 }
