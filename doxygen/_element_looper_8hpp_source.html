171  {
172  ElementGrammar grammar;
173  for(Uint elem = 0; elem != nb_elems; ++elem)

287 template<typename ElementTypesT, typename ExprT>
288 void for_each_element(mesh::Region& root_region, const ExprT& expr)
289 {
290  // Store the variables
291  typedef typename ExpressionProperties<ExprT>::VariablesT VariablesT;
292  VariablesT vars;
293  CopyNumberedVars<VariablesT> ctx(vars); // This is a proto context
294  boost::proto::eval(expr, ctx); // calling eval using the above context stores all variables in vars
295 
296  // Traverse all Elements under the root and evaluate the expression
297  BOOST_FOREACH(mesh::Elements& elements, common::find_components_recursively<mesh::Elements>(root_region))
298  {
299  // We skip order 0 functions in the top-call, because first the support shape function is determined, and order 0 is not allowed there
300  boost::mpl::for_each< boost::mpl::filter_view< ElementTypesT, mesh::IsMinimalOrder<1> > >( ElementLooper<ElementTypesT, ExprT>(elements, expr, vars) );
301  }
