91 void NLog::append_to_log(LogMessage::Type type, bool fromServer,
92  const QString & message)
93 {
94  QString header = "[ %1 ][ %2 ] ";
95  QString msg;
96 
97  header = header.arg(QDateTime::currentDateTime().toString("yyyy-MM-dd HH:mm:ss"));

100  msg = message;
