56 void OptionURI::change_value_impl(const boost::any& value)
57 {

61 
62  if(std::count(m_protocols.begin(), m_protocols.end(), val.scheme() == 0))
63  {
64  std::stringstream error_str;
65  error_str << "Protocol " + URI::Scheme::Convert::instance().to_str(val.scheme()) + " is not supported. Accepted values are:";
