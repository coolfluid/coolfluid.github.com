281 void CNode::modify_options( const QMap<QString, QString> & opts )
282 {
283  QMutexLocker locker(m_mutex);
284 
285  QMap<QString, QString>::const_iterator it = opts.begin();
286 
287  QMap<QString, QString>::const_iterator it2 = opts.begin();
288 
289  // for better readability of the code, a SignalFrame is built even when
290  // modifying local options. It's not the most efficient way to do, but local
291  // options are not meant to be modified 10 times each second so this should
292  // not affect the application performances.
293  // SignalFrame frame("configure", uri(), uri());
294  // SignalOptions options(frame);
295  SignalOptions options;
296 
297  for( ; it != opts.end() ; it++)
298  {
299  std::string sep;
300  std::string name( it.key().toStdString() );
301  std::string value( it.value().toStdString() );
302 
303  if( !this->options().check(name) )
304  throw ValueNotFound(FromHere(), "Could not find an option with name ["+name+"]");
305 
306  Option& option = this->options().option(name);
307  std::string type( option.type() );
308  bool is_array = boost::starts_with(option.type(), "array");
309 
310  // if it is an array, we need to get the element type
311  if(is_array)
312  {
313  type = option.element_type();
314  sep = option.separator();
315  }
316 
317  if(type == common::class_name<bool>()) // bool
318  addValueToXml<bool>(name, value, sep, options);
319  else if(type == common::class_name<int>()) // int
320  addValueToXml<int>(name, value, sep, options);
321  else if(type == common::class_name<Uint>()) // Uint
322  addValueToXml<Uint>(name, value, sep, options);
323  else if(type == common::class_name<Real>()) // Real
324  addValueToXml<Real>(name, value, sep, options);
325  else if(type == common::class_name<std::string>()) // string
326  addValueToXml<std::string>(name, value, sep, options);
327  else if(type == common::class_name<URI>()) // URI
328  {
329  // since OptionT<URI> does not exist, using addValueToXml for
330  // this type would lead to an undefined reference linking error
331  if( sep.empty() )
332  options.add(name, from_str<URI>(value));
333  else
334  {
335  std::vector<URI> data;
336  Map::split_string(value, sep, data);
337  options.add(name, data);
338  }
339  }
340  else
341  throw ValueNotFound(FromHere(), type + ": Unknown type for option [" + name + "]." );
342  }
343 
344  // if there were options to modify
345  if( !options.store.empty() )
346  {
347  SignalFrame frame = options.create_frame("configure", uri(), uri());
348 
349  if( is_local_component() )
350  signal_configure( frame );
351  else
352  NetworkQueue::global()->send( frame );
353  }

596 boost::shared_ptr< CNode > CNode::create_from_xml_recursive( XmlNode & node,
597  QMap<boost::shared_ptr< NLink >, URI> & link_targets )
598 {
599  rapidxml::xml_attribute<>* typeAttr = node.content->first_attribute("atype");
600  rapidxml::xml_attribute<>* nameAttr = node.content->first_attribute("name");
601  rapidxml::xml_attribute<>* mode_attr = node.content->first_attribute("mode");
602 
603  UUCount uuid(node.attribute_value( "uuid" ));
604 
605  cf3_always_assert(typeAttr != nullptr);
606  cf3_always_assert(nameAttr != nullptr);
607 
608  QString type_name = typeAttr->value();
609  char * node_name = nameAttr->value();
610  XmlNode child( node.content->first_node() );
611 
612  cf3_assert( !type_name.isEmpty() );
613  cf3_assert( node_name != nullptr );
614  cf3_assert( type_name.length() > 0 );
615  cf3_assert( std::strlen(node_name) > 0 );
616 
617  boost::shared_ptr< CNode > root_node;
618 
619  if( type_name == "CCore" )
620  return root_node;
621 
622  if( type_name == "Link" )
623  {
624  boost::shared_ptr< NLink > link = boost::shared_ptr<NLink>(new NLink(node_name));
625  root_node = link;
626  link_targets[link] = node.content->value();
627  }
628  else if( CNodeBuilders::instance().has_builder( type_name ) )
629  root_node = CNodeBuilders::instance().build_cnode(type_name, node_name);
630  else if( type_name == "Journal" )
631  root_node = boost::shared_ptr<NJournal>(new NJournal(node_name));
632  else if( type_name == "Root" )
633  root_node = boost::shared_ptr<NRoot>(new NRoot(node_name));
634  else
635  root_node = boost::shared_ptr<NGeneric>(new NGeneric(node_name, type_name));
636 
637  if(mode_attr != nullptr && std::strcmp(mode_attr->value(), "basic") == 0)
638  root_node->mark_basic();
639 
640  if( !uuid.is_nil() )
641  root_node->properties().set( "uuid", uuid );
642  else
643  NLog::global()->add_warning( "Found a Component without no UuiD." );
644 
645  while( child.is_valid() )
646  {
647  try
648  {
649  boost::shared_ptr< CNode > node = create_from_xml_recursive(child, link_targets);
650 
651  if(node.get() != nullptr)
652  {
653  root_node->add_node(node);
654  node->setup_finished();
655  }
656  }
657  catch (Exception & e)
658  {
659  NLog::global()->add_exception(e.msg().c_str());
660  }
661 
662  child.content = child.content->next_sibling();
663  }
664 
665  return root_node;
666 }
