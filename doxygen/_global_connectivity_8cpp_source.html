84  // 1) make std::map<glb_node_idx,loc_node_idx>
85  // 2) Make node2elem connectivity (does not contain elements from other partitions)
86  // 3) foreach ghostnode, store connected owned elements
87  // 4) broadcast (3) to all, and store the map node to ghost elements

91 
92  //1)

94  Uint loc_node_idx(0);

96  node_glb2loc[glb_node_idx]=loc_node_idx++;
97 
98  //2)
99 
100  boost::shared_ptr<NodeElementConnectivity> node2elem_ptr = common::allocate_component<NodeElementConnectivity>("node2elem");
101  NodeElementConnectivity& node2elem = *node2elem_ptr;
102  node2elem.setup(mesh.topology());
103 
104 
105  // 3)
106  Uint nb_ghost(0);
107  for (Uint i=0; i<nodes.size(); ++i)
108  if (nodes.is_ghost(i))
109  ++nb_ghost;
110 
111  std::vector<Uint> ghostnode_glb_idx(nb_ghost);
112  std::vector<Uint> ghostnode_glb_elem_connectivity;
113  std::vector<Uint> ghostnode_glb_elem_connectivity_start(nb_ghost+1);
114  ghostnode_glb_elem_connectivity_start[0]=0;
115  Handle< Component > elem_comp;
116  Uint elem_idx;
117 
118  Uint cnt(0);
119  for (Uint i=0; i<mesh.geometry_fields().size(); ++i)
120  {
121  if (mesh.geometry_fields().is_ghost(i))
122  {
123  ghostnode_glb_idx[cnt] = nodes_glb_idx[i];
124 
125  DynTable<Uint>::ConstRow elems = node2elem.connectivity()[i];
126  boost_foreach(const Uint e, elems)
127  {
128  boost::tie(elem_comp,elem_idx) = node2elem.elements().location(e);
129  ghostnode_glb_elem_connectivity.push_back(dynamic_cast<Elements&>(*elem_comp).glb_idx()[elem_idx]);
130  }
131  ghostnode_glb_elem_connectivity_start[cnt+1] = ghostnode_glb_elem_connectivity_start[cnt] + elems.size();
132 
133  ++cnt;
134  }
135  }
136 
137  // 4)
138  std::vector<std::vector<Uint> > glb_elem_connectivity(nodes.size());
139  nodes_glb_idx.resize(mesh.geometry_fields().size());
140 
141  for (Uint root=0; root<PE::Comm::instance().size(); ++root)
142  {
143  std::vector<Uint> rcv_glb_node_idx(0);//ghostnode_glb_idx.size());
144  PE::Comm::instance().broadcast(ghostnode_glb_idx,rcv_glb_node_idx,root);
145  std::vector<Uint> rcv_glb_elem_connectivity(0);//ghostnode_glb_elem_connectivity.size());
146  PE::Comm::instance().broadcast(ghostnode_glb_elem_connectivity,rcv_glb_elem_connectivity,root);
147  std::vector<Uint> rcv_glb_elem_connectivity_start(0);//ghostnode_glb_elem_connectivity_start.size());
148  PE::Comm::instance().broadcast(ghostnode_glb_elem_connectivity_start,rcv_glb_elem_connectivity_start,root);
149 
150  if (PE::Comm::instance().rank() != root)
151  {
152  for (Uint p=0; p<PE::Comm::instance().size(); ++p)
153  {
154  if (p == PE::Comm::instance().rank())
155  {
156  Uint rcv_idx(0);
157  boost_foreach(const std::size_t glb_node, rcv_glb_node_idx)
158  {
159  if (node_glb2loc.find(glb_node) != node_glb2loc.end())
160  {
161  //std::cout << "["<<PE::Comm::instance().rank() << "] owns ghostnode " << glb_node << " of [" << root << "]" << std::endl;
162  Uint loc_node_idx = node_glb2loc[glb_node];
163  for(Uint l=rcv_glb_elem_connectivity_start[rcv_idx]; l<rcv_glb_elem_connectivity_start[rcv_idx+1]; ++l)
164  glb_elem_connectivity[loc_node_idx].push_back(rcv_glb_elem_connectivity[l]);
165  }
166  ++rcv_idx;
167  }
168  break;
169  }
170  }
171  }
172  }

176 // CFinfo << "nodes_glb_elem_connectivity = " << nodes_glb_elem_connectivity.uri() << CFendl;
177  nodes_glb_elem_connectivity.resize(glb_elem_connectivity.size());

179  {

195  {
196  cf3_assert(cnt < nodes_glb_elem_connectivity[i].size());
