44 HeatCouplingFlux::HeatCouplingFlux(const std::string& name) :
45  ActionDirector(name),
46  m_rhs(options().add("lss", Handle<math::LSS::System>()).pretty_name("LSS").description("The linear system for which the boundary condition is applied"))
47 {
48  options().add("gradient_region", m_gradient_region)

55  .pretty_name("Temperature Field Tag")

63  set_boundary_gradient->options().set("field_tag", std::string("gradient_field"));
64  // Finally set the boundary condition

94 void HeatCouplingFlux::trigger_setup()
95 {
96  // Get the tags for the used fields
97  const std::string temperature_field_tag = options().value<std::string>("temperature_field_tag");
98 
99  Handle<ProtoAction> compute_gradient(get_child("ComputeGradient"));
100  Handle<AdjacentCellToFace> set_boundary_gradient(get_child("SetBoundaryGradient"));
101  Handle<ProtoAction> neumann_heat_flux(get_child("NeumannHeatFlux"));
102 
103  // Represents the temperature field, as calculated
104  FieldVariable<0, ScalarField> T("Temperature", temperature_field_tag);
105  // Represents the gradient of the temperature, to be stored in an (element based) field
106  FieldVariable<1, VectorField> GradT("TemperatureGradient", "gradient_field", mesh::LagrangeP0::LibLagrangeP0::library_namespace());
107 
108  // Expression to calculate the gradient, at the cell centroid:
109  // nabla(T, center) is the shape function gradient matrix evaluated at the element center
110  // T are the nodal values for the temperature
111  compute_gradient->set_expression(elements_expression
112  (
113  boost::mpl::vector2<mesh::LagrangeP0::Quad, mesh::LagrangeP1::Quad2D>(),
114  GradT = nabla(T, gauss_points_1)*nodal_values(T) // Calculate the gradient at the first gauss point, i.e. the cell center
115  ));
116 
117  // Expression for the Neumann BC itself
118  neumann_heat_flux->set_expression(elements_expression
119  (
120  boost::mpl::vector2<mesh::LagrangeP0::Line, mesh::LagrangeP1::Line2D>(), // Valid for surface element types
121  group(m_rhs(T) += - integral<1>(transpose(N(T))*GradT*normal), // Classical Neumann condition formulation for finite elements
122  _cout << "rhs" << integral<1>(transpose(N(T))*GradT*normal) << "\n",
123  _cout << "GradT" << GradT << "\n")
124  ));
125 
126  // Raise an event to indicate that we added a variable (GradT)
127  common::XML::SignalOptions options;
128  common::SignalArgs f = options.create_frame();
129  common::Core::instance().event_handler().raise_event("ufem_variables_added", f);
130 }
