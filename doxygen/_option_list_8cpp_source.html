228 void OptionList::set( const std::string & arg )
229 {
230  // extract: variable_name:type=value or variable_name:array[type]=value1,value2
231  boost::regex expression( "([[:word:]]+)(\\:([[:word:]]+)(\\[([[:word:]]+)\\])?=(.*))?" );
232  boost::match_results<std::string::const_iterator> what;
233 
234  std::string name;
235  std::string type;
236  std::string subtype; // in case of array<type>
237  std::string value;
238 
239  if (regex_search(arg,what,expression))
240  {
241  name=what[1];
242  type=what[3];
243  subtype=what[5];
244  value=what[6];
245 
246  //CFinfo << name << ":" << type << (subtype.empty() ? std::string() : std::string("["+subtype+"]")) << "=" << value << CFendl;
247 
248  if (type == "bool")
249  set_option_to_list< OptionT<bool> >(name, from_str<bool>( value ), *this );
250  else if (type == "unsigned")
251  set_option_to_list< OptionT<Uint> >(name, from_str<Uint>( value ), *this );
252  else if (type == "integer")
253  set_option_to_list< OptionT<int> >(name, from_str<int>( value ), *this );
254  else if (type == "real")
255  set_option_to_list< OptionT<Real> >(name, from_str<Real>( value ), *this );
256  else if (type == "string")
257  set_option_to_list< OptionT<std::string> >(name, value, *this );
258  else if (type == "uri")
259  set_option_to_list< OptionURI >(name, from_str<URI>( value ), *this );
260  else if (type == "array")
261  {
262  std::vector<std::string> array;
263 
264  // the strings could have comma's inside, brackets, etc...
265  Uint in_brackets(0);
266  std::string::iterator first = value.begin();
267  std::string::iterator it = first;
268  for ( ; it!=value.end(); ++it)
269  {
270  if (*it == '(') // opening bracket
271  ++in_brackets;
272  else if (*it == ')') // closing bracket
273  --in_brackets;
274  else if (*it == ',' && in_brackets == 0)
275  {
276  array.push_back(std::string(first,it));
277  boost::algorithm::trim(array.back());
278  first = it+1;
279  }
280  }
281  array.push_back(std::string(first,it));
282  boost::algorithm::trim(array.back());
283 
284  if (subtype == "bool")
285  set_array_to_list<bool>(name, array, *this);
286  else if (subtype == "unsigned")
287  set_array_to_list<Uint>(name, array, *this);
288  else if (subtype == "integer")
289  set_array_to_list<int>(name, array, *this);
290  else if (subtype == "real")
291  set_array_to_list<Real>(name, array, *this);
292  else if (subtype == "string")
293  set_option_to_list< OptionArray<std::string> >(name, array, *this);
294  else if (subtype == "uri")
295  set_array_to_list<URI>(name, array, *this);
296 
297  }
298  else
299  throw ParsingFailed(FromHere(), "The type ["+type+"] of passed argument [" + arg + "] is invalid.\n"+
300  "Format should be:\n"
301  " - for simple types: variable_name:type=value\n"
302  " - for array types: variable_name:array[type]=value1,value2\n"
303  " with possible type: [bool,unsigned,integer,real,string,uri]");
304  }
305  else
306  throw ParsingFailed(FromHere(), "Could not parse [" + arg + "].\n"+
307  "Format should be:\n"
308  " - for simple types: variable_name:type=value\n"
309  " - for array types: variable_name:array[type]=value1,value2\n"
310  " with possible type: [bool,unsigned,integer,real,string,uri]");
311 }
