90 void FieldManager::create_field(const std::string& tag, Dictionary& dict)
91 {
92  boost_foreach(VariablesDescriptor& descriptor, find_components_with_tag<VariablesDescriptor>(m_implementation->variable_manager(), tag))
93  {
94  const Handle< Field > existing_field = find_component_ptr_with_tag<Field>(dict, tag);
95  if(is_not_null(existing_field))
96  {
97  if(descriptor.description() != existing_field->descriptor().description() || descriptor.options().option(common::Tags::dimension()).value<Uint>() != existing_field->descriptor().options().option(common::Tags::dimension()).value<Uint>())
98  {
99  throw SetupError(FromHere(), "Existing field with tag " + tag + " at " + existing_field->uri().string() + " is incompatible with descriptor " + descriptor.uri().string()
100  + ": existing " + existing_field->descriptor().description() + " != required " + descriptor.description());
101  }
102 
103  CFdebug << "Skipping second field creation for tag " << tag << " in fieldgroup " << dict.uri().string() << CFendl;
104  continue;
105  }
106 
107  dict.create_field(tag, descriptor).add_tag(tag);
108 
109  CFdebug << "Creating field with tag " << tag << ": " << descriptor.description() << CFendl;
110  }
111 }
