365 int MainWindow::confirm_close()
366 {
367  int answer = CLOSE_DISC;
368  QMessageBox discBox(this);
369  QPushButton * btDisc = nullptr;
370  QPushButton * btCancel = nullptr;
371  QPushButton * btShutServer = nullptr;
372 
373  if(m_tree_view->try_commit())
374  {
375  btDisc = discBox.addButton("Disconnect", QMessageBox::NoRole);
376  btCancel = discBox.addButton(QMessageBox::Cancel);
377  btShutServer = discBox.addButton("Shutdown server", QMessageBox::YesRole);
378 
379  discBox.setWindowTitle("Confirmation");
380  discBox.setText("You are about to quit this application.");
381  discBox.setInformativeText("What do you want to do ?");
382 
383  // show the message box
384  if(ThreadManager::instance().network().is_connected())
385  {
386  discBox.exec();
387 
388  if(discBox.clickedButton() == btDisc)
389  answer = CLOSE_DISC;
390  else if(discBox.clickedButton() == btShutServer)
391  answer = CLOSE_SHUTDOWN;
392  else
393  answer = CLOSE_CANCEL;
394 
395  delete btDisc;
396  delete btCancel;
397  delete btShutServer;
398  }
399  }
400  else
401  answer = CLOSE_CANCEL;
402 
403  return answer;
404 }
