108 void MeshWriter::config_fields()
109 {
110  if (is_null(m_mesh))
111  throw SetupError(FromHere(),"Mesh was not configured in mesh-writer ["+uri().string()+"]");
112 
113  std::vector<URI> field_uris = options()["fields"].value< std::vector<URI> >();
114  m_fields.clear();
115  m_fields.reserve(field_uris.size());
116  boost_foreach ( const URI& uri, field_uris)
117  {
118  m_fields.push_back(Handle<Field const>(m_mesh->access_component_checked(uri)));
119  if ( is_null(m_fields.back()) )
120  throw ValueNotFound(FromHere(),"Invalid type of field URI ["+uri.string()+"]");
121  }

151 void MeshWriter::execute()
152 {
153  // Check if the mesh was configured
154  if (is_null(m_mesh))
155  throw SetupError(FromHere(),"Mesh was not configured in mesh-writer ["+uri().string()+"]");
156 
157  CFinfo << "Writing mesh " << m_file_path << CFendl;
158 
159  // Configure the fields to write
160  config_fields();
161 
162  // Configure the regions to write
163  config_regions();
164 
165  m_filtered_entities.clear();
166  boost_foreach(const Handle<Region const>& region, m_regions)
167  boost_foreach(const Entities& entities, find_components_recursively_with_filter<Entities>(*region,m_entities_filter))
168  m_filtered_entities.push_back(entities.handle<Entities>());
169 
170  // Call implementation
171  write();
172 }

186 bool MeshWriter::RegionFilter::operator()(const Component& component)
187 {
188  Uint cnt(0);
189 
190  // enable interior cells
191  if (enable_interior_cells)
192  cnt += common::count(common::find_components<Cells>(component));
193 
194  // enable surfaces
195  if (enable_surfaces)
196  cnt += common::count(common::find_components<Faces>(component));
197 
198  // enable interior faces
199  if (enable_interior_faces)
200  cnt += common::count(common::find_components<CellFaces>(component));
201 
202  return (cnt>0);
203 }
