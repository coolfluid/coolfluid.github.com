58 void MergedParallelDistribution::config_nb_obj ()
59 {
60  m_nb_obj = options().value< std::vector<Uint> >("nb_obj");
61  boost_foreach(Handle< ParallelDistribution > hash, m_subhash)
62  remove_component(hash->name());
63  m_subhash.resize(0);
64  Uint i=0;
65  boost_foreach(Uint nb_obj, m_nb_obj)
66  {
67  Handle<ParallelDistribution> hash = create_component<ParallelDistribution>("hash_"+to_str(i));
68  m_subhash.push_back(hash);
69  hash->options().set("nb_obj", nb_obj);
70  hash->options().set("nb_parts", m_nb_parts);
71  ++i;
72  }
73 }
