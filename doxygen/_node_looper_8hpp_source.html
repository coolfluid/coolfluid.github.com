130  void operator()() const
131  {
132  // Create data used for the evaluation
133  mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(m_region);
134  Handle<mesh::Dictionary const> dict;
135  boost::fusion::for_each(m_variables, FindDict(mesh, dict));
136  if(is_null(dict))

138 
139  const mesh::Field& coordinates = dict->coordinates();

142  // Wrap things up so that we can store the intermediate product results
143  do_run(WrapExpression()(m_expr, 0, node_data), node_data, *dict);
