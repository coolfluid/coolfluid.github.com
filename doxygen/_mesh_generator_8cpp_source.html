45 void MeshGenerator::config_mesh()
46 {
47  URI mesh_uri = options().value<URI>("mesh");
48  if ( Handle< Component > found_mesh = Core::instance().root().access_component( mesh_uri ) )
49  {
50  m_mesh = Handle<Mesh>(found_mesh);
51  cf3_assert(m_mesh);
52  }
53  else
54  {
55  URI parent_uri = mesh_uri.base_path();
56  std::string mesh_name = mesh_uri.name();
57  if ( Handle< Component > found_parent = Core::instance().root().access_component( parent_uri ) )
58  {
59  m_mesh = found_parent->create_component<Mesh>(mesh_name);
60  }
61  else
62  {
63  throw ValueNotFound(FromHere(), "Could not find component "+parent_uri.string()+" to create mesh \""+mesh_name+"\" into");
64  }
65  }
