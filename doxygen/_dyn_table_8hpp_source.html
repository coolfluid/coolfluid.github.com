327  void flush()
328  {
329 
330  // get total number of allocated rows
331  Uint allocated_size = total_allocated();
332  Uint old_array_size = m_array.size();
333 
334  // get total number of empty rows
335  Uint nb_emptyRows = m_empty_array_rows.size() + m_empty_buffer_rows.size() + m_new_buffer_rows.size();
336  Uint new_size = allocated_size-nb_emptyRows;
337 
338  if (new_size > old_array_size)
339  {
340  // make m_array bigger
341  m_array.resize(new_size);
342 
343  // copy each buffer into the array
344  Uint array_idx=old_array_size;
345  BOOST_FOREACH (Buffer& buffer, m_buffers)
346  {
347  for (Uint row_idx=0; row_idx<buffer.size(); ++row_idx)
348  {
349  Row row = buffer.rows[row_idx];
350  if (buffer.is_not_empty[row_idx]) // for each non-empty row from all buffers
351  {
352  // first find empty rows inside the old part array
353  if (!m_empty_array_rows.empty())
354  {
355  Row empty_array_row = get_row(m_empty_array_rows.front());
356  m_empty_array_rows.pop_front();
357  empty_array_row = row;
358  }
359  else // then select the new array rows to be filled
360  {
361  cf3_assert(array_idx < m_array.size());
362  Row empty_array_row=m_array[array_idx++];
363  empty_array_row = row;
364  }
365  }
366  }
367  }
368  }
369  else // More rows to be removed than added, now we need to swap rows
370  {
371  // copy all buffer rows in the m_array
372  BOOST_FOREACH (Buffer& buffer, m_buffers)
373  {
374  for (Uint row_idx=0; row_idx<buffer.size(); ++row_idx)
375  {
376  Row row = buffer.rows[row_idx];
377  if (buffer.is_not_empty[row_idx]) // for each non-empty row from all buffers
378  {
379  Uint empty_array_row_idx = m_empty_array_rows.front();
380  m_empty_array_rows.pop_front();
381  Row empty_array_row = get_row(empty_array_row_idx);
382  empty_array_row = row;
383  }
384  }
385  }
386  Uint full_row_idx = new_size;
387  // The part of the table with rows > new_size will be deallocated
388  // The empty rows from the allocated part must be swapped with filled
389  // rows from the part that will be deallocated
390  Uint nb_empty_rows = m_empty_array_rows.size();
391  for (Uint e=0; e<nb_empty_rows; ++e)
392  {
393  Uint empty_row_idx = m_empty_array_rows[e];
394  // swap only necessary if the empty row is in the allocated part
395  if (empty_row_idx < new_size)
396  {
397  // swap this empty row with a full one in the part that will be deallocated
398 
399  // 1) find next full row
400  cf3_assert(full_row_idx<m_array.size());
401  while(is_array_row_empty(full_row_idx))
402  {
403  full_row_idx++;
404  cf3_assert(full_row_idx<m_array.size());
405  }
406 
407  // 2) swap them
408  cf3_assert(empty_row_idx<m_array.size());
409  m_array[empty_row_idx] = m_array[full_row_idx];
410  full_row_idx++;
411  }
412  }
413 
414  // make m_array smaller
415  m_array.resize(new_size);
416  }
417 
418  // clear all buffers
419  reset();
420
