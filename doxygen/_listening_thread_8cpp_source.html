119 void ListeningThread::init()
120 {
121  static int count = 0;
122  m_mutex.lock();
123 
124  std::map<Communicator, ListeningInfo*>::iterator it = m_comms.begin();
125 
126  // non-blocking receive on all communicators
127  for( ; it != m_comms.end() && m_listening ; ++it )
128  {
129  ListeningInfo * info = it->second;
130 
131  if( info->ready )
132  {
133  MPI_Request request;
134  info->request = request;
135 
136  MPI_Irecv(info->data, ListeningInfo::buffer_size(), MPI_CHAR,
137  MPI_ANY_SOURCE, 0, it->first, &info->request);
138 
139  info->ready = false;
140  }

142 
143  m_mutex.unlock();

160 void ListeningThread::check_for_data()
161 {
162  m_mutex.lock();
163 
164  ExceptionManager::instance().ExceptionDumps = false;
165 
166  std::map<Communicator, ListeningInfo*>::iterator it = m_comms.begin();
167 
168  for( ; it != m_comms.end() && m_listening ; ++it )
169  {
170  ListeningInfo * info = it->second;
171 
172  // if the communicator is waiting for data
173  if( !info->ready )
174  {
175  int flag;
176 
177  MPI_Test(&info->request, &flag, MPI_STATUS_IGNORE);
178 
179  // if data arrived, flag is not zero
180  if( flag != 0 )
181  {
182  try
183  {
184  boost::shared_ptr<XmlDoc> doc = XML::parse_cstring( info->data );
185 
186  new_signal( it->first, doc );
187 
188  info->ready = true; // ready to do another non-blocking receive
189  }
190  catch(XmlError & e)
191  {
192  CFerror << e.what() << CFendl;
193  m_listening = false;
194  }
195  }
196  }
197  }
198 
199  m_mutex.unlock();
200 }
