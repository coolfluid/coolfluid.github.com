73  Uint write_data_block(const char* data, const std::streamsize count, const std::string& list_name, const Uint nb_rows, const Uint nb_cols, const std::string& type_name)
74  {
75  cf3_assert(out_file.is_open());

77  // Prefix and suffix markers
78  static const std::string block_prefix("__CFDATA_BEGIN");
79 
80  const Uint block_begin = out_file.tellp();
81 
82  // Write the prefix
83  out_file.write(block_prefix.c_str(), block_prefix.size());
84 
85  if(count != 0)
86  {
87  // Build a compressed stream
88  boost::iostreams::filtering_ostream compressing_stream;
89  compressing_stream.push(boost::iostreams::zlib_compressor());
90  compressing_stream.push(out_file);
91 
92  // Write the data
93  compressing_stream.write(data, count);
94  compressing_stream.pop();
95  }
96 
97  const Uint block_end = out_file.tellp();
98 
99  // Data describing the block on the current CPU
100  const std::vector<Uint> my_block_info = boost::assign::list_of(nb_rows)(nb_cols)(block_begin)(block_end);
101  const Uint block_info_size = my_block_info.size();
102  std::vector<Uint> global_block_info;
103  const Uint root = 0;
104  if(comm.is_active())
105  {
106  comm.gather(my_block_info, global_block_info, root);
107  }
108  else
109  {
110  global_block_info = my_block_info;
111  }
112 
113  // Write out the XML data
114  if(comm.rank() == root)
115  {
116  const Uint nb_procs = comm.size();
117  for(Uint i = 0; i != nb_procs; ++i)
118  {
119  XmlNode block_xml = node_xml_data[i].add_node("block");
120  const Uint j = i*block_info_size;
121  block_xml.set_attribute("name", list_name);
122  block_xml.set_attribute("index", to_str(index));
123  block_xml.set_attribute("type_name", type_name);
124  block_xml.set_attribute("nb_rows", to_str(global_block_info[j]));
125  block_xml.set_attribute("nb_cols", to_str(global_block_info[j+1]));
126  block_xml.set_attribute("begin", to_str(global_block_info[j+2]));
127  block_xml.set_attribute("end", to_str(global_block_info[j+3]));
128  }
129  }
130

133
