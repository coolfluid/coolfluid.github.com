45 // Get the used nodes list of a region, including any nodes that live on the current CPU
46 // but are only linked to on other CPUs
47 std::vector<Uint> used_nodes(const mesh::Region& region, const mesh::Dictionary& dict)
48 {
49  // Start with a used node list of the current CPU
50  boost::shared_ptr< common::List< Uint > > own_used_node_list = build_used_nodes_list(region, dict, false, false);
51  std::vector<Uint> used_node_vec;
52  common::PE::Comm& comm = common::PE::Comm::instance();
53 
54  if(!comm.is_active())
55  {
56  used_node_vec.assign(own_used_node_list->array().begin(), own_used_node_list->array().end());
57  return used_node_vec;
58  }
59 
60  std::vector<Uint> own_gids; own_gids.reserve(own_used_node_list->size());
61  const Uint nb_nodes = dict.size();
62  std::vector<bool> is_added(nb_nodes, false);
63  BOOST_FOREACH(const Uint own_idx, own_used_node_list->array())
64  {
65  own_gids.push_back(dict.glb_idx()[own_idx]);
66  is_added[own_idx] = true; // All local nodes are in the list automatically
67  }
68 
69  std::vector< std::vector<Uint> > recv_gids;
70  comm.all_gather(own_gids, recv_gids);
71 
72  std::set<Uint> global_boundary_gids; // GIDs that reside on other CPUs
73  const Uint nb_procs = comm.size();
74  for(Uint i = 0; i != nb_procs; ++i)
75  {
76  if(i == comm.rank())
77  continue;
78 
79  BOOST_FOREACH(const Uint gid, recv_gids[i])
80  {
81  global_boundary_gids.insert(gid);
82  }
83  }
84 
85  std::list<Uint> extra_nodes;
86  for(Uint i = 0; i != nb_nodes; ++i)
87  {
88  if(is_added[i])
89  continue;
90 
91  if(global_boundary_gids.count(dict.glb_idx()[i]))
92  {
93  is_added[i] = true;
94  extra_nodes.push_back(i);
95  }
96  }
97 
98  used_node_vec.reserve(own_used_node_list->size()+ extra_nodes.size());
99  used_node_vec.insert(used_node_vec.end(), own_used_node_list->array().begin(), own_used_node_list->array().end());
100  used_node_vec.insert(used_node_vec.end(), extra_nodes.begin(), extra_nodes.end());
101  return used_node_vec;
102 }
