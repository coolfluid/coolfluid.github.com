238 template<typename T>
239 void ListBufferT<T>::flush()
240 {
241 
242  // get total number of allocated rows
243  Uint allocated_size = total_allocated();
244  Uint old_array_size = m_array.size();
245 
246  // get total number of empty rows
247  Uint nb_emptyRows = m_empty_array_rows.size() + m_empty_buffer_rows.size() + m_new_buffer_rows.size();
248  Uint new_size = allocated_size-nb_emptyRows;
249  if (new_size >= old_array_size)
250  {
251  // make m_array bigger
252  m_array.resize(boost::extents[new_size]);
253 
254  // copy each buffer into the array
255  Uint array_idx=old_array_size;
256  boost_foreach (Buffer& buffer, m_buffers)
257  {
258  for (Uint row_idx=0; row_idx<buffer.size(); ++row_idx)
259  {
260  value_type& row = buffer.rows[row_idx];
261  if (buffer.is_not_empty[row_idx]) // for each non-empty row from all buffers
262  {
263  // first find empty rows inside the old part array
264  if (!m_empty_array_rows.empty())
265  {
266  value_type& empty_array_row = get_row(m_empty_array_rows.front());
267  m_empty_array_rows.pop_front();
268  empty_array_row = row;
269  }
270  else // then select the new array rows to be filled
271  {
272  cf3_assert(array_idx < m_array.size());
273  value_type& empty_array_row=m_array[array_idx++];
274  empty_array_row = row;
275  }
276  }
277  }
278  }
279  }
280  else // More rows to be removed than added, now we need to swap rows
281  {
282  // copy all buffer rows in the m_array
283  boost_foreach (Buffer& buffer, m_buffers)
284  {
285  for (Uint row_idx=0; row_idx<buffer.size(); ++row_idx)
286  {
287  value_type& row = buffer.rows[row_idx];
288  if (buffer.is_not_empty[row_idx]) // for each non-empty row from all buffers
289  {
290  Uint empty_array_row_idx = m_empty_array_rows.front();
291  m_empty_array_rows.pop_front();
292  value_type& empty_array_row = get_row(empty_array_row_idx);
293  empty_array_row = row;
294  }
295  }
296  }
297  Uint full_row_idx = new_size;
298  // The part of the table with rows > new_size will be deallocated
299  // The empty rows from the allocated part must be swapped with filled
300  // rows from the part that will be deallocated
301  Uint nb_empty_rows = m_empty_array_rows.size();
302  for (Uint e=0; e<nb_empty_rows; ++e)
303  {
304  Uint empty_row_idx = m_empty_array_rows[e];
305  // swap only necessary if the empty row is in the allocated part
306  if (empty_row_idx < new_size)
307  {
308  // swap this empty row with a full one in the part that will be deallocated
309 
310  // 1) find next full row
311  cf3_assert(full_row_idx<m_array.size());
312  while(is_array_row_empty(full_row_idx))
313  {
314  full_row_idx++;
315  cf3_assert(full_row_idx<m_array.size());
316  }
317 
318  // 2) swap them
319  cf3_assert(empty_row_idx<m_array.size());
320  m_array[empty_row_idx] = m_array[full_row_idx];
321  //m_empty_array_rows.push_back(full_row_idx);
322  full_row_idx++;
323  }
324  }
325 
326  // make m_array smaller
327  m_array.resize(boost::extents[new_size]);
328  }
329 
330  // clear all buffers
331  reset();
332 }

336 template<typename T>
337 inline typename ListBufferT<T>::value_type& ListBufferT<T>::get_row(const Uint idx)
338 {
339  Uint cummulative_size = m_array.size();
340  if (idx < cummulative_size)
341  {
342  return m_array[idx];
343  }
344  else
345  {
346  BOOST_FOREACH(Buffer& buffer, m_buffers)
347  {
348  if (idx<cummulative_size+buffer.size())
349  return buffer.rows[idx-cummulative_size];
350  cummulative_size += buffer.size();
351  }
352  }
353  throw common::BadValue(FromHere(),"Trying to access index that is not allocated: ["+common::to_str(idx)+">="+common::to_str(cummulative_size)+"]");
354  return m_array[0];
