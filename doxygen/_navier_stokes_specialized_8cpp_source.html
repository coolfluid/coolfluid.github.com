42  template<typename PT, typename UT, typename NUT, typename LSST>
43  void operator()(const PT& p, const UT& u, const NUT& nu_eff, const Real& rho, const Real& theta, const Real& invdt, LSST& lss, math::LSS::BlockAccumulator& acc) const
44  {
45  typedef mesh::LagrangeP1::Triag2D ElementT;
46  const RealVector2 u_avg = u.value().colwise().mean();
47  const ElementT::NodesT& nodes = u.support().nodes();
48  const Real volume = u.support().volume();
49  const Real fc = 0.5;
50  const Real nu = fabs(nu_eff.value().mean());
51 
52  Eigen::Matrix<Real, 9, 9> A, T;
53  Eigen::Matrix<Real, 9, 1> x;
54  A.setZero();
55  T.setZero();
56  acc.neighbour_indices(u.support().element_connectivity());
57 
58  // Face normals
59  ElementT::NodesT normals;
60  normals(0, XX) = nodes(1, YY) - nodes(2, YY);
61  normals(0, YY) = nodes(2, XX) - nodes(1, XX);
62  normals(1, XX) = nodes(2, YY) - nodes(0, YY);
63  normals(1, YY) = nodes(0, XX) - nodes(2, XX);
64  normals(2, XX) = nodes(0, YY) - nodes(1, YY);
65  normals(2, YY) = nodes(1, XX) - nodes(0, XX);
66 
67  Real tau_ps, tau_su, tau_bulk;
68  ComputeTauImpl compute_tau;
69  compute_tau.compute_coefficients(u, nu, 1./invdt, tau_ps, tau_su, tau_bulk);
70 
71  for(Uint i=0; i<3; ++i)
72  {
73  const Uint Ui = 3*i;
74  const Uint Vi = Ui + 1;
75  const Uint Pi = Ui + 2;
76 
77  const Real u_ni = u_avg[XX]*normals(i, XX)+u_avg[YY]*normals(i, YY);
78 
79  x[Ui] = u.value()(i, XX);
80  x[Vi] = u.value()(i, YY);
81  x[Pi] = p.value()[i];
82 
83  for(Uint j=0; j<3; ++j)
84  {
85  const Uint Uj = 3*j;
86  const Uint Vj = Uj + 1;
87  const Uint Pj = Uj + 2;
88 
89  const Real uk=u_avg[XX];
90  const Real vk=u_avg[YY];
91  const Real uknj=uk*normals(j, XX)+vk*normals(j, YY);
92 
93  // Convection (Standard + SUPG)
94  Real val = 1./6.*uknj;
95  val += tau_su/(4.*volume)*uknj*u_ni;
96  A(Ui,Uj) += val;
97  A(Vi,Vj) += val;
98 
99  // Convection (PSPG)
100  val = tau_ps/(4.*volume)*uknj;
101  A(Pi,Uj) += val*normals(i, XX);
102  A(Pi,Vj) += val*normals(i, YY);
103 
104  // Convection Skewsymm (Standard + SUPG)
105  val = fc/6.;
106  val += fc*tau_su/(4.*volume)*u_ni;
107  A(Ui,Uj) += val*uk*normals(j, XX);
108  A(Ui,Vj) += val*uk*normals(j, YY);
109  A(Vi,Uj) += val*vk*normals(j, XX);
110  A(Vi,Vj) += val*vk*normals(j, YY);
111 
112  // Convection Skewsymm (PSPG)
113  val = fc*tau_ps/(4.*volume);
114  A(Pi,Uj) += val*normals(i, XX)*uk*normals(j, XX);
115  A(Pi,Vj) += val*normals(i, XX)*uk*normals(j, YY);
116  A(Pi,Uj) += val*normals(i, YY)*vk*normals(j, XX);
117  A(Pi,Vj) += val*normals(i, YY)*vk*normals(j, YY);
118 
119  //difusion (Standard)
120  val = nu/(4.*volume);
121  A(Ui,Uj)+=val*(4./3.*normals(i, XX)*normals(j, XX)+ normals(i, YY)*normals(j, YY));
122  A(Ui,Vj)+=val* normals(i, XX)*normals(j, YY)/3.;
123  A(Vi,Uj)+=val* normals(i, YY)*normals(j, XX)/3.;
124  A(Vi,Vj)+=val*( normals(i, XX)*normals(j, XX)+4./3.*normals(i, YY)*normals(j, YY));
125 
126  // Pressure (Standard + SUPG)
127  val = 1./(6.*rho);
128  val += tau_su/(4.*rho*volume)*u_ni;
129  A(Ui,Pj) += normals(j, XX)*val;
130  A(Vi,Pj) += normals(j, YY)*val;
131 
132  // Pressure (PSPG)
133  const Real laplacian = 1./(4.*rho*volume)*(normals(i, XX)*normals(j, XX)+normals(i, YY)*normals(j, YY));
134  A(Pi,Pj) += tau_ps*laplacian;
135 
136  // Continuity (Standard)
137  val = 1./6.;
138  A(Pi,Uj) += val*normals(j, XX);
139  A(Pi,Vj) += val*normals(j, YY);
140 
141  // Bulk viscosity (Standard)
142  val = tau_bulk/(4.*volume);
143  A(Ui,Uj) += val*normals(i, XX)*normals(j, XX);
144  A(Ui,Vj) += val*normals(i, XX)*normals(j, YY);
145  A(Vi,Uj) += val*normals(i, YY)*normals(j, XX);
146  A(Vi,Vj) += val*normals(i, YY)*normals(j, YY);
147 
148  // Time (Standard + SUPG)
149  val = volume/12.*(1.+((i)==(j)?1.:0.));
150  val += tau_su/6.*u_ni;
151  T(Ui,Uj) += val;
152  T(Vi,Vj) += val;
153 
154  // Time (PSPG)
155  val = tau_ps/6.;
156  T(Pi,Uj) += val*normals(i, XX);
157  T(Pi,Vj) += val*normals(i, YY);
158  }
159  }
160  acc.rhs = -A*x;
161  A.row(0) /= theta;
162  A.row(3) /= theta;
163  A.row(6) /= theta;
164  acc.mat = invdt * T + theta*A;
165  lss.matrix().add_values(acc);
166  lss.rhs().add_rhs_values(acc);
167  }
168 };

172 NavierStokesSpecialized::NavierStokesSpecialized ( const std::string& name ) : LSSActionUnsteady( name )
173 {
174  using boost::proto::lit;
175 
176  // Option for the theta scheme
177  options().add("theta", 1.)
178  .pretty_name("Theta")
179  .description("Theta coefficient for the theta-method.")
180  .link_to(&theta);
181 
182  // Size of the block accumulator
183  m_block_accumulator.resize(3, 3);
184 
185  // This determines the name of the field that will be used to store the solution
186  set_solution_tag("navier_stokes_solution");
187 
188  // Create action components that wil be executed in the order they are created here:
189  // 1. Set the linear system matrix and vectors to zero
190  create_component<math::LSS::ZeroLSS>("ZeroLSS");
191 
192  // 2. Assemble the system matrix and RHS using a custom proto expression
193  FieldVariable<0, VectorField> u("Velocity", solution_tag());
194  FieldVariable<1, ScalarField> p("Pressure", solution_tag());
195  FieldVariable<2, ScalarField> nu_eff("EffectiveViscosity", "navier_stokes_viscosity");
196  PhysicsConstant rho("density");
197 
198  Handle<ProtoAction> assembly = create_component<ProtoAction>("Assembly");
199  assembly->set_expression(elements_expression(
200  boost::mpl::vector1< mesh::LagrangeP1::Triag2D>(),
201  assemble_ns_triags(p, u, nu_eff, rho, lit(theta), lit(invdt()), system_matrix, m_block_accumulator)
202  ));
203 
204 
205  // 3. Apply boundary conditions
206  Handle<BoundaryConditions> bc = create_component<BoundaryConditions>("BoundaryConditions");
207  bc->mark_basic();
208  bc->set_solution_tag(solution_tag());
209 
210  // 4. Solve the linear system
211  create_component<math::LSS::SolveLSS>("SolveLSS");
212 
213  // 5. Update the solution
214  create_component<ProtoAction>("Update")->set_expression(nodes_expression(group(p += solution(p), u += solution(u))));
215 }
