32 void array2d_test(const Table<Real>::ArrayT& a, const Table<Real>::ArrayT& b, Accumulator& result, const std::string& context)
33 {
34  const Uint size_a = a.size();
35  const Uint size_b = b.size();
36 
37  for(Uint i = 0, j = 0; i != size_a && j != size_b; ++i, ++j)
38  {
39  Accumulator acc;
40  acc.max_ulps = result.max_ulps;
41  vector_test(a[i], b[j], acc);
42  const bool exact = boost::accumulators::min(acc.exact);
43  const Real ulps = boost::accumulators::max(acc.ulps);
44  if(!exact || ulps > acc.max_ulps)
45  {
46  CFerror << "In " << context << ", row " << i << ":\n";
47  BOOST_FOREACH(const Real& val, a[i])

50  BOOST_FOREACH(const Real& val, b[i])
51  CFerror << " " << val;
52  CFerror << CFendl;
53  if(exact)
54  CFerror << " ulps: " << ulps << CFendl;
55  }
56  result.exact(exact);
57  result.ulps(ulps);
58  }
59 
60  result.exact(size_a == size_b);
