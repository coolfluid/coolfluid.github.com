203 void PythonConsole::key_press_event(QKeyEvent *e){
204  QTextCursor c=textCursor();
205  if (e->text().length() > 0 && !editable_zone(c)){
206  c.movePosition(QTextCursor::End);
207  setTextCursor(c);
208  }

213  c.removeSelectedText();
214  break;
215  }
216  case Qt::Key_Left:
217  if ((c.position() - c.block().position()) > 0 || c.blockNumber() > input_block){
218  QPlainTextEdit::keyPressEvent(e);
219  c=textCursor();
220  if (c.block().length() == 0
221  && c.block().userState()==PythonCodeContainer::PROMPT_2
222  && c.block()==document()->lastBlock()){
223  //user remove all ident of the line we execute it
224  new_line(0);
225  }
226  }
227  break;
228  case Qt::Key_Up:
229  if (c.blockNumber() == input_block && history.size()){
230  if (history_index==history.size()){
231  select_input(c);
232  tmp_command=c.selectedText();
233  tmp_command.replace(8233,'\n');
234  }
235  history_index--;
236  if (history_index < 0)
237  history_index=0;
238  select_input(c);
239  c.removeSelectedText();
240  c.insertText(history.at(history_index));
241  //centerCursor();
242  fix_prompt();
243  }else{
244  QPlainTextEdit::keyPressEvent(e);
245  }
246  break;
247  case Qt::Key_Down:
248  if (c.blockNumber() == document()->blockCount()-1 && history.size()){
249  if (history_index < history.size()){
250  history_index++;
251  select_input(c);
252  c.removeSelectedText();
253  if (history_index > history.size()-1){
254  history_index=history.size();
255  c.insertText(tmp_command);
256  }else{
257  c.insertText(history.at(history_index));
258  }
259  setTextCursor(c);
260  fix_prompt();
261  }
262  }else{
263  QPlainTextEdit::keyPressEvent(e);
264  }
265  break;
266  case Qt::Key_Home:
267  c.movePosition(QTextCursor::StartOfBlock);
268  //c.movePosition(QTextCursor::WordRight);
269  setTextCursor(c);
270  break;
271  default:
272  QPlainTextEdit::keyPressEvent(e);
273  }
274  if (blockCount() != block_count)
275  fix_prompt();
276 }

389 void PythonConsole::execute_code(QString code,bool immediate,QVector<int> break_lines){
390  static QRegExp two_points("^[^#:]*:[ ]*(#[^$]*)?$");
391  static QRegExp extend_statement("^[^#:]*(catch|elif|else|finally)[^#:]*:");
392  static QRegExp empty_line("^[ \\t]*($|#)");
393  QString command;
394  QString line;
395  QVector<int> current_break;
396  int line_number=0;
397  int block_number=0;
398  int ind=-1;
399 
400  bool multi_line=false;
401 
402  foreach (line,code.split('\n')){
403  if (ind=break_lines.indexOf(line_number) > -1){
404  current_break.push_back(line_number-block_number);
405  }
406  if (multi_line){
407  int indent=0;
408  for (int i=0;(line[i]=='\t'|| (line[i]==' ' && line[++i]==' '));indent++,i++);//match tabulation and double space
409  if (indent==0){
410  if (line.contains(extend_statement)){
411  command.append(line).append('\n');
412  }else{//end of multi line commnd
413  command_stack.enqueue(python_command(command,immediate,current_break));
414  block_number=line_number;
415  current_break.clear();
416  command.clear();
417  multi_line=false;
418  }
419  }else{
420  command.append(line).append('\n');
421  }
422  }
423  if (!multi_line){
424  if (line.contains(two_points)){
425  multi_line=true;
426  command=line.append('\n');
427  }else{//simple command
428  if (!line.contains(empty_line)){
429  command_stack.enqueue(python_command(line,immediate,current_break));
430  block_number=line_number;
431  current_break.clear();
432  }
433  }
434  }
435  line_number++;
436  }
437  if (command_stack.size()){
438  stream_next_command();
439  }
440 }

475 void PythonConsole::insert_output(QString output, int fragment){
476  output.replace("\\n","\n");
477  QTextCursor cursor=textCursor();
478  if (fragment < 0){
479  if (input_start_in_text==0){//special case
480  cursor.setPosition(0);
481  cursor.insertText("\n");
482  document()->findBlockByNumber(0).setUserState(1);
483  document()->findBlockByNumber(1).setUserState(PythonCodeContainer::PROMPT_1);
484  cursor.setPosition(0);
485  }else{
486  cursor.setPosition(input_start_in_text);
487  cursor.movePosition(QTextCursor::Left);
488  cursor.insertBlock();
489  }
490  }else{
491  int b=fragment_container[fragment];
492  QTextBlock bl;
493  if (b == 0){
494  bl=document()->findBlockByNumber(input_block);
495  }else{
496  bl=document()->findBlockByNumber(b);
497  bl=bl.next();
498  }
499  while (bl.isValid()){
500  if (bl.userState() == PROMPT_1){
501  cursor.setPosition(bl.position());
502  cursor.movePosition(QTextCursor::Left);
503  cursor.insertBlock();
504  break;
505  }
506  bl=bl.next();
507  }
508  }
509  QTextCursor insert_cursor(cursor);
510  int pos_c=cursor.position();
511  cursor.insertText(output);
512  cursor.movePosition(QTextCursor::Right);
513  input_start_in_text+=output.size()+1;
514  insert_cursor.setPosition(pos_c);
515  //cursor.setPosition(insert_cursor.position()+);
516  int block_diff=(cursor.blockNumber()-insert_cursor.blockNumber());
517  input_block+=block_diff;
518  QTextBlock block=insert_cursor.block().previous();
519  int output_line_number;
520  if (!block.isValid()){
521  output_line_number=1;
522  }else{
523  output_line_number=block.userState()+1;
524  if (output_line_number > PROMPT_1 || output_line_number == 0)
525  output_line_number=1;
526  }
527  block=insert_cursor.block();
528  while (block.blockNumber() < cursor.blockNumber()){
529  if (block.userState() < 0)
530  block.setUserState(output_line_number++);
531  block=block.next();
532  }
533  if (fragment > 0){
534  QMap<int, int>::const_iterator itt_fragment=fragment_container.begin();
535  while (itt_fragment != fragment_container.end()){
536  if (itt_fragment.key() > fragment){
537  fragment_container[itt_fragment.key()]=itt_fragment.value()+block_diff;
538  blocks_fragment.remove(itt_fragment.value());
539  }
540  ++itt_fragment;
541  }
542  itt_fragment=fragment_container.begin();
543  while (itt_fragment != fragment_container.end()){
544  if (itt_fragment.key() > fragment){
545  blocks_fragment.insert(itt_fragment.value(),itt_fragment.key());
546  }
547  ++itt_fragment;
548  }
549  }
550  ensureCursorVisible();
551 }
