133 void VelocityGradient::on_regions_set()
134 {

137  const std::string grad_tag = options().value<std::string>("gradient_tag");

140 
141  FieldVariable<0, VectorField> u(options().value<std::string>("velocity_variable"), options().value<std::string>("velocity_tag"));
142  FieldVariable<1, ScalarField> valence("Valence", "node_valence");
143  FieldVariable<2, ScalarField> div_u("div_"+grad_base, grad_tag);
144 
145  if(dim == 2)
146  {
147  FieldVariable<3, VectorField> grad_ux(grad_name+"x", grad_tag);
148  FieldVariable<4, VectorField> grad_uy(grad_name+"y", grad_tag);
149 
150  set_expression(elements_expression
151  (
152  boost::mpl::vector2<mesh::LagrangeP1::Triag2D, mesh::LagrangeP1::Quad2D>(),
153  detail::set_gradient(u, valence, grad_ux, grad_uy, div_u)
154  ));
155 
156  m_zero_fields->set_expression(nodes_expression(group(grad_ux[_i] = 0., grad_uy[_i] = 0., div_u = 0.)));
157  }
158  else if(dim == 3)
159  {
160  FieldVariable<3, VectorField> grad_ux(grad_name+"x", grad_tag);
161  FieldVariable<4, VectorField> grad_uy(grad_name+"y", grad_tag);
162  FieldVariable<5, VectorField> grad_uz(grad_name+"z", grad_tag);
163 
164  set_expression(elements_expression
165  (
166  boost::mpl::vector3<mesh::LagrangeP1::Tetra3D, mesh::LagrangeP1::Hexa3D, mesh::LagrangeP1::Prism3D>(),
167  detail::set_gradient(u, valence, grad_ux, grad_uy, grad_uz, div_u)
168  ));
169 
170  m_zero_fields->set_expression(nodes_expression(group(grad_ux[_i] = 0., grad_uy[_i] = 0., grad_uz[_i] = 0., div_u = 0.)));
171  }
172  else
173  {
174  throw common::SetupError(FromHere(), "Unsupported dimension " + common::to_str(dim) + " for VelocityGradient");
175  }
176 
177  if(is_not_null(m_node_valence))
178  {
179  setup_node_valence_regions();
180  }
181  m_zero_fields->options().set("regions", options().option("regions").value());
182 }
