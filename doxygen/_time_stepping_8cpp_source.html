156 bool TimeStepping::stop_condition()
157 {
158  Uint nb_criteria = 0;
159  bool finish = false;
160  boost_foreach(Criterion& stop_criterion, find_components<Criterion>(*this))
161  {
162  finish |= stop_criterion();
163  ++nb_criteria;
164  }
165 
166  if (options().value<bool>("time_accurate"))
167  {
168  Real time = options().value<Real>("time");
169  Real end_time = options().value<Real>("end_time");
170  if (time >= end_time)
171  return true; // stop
172  }
173 
174  Uint step = options().value<Uint>("step");
175  Uint max_steps = options().value<Uint>("max_steps");
176  if (step >= max_steps)
177  return true; // stop
178 
179  return finish;
180 }
