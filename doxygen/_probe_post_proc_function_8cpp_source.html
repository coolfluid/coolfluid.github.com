64  std::stringstream vars;

66  for (Uint var_nb=0; var_nb<m_probe->variables()->nb_vars(); ++var_nb)
67  {

69  std::string var_name = m_probe->variables()->user_variable_name(var_nb);

73  vars << mod_var_name;
