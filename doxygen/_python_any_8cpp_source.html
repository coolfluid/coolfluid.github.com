37  std::string result;

39  {
40  std::string elem_type = type_name(pylist[i]);

46  result = elem_type;

283 std::string type_name(const boost::python::api::object& python_object)
284 {
285  // Mapping between the python type name and the coolfluid type
286  typedef std::map<std::string, std::string> TypeMapT;
287  static const TypeMapT python_type_map =
288  boost::assign::map_list_of(std::string(PyBool_Type.tp_name), common::class_name<bool>())
289  (std::string(PyInt_Type.tp_name), common::class_name<int>())
290  (std::string(PyString_Type.tp_name), common::class_name<std::string>())
291  (std::string(PyFloat_Type.tp_name), common::class_name<Real>())
292  (std::string(PyList_Type.tp_name), "array") // Special indication of lists
293  (std::string("URI"), common::class_name<common::URI>()) // "URI" as passed in the boost::python::class_ definition
294  (std::string("UUCount"), common::class_name<common::UUCount>()) // Same as URI
295  (std::string("Component"), "component"); // Same as URI, but needs special treatment afterwards
296 
297  // Look up the type in the map
298  const PyTypeObject& python_type = *python_object.ptr()->ob_type;
299  const TypeMapT::const_iterator it = python_type_map.find(python_type.tp_name);
300  if(it == python_type_map.end())
301  {
302  throw common::ValueNotFound(FromHere(), std::string("Python type ") + python_type.tp_name + " has no coolfluid equivalent");
303  }
304 
305  // Treat special cases
306  if(it->second == "array")
307  {
308  const std::string element_type = python_list_element_type(static_cast<const boost::python::list&>(python_object));
309  return "array[" + element_type + "]";
310  }
311  else if(it->second == "component")
312  {
313  boost::python::extract< ComponentWrapper& > extracted_value(python_object);
314  cf3_assert(extracted_value.check());
315  return "handle[" + extracted_value().component().derived_type_name() + "]";
316  }
317 
318  return it->second;
319 }
