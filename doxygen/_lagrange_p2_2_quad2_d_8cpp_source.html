110 void Quad2D::compute_jacobian<Quad2D::JacobianT>(const MappedCoordsT& mapped_coord, const NodesT& nodes, JacobianT& result)
111 {
112  // get mapped coordinates
113 
114  const Real ksi = mapped_coord[KSI];
115  const Real eta = mapped_coord[ETA];
116 
117  const Real ksi2 = ksi*ksi;
118  const Real eta2 = eta*eta;
119  const Real ksi_eta = ksi*eta;
120  const Real ksi2_eta = ksi2*eta;
121  const Real ksi_eta2 = ksi*eta2;
122 
123  // set shape function derivatives
124  m_shapeFuncDerivs(0,KSI) = 0.25 * (eta - 2.*ksi_eta - eta2 + 2.*ksi_eta2);
125  m_shapeFuncDerivs(1,KSI) = -0.25 * (eta + 2.*ksi_eta - eta2 - 2.*ksi_eta2);
126  m_shapeFuncDerivs(2,KSI) = 0.25 * (eta + 2.*ksi_eta + eta2 + 2.*ksi_eta2);
127  m_shapeFuncDerivs(3,KSI) = -0.25 * (eta - 2.*ksi_eta + eta2 - 2.*ksi_eta2);
128  m_shapeFuncDerivs(4,KSI) = -0.5 * (-2.*ksi_eta + 2.*ksi_eta2);
129  m_shapeFuncDerivs(5,KSI) = 0.5 * (1. - eta2 + 2.*ksi - 2.*ksi_eta2);
130  m_shapeFuncDerivs(6,KSI) = 0.5 * (-2.*ksi_eta - 2.*ksi_eta2);
131  m_shapeFuncDerivs(7,KSI) = -0.5 * (1. - eta2 - 2.*ksi + 2.*ksi_eta2);
132  m_shapeFuncDerivs(8,KSI) = 2.*ksi_eta2 - 2.*ksi;
133 
134  m_shapeFuncDerivs(0,ETA) = 0.25 * (ksi - ksi2 - 2.*ksi_eta + 2.*ksi2_eta);
135  m_shapeFuncDerivs(1,ETA) = -0.25 * (ksi + ksi2 - 2.*ksi_eta - 2.*ksi2_eta);
136  m_shapeFuncDerivs(2,ETA) = 0.25 * (ksi + ksi2 + 2.*ksi_eta + 2.*ksi2_eta);
137  m_shapeFuncDerivs(3,ETA) = -0.25 * (ksi - ksi2 + 2.*ksi_eta - 2.*ksi2_eta);
138  m_shapeFuncDerivs(4,ETA) = -0.5 * (1. - ksi2 - 2.*eta + 2.*ksi2_eta);
139  m_shapeFuncDerivs(5,ETA) = 0.5 * (-2.*ksi_eta - 2.*ksi2_eta);
140  m_shapeFuncDerivs(6,ETA) = 0.5 * (1. - ksi2 + 2.*eta - 2.*ksi2_eta);
141  m_shapeFuncDerivs(7,ETA) = -0.5 * (-2.*ksi_eta + 2.*ksi2_eta);
142  m_shapeFuncDerivs(8,ETA) = 2.*ksi2_eta - 2.*eta;
143 
144  // evaluate Jacobian
145  result.setZero();
146  for (Uint n = 0; n < 9; ++n)
147  {
148  result(KSI,XX) += m_shapeFuncDerivs(n,KSI)*nodes(n,XX);
149  result(ETA,XX) += m_shapeFuncDerivs(n,ETA)*nodes(n,XX);
150 
151  result(KSI,YY) += m_shapeFuncDerivs(n,KSI)*nodes(n,YY);
152  result(ETA,YY) += m_shapeFuncDerivs(n,ETA)*nodes(n,YY);
153  }
154 }

178 void Quad2D::compute_plane_jacobian_normal(const MappedCoordsT& mapped_coord, const NodesT& nodes, const CoordRef orientation, CoordsT& result)
179 {
180  // get mapped coordinates
181  const Real ksi = mapped_coord[KSI];
182  const Real eta = mapped_coord[ETA];
183  const Real ksi2 = ksi*ksi;
184  const Real eta2 = eta*eta;
185  const Real ksi_eta = ksi*eta;
186 
187  if (orientation == 0)
188  {
189  const Real ksi2_eta = ksi2*eta;
190 
192  m_shapeFunc[0] = (ksi - ksi2 - 2.*(ksi_eta - ksi2_eta));
193  m_shapeFunc[1] = -(ksi + ksi2 - 2.*(ksi_eta + ksi2_eta));
194  m_shapeFunc[2] = (ksi + ksi2 + 2.*(ksi_eta + ksi2_eta));
195  m_shapeFunc[3] = -(ksi - ksi2 + 2.*(ksi_eta - ksi2_eta));
196  m_shapeFunc[4] = -2. * (1. - ksi2 - 2.*(eta - ksi2_eta));
197  m_shapeFunc[5] = 4. * (-ksi_eta - ksi2_eta);
198  m_shapeFunc[6] = 2. * (1. - ksi2 + 2.*(eta - ksi2_eta));
199  m_shapeFunc[7] = -4. * (-ksi_eta + ksi2_eta);
200  m_shapeFunc[8] = 8. * (ksi2_eta - eta);
201 
202  result[XX] = +nodes(0,YY)*m_shapeFunc[0];
203  result[YY] = -nodes(0,XX)*m_shapeFunc[0];
204  for (Uint n = 1; n < 9; ++n)
205  {
206  result[XX] += nodes(n,YY)*m_shapeFunc[n];
207  result[YY] -= nodes(n,XX)*m_shapeFunc[n];
208  }
209  }
210  else
211  {
212  const Real ksi_eta2 = ksi*eta2;
213 
215  m_shapeFunc[0] = (eta - eta2 - 2.*(ksi_eta - ksi_eta2));
216  m_shapeFunc[1] = -(eta - eta2 + 2.*(ksi_eta - ksi_eta2));
217  m_shapeFunc[2] = (eta + eta2 + 2.*(ksi_eta + ksi_eta2));
218  m_shapeFunc[3] = -(eta + eta2 - 2.*(ksi_eta + ksi_eta2));
219  m_shapeFunc[4] = -4. * (-ksi_eta + ksi_eta2);
220  m_shapeFunc[5] = 2. * (1. - eta2 + 2.*(ksi - ksi_eta2));
221  m_shapeFunc[6] = 4. * (-ksi_eta - ksi_eta2);
222  m_shapeFunc[7] = -2. * (1. - eta2 - 2.*(ksi - ksi_eta2));
223  m_shapeFunc[8] = 8. * (ksi_eta2 - ksi);
224 
225  result[XX] = -nodes(0,YY)*m_shapeFunc[0];
226  result[YY] = +nodes(0,XX)*m_shapeFunc[0];
227  for (Uint n = 1; n < 9; ++n)
228  {
229  result[XX] -= nodes(n,YY)*m_shapeFunc[n];
230  result[YY] += nodes(n,XX)*m_shapeFunc[n];
231  }
232  }
233  result *= 0.25;
234 }

246 bool Quad2D::is_coord_in_element(const CoordsT& coord, const NodesT& nodes)
247 {
248 
249  static const Real tolerance = 1e-6;
250 
251  if (coord[XX] > nodes.col(XX).maxCoeff()+tolerance)
252  return false;
253  if (coord[XX] < nodes.col(XX).minCoeff()-tolerance)
254  return false;
255  if (coord[YY] > nodes.col(YY).maxCoeff()+tolerance)
256  return false;
257  if (coord[YY] < nodes.col(YY).minCoeff()-tolerance)
258  return false;
259 
260 
261  m_D <<
262  nodes.col(XX).maxCoeff()-nodes.col(XX).minCoeff(),
263  nodes.col(YY).maxCoeff()-nodes.col(YY).minCoeff();
264  m_scale = 1./m_D.minCoeff();
265 
266  if (scp(nodes.row(0),nodes.row(7),coord,m_scale) * scp(nodes.row(0),coord,nodes.row(4),m_scale) < -tolerance)
267  return false;
268  if (scp(nodes.row(4),nodes.row(0),coord,m_scale) * scp(nodes.row(4),coord,nodes.row(1),m_scale) < -tolerance)
269  return false;
270  if (scp(nodes.row(1),nodes.row(4),coord,m_scale) * scp(nodes.row(1),coord,nodes.row(5),m_scale) < -tolerance)
271  return false;
272  if (scp(nodes.row(5),nodes.row(1),coord,m_scale) * scp(nodes.row(5),coord,nodes.row(2),m_scale) < -tolerance)
273  return false;
274  if (scp(nodes.row(2),nodes.row(5),coord,m_scale) * scp(nodes.row(2),coord,nodes.row(6),m_scale) < -tolerance)
275  return false;
276  if (scp(nodes.row(6),nodes.row(2),coord,m_scale) * scp(nodes.row(6),coord,nodes.row(3),m_scale) < -tolerance)
277  return false;
278  if (scp(nodes.row(3),nodes.row(6),coord,m_scale) * scp(nodes.row(3),coord,nodes.row(7),m_scale) < -tolerance)
279  return false;
280  if (scp(nodes.row(7),nodes.row(3),coord,m_scale) * scp(nodes.row(7),coord,nodes.row(0),m_scale) < -tolerance)
281  return false;
282 
283  return true;
284 }
