62 {
63  #define BUFFER_SIZE 500
64 
65  static int i = 0;
66  ++i;
67 
68 
69  std::ostringstream oss;
70  int j, nptrs;
71  void *buffer[BUFFER_SIZE];
72  char **strings;
73 
74  //printf ("dumping %d backtrace ...\n", i);
75  oss << "dumping " << i << " backtrace ...\n";

77  oss << "\nbacktrace() returned " << nptrs << " addresses\n";

80  if (strings == NULL)

85  boost::regex e("([0-9]+)[[:space:]]+(.+)[[:space:]]+(.+)[[:space:]]+(.+)[[:space:]]+\\+[[:space:]]+(.+)");
86  boost::match_results<std::string::const_iterator> what;
87 
88  // iterate over the returned symbol lines. skip the first, it is the
89  // address of this function.
90  for (j = 1; j < nptrs; j++)
91  {
92  std::string trace(strings[j]);
93 
94  if (boost::regex_search(trace,what,e))
95  {
96  trace = std::string(what[4].first,what[4].second);
97  size_t maxName = 1024;
98  int demangleStatus; // will be assigned in abi::__cxa_demangle
99  char* demangledName; // will be allocated in abi::__cxa_demangle

101  &demangleStatus)) && demangleStatus == 0)
102  {
103  trace = demangledName; // the demangled name is now in our trace string
104  }
105  delete_ptr_array(demangledName);
106  }
107  oss << trace << "\n";
108  }
109 #else
110  for (j = 0; j < nptrs; j++)
111  oss << strings[j] << "\n";
112 #endif
113 
114  free(strings);
115  #undef BUFFER_SIZE
116 
117  oss << "\nexit dumping backtrace ...";

193 static int
194 feenableexcept (unsigned int excepts)
195 {
196  static fenv_t fenv;
197  unsigned int new_excepts = excepts & FE_ALL_EXCEPT,
198  old_excepts; // previous masks
199 
200  if ( fegetenv (&fenv) ) return -1;
201  old_excepts = fenv.__control & FE_ALL_EXCEPT;
202 
203  // unmask
204  fenv.__control &= ~new_excepts;
205  fenv.__mxcsr &= ~(new_excepts << 7);
206 
207  return ( fesetenv (&fenv) ? -1 : (int)old_excepts );
208 }
