165 void LSSAction::on_regions_set()
166 {
167  if(m_implementation->m_updating) // avoid recursion
168  {
169  return;
170  }
171 
172  if(m_loop_regions.empty())
173  {
174  return;
175  }
176 
177  Handle<mesh::Mesh> mesh = find_parent_component_ptr<mesh::Mesh>(*m_loop_regions.front());
178  cf3_assert(is_not_null(mesh));
179 
180  if(options().value<bool>("use_geometry_space"))
181  {
182  m_dictionary = mesh->geometry_fields().handle<mesh::Dictionary>();
183  }
184  else
185  {
186  Handle<mesh::Field> field = find_component_ptr_recursively_with_tag<mesh::Field>(*mesh, solution_tag());
187  if(is_null(field))
188  {
189  CFdebug << "Skipping LSS creation on " << uri().path() << " because the non-geometry space field with tag " << solution_tag() << " does not exist yet" << CFendl;
190  return;
191  }
192  m_dictionary = field->dict().handle<mesh::Dictionary>();
193  }
194 
195  cf3_always_assert(is_not_null(m_dictionary));
196 
197  m_implementation->m_updating = true;
198 
199  m_implementation->m_lss = options().value< Handle<LSS::System> >("lss");
200  if(is_null(m_implementation->m_lss))
201  {
202  create_lss();
203  }
204 
205  // Create the LSS if the mesh is set
206  if(!m_implementation->m_lss->is_created())
207  {
208  VariablesDescriptor& descriptor = find_component_with_tag<VariablesDescriptor>(physical_model().variable_manager(), solution_tag());
209 
210  CFdebug << "Creating LSS for " << uri().path() << " using dictionary " << m_dictionary->uri().path() << CFendl;
211 
212  Handle< List<Uint> > gids = m_implementation->m_lss->create_component< List<Uint> >("GIDs");
213  Handle< List<Uint> > ranks = m_implementation->m_lss->create_component< List<Uint> >("Ranks");
214  Handle< List<int> > used_node_map = m_implementation->m_lss->create_component< List<int> >("used_node_map");
215 
216  std::vector<Uint> node_connectivity, starting_indices;
217  boost::shared_ptr< List<Uint> > used_nodes = build_sparsity(m_loop_regions, *m_dictionary, node_connectivity, starting_indices, *gids, *ranks, *used_node_map);
218  if(is_not_null(get_child(used_nodes->name())))
219  remove_component(used_nodes->name());
220  add_component(used_nodes);
221 
222  // This comm pattern is valid only over the used nodes for the supplied regions
223  if(is_not_null(get_child("CommPattern")))
224  remove_component("CommPattern");
225  PE::CommPattern& comm_pattern = *create_component<PE::CommPattern>("CommPattern");
226  comm_pattern.insert("gid",gids->array(),false);
227  comm_pattern.setup(Handle<PE::CommWrapper>(comm_pattern.get_child("gid")),ranks->array());
228 
229  if(is_not_null(m_dictionary->get_child("node_gids")))
230  {
231  Field& node_gids = *(Handle<Field>(m_dictionary->get_child("node_gids")));
232  const Uint nb_nodes = node_gids.size();
233 
234  for(Uint i = 0; i != nb_nodes; ++i)
235  {
236  node_gids[i][0] = -1.;
237  }
238  const Uint nb_used = used_nodes->size();
239  for(Uint i = 0; i != nb_used; ++i)
240  {
241  node_gids[used_nodes->array()[i]][0] = gids->array()[i];
242  }
243  }
244 
245  // Build node periodicity based on the used nodes, if needed
246  std::vector<Uint> periodic_links_nodes_vec;
247  std::vector<bool> periodic_links_active_vec;
248 
249  Handle< List<Uint> > periodic_links_nodes_h(m_dictionary->get_child("periodic_links_nodes"));
250  Handle< List<bool> > periodic_links_active_h(m_dictionary->get_child("periodic_links_active"));
251  if(is_not_null(periodic_links_nodes_h))
252  {
253  const List<Uint>& periodic_links_nodes = *periodic_links_nodes_h;
254  const List<bool>& periodic_links_active = *periodic_links_active_h;
255  const List<Uint>& used_nodes_list = *used_nodes;
256  const Uint nb_used_nodes = used_nodes_list.size();
257  periodic_links_active_vec.resize(nb_used_nodes, false);
258  periodic_links_nodes_vec.resize(nb_used_nodes);
259  for(Uint i = 0; i != nb_used_nodes; ++i)
260  {
261  if(periodic_links_active[used_nodes_list[i]])
262  {
263  periodic_links_active_vec[i] = true;
264  periodic_links_nodes_vec[i] = (*used_node_map)[periodic_links_nodes[used_nodes_list[i]]];
265  }
266  }
267  }
268 
269  do_create_lss(comm_pattern, descriptor, node_connectivity, starting_indices, periodic_links_nodes_vec, periodic_links_active_vec);
270  cf3_always_assert(m_implementation->m_lss->is_created());
271  Handle<math::LSS::SolutionStrategy> solution_strategy = m_implementation->m_lss->solution_strategy();
272  cf3_assert(is_not_null(solution_strategy));
273  // If the solution takes a coordinate list, then generate the list and pass it along
274  solution_strategy->set_coordinates(comm_pattern, m_dictionary->coordinates(), *used_nodes, periodic_links_active_vec);
275 
276  CFdebug << "Finished creating LSS" << CFendl;
277  configure_option_recursively(solver::Tags::regions(), options().option(solver::Tags::regions()).value());
278  configure_option_recursively("lss", m_implementation->m_lss);
279  // Also do links to BCs
280  BOOST_FOREACH(common::Link& link, common::find_components_recursively<common::Link>(*this))
281  {
282  if(link.is_linked() && link.follow()->options().check("lss"))
283  {
284  link.follow()->configure_option_recursively("lss", m_implementation->m_lss);
285  }
286  }
287  }
288  else
289  {
290  CFdebug << "Skipping on_regions_set because LSS is already created" << CFendl;
291  }
292 
293  // Update the regions of any owned initial conditions
294  BOOST_FOREACH(const Handle<Component>& ic, m_created_initial_conditions)
295  {
296  if(is_not_null(ic))
297  ic->options().set(solver::Tags::regions(), options().option(solver::Tags::regions()).value());
298  }
299 
300  cf3_assert(is_not_null(m_implementation->m_lss));
301 
302  m_implementation->m_updating = false;
303 }

317 void LSSAction::trigger_initial_conditions()
318 {
319  if(is_null(m_initial_conditions))
320  {
321  CFwarn << "Initial conditions for " << uri().path() << " were reset to NULL" << CFendl;
322  return;
323  }
324 
325  CFdebug << "Using initial conditions " << m_initial_conditions->uri().path() << " for LSSAction " << uri().path() << CFendl;
326 
327  m_created_initial_conditions.clear();
328 
329  std::set< Handle<Component> > existing_conditions;
330  BOOST_FOREACH(common::Action& ic, find_components<common::Action>(*m_initial_conditions))
331  {
332  existing_conditions.insert(ic.handle());
333  }
334 
335  // Give the concrete solver a chance to add its initial conditions
336  on_initial_conditions_set(*m_initial_conditions);
337 
338  // New initial conditions are the ones we need to track
339  BOOST_FOREACH(common::Action& ic, find_components<common::Action>(*m_initial_conditions))
340  {
341  if(!existing_conditions.count(ic.handle()))
342  {
343  m_created_initial_conditions.push_back(ic.handle());
344  CFdebug << " added initial condition " << ic.name() << CFendl;
345  }
346  }

374  else
375  m_implementation->m_lss->create(cp, vars.size(), node_connectivity, starting_indices, periodic_links_nodes, periodic_links_active);
