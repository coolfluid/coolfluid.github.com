85 void Hexa3D::compute_mapped_coordinate(const CoordsT& coord, const NodesT& nodes, MappedCoordsT& mapped_coord)
86 {
87  // Axes of the local coordinate system, centered around the centroid and going through the center of each face
88  SF::ValueT sf;
89  SF::compute_value(CoordsT(1.,0.,0.), sf);
90  CoordsT ux = (sf*nodes).transpose();
91  SF::compute_value(CoordsT(0.,1.,0.), sf);
92  CoordsT uy = (sf*nodes).transpose();
93  SF::compute_value(CoordsT(0.,0.,1.), sf);
94  CoordsT uz = (sf*nodes).transpose();
95 
96  SF::compute_value(CoordsT(-1.,0.,0.), sf);
97  CoordsT ux_neg = (sf*nodes).transpose();
98  SF::compute_value(CoordsT(0.,-1.,0.), sf);
99  CoordsT uy_neg = (sf*nodes).transpose();
100  SF::compute_value(CoordsT(0.,0.,-1.), sf);
101  CoordsT uz_neg = (sf*nodes).transpose();
102 
103  CoordsT centroid;
104  centroid[XX] = (ux[XX] + ux_neg[XX]) * 0.5;
105  centroid[YY] = (uy[YY] + uy_neg[YY]) * 0.5;
106  centroid[ZZ] = (uz[ZZ] + uz_neg[ZZ]) * 0.5;
107 
108  ux -= ux_neg;
109  uy -= uy_neg;
110  uz -= uz_neg;
111 
112  ux *= 0.5; // because the origin is at the center
113  uy *= 0.5;
114  uz *= 0.5;
115 
116  const Real ux_len_inv = 1. / ux.norm();
117  const Real uy_len_inv = 1. / uy.norm();
118  const Real uz_len_inv = 1. / uz.norm();
119 
120  ux *= ux_len_inv;
121  uy *= uy_len_inv;
122  uz *= uz_len_inv;
123 
124  // Normal vectors
125  CoordsT nyz = uy.cross(uz);
126  CoordsT nxz = ux.cross(uz);
127  CoordsT nxy = ux.cross(uy);
128 
129  // division factors for line-plane intersection
130  const Real fx = ux_len_inv / ux.dot(nyz);
131  const Real fy = uy_len_inv / uy.dot(nxz);
132  const Real fz = uz_len_inv / uz.dot(nxy);
133 
134  CoordsT diff = coord-centroid;
135  CoordsT test;
136  const Real threshold = 1e-24; // 1e-12 squared, because we compare the squared distance
137  Uint nb_iters = 0;
138  // Initial guess will be correct if our element is a parallelepiped
139  mapped_coord[KSI] = diff.dot(nyz) * fx;
140  mapped_coord[ETA] = diff.dot(nxz) * fy;
141  mapped_coord[ZTA] = diff.dot(nxy) * fz;
142  while (nb_iters < 100 && diff.dot(diff) > threshold)
143  {
144  SF::compute_value(mapped_coord, sf);
145  test = (sf*nodes).transpose();
146  diff = coord - test;
147  test[XX] = diff.dot(nyz) * fx; // Transform difference to the relative coordinate system and
148  test[YY] = diff.dot(nxz) * fy; // use it to adjust our initial guess
149  test[ZZ] = diff.dot(nxy) * fz;
150  mapped_coord += test;
151  ++nb_iters;
152  }
153 
154  if(nb_iters > 100)
155  throw common::FailedToConverge(FromHere(), "Failed to find Hexa3DLagrangeP1 mapped coordinates");
156 }

338 bool Hexa3D::is_orientation_inside(const CoordsT& coord, const NodesT& nodes, const Uint face)
339 {
340  static const Real tolerance = 1e-11;
341 
342  //test according to http://graphics.ethz.ch/~peikert/personal/HexCellTest/
343 
344  const Uint a = faces().nodes_range(face)[3];
345  const Uint b = faces().nodes_range(face)[2];
346  const Uint c = faces().nodes_range(face)[1];
347  const Uint d = faces().nodes_range(face)[0];
348 
349  RealMatrix3 M;
350  M.col(0) = nodes.row(b) - nodes.row(a);
351  M.col(1) = nodes.row(d) - nodes.row(a);
352  M.col(2) = nodes.row(c) - nodes.row(a);
353  const CoordsT pp = coord.transpose() - nodes.row(a);
354 
355  const CoordsT bp_x_dp = M.col(0).cross(M.col(1));
356  const Real h = bp_x_dp.dot(M.col(2));
357  if (std::abs(h) > tolerance)
358  {
359  RealMatrix3 T;
360  T << 1, 0, 1,
361  0, 1, 1,
362  0, 0, h;
363 
364  // Do transformation
365  RealVector3 ppp = T*M.inverse()*pp;
366 
367  if (h*ppp[XX]*ppp[YY] - ppp[ZZ] > tolerance)
368  {
369  return false;
370  }
371  }
372  else
373  {
374  if (bp_x_dp.dot(pp) < -tolerance)
375  {
376  return false;
377  }
378  }
379  return true;
380 }

395 void Hexa3D::compute_plane_jacobian_normal(const MappedCoordsT& mapped_coord, const NodesT& nodes, const CoordRef orientation, CoordsT& result)
396 {

398  const Real eta = mapped_coord[ETA];
399  const Real zta = mapped_coord[ZTA];
400 
401  const Real a1 = 1. + xi;
402  const Real a2 = 1. - xi;
403  const Real b1 = 1. + eta;
404  const Real b2 = 1. - eta;
405  const Real c1 = 1. + zta;
406  const Real c2 = 1. - zta;
407 
408  switch (orientation)
409  {
410  case KSI:
411 
412  m_shapeFuncDerivs(0,ETA) = -a2*c2;
413  m_shapeFuncDerivs(1,ETA) = -a1*c2;
414  m_shapeFuncDerivs(2,ETA) = a1*c2;
415  m_shapeFuncDerivs(3,ETA) = a2*c2;
416  m_shapeFuncDerivs(4,ETA) = -a2*c1;
417  m_shapeFuncDerivs(5,ETA) = -a1*c1;
418  m_shapeFuncDerivs(6,ETA) = a1*c1;
419  m_shapeFuncDerivs(7,ETA) = a2*c1;
420 
421  m_shapeFuncDerivs(0,ZTA) = -a2*b2;
422  m_shapeFuncDerivs(1,ZTA) = -a1*b2;
423  m_shapeFuncDerivs(2,ZTA) = -a1*b1;
424  m_shapeFuncDerivs(3,ZTA) = -a2*b1;
425  m_shapeFuncDerivs(4,ZTA) = b2*a2;
426  m_shapeFuncDerivs(5,ZTA) = b2*a1;
427  m_shapeFuncDerivs(6,ZTA) = b1*a1;
428  m_shapeFuncDerivs(7,ZTA) = b1*a2;
429 
430  m_vec1 = m_shapeFuncDerivs(0,ETA)*(nodes.row(0));
431  m_vec2 = m_shapeFuncDerivs(0,ZTA)*(nodes.row(0));
432  for (Uint in = 1; in < 8; ++in)
433  {
434  m_vec1 += m_shapeFuncDerivs(in,ETA)*(nodes.row(in));
435  m_vec2 += m_shapeFuncDerivs(in,ZTA)*(nodes.row(in));
436  }
437  break;
438
