135  void solve()
136  {
137  if(is_null(m_matrix))
138  throw common::SetupError(FromHere(), "Null matrix for " + m_self.uri().path());
139 
140  if(is_null(m_rhs))
141  throw common::SetupError(FromHere(), "Null RHS for " + m_self.uri().path());
142 
143  if(is_null(m_solution))
144  throw common::SetupError(FromHere(), "Null solution vector for " + m_self.uri().path());
145 
146  if(m_lows.is_null())
147  {
148  if(m_self.options().option("print_settings").value<bool>())
149  m_parameter_list->print();
150 
151  m_lows = m_lows_factory->createOp();
152  }
153 
154 
155  if(m_iteration_count % m_preconditioner_reset == 0)
156  {
157  Thyra::initializeOp(*m_lows_factory, m_matrix->thyra_operator(), m_lows.ptr());
158  }
159  else
160  {
161  Thyra::initializeAndReuseOp(*m_lows_factory, m_matrix->thyra_operator(), m_lows.ptr());
162  }
163 
164  Teuchos::RCP< Thyra::VectorBase<Real> const > b = m_rhs->thyra_vector();
165  Teuchos::RCP< Thyra::VectorBase<Real> > x = m_solution->thyra_vector();
166 
167  try
168  {
169  Thyra::SolveStatus<double> status = Thyra::solve<double>(*m_lows, Thyra::NOTRANS, *b, x.ptr());
170  CFinfo << "Thyra::solve finished with status " << status.message << CFendl;
171  }
172  catch(std::exception& e)
173  {
174  std::cout << e.what() << std::endl;
175  }
176 
177  if(m_self.options().option("compute_residual").value<bool>())
178  CFinfo << "Solver residual: " << compute_residual() << CFendl;
179 
180  ++m_iteration_count;
181  }

183  Real compute_residual()
184  {
185  if(is_null(m_matrix))
186  throw common::SetupError(FromHere(), "Null matrix for " + m_self.uri().path());
187 
188  if(is_null(m_rhs))
189  throw common::SetupError(FromHere(), "Null RHS for " + m_self.uri().path());
190 
191  if(is_null(m_solution))
192  throw common::SetupError(FromHere(), "Null solution vector for " + m_self.uri().path());
193 
194  if(m_lows.is_null())
195  throw common::SetupError(FromHere(), "Null linear operator for " + m_self.uri().path());
196 
197  if(m_residual_vec.is_null())
198  {
199  m_residual_vec = m_rhs->thyra_vector()->clone_v();
200  }
201 
202  Thyra::assign(m_rhs->thyra_vector().ptr(), *m_residual_vec);
203  m_lows->apply(Thyra::NOTRANS, *m_solution->thyra_vector(), m_residual_vec.ptr(), 1., -1.);
204  std::vector<Real> residuals(m_residual_vec->domain()->dim());
205  Thyra::norms_2(*m_residual_vec, Teuchos::arrayViewFromVector(residuals));
206  return *std::max_element(residuals.begin(), residuals.end());
207  }
