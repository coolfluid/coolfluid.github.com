110 void SelectPathDialog::path_changed(const QString & path)
111 {
112  int lastSlash = path.lastIndexOf(URI::separator().c_str());
113  QString newPath;
114  QStringList list;
115  Handle< CNode > node;
116 
117  if(m_node_clicked)
118  m_node_clicked = false;
119  else
120  {
121  Handle< Component > root = NTree::global()->tree_root();
122  try
123  {
124  if(Handle<CNode>(root->access_component(path.toStdString())) != nullptr)
125  newPath = path;
126  else
127  newPath = path.left(lastSlash);
128 
129  if(newPath == "/")
130  newPath = root->uri().path().c_str();
131 
132  node = Handle<CNode>(root->access_component(newPath.toStdString()));
133 
134  if(node.get() != nullptr)
135  {
136  node->list_child_paths(list, false);
137 
138  m_model->setStringList(list);
139 
140  m_tree_view->collapseAll();
141 
142  m_tree_view->select_item(newPath.toStdString());
143  }
144  }
