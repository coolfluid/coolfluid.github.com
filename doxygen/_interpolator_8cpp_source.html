63 {
64  if (send_to_pid == PE::Comm::instance().rank() &&

67  receive = send;

216 void Interpolator::stored_interpolation(const Field& source_field, Table<Real>& target)
217 {
218  // Do interpolation on processors that can do the interpolation,
219  // and send back an array of interpolated values
220  for (Uint pid=0; pid<PE::Comm::instance().size(); pid++)
221  {
222  // Trade my requests with processor send and recv
223  const Uint pid_send_interpolated = (PE::Comm::instance().size() + PE::Comm::instance().rank() - pid) %
224  PE::Comm::instance().size();
225  const Uint pid_recv_interpolated = (PE::Comm::instance().rank() + pid) %
226  PE::Comm::instance().size();
227 
228  // number of points to be interpolated
229  const Uint nb_points = m_stored_element[pid_send_interpolated].size();
230 
231  // number of variables for each point to be interpolated
232  const Uint nb_vars = m_source_vars.size();
233 
234  // storage for interpolated variables, which will be sent to the pid that reqests it (pid_recv_interpolated)
235  std::vector<Real> interpolated; interpolated.reserve(nb_points*nb_vars);
236 
237  // Interpolation points and weights
238  const std::vector< std::vector<Uint> >& s_points = m_stored_source_field_points[pid_send_interpolated];
239  const std::vector< std::vector<Real> >& s_weights = m_stored_source_field_weights[pid_send_interpolated];
240 
241  // Do interpolation
242  for (Uint t=0; t<nb_points; ++t)
243  {
244  for (Uint v=0; v<nb_vars; ++v)
245  {
246  interpolated.push_back(0.);
247  for (Uint s=0; s<s_points[t].size(); ++s)
248  {
249  cf3_assert(s_points[t][s]<source_field.size());
250  interpolated.back() += source_field[ s_points[t][s] ][ m_source_vars[v] ] * s_weights[t][s];
251  }
252  }
253  }
254 
255  // Send/Receive interpolated variables
256  std::vector<Real> recv_interpolated;
257  Interpolator_send_receive(pid_send_interpolated , interpolated,
258  pid_recv_interpolated , recv_interpolated);
259 
260  // Fill the target_field with received interpolated variables from requested processor
261  Uint it=0;
262  boost_foreach( const Uint t, m_expect_recv[pid_recv_interpolated] )
263  {
264  for (Uint v=0; v<nb_vars; ++v)
265  {
266  cf3_assert(t<target.size());
267  target[t][ m_target_vars[v] ] = recv_interpolated[it++];
268  }
269  }
270  }

283

286 
287  std::vector<int> not_found; not_found.reserve(nb_coords);
288  for (Uint i=0; i<nb_coords; ++i)
289  not_found.push_back(i);
290 
291  // Now find missing on other procs.
292  for (Uint pid=0; pid<PE::Comm::instance().size(); pid++)
293  {
294  // Trade my requests with processor send and recv
295  const Uint pid_send_coords = (PE::Comm::instance().rank() + pid) %
296  PE::Comm::instance().size();
297  const Uint pid_recv_coords = (PE::Comm::instance().size() + PE::Comm::instance().rank() - pid) %
298  PE::Comm::instance().size();
299 
300  std::vector<Real> send_coords; send_coords.reserve(not_found.size()*target_coords.row_size());
301  std::vector<Uint> unknown_ids; unknown_ids.reserve(not_found.size());
302  std::vector<Real> received_coords;
303 
304  // fill in coords to send
305  boost_foreach (const int t, not_found)
306  {

308  {
309  unknown_ids.push_back(t);
310  boost_foreach (const Real& xyz, target_coords[t])
311  {
312  send_coords.push_back(xyz);
313  }
314  }
315  }
316 
317  // send coords, and receive coords
318  Interpolator_send_receive(pid_send_coords, send_coords,
319  pid_recv_coords, received_coords);
320 
321  Uint nb_received_coords = received_coords.size()/dim;
322 
323  // Find interpolated
324 
325  std::vector<Uint> send_found_coords; send_found_coords.reserve(nb_received_coords);
326 
327  // number of variables for each point to be interpolated
328  const Uint nb_vars = m_source_vars.size();
329 
330  // storage for interpolated variables, which will be sent to the pid that reqests it (pid_recv_interpolated)
331  std::vector<Real> send_interpolated; send_interpolated.reserve(nb_received_coords*nb_vars);
332 
333  Uint dim = target_coords.row_size();
334  RealVector t_point(dim);
335  RealVector t_val(source_field.row_size());
336 
337  for (Uint t=0; t<nb_received_coords; ++t)
338  {
339  t_point = RealVector::MapType(&received_coords[t*dim],dim);
340  bool interpolation_possible_on_this_proc =
341  m_point_interpolator->interpolate(source_field,t_point,t_val);
342  if (interpolation_possible_on_this_proc)
343  {
344  // mark found
345  send_found_coords.push_back(t);
346 
347  for (Uint v=0; v<nb_vars; ++v)
348  send_interpolated.push_back(t_val[ m_source_vars[v] ] );
349  }
350  }
351 
352  std::vector<Uint> recv_found_coords;
353  std::vector<Real> recv_interpolated;
354 
355  const Uint pid_send_back = pid_recv_coords;
356  const Uint pid_recv_back = pid_send_coords;
357 
358  Interpolator_send_receive (pid_send_back, send_found_coords,
359  pid_recv_back, recv_found_coords);
360  Interpolator_send_receive (pid_send_back, send_interpolated,
361  pid_recv_back, recv_interpolated);
362 
363  Uint it=0;
364  boost_foreach(const Uint i, recv_found_coords)
365  {
366  cf3_assert(i<unknown_ids.size());
367  const int t = unknown_ids[i];
368  not_found[t] = -1;
369  cf3_assert_desc(common::to_str(t)+'<'+common::to_str(nb_coords),t<(int)nb_coords);
370  for (Uint v=0; v<nb_vars; ++v)
371  {
372  cf3_assert(t<target.size());
373  target[t][ m_target_vars[v] ] = recv_interpolated[it++];
374  }
375  }
376  }
