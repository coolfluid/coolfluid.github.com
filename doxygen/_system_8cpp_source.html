147 void LSS::System::swap( const boost::shared_ptr< LSS::Matrix >& matrix, const boost::shared_ptr< LSS::Vector >& solution, const boost::shared_ptr< LSS::Vector >& rhs )
148 {
149  if (m_mat->is_swappable(*solution,*rhs))
150  {
151  if ((matrix->is_created()!=solution->is_created())||(matrix->is_created()!=rhs->is_created()))
152  throw common::SetupError(FromHere(),"Inconsistent states.");
153  if ((matrix->solvertype()!=solution->solvertype())||(matrix->solvertype()!=rhs->solvertype()))
154  throw common::NotSupported(FromHere(),"Inconsistent linear solver types.");
155  if ((matrix->neq()!=solution->neq())||(matrix->neq()!=rhs->neq()))
156  throw common::BadValue(FromHere(),"Inconsistent number of equations.");
157  if ((matrix->blockcol_size()!=solution->blockrow_size())||(matrix->blockcol_size()!=rhs->blockrow_size()))
158  throw common::BadValue(FromHere(),"Inconsistent number of block rows.");
159 
160  if(is_not_null(get_child("Matrix")))
161  remove_component("Matrix");
162  if(is_not_null(get_child("Solution")))
163  remove_component("Solution");
164  if(is_not_null(get_child("RHS")))
165  remove_component("RHS");
166 
167  m_mat = make_handle(matrix);
168  m_rhs = make_handle(rhs);
169  m_sol = make_handle(solution);
170 
171  add_component(matrix);
172  add_component(solution);
173  add_component(rhs);
174 
175  options().option("matrix_builder").change_value(matrix->solvertype());
176  } else {
177  throw common::NotSupported(FromHere(),"System of '" + matrix->name() + "' x '" + solution->name() + "' = '" + rhs->name() + "' is incompatible." );
178  }
179 }

259 void LSS::System::periodicity (const Uint iblockrow_to, const Uint iblockrow_from)
260 {
261  cf3_assert(is_created());
262  LSS::BlockAccumulator ba;
263  const int neq=m_mat->neq();
264  ba.resize(2,neq);
265  ba.indices[0]=iblockrow_to;
266  ba.indices[1]=iblockrow_from;
267  m_mat->tie_blockrow_pairs(iblockrow_to,iblockrow_from);
268  m_rhs->get_rhs_values(ba);
269  for (int i=0; i<(const int)neq; i++)
270  {
271  ba.rhs[i]+=ba.rhs[neq+i];
272  ba.rhs[neq+i]=0.;
273  }
274  m_rhs->set_rhs_values(ba);
275  m_sol->get_sol_values(ba);
276  for (int i=0; i<(const int)neq; i++)
277  {
278  ba.sol[neq+i]=0.5*(ba.sol[i]+ba.sol[neq+i]);
279  ba.sol[i]=ba.sol[neq+i];
280  }
281  m_sol->set_sol_values(ba);
282 }

388 const bool LSS::System::is_created()
389 {
390  int numcreated=0;
391  if (m_mat!=nullptr) if (m_mat->is_created()) numcreated+=1;
392  if (m_sol!=nullptr) if (m_sol->is_created()) numcreated+=2;
393  if (m_rhs!=nullptr) if (m_rhs->is_created()) numcreated+=4;
394  if (m_solution_strategy!=nullptr) numcreated+=8;
395  switch (numcreated) {
396  case 0 : return false;
397  case 15 : return true;
398  default: throw common::SetupError(FromHere(),"LSS System is in inconsistent state: " + common::to_str(numcreated));
399  }
400 }
