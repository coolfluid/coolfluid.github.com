118 void Manager::new_signal ( const ::MPI::Intercomm&, boost::shared_ptr<XML::XmlDoc> sig)
119 {
120  if( Comm::instance().instance().get_parent() == MPI_COMM_NULL )
121  {
122  SignalFrame frame( sig );
123 
124  //std::cout << "forwarding frame " << frame.node.attribute_value("key") <<std::endl;
125  call_signal( "signal_to_forward", frame );
126  }
127  else
128  {
129  SignalFrame signal_frame( sig );
130  bool success = false;
131  std::string message;
132 
133  std::string target = signal_frame.node.attribute_value("target");
134  std::string receiver = signal_frame.node.attribute_value("receiver");
135 
136  try
137  {
138 
139  if( target.empty() )
140  throw ValueNotFound(FromHere(), "Could not find a valid target.");
141 
142  if( receiver.empty() )
143  throw ValueNotFound(FromHere(), "Could not find a valid receiver.");
144 
145  std::string str;
146  to_string( signal_frame.node, str);
147 
148  Handle<Component> comp = access_component_checked( receiver );
149 
150  comp->call_signal(target, signal_frame);
151 
152  if( Comm::instance().rank() == 0 ) // only rank 0 sends the reply
153  {
154  SignalFrame reply = signal_frame.get_reply();
155 
156  if( reply.node.is_valid() && !reply.node.attribute_value("target").empty() )
157  {
158  send_to_parent( signal_frame );
159 
160  //std::cout << "sending reply to frame " << signal_frame.node.attribute_value("key") <<std::endl;
161  }
162  }
163 
164  success = true;
165  }
166  catch( Exception & cfe )
167  {
168  message = cfe.what();
169  }
170  catch( std::exception & stde )
171  {
172  message = stde.what();
173  }
174  catch(...)
175  {
176  message = "Unhandled exception.";
177  }
178 
179  if( Comm::instance().rank() == 0 )
180  {
182  SignalFrame frame("ack", uri(), "cpath:/UI/NetworkQueue");
183  SignalOptions & options = frame.options();
184  std::string frameid = signal_frame.node.attribute_value("frameid");
185 
186  options.add("frameid", frameid );
187  options.add("success", success );
188  options.add("message", message );
189 
190  options.flush();
191 
192  m_queue->flush();
193 
194  send_to_parent( frame );
195  }
196 
197  // synchronize with other buddies
198  Comm::instance().barrier();
199  }

212 void Manager::spawn_group ( const std::string & name,
213  Uint nb_workers,
214  const char * command,
215  const std::string & forward,
216  const char * hosts )
217 {
218  if( m_groups.find(name) != m_groups.end())
219  throw ValueExists(FromHere(), "A group of name " + name + " already exists.");
220 
221  boost::filesystem::path path;
222  std::string forw = "--forward=" + forward;
223 
224  path = boost::filesystem::system_complete( command );
225 
226  // MPI wants the arguments to be 'char *' and not 'const char *'
227  // thus, we need to make a copy since std::string::c_str() returns a const.
228  char * forw_cstr = new char[forw.length() + 1];
229 
230  std::strcpy( forw_cstr, forw.c_str() );
231 
232  char * args[] = { forw_cstr, nullptr };
233 
234  Communicator comm = Comm::instance().spawn(nb_workers, command, args, hosts);
235  m_groups[name] = comm;
236  m_listener->add_communicator( comm );
237 
238  Handle<WorkerGroup> wg = create_component<WorkerGroup>(name);
239  wg->set_communicator(comm);
240  wg->mark_basic();
241 
242  Comm::instance().barrier( comm );
243 
244  // if it is the first group, we start listening
245  if( m_groups.size() == 1 )
246  m_listener->start_listening();
247 
248 
249  delete forw_cstr;
250 }

254 void Manager::kill_group ( const std::string & name )
255 {
256  SignalFrame frame("exit", uri(), uri());
257  std::map<std::string, Communicator>::iterator it = m_groups.find(name);
258 
259  if( it == m_groups.end() )
260  throw ValueNotFound(FromHere(), "Group [" + name + "] does not exist.");
261 
262  send_to( it->second, frame );
263 
264  // workers have a barrier on their parent comm just before calling MPI_Finalize
265  Comm::instance().barrier( it->second );
266  m_listener->remove_comunicator( it->second );
267 
268  m_groups.erase(it);
269 
270  // if there are no groups anymore, we stop the listening
271  if( m_groups.empty() )
272  m_listener->stop_listening();
273 
274  remove_component(it->first);
275 
276  CFinfo << "Group " << name << " was killed." << CFendl;
277 }
