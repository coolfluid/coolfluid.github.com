82 void TCPConnection::process_header( boost::system::error_code & error )
83 {
84  std::string header_str = std::string( m_incoming_header, HEADER_LENGTH );
85 
86  try
87  {
88  // trim the string to remove the leading spaces (cast fails if spaces are present)
89  boost::algorithm::trim( header_str );
90  m_incoming_data_size = boost::lexical_cast<cf3::Uint> ( header_str );
91 
92  // destroy old buffer and allocate the new one
93  delete[] m_incoming_data;
94  m_incoming_data = new char[m_incoming_data_size];
95  }
96  catch ( boost::bad_lexical_cast & blc ) // thrown by from_str()
97  {
98  notify_error( "Could not cast frame header to unsigned int (header content was ["
99  + header_str + "]).");
100  error = asio::error::invalid_argument;
101  }
102  catch ( cf3::common::Exception & cfe )
103  {
104  notify_error(cfe.what());
105  error = asio::error::invalid_argument;
106  }
107  catch ( std::exception & stde )
108  {
109  notify_error(stde.what());
110  error = asio::error::invalid_argument;
111  }
112  catch ( ... ) // this function should catch all exception, since it is
113  { // called by some kind of event handler from boost.
114  notify_error("An unknown exception has been raised during frame header processsing.");
115  error = asio::error::invalid_argument;
116  }
117 }

150 void TCPConnection::disconnect()
151 {
152  if( m_socket.is_open() )
153  {
154  try
155  {
156  // shutdown both reading and writing async ops => "shutdown_both"
157  // node: this doesn't stop running async ops. It only prevents from

160  m_socket.close();
161  }
162  catch( boost::system::error_code & e )
163  {
164  notify_error("An error occured while the socket was shutting down: " + e.message());
165  }
