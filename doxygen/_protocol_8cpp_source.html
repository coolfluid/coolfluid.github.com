72  XmlNode Protocol::goto_doc_node ( const XmlNode& node )
73  {
74  XmlNode fnode(node);
75 
76  if( !fnode.is_valid() )
77  throw XmlError(FromHere(), "Node is not valid");
78 
79  if ( fnode.content->type() == rapidxml::node_document )
80  {
81  fnode.content = fnode.content->first_node();
82  if ( !fnode.is_valid() )
83  throw XmlError( FromHere(), "XML document is empty" );
84  }
85 
86  if ( fnode.content->type() == rapidxml::node_document )
87  {
88  fnode.content = fnode.content->next_sibling();
89  if ( !fnode.is_valid() )

91  }
92 
93  // find the first doc node
94  if ( strcmp(fnode.content->name() , Tags::node_doc()) ) /* are not equal */
95  fnode.content = fnode.content->next_sibling( Tags::node_doc() );

97  if ( !fnode.is_valid() )
98  throw common::XmlError( FromHere(), "No xml doc found" );
99 
100  return fnode;

143 
144  replynode.set_attribute( "type", Tags::node_type_reply() );
145 
146  // reply with same target
147  rapidxml::xml_attribute<>* target_att = xml_node->first_attribute("target");
148  std::string target = is_not_null(target_att) ? target_att->value() : "";
149  replynode.set_attribute("target", target);
150 
151  // the sender becomes the receiver
152  rapidxml::xml_attribute<>* sender_att = xml_node->first_attribute("sender");
153  std::string receiver = is_not_null(sender_att) ? sender_att->value() : "";
154  replynode.set_attribute("receiver", receiver);
155 
156  // same transaction type
157  rapidxml::xml_attribute<>* trans_att = xml_node->first_attribute("transaction");
158  std::string trans = is_not_null(trans_att) ? trans_att->value() : "auto";
159  replynode.set_attribute("transaction", trans);
160 
161  // copy uuids, if any
162  rapidxml::xml_attribute<>* client_uuid_att = xml_node->first_attribute( Tags::attr_clientid() );
163  rapidxml::xml_attribute<>* frame_uuid_att = xml_node->first_attribute( Tags::attr_frameid() );
164 
165  if( is_not_null(client_uuid_att) )
166  replynode.set_attribute(Tags::attr_clientid(), client_uuid_att->value());
167 
168  if( is_not_null(frame_uuid_att) )
169  replynode.set_attribute(Tags::attr_frameid(), frame_uuid_att->value() );
170
