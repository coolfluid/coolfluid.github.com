84 void BasicCommands::call(const std::vector<std::string>& params)
85 {
86  if (params.size() == 0)
87  throw SetupError(FromHere(),"executable name needed + possible options");
88 
89  std::string executable_path = params[0];
90 
91  if ( Handle< Component > executable = current_component->access_component(executable_path) )
92  dispatcher->dispatch_empty_signal( "execute", executable->uri() );
93  else
94  {
95  Handle< Component > signaling_component = current_component->access_component(URI(executable_path).base_path());
96  if ( is_null(signaling_component) )
97  throw ValueNotFound(FromHere(), "component " + URI(executable_path).base_path().path() + " was not found in " + current_component->uri().path());
98 
99  std::string name = URI(executable_path).name();
100  std::vector<std::string> signal_options(params.size()-1);
101  for (Uint i=0; i<signal_options.size(); ++i)
102  signal_options[i] = params[i+1];
103 
104  // The following 3 lines are added/adapted to get default signature values
105  SignalArgs signal_args;
106  (*signaling_component->signal(name)->signature()) (signal_args);
107  SignalOptions options(signal_args);
108 
109  options.set(signal_options);
110  options.flush();
111 
112 
113  XML::SignalFrame frame = options.create_frame(name, signaling_component->uri(), signaling_component->uri() );
114 
115 
116  dispatcher->dispatch_signal( name, signaling_component->uri(), frame );
117  }

148 std::string BasicCommands::filter_env_vars(const std::string &line)
149 {
150  std::string filtered_line = line;
151  // Check for environment variables
152  boost::regex re("\\$\\{(\\w+)\\}");
153  boost::sregex_iterator i(filtered_line.begin(), filtered_line.end(), re);
154  boost::sregex_iterator j;
155  for(; i!=j; ++i)
156  {
157  if (current_component->properties().check((*i)[1]) )
158  boost::algorithm::replace_all(filtered_line,std::string((*i)[0]),current_component->properties().value_str((*i)[1]));
159 
160  if (environment_component.properties().check((*i)[1]) )
161  boost::algorithm::replace_all(filtered_line,std::string((*i)[0]),environment_component.properties().value_str((*i)[1]));
162  }
163  return filtered_line;
164 }

213  // ls this_component

229  }
230  }
231  else if (arg == "s")
232  {
233  boost_foreach( SignalPtr sig, current_component->signal_list() )
234  {
235  if (hidden_signals.find(sig->name()) == hidden_signals.end())
236  {
237  if ( sig->is_hidden() == false )
238  CFinfo << sig->name() << CFendl;
239  }

241  }

252  CFinfo << "-";

256  }

258  {
259  if (hidden_signals.find(sig->name()) == hidden_signals.end())
260  {
261  if ( sig->is_hidden() == false )
262  {
263  if ( sig->is_read_only() )
264  CFinfo << "r-s " << sig->name() << CFendl;
265  else
266  CFinfo << "rws " << sig->name() << CFendl;
267  }
268  }
269  }
270  }
271  else
272  {
273  std::string cpath = params.back();

275  {
276  Handle<Component> parent = current_component->access_component(URI(cpath));
277  boost_foreach(Component& sub_comp, find_components(*parent))
278  {
279  CFinfo << sub_comp.name() << CFendl;
280  }
281  }
282  }
283  }
284  else if (params.size() == 2)
285  {
286  // ls -x path
287  std::string arg = params[0];
288  std::string cpath = params.back();
289  Handle<Component> parent = current_component->access_component(URI(cpath));
290 
291  if (arg == "l")

293  // ls this_component

309  }
310  }
311  else if (arg == "s")
312  {
313  boost_foreach( SignalPtr sig, current_component->signal_list() )
314  {
315  if (hidden_signals.find(sig->name())==hidden_signals.end())
316  {
317  if ( sig->is_hidden() == false)
318  CFinfo << sig->name() << CFendl;
319  }
320  }
321  }
322  else if (arg == "a")
323  {
324  // ls this_component

326  {
327  if(sub_comp.has_tag(common::Tags::static_component()))
328  CFinfo << "r-";

340  {
341  if ( ! sig->is_hidden() )
342  {
343  if ( sig->is_read_only() )
344  CFinfo << "r-s " << sig->name() << CFendl;
345  else
346  CFinfo << "rws " << sig->name() << CFendl;
347  }
348  }
349  }
350  }
351  else
352  {
353  throw ParsingFailed(FromHere(),"unrecognized ls option "+arg);
354  }
355  }
356  else
357  throw ParsingFailed (FromHere(), "More than 2 arguments for ls not supported");
358 }
