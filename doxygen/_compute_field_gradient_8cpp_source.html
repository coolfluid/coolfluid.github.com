70 void ComputeFieldGradient::execute()
71 {
72  // Check correct configuration
73  if (is_null(m_field)) throw SetupError(FromHere(), "Option field not set in "+uri().string());
74  if (is_null(m_field_gradient)) throw SetupError(FromHere(), "Option field_gradient not set in "+uri().string());
75 
76  const Uint ndim = m_field->coordinates().row_size();
77 
78  if (m_field_gradient->row_size() != ndim*m_field->row_size())
79  {
80  throw SetupError(FromHere(), "Field "+m_field_gradient->uri().string()+" must have row-size of "+to_str(ndim*m_field->row_size())+". Currently it is "+to_str(m_field_gradient->row_size()));
81  }
82 
83  RealMatrix n(ndim,ndim);
84 
85  if (m_normal.size() == 0)
86  {
87  n.setZero();
88  for (Uint d=0; d<ndim; ++d)
89  {
90  n(d,d) = 1.;
91  }
92  }
93  else
94  {
95  if (ndim == 1)
96  {
97  n << m_normal[XX];
98  }
99  else if (ndim == 2)
100  {
101  n << // rows are n, s
102  m_normal[XX], m_normal[YY],
103  m_normal[YY], -m_normal[XX];
104  }
105  else if (ndim == 3)
106  {
107  throw NotImplemented(FromHere(), "3d normal not implemented yet");
108  }
109  }
110  // Dereference the handles
111  const Field& field = *m_field;
112  Field& grad = *m_field_gradient;
113 
114  std::vector<Uint> shared_nodes;
115  if (grad.continuous())
116  {
117  shared_nodes.resize(grad.size(),0);
118  grad = 0.;
119  }
120 
121  boost_foreach(const Handle<Space>& grad_space_handle, grad.spaces())
122  {
123  Space& grad_space = *grad_space_handle;
124 
125  if (grad_space.shape_function().dimensionality() == ndim) // if volume element
126  {
127  if (field.dict().defined_for_entities( grad_space.support().handle<Entities>() ) == false )
128  throw SetupError(FromHere(), "Field "+field.uri().string()+" is not defined for elements "+grad_space.support().uri().string());
129  Space& field_space = grad_space.support().space(field.dict());
130 
131 
132  // Compute interpolation matrices to compute the gradient of
133  // field_space in coordinates of grad_space
134  std::vector<RealMatrix> gradient_matrix_per_point (grad_space.shape_function().nb_nodes());
135  for (Uint grad_pt=0; grad_pt<gradient_matrix_per_point.size(); ++grad_pt)
136  {
137  gradient_matrix_per_point[grad_pt].resize(ndim,field_space.shape_function().nb_nodes());
138  field_space.shape_function().compute_gradient(grad_space.shape_function().local_coordinates().row(grad_pt),
139  gradient_matrix_per_point[grad_pt]);
140  }
141 
142  RealMatrix field_element_values( field_space.shape_function().nb_nodes() , field.row_size() );
143  RealMatrix grad_values (ndim,field.row_size());
144 
145  RealMatrix jacobian(field_space.shape_function().dimensionality(),ndim);
146  RealMatrix cell_coords;
147  Entities& entities = field_space.support();
148  entities.geometry_space().allocate_coordinates(cell_coords);
149  // Compute the actual gradients for each element
150  for (Uint e=0; e<grad_space.size(); ++e)
151  {
152  entities.geometry_space().put_coordinates(cell_coords,e);
153 
154  // Assemble field values in a matrix
155  for (Uint node=0; node<field_space.shape_function().nb_nodes(); ++node)
156  {
157  Uint p = field_space.connectivity()[e][node];
158  for (Uint v=0; v<field.row_size(); ++v)
159  {
160  field_element_values(node,v) = field[p][v];
161  }
162  }
163 
164  for (Uint grad_pt=0; grad_pt<grad_space.shape_function().nb_nodes(); ++grad_pt)
165  {
166  // Compute jacobian of transformation to local coordinates in grad_pt
167  entities.element_type().compute_jacobian(grad_space.shape_function().local_coordinates().row(grad_pt),
168  cell_coords,
169  jacobian);
170  // Compute gradient
171  grad_values.noalias() = n * jacobian.inverse() * gradient_matrix_per_point[grad_pt] * field_element_values;
172 
173  Uint p = grad_space.connectivity()[e][grad_pt];
174 
175  if (grad.continuous()) // just sum up, and divide by shared_nodes[p] later to average
176  {
177  shared_nodes[p] += 1;
178  for (Uint d=0; d<ndim; ++d)
179  {
180  for (Uint v=0; v<field.row_size(); ++v)
181  {
182  grad[p][v+d*field.row_size()] += grad_values(d,v);
183  }
184  }
185  }
186  else // if discontinuous
187  {
188  for (Uint d=0; d<ndim; ++d)
189  {
190  for (Uint v=0; v<field.row_size(); ++v)
191  {
192  grad[p][v+d*field.row_size()] = grad_values(d,v);
193  }
194  }
195  }
196  }
197  }
198  }
199  }
200 
201  // Divide by number of times the node was visited, to get the average
202  if (grad.continuous())
203  {
204  for (Uint n=0; n<grad.size(); ++n)
205  {
206  if (shared_nodes[n] > 0)
207  {
208  for (Uint v=0; v<grad.row_size(); ++v)
209  {
210  grad[n][v] /= shared_nodes[n];
211  }
212  }
213  }
214  }
215 }
