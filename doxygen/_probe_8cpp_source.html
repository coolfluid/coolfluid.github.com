84 void Probe::execute()
85 {
86  if ( is_null(m_dict) )
87  throw SetupError(FromHere(), "Option \"dict\" was not configured in "+uri().string());
88 
89  // Take the coordinate from the options
90  std::vector<Real> opt_coord = options().value< std::vector<Real> >("coordinate");
91  RealVector coord(opt_coord.size());
92  math::copy(opt_coord,coord);
93 
94  // Find interpolation data for this coordinate
95  SpaceElem m_element;
96  std::vector<SpaceElem> m_stencil;
97  std::vector<Uint> m_points;
98  std::vector<Real> m_weights;
99 
100  int found = m_point_interpolator->compute_storage(coord,m_element,m_stencil,m_points,m_weights);
101 
102 // std::cout << PE::Comm::instance().rank() << ": found = " << found << std::endl;
103 
104  int found_on_proc = found ? PE::Comm::instance().rank() : -1;
105 
106  if (PE::Comm::instance().is_active())
107  PE::Comm::instance().all_reduce(PE::max(), &found_on_proc, 1, &found_on_proc);
108 
109 // std::cout << PE::Comm::instance().rank() << ": found_on_proc = " << found_on_proc << std::endl;
110 
111  if (found_on_proc<0)
112  throw SetupError(FromHere(),"Cannot probe: coordinate ("+to_str(opt_coord)+") lies outside the domain");
113 
114  PE::Buffer elem_comp_buffer;
115  if (found)
116  {
117  elem_comp_buffer << m_element.comp->uri().path() << m_element.glb_idx();
118  }
119  elem_comp_buffer.broadcast(found_on_proc);
120  std::string elem_comp;
121  Uint glb_idx;
122  elem_comp_buffer >> elem_comp >> glb_idx;
123 
124  properties()["space"]=elem_comp;
125  properties()["glb_elem_idx"]=glb_idx;
126 
127 // std::cout << PE::Comm::instance().rank() << ": elem_comp = " << elem_comp << std::endl;
128 // std::cout << PE::Comm::instance().rank() << ": glb_idx = " << glb_idx << std::endl;
129 
130  boost_foreach (const Handle<Field>& field, m_dict->fields())
131  {
132 
133  // Interpolate each field to the given point
134  std::vector<Real> interpolated(field->row_size());
135 
136  if (found)
137  {
138  for(Uint v=0; v<interpolated.size(); ++v)
139  {
140  interpolated[v]=0.;
141  for(Uint i=0; i<m_points.size(); ++i)
142  {
143  interpolated[v] += field->array()[m_points[i]][v] * m_weights[i];
144  }
145  }
146  }
147 
148  PE::Comm::instance().broadcast(interpolated,interpolated,found_on_proc);
149 
150  // Set interpolated variables as properties
151  for (Uint var_idx=0; var_idx<field->nb_vars(); ++var_idx)
152  {
153  Uint var_begin = field->descriptor().offset(var_idx);
154  Uint var_length = field->descriptor().var_length(var_idx);
155  if (var_length==1)
156  {
157  set(field->descriptor().user_variable_name(var_idx) , interpolated[var_begin]);
158  }
159  else
160  {
161  for (Uint i=0; i<var_length; ++i)
162  {
163  set(field->descriptor().user_variable_name(var_idx)+"["+to_str(i)+"]" , interpolated[var_begin+i]);
164  }
165  }
166  }
167  }
168 
169  // Do all post-processing actions, which could add more properties to the probe,
170  // or other things, such as log a variable in a History component, ...
171  boost_foreach (common::Action& action, find_components<common::Action>(*this))
172  {
173  action.execute();
174  }
