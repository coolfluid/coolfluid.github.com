96  void DiscontinuousDictionary_send_receive(const Uint send_to_pid, std::vector<T>& send, const Uint receive_from_pid, std::vector<T>& receive)
97  {
98  if (send_to_pid == PE::Comm::instance().rank() &&
99  receive_from_pid == PE::Comm::instance().rank())
100  {
101  receive = send;
102  return;
103  }
104  size_t recv_size;
105  size_t send_size = send.size();
106  MPI_Sendrecv(&send_size, 1, PE::get_mpi_datatype<size_t>(), (int)send_to_pid, 0,
107  &recv_size, 1, PE::get_mpi_datatype<size_t>(), (int)receive_from_pid, 0,
108  PE::Comm::instance().communicator(), MPI_STATUS_IGNORE);
109  receive.resize(recv_size);
110  MPI_Sendrecv(&send[0], (int)send.size(), PE::get_mpi_datatype<T>(), (int)send_to_pid, 0,
111  &receive[0], (int)receive.size(), PE::get_mpi_datatype<T>(), (int)receive_from_pid, 0,
112  PE::Comm::instance().communicator(), MPI_STATUS_IGNORE);
113  }

130 void DiscontinuousDictionary::rebuild_spaces_from_geometry()
131 {
132  // STEP 1: Assign the space connectivity table in the topology
133  // -----------------------------------------------------------
134  Uint field_idx = 0;
135  boost_foreach(const Handle<Entities>& entities, entities_range())
136  {
137  const Space& entities_space = *space(entities);
138  Connectivity& space_connectivity = const_cast<Connectivity&>(entities_space.connectivity());
139  space_connectivity.resize(entities->size());
140  Uint nb_nodes_per_elem = entities_space.shape_function().nb_nodes();
141  const Uint nb_elems = entities->size();
142  for (Uint elem=0; elem<nb_elems; ++elem)
143  {
144  for (Uint node=0; node<nb_nodes_per_elem; ++node)
145  space_connectivity[elem][node] = field_idx++;
146  }
147  }
148 
149  // STEP 2: Resize this space
150  // -------------------------
151  resize(field_idx);
152 
153  // STEP 3: fix unknown glb_idx of nodes
154  // ------------------------------------
155  // (1) Count the number of owned entries per process (owned when element it belongs to is owned)
156  // (2) glb_idx for owned nodes is filled in
157  // (3) Create map< glb_idx , element > of all elements. It will be used to match different cpu-elems
158  // (4) ghost elements entries are communicated for lookup
159  // (5) glb_idx of ghost entries are received back
160 
161  // (1)
162 
163  Uint nb_owned(0);
164  std::vector<Uint> nb_ghosts_belonging_to_rank(PE::Comm::instance().size());
165  std::map<boost::uint64_t, Entity > glb_elem_to_loc;
166  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
167  {
168  Entities& entities = *entities_handle;
169  const Connectivity& space_connectivity = space(entities).connectivity();
170  Uint nb_states_per_cell = space(entities).shape_function().nb_nodes();
171 
172  for (Uint e=0; e<entities.size(); ++e)
173  {
174  glb_elem_to_loc[ entities.glb_idx()[e] ] = Entity(entities,e);
175  if (entities.rank()[e] == PE::Comm::instance().rank())
176  {
177  nb_owned+=nb_states_per_cell;
178  }
179  else
180  {
181  ++nb_ghosts_belonging_to_rank[ entities.rank()[e] ];
182  }
183  boost_foreach(const Uint idx, space_connectivity[e])
184  {
185  rank()[idx] = entities.rank()[e];
186  glb_idx()[idx] = entities.glb_idx()[e]; // temporarily store element-glb-idx in node-glb-idx
187  }
188  }
189  }
190 
191  std::vector<Uint> nb_owned_per_proc(Comm::instance().size(),nb_owned);
192  if (Comm::instance().is_active())
193  Comm::instance().all_gather(nb_owned, nb_owned_per_proc);
194 
195  std::vector<Uint> start_id_per_proc(Comm::instance().size(),0);
196  for (Uint i=0; i<Comm::instance().size(); ++i)
197  {
198  start_id_per_proc[i] = (i==0? 0 : start_id_per_proc[i-1]+nb_owned_per_proc[i-1]);
199  }
200 
201  // (2)
202  Uint id = start_id_per_proc[Comm::instance().rank()];
203  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
204  {
205  Entities& entities = *entities_handle;
206  const Connectivity& space_connectivity = space(entities).connectivity();
207  for (Uint e=0; e<entities.size(); ++e)
208  {
209  boost_foreach(const Uint idx, space_connectivity[e])
210  {
211  if (rank()[idx] == PE::Comm::instance().rank() || PE::Comm::instance().size() == 1)
212  {
213  glb_idx()[idx] = id++;
214  }
215  }
216  }
217  }
218 
219  // (3)
220  std::vector< std::vector<boost::uint64_t> > ghosts_elem_idx(PE::Comm::instance().size());
221  for(Uint p=0; p<PE::Comm::instance().size(); ++p)
222  ghosts_elem_idx[p].reserve( nb_ghosts_belonging_to_rank[p] );
223 
224 
225  if (PE::Comm::instance().size() > 1)
226  {
227  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
228  {
229  Entities& entities = *entities_handle;
230  const Space& entities_space = space(entities);
231  Uint nb_states_per_elem = entities_space.shape_function().nb_nodes();
232 
233  for (Uint e=0; e<entities.size(); ++e)
234  {
235  if (entities.is_ghost(e)) // if is ghost
236  {
237  cf3_always_assert(entities.rank()[e] != UNKNOWN);
238  ghosts_elem_idx[entities.rank()[e]].push_back(entities.glb_idx()[e]);
239  }
240  }
241  }
242 
243  // (5) Search if this process contains the unknown ghosts of other processes
244  std::vector< std::vector<boost::uint64_t> > received_glb_elem_node_indices(PE::Comm::instance().size());
245 
246  for (Uint pid=0; pid<Comm::instance().size(); ++pid)
247  {
248  const Uint pid_send = (PE::Comm::instance().rank() + pid) %
249  PE::Comm::instance().size();
250  const Uint pid_recv = (PE::Comm::instance().size() + PE::Comm::instance().rank() - pid) %
251  PE::Comm::instance().size();
252 
253  std::vector<boost::uint64_t>& send_glb_ghost_elem_indices = ghosts_elem_idx[pid_send];
254  std::vector<boost::uint64_t> received_glb_ghost_elem_indices;
255  detail::DiscontinuousDictionary_send_receive( pid_send, send_glb_ghost_elem_indices,
256  pid_recv, received_glb_ghost_elem_indices );
257 
258  std::vector<boost::uint64_t> send_glb_elem_node_indices;
259  send_glb_elem_node_indices.reserve(received_glb_ghost_elem_indices.size());
260 
261  boost_foreach( const boost::uint64_t& recv_glb_idx, received_glb_ghost_elem_indices )
262  {
263  cf3_assert( glb_elem_to_loc.count( recv_glb_idx ) > 0);
264  const Entity& found_entity = glb_elem_to_loc[recv_glb_idx];
265  const Connectivity& space_connectivity = space(*found_entity.comp).connectivity();
266  send_glb_elem_node_indices.push_back( glb_idx()[space_connectivity[found_entity.idx][0]] );
267  }
268 
269  const Uint pid_send_glb_elem_node_indices = pid_recv;
270  const Uint pid_recv_glb_elem_node_indices = pid_send;
271 
272  detail::DiscontinuousDictionary_send_receive( pid_send_glb_elem_node_indices, send_glb_elem_node_indices,
273  pid_recv_glb_elem_node_indices, received_glb_elem_node_indices[pid_recv_glb_elem_node_indices] );
274  }
275 
276  std::vector<Uint> count(PE::Comm::instance().size(), 0);
277  boost_foreach(const Handle<Entities>& entities_handle, entities_range())
278  {
279  Entities& entities = *entities_handle;
280  const Space& entities_space = space(entities);
281  const Connectivity& space_connectivity = space(entities).connectivity();
282  Uint nb_states_per_elem = entities_space.shape_function().nb_nodes();
283  for (Uint e=0; e<entities.size(); ++e)
284  {
285  if (entities.is_ghost(e) && PE::Comm::instance().size() > 1) // if is ghost
286  {
287  const Uint p = entities.rank()[e];
288  const Uint start_id = received_glb_elem_node_indices[entities.rank()[e]][count[p]];
289  for (Uint n=0; n<nb_states_per_elem; ++n)
290  {
291  glb_idx()[space_connectivity[e][n]] = start_id + n;
292  }
293  ++count[p];
294  }
295  }
296  }
297  }
298  create_coordinates();
299 }
