74 std::string find_home_path()
75 {
76  std::string home_path;
77 
78 #if defined CF3_OS_LINUX || defined CF3_OS_MACOSX // if we are on a POSIX system...
79  char * home = getenv("HOME");
80 
81  if( is_not_null(home) )
82  home_path = home;
83  else
84  {
85  // the $HOME environment variable may not exist (i.e. the application
86  // was not lauched from a terminal on OSX). We need to use getpwuid()
87  // POSIX function then.
88  passwd * pwd = getpwuid( getuid() );
89 
90  if( is_not_null(pwd) )
91  home_path = pwd->pw_dir; // get home directory
92  }
93 #else
94 #ifdef CF3_OS_WINDOWS
95  char * user_profile = getenv("USERPROFILE");
96 
97  if( is_not_null(user_profile) )
98  home_path = user_profile;
99 
100  // note: some WebSites recommend to use %HOMEDRIVE% + %HOMEPATH% if
101  // %USERPROFILE% does not exist. This does not seem to be a good solution to me
102  // because - according to Microsoft documentation - %HOMEDRIVE% may not
103  // contain a useful value in the case of a network home directory (in which
104  // case %HOMESHARE% should also have to be used somewhere).
105 #endif
106 #endif
107 
108  if( home_path.empty() )
109  throw ValueNotFound( FromHere(), "Could not retrieve home directory." );
110 
111  return home_path;
112 }

197 bool CCore::get_dir_content( const std::string &directory,
198  const std::vector<std::string> & extensions,
199  bool include_files,
200  bool include_no_ext,
201  DirContent & content) const
202 {
203  using namespace boost::filesystem;
204 
205  path p(directory);
206  bool dir_exists = exists(p) && is_directory(p);
207 
208  if( dir_exists )
209  {
210  boost::regex regexp(".*\\..*"); // matches file that have an extension
211 
212  if( include_files && !extensions.empty() )
213  {
214  /* build the regex pattern string.
215  For example, if the vector contains "xml" and "CFcase" extensions,
216  the resulting string will be : "^.+\\.((xml)|(CFcase))$"
217  */
218 
219  std::string pattern = boost::algorithm::join( extensions, ")|(" );
220 
221  pattern = "^.+\\.((" + pattern + "))$";
222  regexp = boost::regex( pattern );
223  }
224 
225  // iterate through the directory contents
226  for( directory_iterator it_dir(p) ; it_dir != directory_iterator() ; ++it_dir )
227  {
228  directory_entry entry = *it_dir;
229  path entry_path = entry.path();
230  std::string filename = entry_path.filename().string();
231  std::time_t modif_time = last_write_time(entry_path);
232  std::string time_str( std::ctime(&modif_time) );
233 
234  time_str.resize( time_str.length()-1 ); // remove the ending '\n'
235 
236  if( is_directory( entry_path ) )
237  {
238  content.dirs.push_back( filename );
239  content.dir_dates.push_back( time_str );
240  }
241  else if ( is_regular_file( entry_path ) && include_files )
242  {
243  // (1) filename matches the regular expression
244  // (2) files w/out extension are allowed and filename doesn't have one
245  if( boost::regex_match( filename, regexp ) || // (1)
246  ( include_no_ext && !entry_path.has_extension() ) ) // (2)
247  {
248  content.files.push_back( filename );
249  content.file_sizes.push_back( file_size(entry_path) );
250  content.file_dates.push_back( time_str );
251  }
252  }
253  }
254 
255  }
256 
257  return dir_exists;
258 }

463 void wait_all_pids(common::SignalArgs node){
464  boost::unique_lock<boost::mutex> locker(*pid_queue_mutex,boost::defer_lock_t());
465  int status;
466  while (pid_still_coming || pid_queue.size()){
467  locker.lock();
468  if (pid_queue.size()){
469  pid_t pid=pid_queue.front();
470  pid_queue.pop();
471  std::cout << "waiting for pid :" << pid << std::endl;
472  locker.unlock();
473  waitpid(pid,&status,0);
474  }else{
475  pid_queue_condition->wait(locker);
476  locker.unlock();
477  }
478  }
479  delete pid_queue_mutex;
480  delete pid_queue_condition;
481  std::cout << "wait finished" << std::endl;
482  node.create_reply();
483 }
