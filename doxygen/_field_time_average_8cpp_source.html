54 void FieldTimeAverage::execute()
55 {
56  if(is_null(m_source_field))
57  throw common::SetupError(FromHere(), "No field configured for " + uri().path());
58 
59  const mesh::Field::ArrayT& source_array = m_source_field->array();
60  mesh::Field::ArrayT& avg_array = m_statistics_field->array();
61  const Uint nb_rows = m_source_field->size();
62  const Uint row_size = m_source_field->row_size();
63 
64  for(Uint i = 0; i != nb_rows; ++i)
65  {
66  const Eigen::Map<RealVector const> source_row(&source_array[i][0], row_size);
67  Eigen::Map<RealVector> avg_row(&avg_array[i][0], row_size);
68  avg_row = (avg_row*static_cast<Real>(m_count) + source_row) / static_cast<Real>(m_count +1);
69  }
70 
71  options().set("count", m_count+1u);
72 }
