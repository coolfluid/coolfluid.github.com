125  Bits ulp_diff(const Bits& sam1, const Bits& sam2) const
126  {
127  const Bits biased1 = SignAndMagnitudeToBiased(sam1);
128  const Bits biased2 = SignAndMagnitudeToBiased(sam2);
129 
130  return (biased1 >= biased2) ? (biased1 - biased2) : (biased2 - biased1);
131  }

133  Bits SignAndMagnitudeToBiased(const Bits& sam) const
134  {
135  if (kSignBitMask & sam) {
136  return ~sam + 1; // two's complement
137  } else {
138  return kSignBitMask | sam; // * 2
