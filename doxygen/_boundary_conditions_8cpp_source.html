137  void configure_bc(Action& bc_action, const std::string& region_name)
138  {
139  std::vector<URI> bc_regions;
140  // find the URIs of all the regions that match the region_name
141  boost_foreach(const URI& region_uri, m_region_uris)
142  {
143  Handle< Component > region_comp = m_component.access_component(region_uri);
144  if(!region_comp)
145  throw SetupError(FromHere(), "No component found at " + region_uri.string() + " when reading regions from " + m_component.uri().string());
146  Handle< Region > root_region(region_comp);
147  if(!root_region)
148  throw SetupError(FromHere(), "Component at " + region_uri.string() + " is not a region when reading regions from " + m_component.uri().string());
149 
150  Handle< Region > region = find_component_ptr_recursively_with_name<Region>(*root_region, region_name);
151  if(region)
152  bc_regions.push_back(region->uri());
153  }
154 
155  // debug output
156  if(bc_regions.empty())
157  {
158  CFdebug << "No default regions found for region name " << region_name << CFendl;
159  }
160  else
161  {
162  CFdebug << "Region name " << region_name << " resolved to:\n";
163  boost_foreach(const URI& uri, bc_regions)
164  {
165  CFdebug << " " << uri.string() << CFendl;
166  }
167  }
168 
169  bc_action.options().set(solver::Tags::regions(), bc_regions);
170  bc_action.configure_option_recursively(solver::Tags::physical_model(), m_physical_model);
171  bc_action.options().option(solver::Tags::regions()).add_tag("norecurse");
172  }

228 Handle<common::Action> BoundaryConditions::add_constant_bc(const std::string& region_name, const std::string& variable_name)
229 {
230  const VariablesDescriptor& descriptor = find_component_with_tag<VariablesDescriptor>(m_implementation->physical_model().variable_manager(), m_implementation->m_solution_tag);
231 
232  boost::shared_ptr< common::Action > result = descriptor.dimensionality(variable_name) == VariablesDescriptor::Dimensionalities::SCALAR ?
233  m_implementation->create_constant_scalar_bc(region_name, variable_name) :
234  m_implementation->create_constant_vector_bc(region_name, variable_name);
235 
236 
237  add_component(result); // Append action
238 
239  m_implementation->configure_bc(*result, region_name);
240 
241  return Handle<Action>(result);
242 }
