58 void ParsedFunctionExpression::trigger_value()
59 {
60  if(m_loop_regions.empty())
61  return;
62 
63  const std::vector<std::string> functions = options().value< std::vector<std::string> >("value");
64 
65  const Uint dim = find_parent_component<mesh::Mesh>(*regions().front()).dimension();
66 
67  // Set variables according to the dimension
68  std::vector<std::string> vars;
69  if(dim > 0)
70  vars.push_back("x");
71  if(dim > 1)
72  vars.push_back("y");
73  if(dim > 2)
74  vars.push_back("z");
75 
76  vars.push_back("t");
77 
78  m_function.variables(vars);
79  m_function.functions(functions);
80  m_function.parse();
81  const Real time = m_function.predefined_values.back();
82  m_function.predefined_values.assign(vars.size(), 0.);
83  m_function.predefined_values.back() = time;
