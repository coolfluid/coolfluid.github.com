107 void PDE::create_fields()
108 {
109  if (m_nb_eqs == 0)
110  throw InvalidStructure(FromHere(), "PDE "+derived_type_name()+" does not have any equations defined");
111  if (is_null(m_fields))
112  throw SetupError(FromHere(), "Dictionary in PDE "+uri().string()+" is not setup correctly");
113  if ( is_null(m_solution) || ( &m_solution->dict() != m_fields.get() ) )
114  {
115  if ( Handle<Component> found = m_fields->get_child("solution") )
116  {
117  m_solution = found->handle<Field>();
118  m_solution->parallelize();
119  }
120  else
121  {
122  m_solution = m_fields->create_field("solution",solution_variables()).handle<Field>();
123  m_solution->parallelize();
124  }
125  options().set("solution",m_solution);
126  }
127  if ( is_null(m_rhs) || ( &m_rhs->dict() != m_fields.get() ) )
128  {
129  if ( Handle<Component> found = m_fields->get_child("rhs") )
130  {
131  m_rhs = found->handle<Field>();
132  m_rhs->parallelize();
133  }
134  else
135  {
136  m_rhs = m_fields->create_field("rhs",m_nb_eqs).handle<Field>();
137  m_rhs->parallelize();
138  }
139  options().set("rhs",m_rhs);
140  }
141  if ( is_null(m_wave_speed) || ( &m_wave_speed->dict() != m_fields.get() ) )
142  {
143  if ( Handle<Component> found = m_fields->get_child("wave_speed") )
144  {
145  m_wave_speed = found->handle<Field>();
146  m_wave_speed->parallelize();
147  }
148  else
149  {
150  m_wave_speed = m_fields->create_field("wave_speed",1u).handle<Field>();
151  m_wave_speed->parallelize();
152  }
153  options().set("wave_speed",m_wave_speed);
154  }
155  m_rhs_computer->options().set("rhs",m_rhs);
156  m_rhs_computer->options().set("wave_speed",m_wave_speed);
157 }

201 void PDE::configure(const Handle<Component>& component)
202 {
203  cf3_assert( is_not_null(component) );
204  foreach_container( (const std::string& opt_name) (boost::shared_ptr<Option>& opt) , options() )
205  {
206  if ( component->options().check(opt_name) )
207  {
208  opt->link_option(component->options().option_ptr(opt_name));
209  component->options().set(opt_name,opt->value());
210  }
211  boost_foreach( Component& subcomponent, find_components_recursively(*component) )
212  {
213  if ( subcomponent.options().check(opt_name) )
214  {
215  opt->link_option(subcomponent.options().option_ptr(opt_name));
216  subcomponent.options().set(opt_name,opt->value());
217  }
218  }
219  }
