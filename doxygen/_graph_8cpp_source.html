425  void Graph::zoomWheel(QWheelEvent* event){
426 
427  //set scroll enable only while not in zoom mode
428  if(!m_zoomer->isEnabled()){
429 
430  double x,y,x2,y2,weight,height,posx,posy,weightMaj,heightMaj;
431 
432  posx = m_plot->invTransform(QwtPlot::xBottom, m_picker->trackerPosition().x());
433  posy = m_plot->invTransform(QwtPlot::yLeft, m_picker->trackerPosition().y());
434 
435  x = m_plot->axisScaleDiv(QwtPlot::xBottom)->lowerBound();
436  x2 = m_plot->axisScaleDiv(QwtPlot::xBottom)->upperBound();
437  y = m_plot->axisScaleDiv(QwtPlot::yLeft)->lowerBound();
438  y2 = m_plot->axisScaleDiv(QwtPlot::yLeft)->upperBound();
439 
440  weight = (std::abs(x2-x) / 2);
441  height = (std::abs(y2-y) / 2);
442 
443  weightMaj = weight + ((event->delta() / 120) * 0.02 * weight);
444  heightMaj = height + ((event->delta() / 120) * 0.02 * height);
445 
446  x = posx - weightMaj;
447  x2 = posx + weightMaj;
448 
449  y = posy - heightMaj;
450  y2 = posy + heightMaj;
451 
452 
453  m_plot->setAxisScale(QwtPlot::xBottom,x,x2);
454  m_plot->setAxisScale(QwtPlot::yLeft,y,y2);
455 
456  m_plot->replot();
457 
458  //put the mouse on the center of the widget to avoid wrong "destination"
459  QPoint posMouse( m_plot->canvas()->pos().x() + 8.999999 + (m_plot->canvas()->width()/2),
460  m_plot->canvas()->pos().y() + 69 + (m_plot->canvas()->height()/2));
461 
462  QPoint posMouseGlobale = QWidget::mapToGlobal(posMouse);
463  QCursor::setPos(posMouseGlobale);
464 
465  }else{
466  //zoom button triggered
467  }
468  }
