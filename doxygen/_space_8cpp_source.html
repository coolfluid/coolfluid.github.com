34 Space::Space ( const std::string& name ) :
35  Component ( name ),
36  m_dict_idx(0u)
37 {
38  mark_basic();
39 
40  properties()["brief"] = std::string("Spaces are other views of Entities, for instance a higher-order representation");
41 
42  std::stringstream msg;
43  msg << "A Space component uniquely relates to 1 Entities component. \n"
44  << "The concept of \"space\" is here introduced as an invisible mesh parallel\n"
45  << "to the original mesh. It has exactly the same elements as the original mesh,\n"
46  << "but every element is defined by a different shape function.\n"
47  << "A default space that is always created is the \"geometry\" space, defined by\n"
48  << "the mesh (from mesh-readers / mesh-generators).\n"
49  << "Space is a concept that allows to create fields in the same mesh or parts of the\n"
50  << "mesh, with different shape functions than prescribed by the mesh.\n"
51  << "This is useful for e.g. high-order discretization methods, without having to\n"
52  << "generate a high-order mesh.\n"
53  << "A class Dictionary is responsible for managing multiple Space components.\n"
54  << "Fields are created in the dictionary. More than one field can be created in he same\n"
55  << "dictionary, ensuring they have the same space and other common definitions.\n"
56  << "A connectivity table which is held inside this Space component refers to entries\n"
57  << "in the dictionary.\n"
58  << "\n"
59  << "Example:\n"
60  << "The coordinates of mesh element vertices is e.g. a field in the dictionary \"geometry\",\n"
61  << "and the connectivity table of the space \"geometry\" refers to the\n"
62  << "vertices connected to the mesh-elements.\n"
63  << "\n"
64  << "Notes:\n"
65  << "- Newly created spaces always have a coordinates field in their dictionary.\n";
66 
67  properties()["description"] = msg.str();
68 
69  options().add("shape_function", std::string())
70  .description("Shape Function defined in this space")
71  .pretty_name("Shape Function")
72  .attach_trigger(boost::bind(&Space::configure_shape_function, this))
73  .mark_basic();
74 }

111 void Space::configure_shape_function()
112 {
113  if (is_null(m_dict))
114  throw SetupError(FromHere(), "Space "+uri().string()+" must be created using Entities::create_space()");
115 
116  const std::string sf_name = options().value<std::string>("shape_function");
117  if (is_not_null(m_shape_function))
118  {
119  if (m_shape_function->derived_type_name() != sf_name)
120  {
121  if ( m_dict->size() !=0 )
122  throw NotImplemented(FromHere(), "Changing the shape-function will affect the dictionary, which is not yet supported");
123  }
124  }
125  else
126  {
127  m_shape_function = create_component<ShapeFunction>(sf_name, sf_name);
128  m_shape_function->rename(m_shape_function->derived_type_name());
129  m_connectivity->set_row_size(m_shape_function->nb_nodes());
130  m_connectivity->resize(m_support->size());
131  }

136 RealMatrix Space::compute_coordinates(const Uint elem_idx) const
137 {
138  const ShapeFunction& space_sf = shape_function();
139  const Entities& geometry = support();
140  const ElementType& geometry_etype = geometry.element_type();
141  const ShapeFunction& geometry_sf = geometry_etype.shape_function();
142  RealMatrix geometry_coordinates = geometry.geometry_space().get_coordinates(elem_idx);
143  RealMatrix space_coordinates(space_sf.nb_nodes(),geometry_etype.dimension());
144  for (Uint node=0; node<space_sf.nb_nodes(); ++node)
145  {
146  space_coordinates.row(node) = geometry_sf.value( space_sf.local_coordinates().row(node) ) * geometry_coordinates;
147  }
148  return space_coordinates;
149 }
