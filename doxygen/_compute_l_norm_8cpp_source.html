30 void ComputeLNorm::compute_L2( const Field& field, std::vector<Real>& norms ) const
31 {
32 
33  std::vector<Real> loc_norm(norms.size(),0.); // norm on local processor
34  std::vector<Real> glb_norm(norms.size(),0.); // norm summed over all processors
35  Uint N=0;
36  if (field.discontinuous())
37  {
38  // loop over all elements
39  boost_foreach (const Handle<Space>& space, field.spaces() )
40  {
41  // only if the elements are volume elements
42  if (space->support().element_type().dimension() == space->support().element_type().dimensionality())
43  {
44  Uint nb_nodes_per_elem = space->shape_function().nb_nodes();
45 
46  for (Uint e=0; e<space->size(); ++e)
47  {
48  if (!space->support().is_ghost(e))
49  {
50  N += nb_nodes_per_elem;
51 
52  // compute norm for these nodes
53  boost_foreach( const Uint node, space->connectivity()[e] )
54  {
55  for (Uint i=0; i<norms.size(); ++i)
56  loc_norm[i] += field[node][i]*field[node][i];
57  }
58  }
59  }
60  }
61  }
62  }
63  else if (field.continuous())
64  {
65  for (Uint n=0; n<field.size(); ++n)
66  {
67  if (!field.is_ghost(n))
68  {
69  ++N;
70  for (Uint i=0; i<norms.size(); ++i)
71  loc_norm[i] += field[n][i]*field[n][i];
72  }
73  }
74  }
75 
76  PE::Comm::instance().all_reduce( PE::plus(), &loc_norm[0], norms.size(), &glb_norm[0] );
77 
78  if( options().value<bool>("scale") )
79  PE::Comm::instance().all_reduce( PE::plus(), &N, 1, &N );
80  else
81  N = 1;
82 
83  for (Uint i=0; i<norms.size(); ++i)
84  norms[i] = std::sqrt(glb_norm[i]/static_cast<Real>(N));
85 }

302 std::vector<Real> ComputeLNorm::compute_norm(Field& field) const
303 {
304 
305  const Uint loc_nb_rows = compute_nb_rows(field); // table size on local processor
306  Uint nb_rows = 0; // table size over all processors
307 
308  PE::Comm::instance().all_reduce( PE::plus(), &loc_nb_rows, 1u, &nb_rows );
309 
310  if ( !nb_rows ) throw SetupError(FromHere(), "Table is empty");
311 
312  std::vector<Real> norm(field.row_size(), 0.);
313 
314  const Uint order = options().value<Uint>("order");
315 
316  // sum of all processors
317 
318  switch(order) {
319 
320  case 2: compute_L2( field, norm ); break;
321 
322  case 1: compute_L1( field, norm ); break;
323 
324  case 0: compute_Linf( field, norm ); break; // consider order 0 as Linf
325 
326  default: compute_Lp( field, norm, order ); break;
327 
328  }
329 
330  field.properties()["norm"] = norm;
331 
332  return norm;
333 }
