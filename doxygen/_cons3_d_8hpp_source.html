53  template < typename CV, typename SV, typename GM >
54  static void compute_properties ( const CV& coord,
55  const SV& sol,
56  const GM& grad_vars,
57  MODEL::Properties& p )
58  {
60  p.coords = coord; // cache the coordiantes locally
61  p.vars = sol; // cache the variables locally
62  p.grad_vars = grad_vars; // cache the gradient of variables locally
63 
64  p.rho = sol[Rho]; // acoustic density
65  p.rho0u = sol[Rho0U]; // rho0.u
66  p.rho0v = sol[Rho0V]; // rho0.v
67  p.rho0w = sol[Rho0W]; // rho0.w
68  p.p = sol[P]; // acoustic pressure
69 
70  p.u = p.rho0u / p.rho0; // velocity along XX, rho0.u / rho0
71  p.v = p.rho0v / p.rho0; // velocity along YY, rho0.v / rho0
72  p.w = p.rho0w / p.rho0; // velocity along ZZ, rho0.w / rho0
73 
74  p.H = p.p/( p.rho+p.rho0 ) + 0.5*(p.u*p.u + p.v*p.v + p.w*p.w);
75 
76  if (p.P0 + p.p <= 0)
77  {
78  std::cout << "rho : " << p.rho << std::endl;
79  std::cout << "rho0u : " << p.rho0u << std::endl;
80  std::cout << "rho0v : " << p.rho0v << std::endl;
81  std::cout << "rho0w : " << p.rho0w << std::endl;
82  std::cout << "P : " << p.p << std::endl;
83  std::cout << "u : " << p.u << std::endl;
84  std::cout << "v : " << p.v << std::endl;
85  std::cout << "w : " << p.w << std::endl;
86  std::cout << "H : " << p.H << std::endl;

88  throw common::FailedToConverge( FromHere(), "Pressure is negative at coordinates ["

198  template < typename GV, typename EM, typename EV >
199  static void flux_jacobian_eigen_structure(const MODEL::Properties& p,
200  const GV& direction,
201  EM& Rv,
202  EM& Lv,
203  EV& Dv)
204  {
206  const Real nx = direction[XX];
207  const Real ny = direction[YY];
208  const Real nz = direction[ZZ];
209 
210  // state is not used as Linearized Euler is, well, linear
211 
212  const Real u0n = p.u0[XX] * direction[XX] +
213  p.u0[YY] * direction[YY] +
214  p.u0[ZZ] * direction[ZZ] ;
215 
216  const Real inv_c2 = p.inv_c*p.inv_c;
217 
218  Rv << nx, ny, nz, 0.5*p.inv_c, 0.5*p.inv_c,
219  0., -nz, ny, 0.5*nx, -0.5*nx,
220  nz, 0., -nx, 0.5*ny, -0.5*ny,
221  -ny, nx, 0., 0.5*nz, -0.5*nz,
222  0., 0., 0., 0.5*p.c, 0.5*p.c;
223 
224  Lv << nx, 0., nz, -ny, -nx*inv_c2,
225  ny, -nz, 0., nx, -ny*inv_c2,
226  nz, ny, -nx, 0., -nz*inv_c2,
227  0., nx, ny, nz, p.inv_c,
228  0., -nx, -ny, -nz, p.inv_c;
229 
230 // Rv << nx, ny, nz, 0.5*p.inv_c, 0.5*p.inv_c,
231 // 0., -nz, ny, 0.5*nx, -0.5*nx,
232 // nz, 0., -nx, 0.5*ny, -0.5*ny,
233 // -ny, nx, 0., 0.5*nz, -0.5*nz,
234 // 0., 0., 0., 0.5*p.c, 0.5*p.c;
235 
236 // Lv << nx, 0., nz, -ny, -nx*inv_c2,
237 // ny, -nz, 0., nx, -ny*inv_c2,
238 // nz, ny, -nx, 0., -nz*inv_c2,
239 // 0., nx, ny, nz, p.inv_c,
240 // 0., -nx, -ny, -nz, p.inv_c;
241 
242  Dv[0] = u0n;
243  Dv[1] = u0n;
244  Dv[2] = u0n;
245  Dv[3] = u0n + p.c;
246  Dv[4] = u0n - p.c;
