76 bool NetworkThread::connect_to_host(const std::string &host_address, unsigned short port)
77 {
78  if(!isRunning())
79  {
80  m_port = port;
81  m_hostname = host_address.c_str();
82 
83  // check whether the host address contains an IP address or a hostname
84  boost::regex expression( "^[0-9]{1,3}\\.[0-9]{1,3}\\.[0-9]{1,3}\\.[0-9]{1,3}$" );
85 
86  if( !regex_match( host_address, expression ) )
87  {
88  // it's not an IP address so we need to resolve the name
89  tcp::resolver resolver( *m_io_service );
90  tcp::resolver::query query ( tcp::v4(), host_address, to_str<cf3::Uint>(port) );
91  tcp::resolver::iterator it_begin = resolver.resolve( query );
92  tcp::resolver::iterator it_end;
93 
94  if( it_begin == it_end ) // if no addresses were found
95  throw BadValue( FromHere(), "Could not resolve hostname [" + host_address + "].");
96 
97  m_endpoint = new tcp::endpoint ( it_begin->endpoint() );
98  }
99  else
100  m_endpoint = new tcp::endpoint( address::from_string(host_address), port );
101 
102  start();
103  return true;
104  }
105 
106  return false;
107 }

159 void NetworkThread::run()
160 {
161  // [thread execution starts here]
162 
163  m_connection = TCPConnection::create( *m_io_service );
164  m_connection->socket().async_connect( *m_endpoint,
165  boost::bind( &NetworkThread::callback_connect,
166  this,
167  asio::placeholders::error
168  )
169  );
170 
171  // the above call does not return before all asynchronous operation are done.
172  // since the client constantly reads on the socket, this function will return
173  // if the connection has been lost or disconnect_from_server() was called.
174  m_io_service->run();
175 
176  // if we arrive here, it means that all asynchronous operations are finished
177  // hence, we can disconnect from the server an destroy the endpoint object
178  delete_ptr(m_endpoint);
179 
180  m_connection.reset();
181 
182  // [thread execution ends here]
