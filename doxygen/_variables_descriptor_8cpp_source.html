163  void set_variables(const std::string& description)
164  {
165  const boost::regex e_variable("([[:word:]]+)[[:space:]]*(\\[[[:space:]]*[[:word:]]+[[:space:]]*\\])?");
166  const boost::regex e_scalar ("((s(cal(ar)?)?)?)|1" ,boost::regex::perl|boost::regex::icase);
167  const boost::regex e_vector("v(ec(tor)?)?",boost::regex::perl|boost::regex::icase);
168  const boost::regex e_tensor("t(ens(or)?)?",boost::regex::perl|boost::regex::icase);
169  const boost::regex e_array("[1-9][0-9]*");
170 
171  std::vector<std::string> tokenized_variables;
172  boost::split(tokenized_variables, description, boost::is_any_of(","));
173  const Uint nb_vars = tokenized_variables.size();
174 
175  std::vector<std::string> names_to_add; names_to_add.reserve(nb_vars);
176  std::vector<Dimensionalities::Type> types_to_add; types_to_add.reserve(nb_vars);
177 
178  BOOST_FOREACH(std::string var, tokenized_variables)
179  {
180  boost::match_results<std::string::const_iterator> what;
181  if (boost::regex_search(var,what,e_variable))
182  {
183  names_to_add.push_back(what[1]); boost::trim(names_to_add.back());
184  std::string var_type = what[2];
185  boost::trim_if(var_type, boost::is_any_of(" []"));
186 
187  if(boost::regex_match(var_type,e_scalar))
188  {
189  types_to_add.push_back(Dimensionalities::SCALAR);
190  }
191  else if(boost::regex_match(var_type,e_vector))
192  {
193  types_to_add.push_back(Dimensionalities::VECTOR);
194  }
195  else if(boost::regex_match(var_type,e_tensor))
196  {
197  types_to_add.push_back(Dimensionalities::TENSOR);
198  }
199  else if(boost::regex_match(var_type,e_array)) // Match varname[n] and create n scalars as a result
200  {
201  Uint nb_scalars = from_str<Uint>(var_type);
202  const std::string basename = names_to_add.back();
203  names_to_add.back() = basename + "1";
204  types_to_add.push_back(Dimensionalities::SCALAR);
205  for(Uint i = 1; i != nb_scalars;)
206  {
207  names_to_add.push_back(basename + to_str(++i));
208  types_to_add.push_back(Dimensionalities::SCALAR);
209  }
210  }
211  else
212  {
213  throw ParsingFailed(FromHere(), "Type " + var_type + " deduced from " + var + " is not recognized");
214  }
215  }
216  else
217  {
218  throw ParsingFailed(FromHere(), "Invalid variable: " + var);
219  }
220  }
221 
222  for(Uint i = 0; i != names_to_add.size(); ++i)
223  {
224  push_back(names_to_add[i], types_to_add[i]);
225  }
226  }

242  void prefix_variable_names(const std::string& prefix)
243  {
244  boost_foreach(std::string& name, m_user_names)

246  name = prefix+name;

249  std::vector<Uint> indices; indices.reserve(m_indices.size());
250  std::vector<std::string> internal_names; internal_names.reserve(m_indices.size());
251  foreach_container((const std::string& internal_name) (const Uint idx) , m_indices)
252  {
253  internal_names.push_back(internal_name);
