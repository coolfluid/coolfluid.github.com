67 void AInterpolator::signal_interpolate ( common::SignalArgs& node )
68 {
69  common::XML::SignalOptions options( node );
70 
71  URI source_uri = options.value<URI>("source");
72  URI target_uri = options.value<URI>("target");
73  URI coordinates_uri = options.value<URI>("coordinates");
74 
75  Field& source = *Handle<Field>(Core::instance().root().access_component(source_uri));
76  common::Table<Real>& target = *Handle< common::Table<Real> >(Core::instance().root().access_component(target_uri));
77 
78  Handle< common::Table<Real> > coordinates;
79  if (coordinates_uri.string() == URI().string())
80  {
81  if ( Handle< Field > target_field = Handle<Field>(target.handle<Component>()) )
82  {
83  coordinates = Handle< common::Table<Real> >(target_field->dict().coordinates().handle<Component>());
84  }
85  else
86  {
87  throw SetupError(FromHere(),"Argument \"coordinates\" not passed to interpolate() in "+uri().string());
88  }
89  }
90  else
91  {
92  coordinates = Handle< common::Table<Real> >(Core::instance().root().access_component(coordinates_uri));
93  }
94 
95  std::vector<Uint> source_vars = options.value< std::vector<Uint> >("source_vars");
96  std::vector<Uint> target_vars = options.value< std::vector<Uint> >("target_vars");
97  if (source_vars.size())
98  {
99  interpolate_vars(source,*coordinates,target,source_vars,target_vars);
100  }
101  else
102  {
103  interpolate(source,*coordinates,target);
104  }
105 }
