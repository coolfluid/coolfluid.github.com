187 void Octtree::find_cell_ranks( const boost::multi_array<Real,2>& coordinates, std::vector<Uint>& ranks )
188 {
189  ranks.resize(coordinates.size());
190 
191  Entity dummy;
192  std::deque<Uint> missing_cells;
193 
194  RealVector coord(m_dim);
195 
196  for(Uint i=0; i<coordinates.size(); ++i)
197  {
198  for (Uint d=0; d<m_dim; ++d)
199  coord[d] = coordinates[i][d];
200  if( find_element(coord,dummy) ) // if element is found on this rank
201  {
202  ranks[i] = Comm::instance().rank();
203  }
204  else
205  {
206  ranks[i] = math::Consts::uint_max();
207  missing_cells.push_back(i);
208  }
209  }
210 
211  std::vector<Real> send_coords(m_dim*missing_cells.size());
212  std::vector<Real> recv_coords;
213 
214  Uint c(0);
215  boost_foreach(const Uint i, missing_cells)
216  {
217  for(Uint d=0; d<m_dim; ++d)
218  send_coords[c++]=coordinates[i][d];
219  }
220 
221  for (Uint root=0; root<PE::Comm::instance().size(); ++root)
222  {
223 
224  recv_coords.resize(0);
225  PE::Comm::instance().broadcast(send_coords,recv_coords,root,m_dim);
226 
227  // size is only because it doesn't get resized for this rank
228  std::vector<Uint> send_found(missing_cells.size(),math::Consts::uint_max());
229 
230  if (root!=Comm::instance().rank())
231  {
232  std::vector<RealVector> recv_coordinates(recv_coords.size()/m_dim) ;
233  boost_foreach(RealVector& realvec, recv_coordinates)
234  realvec.resize(m_dim);
235 
236  c=0;
237  for (Uint i=0; i<recv_coordinates.size(); ++i)
238  {
239  for(Uint d=0; d<m_dim; ++d)
240  recv_coordinates[i][d]=recv_coords[c++];
241  }
242 
243  send_found.resize(recv_coordinates.size());
244  for (Uint i=0; i<recv_coordinates.size(); ++i)
245  {
246  if( find_element(recv_coordinates[i],dummy) ) // if element found on this rank
247  {
248  send_found[i] = Comm::instance().rank();
249  }
250  else
251  send_found[i] = math::Consts::uint_max();
252  }
253  }
254 
255  std::vector<Uint> recv_found(missing_cells.size()*Comm::instance().size());
256  PE::Comm::instance().gather(send_found,recv_found,root);
257 
258  if( root==Comm::instance().rank())
259  {
260  const Uint stride = missing_cells.size();
261  for (Uint i=0; i<missing_cells.size(); ++i)
262  {
263  for(Uint p=0; p<Comm::instance().size(); ++p)
264  {
265  ranks[missing_cells[i]] = std::min(recv_found[i+p*stride] , ranks[missing_cells[i]]);
266  }
267  }
268  }
269  }
270 }

275 {

300 void Octtree::gather_elements_around_idx(const std::vector<Uint>& octtree_idx, const Uint ring, std::vector<Entity>& elements)
301 {
302  int i(0), j(0), k(0);
303  int imin, jmin, kmin;
304  int imax, jmax, kmax;
305 
306  int irmin, jrmin, krmin;
307  int irmax, jrmax, krmax;
308 
309  if (ring == 0)
310  {
311  boost_foreach(const Entity& element, m_octtree[octtree_idx[XX]][octtree_idx[YY]][octtree_idx[ZZ]])
312  {
313  cf3_assert(element.comp);
314  elements.push_back(element);
315  }
316  return;
317  }
318  else
319  {
320  switch (m_dim)
321  {
322  case DIM_3D:
323  irmin = int(octtree_idx[XX])-int(ring); irmax = int(octtree_idx[XX])+int(ring);
324  jrmin = int(octtree_idx[YY])-int(ring); jrmax = int(octtree_idx[YY])+int(ring);
325  krmin = int(octtree_idx[ZZ])-int(ring); krmax = int(octtree_idx[ZZ])+int(ring);
326 
327  imin = std::max(irmin, 0); imax = std::min(irmax,int(m_N[XX])-1);
328  jmin = std::max(jrmin, 0); jmax = std::min(jrmax,int(m_N[YY])-1);
329  kmin = std::max(krmin, 0); kmax = std::min(krmax,int(m_N[ZZ])-1);
330 
331  // imin:
332  i = imin;
333  for (i = imin; i <= imax; ++i)
334  {
335  for (j = jmin; j <= jmax; ++j)
336  {
337  for (k = kmin; k <= kmax; ++k)
338  {
339  if ( i == irmin || i == irmax || j == jrmin || j == jrmax || k == krmin || k == krmax)
340  {
341  boost_foreach(const Entity& element, m_octtree[i][j][k])
342  {
343  cf3_assert(element.comp);
344  elements.push_back(element);
345  }
346  }
347  }
348  }
349  }
350 
351  break;
352  case DIM_2D:
353 
354  irmin = int(octtree_idx[XX])-int(ring); irmax = int(octtree_idx[XX])+int(ring);
355  jrmin = int(octtree_idx[YY])-int(ring); jrmax = int(octtree_idx[YY])+int(ring);
356 
357  imin = std::max(irmin, 0); imax = std::min(irmax,int(m_N[XX])-1);
358  jmin = std::max(jrmin, 0); jmax = std::min(jrmax,int(m_N[YY])-1);
359 
360  for (i = imin; i <= imax; ++i)
361  {
362  for (j = jmin; j <= jmax; ++j)
363  {
364  if ( i == irmin || i == irmax || j == jrmin || j == jrmax )
365  {
366  boost_foreach(const Entity& element, m_octtree[i][j][k])
367  {
368  cf3_assert(element.comp);
369  elements.push_back(element);
370  }
371  }
372  }
373  }
374 
375  break;
376 
377  case DIM_1D:
378  irmin = int(octtree_idx[XX])-int(ring); irmax = int(octtree_idx[XX])+int(ring);
379  imin = std::max(int(octtree_idx[XX])-int(ring), 0); imax = std::min(int(octtree_idx[XX])+int(ring),int(m_N[XX])-1);
380 
381  for (i = imin; i <= imax; ++i)
382  {
383  if ( i == irmin || i == irmax)
384  {
385  boost_foreach(const Entity& element, m_octtree[i][j][k])
386  {
387  cf3_assert(element.comp);
388  elements.push_back(element);
389  }
390  }
391  }
392 
393 
394  break;
395  }
396  }

414 bool Octtree::find_element(const RealVector& target_coord, Entity& element)
415 {
416  if ( !is_created() )
417  create_octtree();
418 
419  cf3_assert(target_coord.size() <= (long)m_dim);
420  RealVector t_coord(m_dim);
421  for (Uint d=0; d<target_coord.size(); ++d)
422  t_coord[d] = target_coord[d];
423 
424  for (Uint t=0; t<target_coord.size(); ++t)
425 
426  if (find_octtree_cell(t_coord,m_octtree_idx))
427  {
428  m_elements_pool.clear();
429  Uint pool_size = 0;
430  Uint rings=0;
431  for ( ; pool_size==0 ; ++rings)
432  {
433  pool_size=m_elements_pool.size();
434  gather_elements_around_idx(m_octtree_idx,rings,m_elements_pool);
435  boost_foreach(const Entity& pool_elem, boost::make_iterator_range(m_elements_pool.begin()+pool_size,m_elements_pool.end()))
436  {
437  cf3_assert(is_not_null(pool_elem.comp));
438  const RealMatrix elem_coordinates = pool_elem.get_coordinates();
439  if (pool_elem.element_type().is_coord_in_element(t_coord,elem_coordinates))
440  {
441  element = pool_elem;
442  return true;
443  }
444  }
445  }
446  // if arrived here, keep searching
447  // it means no element has been found. The search is enlarged with one more ring, for possible misses.
448  gather_elements_around_idx(m_octtree_idx,rings,m_elements_pool);
449  boost_foreach(const Entity& pool_elem, boost::make_iterator_range(m_elements_pool.begin()+pool_size,m_elements_pool.end()))
450  {
451  cf3_assert(is_not_null(pool_elem.comp));
452  const RealMatrix elem_coordinates = pool_elem.get_coordinates();
453  if (pool_elem.element_type().is_coord_in_element(t_coord,elem_coordinates))
454  {
455  element = pool_elem;
456  return true;
457  }
458  }
459  }
460  // if arrived here, it means no element has been found in the octtree cell. Give up.
461  element = Entity();
462  CFdebug << "coord";
463  for(Uint i = 0; i != m_dim; ++i)
464  {
465  CFdebug << " " << common::to_str(t_coord[i]);
466  }
467  CFdebug << " has not been found in the octtree cell" << CFendl;
468  return false;
469 }
