153 void VectorialFunction::parse()
154 {
155  clear();
156 
157  for(Uint i = 0; i < m_functions.size(); ++i)
158  {
159  FunctionParser* ptr = new FunctionParser();
160  ptr->AddConstant("pi", Consts::pi());
161  m_parsers.push_back(ptr);
162 
163  // CFinfo << "Parsing Function: \'" << m_functions[i] << "\' Vars: \'" << m_vars << "\'\n" << CFendl;
164  ptr->Parse(m_functions[i],m_vars);
165 
166  if ( ptr->GetParseErrorType() != FunctionParser::FP_NO_ERROR )
167  {
168  std::string msg("ParseError in VectorialFunction::parse(): ");
169  msg += " Error [" +std::string(ptr->ErrorMsg()) + "]";
170  msg += " Function [" + m_functions[i] + "]";
171  msg += " Vars: [" + m_vars + "]";
172  throw common::ParsingFailed (FromHere(),msg);
173  }
174  }
175 
176  m_result.resize(m_functions.size());
177  m_is_parsed = true;
