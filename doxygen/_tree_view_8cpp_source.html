175 void TreeView::mousePressEvent(QMouseEvent * event)
176 {
177  QTreeView::mousePressEvent(event);
178  QPoint mousePosition = event->pos() + this->geometry().topLeft();
179 
180  QModelIndex index = this->indexAt(mousePosition);
181  Handle< NTree > tree = NTree::global();
182 
183  QModelIndex indexInModel = m_model_filter->mapToSource(this->currentIndex());
184 
185  Qt::MouseButton button = event->button();
186 
187  try
188  {
189  if(m_context_menu_allowed && indexInModel.isValid())
190  {
191  if(button == Qt::RightButton && this->confirm_change_options(index))
192  {
193  QList<ActionInfo> actions;
194  URI path;
195 
196  tree->set_current_index(indexInModel);
197  tree->list_node_actions(indexInModel, actions);
198  path = tree->current_path();
199 
200  m_signal_manager->show_menu(QCursor::pos(), tree->node_by_path(path), actions);
201  }
202  else if(!tree->are_from_same_node(indexInModel, tree->current_index()))
203  {
204  if(this->confirm_change_options(index))
205  tree->set_current_index(indexInModel);
206  else
207  this->current_index_changed(tree->current_index(), tree->current_index());
208  }
209  }
210  }
211  catch(Exception & e)
212  {
213  NLog::global()->add_exception(e.what());
214  }
215 }

266 bool TreeView::confirm_change_options(const QModelIndex & index, bool okIfSameIndex)
267 {
268  bool confirmed = true;
269  Handle< NTree > tree = NTree::global();
270 
271  if(!okIfSameIndex && tree->are_from_same_node(tree->current_index(), index))
272  return confirmed;
273 
274  if(m_central_panel->is_modified())
275  {
276  CommitDetails commitDetails;
277 
278  ConfirmCommitDialog dlg;
279 
280  m_central_panel->list_modified_options(commitDetails);
281 
282  ConfirmCommitDialog::CommitConfirmation answer = dlg.show(commitDetails);
283 
284  if(answer == ConfirmCommitDialog::COMMIT)
285  m_central_panel->bt_apply_clicked();
286 
287  confirmed = answer != ConfirmCommitDialog::CANCEL;
288  }
289 
290  return confirmed;
291 }
