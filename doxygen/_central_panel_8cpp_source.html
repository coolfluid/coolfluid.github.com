125 void CentralPanel::set_options(const QList<boost::shared_ptr< Option > > & list)
126 {
127  QList<boost::shared_ptr< Option > >::const_iterator it = list.begin();
128  const Handle< NTree > & tree = NTree::global();
129 
130  // delete old options
131  m_basic_option_layout->clear_options();
132  m_advanced_option_layout->clear_options();
133 
134  // if there is at least one option, we set the group boxes title
135  if(!list.isEmpty())
136  {
137  // get a UNIX-like path for the node
138  QString parentPath = tree->node_path(tree->current_index());
139 
140  m_gbox_basic_options->setTitle(QString("Basic options of %1").arg(parentPath));
141  m_gbox_advanced_options->setTitle(QString("Advanced options of %1").arg(parentPath));
142  m_current_path = parentPath;
143  }
144 
145  while(it != list.end())
146  {
147  // create the option
148  try
149  {
150  boost::shared_ptr< Option > option = *it;
151  bool basic = option->has_tag("basic");
152 
153  if (basic)
154  m_basic_option_layout->add(option);
155  else
156  m_advanced_option_layout->add(option);
157  }
158  catch(Exception e)
159  {
160  NLog::global()->add_exception(e.what());
161  }
162 
163  it++;
164 
165  } // end of "while()"
166 
167  // change row stretch and panel visibilities
168  this->advanced_mode_changed(tree->is_advanced_mode());
169  this->set_buttons_visible(!list.isEmpty());
170  this->set_buttons_enabled(false);
171 
172  // set options to enabled or disabled (depending on their mode)
173 // this->setEnabled(m_basicOptionsNodes, m_basicOptions);
174 // this->setEnabled(m_advancedOptionsNodes, m_advancedOptions);
175 }

230 void CentralPanel::bt_apply_clicked()
231 {
232  QMap<QString, QString> options;
233 
234  m_basic_option_layout->options(options, false);
235  m_advanced_option_layout->options(options, false);
236 
237  // if there is at least one option that has been modified
238  if(!options.isEmpty())
239  {
240  try
241  {
242  QModelIndex currentIndex = NTree::global()->current_index();
243 
244  NTree::global()->modify_options(currentIndex, options);
245 
246  m_basic_option_layout->commit_options();
247  m_advanced_option_layout->commit_options();
248  }
249  catch (ValueNotFound & vnf)
250  {
251  NLog::global()->add_exception(vnf.msg().c_str());
252  }
253  }
254 }

268 void CentralPanel::advanced_mode_changed(bool advanced)
269 {
270  Handle< NTree > tree = NTree::global();
271 
272  // if the node went to a hidden state, we clear everything
274  if(!tree->check_index_visible(tree->current_index()))
275  {
276  m_basic_option_layout->clear_options();
277  m_advanced_option_layout->clear_options();
278  }
279 
280  m_scroll_advanced_options->setVisible(advanced && m_advanced_option_layout->has_options());
281 
282  // To avoid confusion, basic option panel is always showed if there is at
283  // least one option for the selected object, even if all options are advanced.
284  // Doing so, we ensure that the advanced options panel is *never* the
285  // top one (if visible).
286  m_scroll_basic_options->setVisible(m_basic_option_layout->has_options() ||
287  m_scroll_advanced_options->isVisible());
288 
289  set_buttons_visible(m_scroll_basic_options->isVisible());
290 }
