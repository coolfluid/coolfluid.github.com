92 void RandomizeField::execute()
93 {
94  Handle<mesh::Field> field_handle = options().value< Handle<mesh::Field> >("field");
95  if(is_null( field_handle ))
96  throw common::SetupError(FromHere(), "field option is not set for " + uri().path());
97 
98  mesh::Field& field = *field_handle;
99 
100  const std::string var_name = options().value<std::string>("variable_name");
101  const Uint reference_component = options().value<Uint>("reference_component");
102  const Uint var_begin = field_handle->var_offset(var_name);
103  const Uint var_length = field_handle->var_length(var_name);
104 
105  if(reference_component > var_length)
106  throw common::SetupError(FromHere(), "Value for reference_component is out of range");
107 
108  const std::vector<Real> maximum_variations = detail::get_vector(*this, "maximum_variations", var_length);
109  const std::vector<Real> maximum_values = detail::get_vector(*this, "maximum_values", var_length);
110  const std::vector<Real> minimum_values = detail::get_vector(*this, "minimum_values", var_length);
111 
112  boost::random::mt19937 gen(common::PE::Comm::instance().rank() + options().value<Uint>("seed"));
113  boost::random::uniform_real_distribution<Real> dist(-1., 1.);
114 
115  const Uint nb_rows = field.size();
116  for(Uint i = 0; i != nb_rows; ++i)
117  {
118  const Real ref_val = field[i][var_begin + reference_component];
119  for(Uint j = 0; j != var_length; ++j)
120  {
121  Real& value = field[i][j+var_begin];
122  const Real new_value = value + ref_val*dist(gen)*maximum_variations[j];
123  value = std::max(std::min(new_value, maximum_values[j]), minimum_values[j]);
124  }
125  }
126 
127  // Sync periodic nodes
128  Handle< common::List<Uint> > periodic_links_nodes_h(field.parent()->get_child("periodic_links_nodes"));
129  Handle< common::List<bool> > periodic_links_active_h(field.parent()->get_child("periodic_links_active"));
130  if(is_not_null(periodic_links_active_h) && is_not_null(periodic_links_nodes_h))
131  {
132  const common::List<Uint>& periodic_links_nodes = *periodic_links_nodes_h;
133  const common::List<bool>& periodic_links_active = *periodic_links_active_h;
134  cf3_assert(nb_rows == periodic_links_nodes.size());
135  for(Uint i = 0; i != nb_rows; ++i)
136  {
137  Uint target_node = i;
138  while(periodic_links_active[target_node])
139  {
140  target_node = periodic_links_nodes[target_node];
141  }
142  if(i != target_node)
143  {
144  for(Uint j = 0; j != var_length; ++j)
145  field[i][var_begin+j] = field[target_node][var_begin+j];
146  }
147  }
148  }
149 
150  // Parallel sync
151  if(common::PE::Comm::instance().is_active())
152  {
153  field.parallelize();
154  field.synchronize();
155  }
156 }
