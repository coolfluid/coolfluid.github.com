47  struct ChangeArrayValue<Uint>
48  {
49  inline void operator()(boost::any& to_set, const boost::any& new_value)
50  {
51  if(new_value.type() == to_set.type())
52  {
53  to_set = new_value;
54  }
55  else
56  {
57  try
58  {
59  std::vector<int> int_vals = boost::any_cast< std::vector<int> >(new_value);
60  const Uint nb_vals = int_vals.size();
61  std::vector<Uint> result(nb_vals);
62  for(Uint i = 0; i != nb_vals; ++i)
63  {
64  if(int_vals[i] < 0)
65  {
66  std::stringstream err;
67  err << "Tried to store a negative value in an unsigned int option array at index " << i;
68  throw BadValue(FromHere(), err.str());
69  }
70  result[i] = static_cast<Uint>(int_vals[i]);
71  }
72  to_set = result;
73  }
74  catch(boost::bad_any_cast& e)
75  {
76  throw CastingFailed(FromHere(), std::string("Failed to cast object of type ") + new_value.type().name() + " to type std::vector<Uint>");
