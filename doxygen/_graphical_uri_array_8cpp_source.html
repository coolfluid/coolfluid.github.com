164 void GraphicalUriArray::bt_add_clicked()
165 {
166 
167  if(m_edit_add->text().isEmpty())
168  {
169  if(m_combo_type->currentText() == "cpath")
170  {
171  SelectPathDialog spd;
172  QString modified_path = m_edit_add->text();
173 
174  URI path = spd.show(modified_path.toStdString());
175 
176  if(!path.empty())
177  m_edit_add->setText( path.string().c_str() );
178  }
179  else if(m_combo_type->currentText() == "file")
180  {
181  BrowserDialog nro;
182  QVariant val;
183 
184  if( nro.show(true, val) )
185  {
186  QStringList file_list = val.toStringList();
187  QStringList::iterator file = file_list.begin();
188 
189  for( ; file != file_list.end() ; ++ file)
190  file->prepend( m_combo_type->currentText() + ':' );
191 
192  if(!file_list.isEmpty())
193  m_model->setStringList( m_model->stringList() << file_list );
194  }
195  }
196  }
197 
198  if(!m_edit_add->text().isEmpty())
199  {
200  QString pathStr = m_edit_add->text();
201 
202  if( !pathStr.startsWith(m_combo_type->currentText()) )
203  pathStr.prepend(m_combo_type->currentText() + ':');
204 
205  m_model->setStringList( m_model->stringList() << pathStr );
206  m_edit_add->clear();
207  }
208 
209  emit value_changed();
210 }
