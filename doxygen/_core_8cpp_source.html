45 Core::Core()
46 {
47  // this is most likely the first part of the code to be executed
48  // due to library objects registration preceding all other execution.
49  // that registration will call Core::instance() to get access to libraries and factories
50 
51  // here we ensure that all singletons get created and exist from now on
52  // to avoid cyclic creation dependencies
53  TypeInfo::instance();
54  Logger::instance();
55  AssertionManager::instance();
56  OSystem::instance().layer()->platform_name();
57  PE::Comm::instance();
58  EventHandler::instance();
59 
60  // create singleton objects inside core
61  m_build_info.reset ( new BuildInfo() );
62  m_network_info.reset ( new NetworkInfo() );
63 
64 
65  // this types must be registered immediately on creation,
66  // registration could be defered to after the Core has been inialized.
67  RegistTypeInfo<Environment,LibCommon>();
68  RegistTypeInfo<Libraries,LibCommon>();
69  RegistTypeInfo<Factories,LibCommon>();
70 
71  // create the root component and its structure structure
72  m_environment = allocate_component<Environment>( "Environment" );
73  m_libraries = allocate_component<Libraries>("Libraries");
74  m_factories = allocate_component<Factories>("Factories");
75 
76  m_root = allocate_component<Group>( "Root" );
77  m_root->mark_basic();
78  m_root->add_component(m_environment);
79  m_root->add_component(m_libraries);
80  m_root->add_component(m_factories);
81 
82  // create tools
83  Handle<Group> tools = m_root->create_component<Group>("Tools");
84  tools->mark_basic();
85  tools->properties()["brief"] = std::string("Generic tools");
86  tools->properties()["description"] = std::string("");
87 
88 }

98 void Core::initiate ( int argc, char** argv )
99 {
100  m_argc = argc;
101  m_argv = argv;
102 
103  if( environment().options().value<bool>("regist_signal_handlers") )
104  OSystem::instance().layer()->regist_os_signal_handlers();
105 
106  // initiate the logging facility
107  Logger::instance().initiate();
108 
109  // load libraries listed in the COOLFLUID_PLUGINS environment variable
110 
111  char* env_var = std::getenv("COOLFLUID_PLUGINS");
112  if (env_var != NULL)
113  {
114  std::string environment_variable_coolfluid_plugins = env_var;
115  typedef boost::tokenizer<boost::char_separator<char> > Tokenizer;
116  boost::char_separator<char> sep(":");
117  Tokenizer tokens(environment_variable_coolfluid_plugins, sep);
118 
119  for (Tokenizer::iterator tok_iter = tokens.begin(); tok_iter != tokens.end(); ++tok_iter)
120  OSystem::instance().lib_loader()->load_library(*tok_iter);
121  }
122 
123  // initiate here all the libraries which the kernel was linked to
124 
125  libraries().initiate_all_libraries();
126 
127 }
