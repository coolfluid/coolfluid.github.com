53 void Plotter::signal_create_xyplot(SignalArgs &args)
54 {
55  SignalOptions options( args );
56 
57  std::string name = options.value<std::string>("Plot name");
58  URI parent = options.value<URI>("Parent");
59 
60  // some checks
61  if(name.empty())
62  throw BadValue(FromHere(), "The plot name is empty.");
63 
64  if(parent.empty())
65  throw BadValue(FromHere(), "The parent is empty.");
66 
67  if(parent.scheme() != URI::Scheme::CPATH)
68  throw InvalidURI(FromHere(), "The parent scheme is not CPATH");
69 
70  // create and add the component
71  Handle< Component > parent_comp = Core::instance().root().access_component(parent);
72  boost::shared_ptr< PlotXY > plot(common::allocate_component<PlotXY>(name));
73  parent_comp->add_component( plot );
74  plot->mark_basic();
75  plot->set_data(m_data);
76 }
