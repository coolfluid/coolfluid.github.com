482 void TrilinosFEVbrMatrix::symmetric_dirichlet(const Uint blockrow, const Uint ieq, const Real value, Vector& rhs)
483 {

485  const int columns_end = m_starting_indices[blockrow+1];
486 
487  Epetra_SerialDenseMatrix **val;
488  int* colindices;
489  int blockrowsize;
490  int dummy_neq;
491 
492  const Uint bc_col = m_p2m[blockrow];
493 
494  for(int col_idx = columns_begin; col_idx != columns_end; ++col_idx)
495  {
496  const int col = m_node_connectivity[col_idx];
497  const Uint other_row = m_p2m[col];
498  if(other_row >= m_blockrow_size)
499  continue;
500 
501  const int other_cols_begin = m_starting_indices[col];
502  const int other_cols_end = m_starting_indices[col+1];
503  bool row_has_node = false;
504  for(int i = other_cols_begin; i != other_cols_end; ++i)
505  {
506  if(m_node_connectivity[i] == blockrow)
507  {
508  row_has_node = true;
509  break;
510  }
511  }
512 
513  if(!row_has_node)
514  continue;
515 
516  TRILINOS_THROW(m_mat->ExtractMyBlockRowView(other_row,dummy_neq,blockrowsize,colindices,val));
517  const int nb_entries_const = blockrowsize;
518  Uint i;
519 
520  for(i = 0; i != nb_entries_const; )
521  {
522  if(colindices[i] == bc_col)
523  {
524  for(int j = 0; j != m_neq; ++j)
525  {
526  rhs.add_value(col, j, -val[i][0](j, ieq) * value);
527  val[i][0](j, ieq) = 0;
528  }
529  break;
530  }
531  ++i;
532  }
533 
534  cf3_assert(i != nb_entries_const);
535 
536  if(other_row == bc_col)
537  {
538  for(i = 0; i != nb_entries_const; ++i)
539  {
540  for(int j = 0; j != m_neq; ++j)
541  {
542  val[i][0](ieq, j) = 0;
543  }
544  if(colindices[i] == bc_col)
545  {
546  val[i][0](ieq, ieq) = 1.;
547  }
548  }
549  }
550  }
551 
552  rhs.set_value(blockrow, ieq, value);
553 }
