210 void parse_blockmesh_dict(std::istream& file, BlockArrays& blocks)
211 {
212  std::string storage; // We will read the contents here.
213  file.unsetf(std::ios::skipws); // No white space skipping!
214  std::copy(
215  std::istream_iterator<char>(file),
216  std::istream_iterator<char>(),
217  std::back_inserter(storage));
218 
219  typedef BlockMeshGrammar<std::string::const_iterator> GrammarT;
220  GrammarT blockMesh; // Our grammar
221  WhiteSpace<std::string::const_iterator> ws; // whitespace skipper
222 
223  BlockData block_data;
224  std::string::const_iterator iter = storage.begin();
225  std::string::const_iterator end = storage.end();
226  bool r = phrase_parse(iter, end, blockMesh, ws, block_data);
227  if(!r)
228  throw ParsingFailed(FromHere(), "Error parsing blockMeshDict file");
229  expand_simple_gradings(block_data.block_gradings);
230 
231  // Translate to the BlockArrays structure
232  const Uint nb_points = block_data.points.size();
233  Table<Real>& points = *blocks.create_points(3, nb_points);
234  for(Uint i = 0; i != nb_points; ++i)
235  {
236  points.set_row(i, block_data.points[i]);
237  }
238 
239  const Uint nb_blocks = block_data.block_points.size();
240  Table<Uint>& block_nodes = *blocks.create_blocks(nb_blocks);
241  Table<Uint>& block_subdivisions = *blocks.create_block_subdivisions();
242  Table<Real>& block_gradings = *blocks.create_block_gradings();
243  for(Uint i = 0; i != nb_blocks; ++i)
244  {
245  block_nodes.set_row(i, block_data.block_points[i]);
246  block_subdivisions.set_row(i, block_data.block_subdivisions[i]);
247  block_gradings.set_row(i, block_data.block_gradings[i]);
248  }
249 
250  const Uint nb_patches = block_data.patch_names.size();
251  for(Uint i = 0; i != nb_patches; ++i)
252  {
253  const Uint patch_nb_elems = block_data.patch_points[i].size();
254  Handle< Table<Uint> > patch_tbl = blocks.create_patch(block_data.patch_names[i], patch_nb_elems);
255  patch_tbl->seekp(0);
256  for(Uint j = 0; j != patch_nb_elems; ++j)
257  (*patch_tbl) << block_data.patch_points[i][j];
258  patch_tbl->seekp(0);
259  }
260 }
