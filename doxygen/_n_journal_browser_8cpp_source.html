222 void NJournalBrowser::send_exec_signal(const QModelIndex & index)
223 {
224  if(!index.isValid())
225  throw ValueNotFound(FromHere(), "Index is not valid.");
226 
227  if(!hasIndex(index.row(), index.column(), index.parent()) )
228  throw ValueNotFound(FromHere(), "Index is not part of this model");
229 
230  std::stringstream ss;
231  SignalFrame frame;
232 
233  SignalArgs * signal_node = index_to_xml_node(index);
234  signal_node->node.deep_copy(frame.node);
235 
236  rapidxml::xml_attribute<> * clientIdAttr;
237 
238  clientIdAttr = frame.node.content->first_attribute( "clientid" );
239 
240  // if found, we remove the old client UuiD.

243  frame.node.content->remove_attribute(clientIdAttr);
244 
245  // modify the frame UuiD
246  ss << UUCount().string();
247  frame.node.set_attribute( Protocol::Tags::attr_frameid(), ss.str());
