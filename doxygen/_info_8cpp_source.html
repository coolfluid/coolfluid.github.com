125 std::string Info::print_elements(const Component& region, Uint level)
126 {

131  tree += " ";
132  std::string dimensionality;

134  dimensionality = "volume";
135  else if ( elements_region.element_type().dimension() == elements_region.element_type().dimensionality() + 1 )
136  dimensionality = "surface";
137  else if ( elements_region.element_type().dimension() == elements_region.element_type().dimensionality() + 2 )
138  dimensionality = "edge";
139  else if ( elements_region.element_type().dimension() == elements_region.element_type().dimensionality() + 3 )
140  dimensionality = "points";
141  else
142  throw InvalidStructure(FromHere(), "dimensionality of element is not physical");
143 
144  tree += elements_region.name() + " -- " + dimensionality + " (" + to_str(elements_region.size()) + ")\n";
145  }
