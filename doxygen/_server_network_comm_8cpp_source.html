209 void ServerNetworkComm::callback_read( TCPConnection::Ptr conn,
210  const boost::system::error_code & error )
211 {
212  if( !error )
213  {
214  std::string error_msg;
215  ClientInfo& info = m_clients[conn];
216  SignalFrame & buffer = info.buffer;
217 
218  std::string target = buffer.node.attribute_value( "target" );
219  std::string receiver = buffer.node.attribute_value( "receiver" );
220  std::string clientid = buffer.node.attribute_value( "clientid" );
221  std::string frameid = buffer.node.attribute_value( "frameid" );
222 
223  // check if the client is attempting to register
224  if( target == "client_registration" )
225  {
226  if( !info.uuid.empty() )
227  error_msg = "This client has already been registered.";
228  else
229  {
230  info.uuid = clientid;
231 
232  // Build and send the reply
233  SignalFrame reply = buffer.create_reply();
234  SignalOptions & roptions = reply.options();
235 
236  roptions.add("accepted", true);
237  roptions.flush();
238 
239  this->init_send(conn, buffer );
240 
241  // tell listeners a new client has arrived
242  SignalFrame frame("new_client_connected", "cpath:/", "cpath:/");
243  frame.options().add( "clientid", clientid );
244  call_signal( "new_client_connected", frame );
245  }
246  }
247  else
248  {
249  if( info.uuid.empty() )
250  error_msg = "The signal came from an unregistered client.";
251  else if( info.uuid != clientid )
252  error_msg = "The client id '" + info.uuid + "' (used for registration) "
253  + "and '" + clientid + "' (used for identification) do not match.";
254  else
255  ServerRoot::instance().process_signal(target, receiver, clientid, frameid, buffer);
256  }
257 
258  if( !error_msg.empty() )
259  this->send_frame_rejected( conn, frameid, SERVER_CORE_PATH, error_msg );
260 
261  init_read( info );
262  }
263  else if( error != boost::asio::error::eof )
264  CFerror << "Could not read from [" << conn->socket().remote_endpoint()
265  << "]: " << error.message() << CFendl;
266 
267  if( error )
268  {
269  std::string uuid = m_clients[conn].uuid;
270  conn->disconnect();
271  m_clients.erase( conn );
272 
273  if( error == boost::asio::error::eof )
274  CFinfo << "Cliemt [" << uuid << "] has disconnected. ("
275  << m_clients.size() << " left)." << CFendl;
276  }
277 
278 }
