38  void apply(const mesh::LagrangeP1::Triag2D&, const PT& p, const UT& u, const UADVT& u_adv, const NUT& nu_eff, const Real& dt, MatrixT& A, MatrixT& T) const
39  {
40  typedef mesh::LagrangeP1::Triag2D ElementT;
41  const RealVector2 u_avg = u_adv.value().colwise().mean();
42  const ElementT::NodesT& nodes = u.support().nodes();
43  const Real volume = u.support().volume();
44  const Real fc = 0.5;
45  const Real nu = fabs(nu_eff.value().mean());
46 
47  // Face normals
48  ElementT::NodesT normals;
49  normals(0, XX) = nodes(1, YY) - nodes(2, YY);
50  normals(0, YY) = nodes(2, XX) - nodes(1, XX);
51  normals(1, XX) = nodes(2, YY) - nodes(0, YY);
52  normals(1, YY) = nodes(0, XX) - nodes(2, XX);
53  normals(2, XX) = nodes(0, YY) - nodes(1, YY);
54  normals(2, YY) = nodes(1, XX) - nodes(0, XX);
55 
56  Real tau_ps, tau_su, tau_bulk;
57  ComputeTauImpl compute_tau;
58  compute_tau.compute_coefficients(u, nu, dt, tau_ps, tau_su, tau_bulk);
59 
60  for(Uint i=0; i<3; ++i)
61  {
62  const Uint Pi = i + 3*p.offset;
63  const Uint Ui = i + 3*u.offset;
64  const Uint Vi = Ui + 3;
65 
66  const Real u_ni = u_avg[XX]*normals(i, XX)+u_avg[YY]*normals(i, YY);
67 
68  for(Uint j=0; j<3; ++j)
69  {
70  const Uint Pj = j + 3*p.offset;
71  const Uint Uj = j + 3*u.offset;
72  const Uint Vj = Uj + 3;
73 
74  const Real uk=u_avg[XX];
75  const Real vk=u_avg[YY];
76  const Real uknj=uk*normals(j, XX)+vk*normals(j, YY);
77 
78  // Convection (Standard + SUPG)
79  Real val = 1./6.*uknj;
80  val += tau_su/(4.*volume)*uknj*u_ni;
81  A(Ui,Uj) += val;
82  A(Vi,Vj) += val;
83 
84  // Convection (PSPG)
85  val = tau_ps/(4.*volume)*uknj;
86  A(Pi,Uj) += val*normals(i, XX);
87  A(Pi,Vj) += val*normals(i, YY);
88 
89  // Convection Skewsymm (Standard + SUPG)
90  val = fc/6.;
91  val += fc*tau_su/(4.*volume)*u_ni;
92  A(Ui,Uj) += val*uk*normals(j, XX);
93  A(Ui,Vj) += val*uk*normals(j, YY);
94  A(Vi,Uj) += val*vk*normals(j, XX);
95  A(Vi,Vj) += val*vk*normals(j, YY);
96 
97  // Convection Skewsymm (PSPG)
98  val = fc*tau_ps/(4.*volume);
99  A(Pi,Uj) += val*normals(i, XX)*uk*normals(j, XX);
100  A(Pi,Vj) += val*normals(i, XX)*uk*normals(j, YY);
101  A(Pi,Uj) += val*normals(i, YY)*vk*normals(j, XX);
102  A(Pi,Vj) += val*normals(i, YY)*vk*normals(j, YY);
103 
104  //difusion (Standard)
105  val = nu/(4.*volume);
106  A(Ui,Uj)+=val*(4./3.*normals(i, XX)*normals(j, XX)+ normals(i, YY)*normals(j, YY));
107  // A(Ui,Vj)+=val* normals(i, YY)*normals(j, XX);
108  // A(Vi,Uj)+=val* normals(i, XX)*normals(j, YY);
109  A(Ui,Vj)+=val* normals(i, XX)*normals(j, YY)/3.;
110  A(Vi,Uj)+=val* normals(i, YY)*normals(j, XX)/3.;
111  A(Vi,Vj)+=val*( normals(i, XX)*normals(j, XX)+4./3.*normals(i, YY)*normals(j, YY));
112 
113  // Pressure (Standard + SUPG)
114  val = 1./6.;
115  val += tau_su/(4.*volume)*u_ni;
116  A(Ui,Pj) += normals(j, XX)*val;
117  A(Vi,Pj) += normals(j, YY)*val;
118 
119  // Pressure (PSPG)
120  const Real laplacian = 1./(4.*volume)*(normals(i, XX)*normals(j, XX)+normals(i, YY)*normals(j, YY));
121  A(Pi,Pj) += tau_ps*laplacian;
122 
123  // Continuity (Standard)
124  val = 1./6.;
125  A(Pi,Uj) += val*normals(j, XX);
126  A(Pi,Vj) += val*normals(j, YY);
127 
128  // Bulk viscosity (Standard)
129  val = tau_bulk/(4.*volume);
130  A(Ui,Uj) += val*normals(i, XX)*normals(j, XX);
131  A(Ui,Vj) += val*normals(i, XX)*normals(j, YY);
132  A(Vi,Uj) += val*normals(i, YY)*normals(j, XX);
133  A(Vi,Vj) += val*normals(i, YY)*normals(j, YY);
134 
135  // Time (Standard + SUPG)
136  val = volume/12.*(1.+((i)==(j)?1.:0.));
137  val += tau_su/6.*u_ni;
138  T(Ui,Uj) += val;
139  T(Vi,Vj) += val;
140 
141  // Time (PSPG)
142  val = tau_ps/6.;
143  T(Pi,Uj) += val*normals(i, XX);
144  T(Pi,Vj) += val*normals(i, YY);
145  }
146  }
147  }
