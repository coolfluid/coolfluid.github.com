86  void add_options(common::OptionList& options)
87  {
88  // Add scalar options
89  for(ConstantStorage::ScalarsT::iterator it = m_constant_values.m_scalars.begin(); it != m_constant_values.m_scalars.end(); ++it)
90  {
91  const std::string& name = it->first;
92 
93  if(options.check(name))
94  {
95  options.erase(name);
96  }
97 
98  common::Option& option = options.add(name, it->second);
99  option.description(m_constant_values.descriptions[name]);
100  option.link_to(&it->second);
101  option.mark_basic();
102  }
103 
104  // Add vector options
105  for(ConstantStorage::VectorsT::iterator it = m_constant_values.m_vectors.begin(); it != m_constant_values.m_vectors.end(); ++it)
106  {
107  // Initialize proxy with the default value
108  std::vector<Real>& vec_proxy = m_constant_values.m_vector_proxies[it->first];
109  const RealVector& real_vec = it->second;
110  const Uint nb_comps = real_vec.size();
111  vec_proxy.resize(nb_comps);
112  for(Uint i = 0; i != nb_comps; ++i)
113  {
114  vec_proxy[i] = real_vec[i];
115  }
116 
117  const std::string& name = it->first;
118 
119  if(options.check(name))
120  {
121  options.erase(name);
122  }
123  common::Option& option = options.add(name, vec_proxy);
124 
125  option.description(m_constant_values.descriptions[name]);
126  option.link_to(&vec_proxy);
127  option.attach_trigger(boost::bind(&ConstantStorage::convert_vector_proxy, &m_constant_values));
128  option.mark_basic();
129  }
130  }
