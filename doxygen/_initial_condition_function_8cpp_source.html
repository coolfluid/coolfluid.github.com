78 void InitialConditionFunction::trigger()
79 {
80  const std::string field_tag = options().value<std::string>("field_tag");
81  const std::string var_name = options().value<std::string>("variable_name");
82  const std::string field_space_name = options().value<std::string>("field_space_name");
83  if(field_tag.empty() || var_name.empty())
84  return;
85 
86  VariablesDescriptor& descriptor = find_component_with_tag<VariablesDescriptor>(physical_model().variable_manager(), field_tag);
87 
88  if(!descriptor.has_variable(var_name))
89  throw SetupError(FromHere(), "No variable named " + var_name + " found in field with tag " + field_tag + " in setup of " + uri().path());
90 
91  if(descriptor.var_length(var_name) == 1)
92  {
93  FieldVariable<0, ScalarField> v(var_name, field_tag, field_space_name);
94  set_expression(nodes_expression(v = scalar_function()));
95  }
96  else
97  {
98  FieldVariable<0, VectorField> v(var_name, field_tag, field_space_name);
99  set_expression(nodes_expression(v = vector_function()));
100  }
