41 public:

48  virtual void execute()
49  {
50  if(is_null(m_lss))
51  throw common::SetupError(FromHere(), "LSS not set for " + uri().path());
52 
53  math::LSS::Matrix& lss_mat = *m_lss->matrix();
54  math::LSS::Vector& lss_rhs = *m_lss->rhs();
55 
56  BOOST_FOREACH(const Handle<mesh::Region>& region, m_loop_regions)
57  {
58  BOOST_FOREACH(const mesh::Elements& elements, find_components_recursively_with_filter<Elements>(*region, IsElementType<mesh::LagrangeP1::Triag2D>()))
59  {
60  const Uint nb_elems = elements.size();
61  const mesh::Connectivity& connectivity = elements.geometry_space().connectivity();
62  const mesh::Field& coordinates = elements.geometry_fields().coordinates();
63  Eigen::Matrix<Real, 3, 2> nodes, normals;
64  math::LSS::BlockAccumulator acc;
65  acc.resize(3, 1);
66 
67  Eigen::Matrix<Real, 3, 1> f;
68 
69  const mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(elements);
70  const mesh::Field& source_term = common::find_component_recursively_with_tag<mesh::Field>(mesh, "source_term");
71 
72  for(Uint i = 0; i != nb_elems; ++i)
73  {
74  acc.neighbour_indices(connectivity[i]);
75  mesh::fill(nodes, coordinates, connectivity[i]);
76  mesh::fill(f, source_term, connectivity[i]);
77 
78  normals(0, XX) = nodes(1, YY) - nodes(2, YY); normals(0, YY) = nodes(2, XX) - nodes(1, XX);
79  normals(1, XX) = nodes(2, YY) - nodes(0, YY); normals(1, YY) = nodes(0, XX) - nodes(2, XX);
80  normals(2, XX) = nodes(0, YY) - nodes(1, YY); normals(2, YY) = nodes(1, XX) - nodes(0, XX);
81 
82  // Jacobian determinant
83  const Real det_jac = normals(2, YY)*normals(1, XX) - normals(1, YY)*normals(2, XX);
84  const Real c = 1. / (2.*det_jac);
85 
86  for(Uint i = 0; i != 3; ++i)
87  for(Uint j = 0; j != 3; ++j)
88  acc.mat(i, j) = c * (normals(i, XX)*normals(j, XX) + normals(i, YY)*normals(j, YY));
89 
90 
91  acc.rhs[0] = (2*f[0] + f[1] + f[2]);
92  acc.rhs[1] = (f[0] + 2*f[1] + f[2]);
93  acc.rhs[2] = (f[0] + f[1] + 2*f[2]);
94  acc.rhs *= det_jac/24.;
95 
96  lss_mat.add_values(acc);
97  lss_rhs.add_rhs_values(acc);
98  }
99  }
100  }
101  }

104 };

108 PoissonManual::PoissonManual ( const std::string& name ) : LSSAction ( name )
109 {
110  // This determines the name of the field that will be used to store the solution
111  set_solution_tag("poisson_solution");
112 
113  // Create action components that wil be executed in the order they are created here:
114  // 1. Set the linear system matrix and vectors to zero
115  create_component<math::LSS::ZeroLSS>("ZeroLSS");
116 
117  // 2. Assemble the system matrix and RHS using an action component
118  Handle<PoissonManualAssembly> assembly = create_component<PoissonManualAssembly>("Assembly");
119  options().option("lss").link_to(&assembly->m_lss);
120 
121  // 3. Apply bondary conditions
122  Handle<BoundaryConditions> bc = create_component<BoundaryConditions>("BoundaryConditions");
123  bc->mark_basic();
124  bc->set_solution_tag(solution_tag());
125 
126  // 4. Solve the linear system
127  create_component<math::LSS::SolveLSS>("SolveLSS");
128 
129  // 5. Update the solution
130  // The unknown function. The first template argument is a constant to distinguish each variable at compile time
131  FieldVariable<0, ScalarField> u("u", solution_tag());
132  // The source term, to be set at runtime using an initial condition
133  FieldVariable<1, ScalarField> f("f", "source_term");
134  create_component<ProtoAction>("Update")->set_expression(nodes_expression(u = solution(u)));
135  // Dummy action to have f created automatically
136  create_component<ProtoAction>("DummyF")->set_expression(nodes_expression(f));
137 }
