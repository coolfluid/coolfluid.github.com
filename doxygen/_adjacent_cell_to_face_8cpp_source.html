68 void AdjacentCellToFace::on_regions_set()
69 {
70  if(m_loop_regions.empty())
71  return;
72 
73  mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(*m_loop_regions.front());
74  m_node_connectivity->initialize(find_components_recursively_with_filter<Elements>(mesh, IsElementsVolume()));
75 
76  std::set<std::string> adjacent_regions;
77 
78  BOOST_FOREACH(Handle<Region> region, m_loop_regions)
79  {
80  BOOST_FOREACH(Elements& elements, find_components_recursively_with_filter<Elements>(*region, IsElementsSurface()))
81  {
82  Handle<FaceConnectivity> face_conn(find_component_ptr_with_tag(elements, "face_to_cell_connectivity"));
83  if(is_null(face_conn))
84  {
85  face_conn = elements.create_component<FaceConnectivity>("FaceToCellConnectivity");
86  face_conn->add_tag("face_to_cell_connectivity");
87  face_conn->initialize(*m_node_connectivity);
88  }
89 
90  const Uint nb_elems = elements.size();
91  for(Uint i = 0; i != nb_elems; ++i)
92  {
93  cf3_assert(face_conn->has_adjacent_element(i, 0));
94  const FaceConnectivity::ElementReferenceT adj_elem = face_conn->adjacent_element(i, 0);
95  const mesh::Entities& adjacent_entities = *face_conn->node_connectivity().entities()[adj_elem.first];
96  adjacent_regions.insert(adjacent_entities.parent()->name());
97  }
98  }
99  }
100 
101  CFdebug << "AdjacentCellToFace at " << uri().path() << " uses adjacent regions ";
102  BOOST_FOREACH(const std::string& region, adjacent_regions)
103  {
104  CFdebug << region << " ";
105  }

110 void AdjacentCellToFace::execute()
111 {
112  const std::string field_tag = options()["field_tag"].value<std::string>();
113  if(field_tag.empty())
114  throw common::SetupError(FromHere(), "field_tag option is not set for " + uri().path());
115 
116  BOOST_FOREACH(const Handle<Region>& region, m_loop_regions)
117  {
118  mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(*region);
119  mesh::Field& field = common::find_component_recursively_with_tag<mesh::Field>(mesh, field_tag);
120  BOOST_FOREACH(Elements& elements, find_components_recursively_with_filter<Elements>(*region, IsElementsSurface()))
121  {
122  FaceConnectivity& face_conn = find_component_with_tag<FaceConnectivity>(elements, "face_to_cell_connectivity");
123  const Uint own_field_elements_begin = field.dict().space(elements).connectivity()[0][0];
124  const Uint nb_elems = elements.size();
125  for(Uint i = 0; i != nb_elems; ++i)
126  {
127  cf3_assert(face_conn.has_adjacent_element(i, 0));
128  const FaceConnectivity::ElementReferenceT adj_elem = face_conn.adjacent_element(i, 0);
129  const mesh::Entities& adjacent_entities = *face_conn.node_connectivity().entities()[adj_elem.first];
130  const Uint other_field_elements_begin = field.dict().space(adjacent_entities).connectivity()[0][0];
131  field.set_row(own_field_elements_begin + i, field[other_field_elements_begin + adj_elem.second]);
132  }
133  }
134  }
135 }
