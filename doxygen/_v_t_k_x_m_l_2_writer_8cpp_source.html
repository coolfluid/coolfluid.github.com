78  void start_array(const Uint nb_elems, const Uint wordsize)
79  {
80  m_wordsize = wordsize;
81 
82  const Uint nb_bytes = nb_elems * wordsize;
83  m_header.last_blocksize = nb_bytes % m_header.blocksize;
84  if(m_header.last_blocksize)
85  {
86  m_header.nb_blocks = nb_bytes / m_header.blocksize + 1;
87  }
88  else
89  {
90  m_header.last_blocksize = m_header.blocksize;
91  m_header.nb_blocks = nb_bytes / m_header.blocksize;
92  }
93 
94  m_header.compressed_blocksizes.clear();
95  m_header.compressed_blocksizes.reserve(m_header.nb_blocks);
96 
97  // Write known header info
98  data_stream.write(reinterpret_cast<const char*>(&m_header.nb_blocks), 4);
99  data_stream.write(reinterpret_cast<const char*>(&m_header.blocksize), 4);
100  data_stream.write(reinterpret_cast<const char*>(&m_header.last_blocksize), 4);
101 
102  // save filepointer
103  m_compressed_sizes_start = data_stream.tellp();
104 
105  // Reserve space for compressed block sizes
106  for(Uint i = 0; i != m_header.nb_blocks; ++i)
107  data_stream.write(reinterpret_cast<const char*>(&m_header.nb_blocks), 4);
108  }

111  void finish_array()
112  {
113  // Write the last block
114  cf3_assert(static_cast<Uint>(m_current_block.tellp()) == static_cast<Uint>(m_header.last_blocksize));
115  compress_block();
116 
117  // go back to the header
118  const Uint stream_end = data_stream.tellp();
119  data_stream.seekp(m_compressed_sizes_start);
120 
121  // Write actual compressed block sizes
122  for(Uint i = 0; i != m_header.nb_blocks; ++i)
123  data_stream.write(reinterpret_cast<const char*>(&m_header.compressed_blocksizes[i]), 4);
124 
125  // go back to the stream end
126  data_stream.seekp(stream_end);
127  }

147  void compress_block()
148  {
149  // set up zlib compressor
150  boost::iostreams::filtering_stream<boost::iostreams::input> compressed_stream;
