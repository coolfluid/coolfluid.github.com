107 void NetworkXPython::signal_get_detailed_info(SignalArgs& args)
108 {
109  // get target component's uri
110  SignalOptions options( args );
111  Handle<Component> c = access_component_checked(options.option("uri").value<URI>());
112  std::string coll("");
113 
114  // header
115  coll += "-----NAME---------------\n " + c->name() + "\n";
116  coll += "-----TYPE---------------\n " + c->derived_type_name() + "\n";
117  coll += "-----PATH---------------\n " + c->uri().path() + "\n";
118 
119  // tags
120  coll += "-----TAGS---------------\n";
121  BOOST_FOREACH(const std::string &t, c->get_tags())
122  coll += " " + t + "\n";
123 
124  // options
125  coll += "-----OPTIONS------------\n";
126  BOOST_FOREACH(const OptionList::OptionStorage_t::value_type &ot, c->options())
127  {
128  common::Option &o = *ot.second;
129  coll += " " + o.name()+ " (" + o.type() + ")=" + o.value_str() + "\n";
130  if (o.has_restricted_list())
131  {
132  coll += " restricted to:";
133  std::vector<boost::any>::iterator ir = o.restricted_list().begin();
134  for( ; ir != o.restricted_list().end() ; ++ir )
135  coll += " " + boost::any_cast<std::string>(*ir);
136  coll += "\n";
137  }
138  }
139 
140  // properties
141  coll += "-----PROPERTIES---------\n";
142  BOOST_FOREACH(const PropertyList::PropertyStorage_t::value_type &pt, c->properties())
143  {
144  std::string name = pt.first;
145  boost::any value = pt.second;
146  std::string valuestr="<String Conversion Failed>";
147  try {
148  valuestr=boost::any_cast<std::string>(value);
149  } catch(...) {}
150  coll += " " + name + " (" + value.type().name() + ")=" + valuestr + "\n";
151  }
152 
153  // child components
154  coll += "-----COMPONENTS---------\n";
155  BOOST_FOREACH(const Component& subc, *c )
156  coll += " " + subc.name() + " (" + subc.derived_type_name() + ")\n";
157 
158  // signals
159  coll += "-----SIGNALS------------\n";
160  BOOST_FOREACH(const common::SignalPtr s, c->signal_list())
161  {
162  coll += " " + s->name() + ": " + s->description() + "\n";
163  common::SignalArgs node;
164  ( * s->signature() ) ( node );
165  common::XML::SignalOptions options(node);
166  for(common::OptionList::iterator iopt = options.begin(); iopt != options.end(); ++iopt)
167  {
168  boost::shared_ptr<common::Option> o=iopt->second;
169  coll += " " + iopt->first + " (" + o->type() + ")=" + o->value_str() + ": " + o->description() + "\n";
170  if (o->has_restricted_list())
171  {
172  coll += " restricted to:";
173  std::vector<boost::any>::iterator ir = o->restricted_list().begin();
174  for( ; ir != o->restricted_list().end() ; ++ir )
175  coll += " " + boost::any_cast<std::string>(*ir);
176  coll += "\n";
177  }
178  }
179  }
180 
181  // send coll to python
182  SignalFrame reply = args.create_reply(uri());
183  SignalOptions reply_options(reply);
184  reply_options.add("return_value", coll);
185 }

473 void NetworkXPython::append_link_nodes_recursive(const Component &c, std::string &coll, const int depthlimit, const int depth, std::string printroot)
474 {
475  // has to address by c: because l: and c: can be mixed, the edge does not know if target node is in scope with c: or out of scope with l:
476  // this is tricky, there are two reasons why a node needs to be added
477  // 1. target component is outside the subtree
478  // 2. target component is behind the depthlimit
479  if (IsComponentType<common::Link>()(c))
480  {
481  const common::Link& l=dynamic_cast<const common::Link&>(c);
482  std::string t=l.follow()->uri().path();
483  if (t.find(printroot)==t.npos)
484  {
485  coll.append("G.add_node('c:" + t + "',depth=" + boost::lexical_cast<std::string>(depth+1) + ",tag='link',hidden=" + pybool(!l.has_tag("basic")) + ")\n");
486  coll.append("nodecaption.update({'c:" + t + "':'" + t + "'})\n");
487  }
488  else
489  {
490  Handle<const Component> d=l.follow();
491  for (int i=1; i>-1; ++i)
492  {
493  d=d->parent();
494  if (d==nullptr) break;
495  if ((d->uri().path()==printroot)&&(i>depthlimit))
496  {
497  coll.append("G.add_node('c:" + t + "',depth=" + boost::lexical_cast<std::string>(depth+1) + ",tag='link',hidden=" + pybool(!l.has_tag("basic")) + ")\n");
498  coll.append("nodecaption.update({'c:" + t + "':'" + t + "'})\n");
499  break;
500  }
501  }
502  }
503  }
504  if (depth<depthlimit) BOOST_FOREACH(const Component& subc, c )
505  append_link_nodes_recursive(subc, coll, depthlimit, depth+1, printroot);
506 }
