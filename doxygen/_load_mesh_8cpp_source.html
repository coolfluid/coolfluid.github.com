112 void LoadMesh::load_multiple_files(const std::vector<URI>& files, Mesh& mesh)
113 {
114  if (files.size() == 0)
115  {
116  throw SetupError(FromHere(),"No files are specified to be loaded");
117  }
118 
119  if (files.size() == 1 && mesh.dimension()==0)
120  {
121  load_mesh_into(files[0],mesh);
122  }
123  else // files need to be combined
124  {
125  MeshAdaptor mesh_adaptor(mesh);
126 
127  for (Uint f=0; f<files.size(); ++f)
128  {
129  const URI& file = files[f];
130  boost::shared_ptr<Mesh> tmp_mesh = allocate_component<Mesh>(file.base_name());
131  load_mesh_into(file,*tmp_mesh);
132  mesh_adaptor.combine_mesh(*tmp_mesh);
133  }
134  mesh_adaptor.remove_duplicate_elements_and_nodes();
135  mesh_adaptor.fix_node_ranks();
136  mesh_adaptor.finish();
137  }
138 }

141 void LoadMesh::load_mesh_into(const URI& file, Mesh& mesh)
142 {
143  update_list_of_available_readers();
144 
145  const std::string extension = file.extension();
146 
147  if ( m_extensions_to_readers.count(extension) == 0 )
148  {
149  throw FileFormatError (FromHere(), "No meshreader exists for files with extension " + extension);
150  }
151  else
152  {
153  if (m_extensions_to_readers[extension].size()>1)
154  {
155  std::string msg;
156  msg = file.string() + " has ambiguous extension " + extension + "\n"
157  + "Possible readers for this extension are: \n";
158  boost_foreach(const Handle< MeshReader > reader , m_extensions_to_readers[extension])
159  msg += " - " + reader->name() + "\n";
160  throw FileFormatError( FromHere(), msg);
161  }
162  else
163  {
164  Handle< MeshReader > meshreader = m_extensions_to_readers[extension][0];
165  meshreader->options().set("mesh",mesh.handle<Mesh>());
166  meshreader->options().set("file",file);
167  meshreader->options().set("dimension",options().value<Uint>("dimension"));
168  meshreader->execute();
169  }
170  }

175 void LoadMesh::signal_load_mesh ( common::SignalArgs& node )
176 {
177  update_list_of_available_readers();
178 
179  SignalOptions options( node );
180 
181  URI path = options.value<URI>("mesh");
182 
183  if( path.scheme() != URI::Scheme::CPATH )
184  throw ProtocolError( FromHere(), "Wrong protocol to access the Parent Component, expecting a \'cpath\' but got \'" + path.string() +"\'");
185 
186  // std::vector<URI> files = option("Files").value<std::vector<URI> >();
187  std::vector<URI> files = options.value< std::vector<URI> >("files");
188 
189  // check protocol for file loading
190  boost_foreach(URI file, files)
191  {
192  if( file.empty() || file.scheme() != URI::Scheme::FILE )
193  throw ProtocolError( FromHere(), "Wrong protocol to access the file, expecting a \'file\' but got \'" + file.string() + "\'" );
194  }
195 
196  // create a mesh in the domain
197  if( files.empty() )
198  {
199  throw BadValue( FromHere(), "No mesh was read because no files were selected." );
200  }
201 
202  Handle<Mesh> mesh;
203  if (Handle<Component> found_mesh = access_component(path))
204  {
205  mesh = found_mesh->handle<Mesh>();
206  }
207  else
208  {
209  Handle<Component> parent = access_component(path.base_path());
210  if (is_null(parent))
211  throw SetupError(FromHere(),"Component "+path.base_path().string()+" does dont exist");
212  mesh = parent->create_component<Mesh>(path.base_name());
213  }
214 
215  load_multiple_files(files,*mesh);
216 
217  SignalFrame reply = node.create_reply(uri());
218  SignalOptions reply_options(reply);
219  reply_options.add("created_component", mesh->uri());
220 }
