31 void NodeConnectivity::initialize ( const NodeConnectivity::EntitiesT& entities )
32 {
33  if(entities.empty())
34  {
35  initialize(0, entities);
36  return;
37  }
38 
39  std::set<const Dictionary*> nodes_set;
40  BOOST_FOREACH(const Handle<Entities const>& elements, entities)
41  nodes_set.insert(&elements->geometry_fields());
42 
43  // All elements in the range must use the same dictionary
44  if(nodes_set.size() != 1)
45  {
46  std::stringstream err;
47  err << "Elements span different geometry dicts:";
48  BOOST_FOREACH(const Dictionary* dict, nodes_set)
49  {
50  err << " " << dict->uri().path();
51  }
52  throw common::SetupError(FromHere(), err.str());
53  }
54 
55  // Total number of nodes in the mesh
56  Uint nb_nodes = (*nodes_set.begin())->size();
57 
58  initialize(nb_nodes, entities);
59 }

142 void create_node_element_connectivity(const Uint nb_nodes,
143  const NodeConnectivity::EntitiesT& entities,
144  NodeConnectivity::IndicesT& node_first_elements,
145  NodeConnectivity::CountsT& node_element_counts,
146  NodeConnectivity::NodeElementsT& node_elements)
147 {
148  node_first_elements.resize(nb_nodes);
149  node_element_counts.resize(nb_nodes, 0);
150 
151  // Count the node usage
152  const Uint celement_count = entities.size();
153  for(Uint celement_idx = 0; celement_idx != celement_count; ++celement_idx)
154  {
155  const Connectivity::ArrayT& element_connectivity = entities[celement_idx]->geometry_space().connectivity().array();
156  const Uint elem_count = element_connectivity.size();
157  for(Uint elem_idx = 0; elem_idx != elem_count; ++elem_idx)
158  {
159  BOOST_FOREACH(const Uint node_idx, element_connectivity[elem_idx])
160  {
161  ++node_element_counts[node_idx];
162  }
163  }
164  }
165 
166  // Calculate the size of node_elements, and build node_first_elements
167  Uint sum = 0;
168  for(Uint node_idx = 0; node_idx != nb_nodes; ++node_idx)
169  {
170  node_first_elements[node_idx] = sum;
171  sum += node_element_counts[node_idx];
172  }
173 
174  // Build node_elements
175  node_elements.resize(sum);
176  FaceConnectivity::IndicesT last_added_element(nb_nodes, 0); // helper array to keep track of where we are in node_elements
177  for(Uint entities_idx = 0; entities_idx != celement_count; ++entities_idx)
178  {
179  const Connectivity::ArrayT& element_connectivity = entities[entities_idx]->geometry_space().connectivity().array();
180  const Uint elem_count = element_connectivity.size();
181  for(Uint elem_idx = 0; elem_idx != elem_count; ++elem_idx)
182  {
183  BOOST_FOREACH(const Uint node_idx, element_connectivity[elem_idx])
184  {
185  const Uint insert_idx = node_first_elements[node_idx] + last_added_element[node_idx];
186  node_elements[insert_idx] = std::make_pair(entities_idx, elem_idx);
187  ++last_added_element[node_idx];
188  }
189  }
190  }
191 }

193 void create_face_element_connectivity(const Entities& own_celements,
194  const NodeConnectivity::EntitiesT& entities,
195  const NodeConnectivity::IndicesT& node_first_elements,
196  const NodeConnectivity::CountsT& node_element_counts,
197  const NodeConnectivity::NodeElementsT& node_elements,
198  FaceConnectivity::BoolsT& face_has_neighbour,
199  NodeConnectivity::NodeElementsT& face_element_connectivity,
200  NodeConnectivity::IndicesT& face_face_connectivity
201  )
202 {
203  // Cache some commonly accessed data
204  const Connectivity::ArrayT& element_connectivity = own_celements.geometry_space().connectivity().array();
205  const Uint elem_count = element_connectivity.size();
206  const ElementType& etype = own_celements.element_type();
207  const ElementType::FaceConnectivity& face_connectivity = etype.faces();
208  const Uint face_count = etype.nb_faces();
209  const Uint total_face_count = face_count * elem_count;
210 
211  // Init output arrays to the correct size
212  face_has_neighbour.resize(total_face_count, false);
213  face_element_connectivity.resize(total_face_count);
214  face_face_connectivity.resize(total_face_count);
215 
216  // Find out if own_celements is part of entities
217  bool vector_has_own_celements = false;
218  NodeConnectivity::ElementReferenceT own_element;
219  for(Uint i = 0; i != entities.size(); ++i)
220  {
221  if(&own_celements == entities[i].get())
222  {
223  vector_has_own_celements = true;
224  own_element.first = i;
225  break;
226  }
227  }
228 
229  // fill the output arrays
230  for(Uint elem_idx = 0; elem_idx != elem_count; ++elem_idx)
231  {
232  own_element.second = elem_idx;
233  // loop over all faces of the element
234  for(Uint face_idx = 0; face_idx != face_count; ++face_idx)
235  {
236  // collect the nodes used by this face and look up the elements that use them
237  FaceConnectivity::IndicesT face_nodes; face_nodes.reserve(face_connectivity.stride[face_idx]);
238  NodeConnectivity::NodeElementsT adjacent_elements;
239  adjacent_elements.reserve(32); // 32 in the case of a structured hexahedral mesh, just to avoid too many allocations in common cases
240  BOOST_FOREACH(const Uint local_face_node, face_connectivity.nodes_range(face_idx))
241  {
242  // global node index
243  const Uint face_node = element_connectivity[elem_idx][local_face_node];
244  face_nodes.push_back(face_node);
245 
246  // collect other elements that use this node
247  const Uint node_elements_begin = node_first_elements[face_node];
248  const Uint node_elements_end = node_element_counts[face_node] + node_elements_begin;
249  adjacent_elements.insert(adjacent_elements.end(), node_elements.begin() + node_elements_begin, node_elements.begin() + node_elements_end);
250  }
251 
252  // Find the adjacent element, if any
253  std::sort(adjacent_elements.begin(), adjacent_elements.end());
254  const Uint adjacent_count = adjacent_elements.size();
255  for(Uint i = 0; i != adjacent_count;)
256  {
257  const NodeConnectivity::ElementReferenceT adjacent_element = adjacent_elements[i];
258  const Uint start_idx = i;
259  while(i != adjacent_count && adjacent_element == adjacent_elements[i])
260  ++i;
261  if(!vector_has_own_celements || (vector_has_own_celements && adjacent_element != own_element))
262  {
263  // If any other element was found to use every node of this face, it is adjacent to the current face
264  if((i - start_idx) == face_connectivity.stride[face_idx])
265  {
266  // Verify that the face orientation is opposite to our own
267  const Entities& adjacent_entities = *entities[adjacent_element.first];
268  const Uint adjacent_local_elem = adjacent_element.second;
269  const Connectivity::ArrayT& adjacent_connectivity_table = adjacent_entities.geometry_space().connectivity().array();
270  const Uint adjacent_nb_faces = adjacent_entities.element_type().nb_faces();
271  const ElementType::FaceConnectivity& faces = adjacent_entities.element_type().faces();
272  for(Uint adj_face_idx = 0; adj_face_idx != adjacent_nb_faces; ++adj_face_idx)
273  {
274  FaceConnectivity::IndicesT adjacent_face_nodes;
275  adjacent_face_nodes.reserve(faces.stride[adj_face_idx]);
276  BOOST_FOREACH(const Uint local_node, faces.nodes_range(adj_face_idx))
277  {
278  adjacent_face_nodes.push_back(adjacent_connectivity_table[adjacent_local_elem][local_node]);
279  }
280 
281  // A face will match if its node ordering is reversed relative to our own for internal faces, or if the ordering is the same on boundaries
282  if(own_celements.element_type().dimensionality() == adjacent_entities.element_type().dimensionality())
283  std::reverse(adjacent_face_nodes.begin(), adjacent_face_nodes.end());
284  // We need to start at the same node, except for line segments
285  if(adjacent_face_nodes.size() > 2)
286  {
287  // Try to find the first node of the face in the candidate-adjacent-face
288  const FaceConnectivity::IndicesT::iterator first_matching_node = std::find(adjacent_face_nodes.begin(), adjacent_face_nodes.end(), face_nodes.front());
289  // Roll the node list of the candidate match to start at the same node
290  std::rotate(adjacent_face_nodes.begin(), first_matching_node, adjacent_face_nodes.end());
291  }
292 
293  if(adjacent_face_nodes == face_nodes) // now we have a match if the node lists are identical
294  {
295  const Uint global_face_idx = elem_idx * face_count + face_idx;
296 
297 // WD: DISABLED FOLLOWING ASSERTION BECAUSE:
298 // a cell-face might be adjacent to a inner surface as well as an adjacent cell.
299 // Probably it is the adjacent element that is sought after, and this one should be selected
300 // in that case.
301 // cf3_assert(!face_has_neighbour[global_face_idx]); // We must find only one matching face
302  face_has_neighbour[global_face_idx] = true;
303  face_element_connectivity[global_face_idx] = adjacent_element;
304  face_face_connectivity[global_face_idx] = adj_face_idx;
305  }
306  }
307  }
308  }
309  }
310  }
311  }

325 void create_face_face_connectivity(const Entities& own_celements, const NodeConnectivity::EntitiesT& entities, const FaceConnectivity::BoolsT& face_has_neighbour, const NodeConnectivity::NodeElementsT& face_element_connectivity, NodeConnectivity::IndicesT& face_face_connectivity)
326 {
327  // Init to correct size
328  face_face_connectivity.resize(face_has_neighbour.size());
329 
330  // Find out if own_celements is part of entities
331  bool vector_has_own_celements = false;
332  NodeConnectivity::ElementReferenceT own_element;
333  for(Uint i = 0; i != entities.size(); ++i)
334  {
335  if(&own_celements == entities[i].get())
336  {
337  vector_has_own_celements = true;
338  own_element.first = i;
339  break;
340  }
341  }
342 
343  const Connectivity::ArrayT& connectivity_table = own_celements.geometry_space().connectivity().array();
344  const Uint elements_begin = 0;
345  const Uint elements_end = connectivity_table.size();
346  const Uint nb_faces = own_celements.element_type().nb_faces();
347  for(Uint element_idx = elements_begin; element_idx != elements_end; ++element_idx)
348  {
349  own_element.second = element_idx;
350  //const Uint global_element_idx = global_offset + element_idx;
351  for(Uint face_idx = 0; face_idx != nb_faces; ++face_idx)
352  {
353  const Uint global_face_idx = nb_faces * element_idx + face_idx;
354  if(!face_has_neighbour[global_face_idx])
355  continue;
356 
357  // find out what our adjacent element is
358  const NodeConnectivity::ElementReferenceT adjacent_global_elem = face_element_connectivity[global_face_idx];
359  const Uint adjacent_entities_idx = adjacent_global_elem.first;
360  const Entities& adjacent_celements = *entities[adjacent_entities_idx];
361  const Uint adjacent_nb_faces = adjacent_celements.element_type().nb_faces();
362  // if we have a neighbour inside own_celements, we can use the face_element_connectivity table for fast lookup of the connecting face
363  if(vector_has_own_celements && adjacent_entities_idx == own_element.first)
364  {
365  const Uint adjacent_local_elem = adjacent_global_elem.second;
366  const Uint adjacent_faces_begin = adjacent_local_elem * adjacent_nb_faces;
367  const Uint adjacent_faces_end = adjacent_faces_begin + adjacent_nb_faces;
368  for(Uint adjacent_face = adjacent_faces_begin; adjacent_face != adjacent_faces_end; ++adjacent_face)
369  {
370  if(face_has_neighbour[adjacent_face] && face_element_connectivity[adjacent_face].second == element_idx)
371  {
372  face_face_connectivity[global_face_idx] = adjacent_face - adjacent_faces_begin;
373  break;
374  }
375  }
376  }
377  else // otherwise we need to compare face node indices, which is slower
378  {
379  FaceConnectivity::IndicesT face_nodes;
380  sorted_face_nodes(own_celements, connectivity_table, element_idx, face_idx, face_nodes);
381  const Uint adjacent_local_elem = adjacent_global_elem.second;
382  const Connectivity::ArrayT& adjacent_connectivity_table = adjacent_celements.geometry_space().connectivity().array();
383  for(Uint adjacent_face = 0; adjacent_face != adjacent_nb_faces; ++adjacent_face)
384  {
385  FaceConnectivity::IndicesT adjacent_face_nodes;
386  sorted_face_nodes(adjacent_celements, adjacent_connectivity_table, adjacent_local_elem, adjacent_face, adjacent_face_nodes);
387 
388  if(face_nodes == adjacent_face_nodes)
389  {
390  face_face_connectivity[global_face_idx] = adjacent_face;
391  break;
392  }
393  }
394  }
395  }
396  }
