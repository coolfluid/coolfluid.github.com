211 template <class CONCRETE, class BASE, class LIB >
212 struct ComponentBuilder
213 {
215  ComponentBuilder(const std::string& name =
216  std::string( LIB::library_namespace() + "." + CONCRETE::type_name()) )
217  {
218  // verify that LIB derives from Library
219  BOOST_STATIC_ASSERT( (boost::is_base_of<common::Library,LIB>::value) );
220  // verify that BASE derives or is same type of Component
221  BOOST_STATIC_ASSERT( (boost::is_base_of<common::Component,BASE>::value) );
222  // verify that CONCRETE derives from BASE
223  BOOST_STATIC_ASSERT( (boost::is_base_of<BASE,CONCRETE>::value) );
224  // verify that BuilderT derives from Builder
225  BOOST_STATIC_ASSERT( (boost::is_base_of<Builder,BuilderT<BASE,CONCRETE> >::value) );
226 
227  // give some info
228  //CFdebug << "lib [" << LIB::library_namespace() << "] : factory of \'" << BASE::type_name() << "\' registering builder of \'" << CONCRETE::type_name() << "\' with name \'" << name << "\'" << CFendl;
229 
230  // regist the concrete type in TypeInfo
231  RegistTypeInfo<CONCRETE,LIB> regist(name);
232  cf3::common::TypeInfo::instance().
233  regist< BuilderT<BASE,CONCRETE> >( BuilderT<BASE,CONCRETE>::type_name() );
234 
235  // put builder in correct factory
236  Handle<Factory> factory(common::Core::instance().factories().get_factory< BASE >());
237 
238  cf3_assert_desc ( BASE::type_name()+" is not a valid factory type", is_not_null(factory) );
239 
240  Handle< common::BuilderT<BASE,CONCRETE> > builder =
241  factory->create_component< BuilderT<BASE,CONCRETE> >( name );
242 
243  // check that BuilderT can cast to Builder
246  cf3_assert ( is_not_null(Handle<Builder>(builder)) );
247 
248  // put a Link to the builder in the respective Library
249  Handle<Library> lib(Core::instance().libraries().library<LIB>());
250  cf3_assert ( is_not_null(lib) );
251 
252  Handle<Link> builder_link = lib->create_component<Link>( name );
253  cf3_assert ( is_not_null(builder_link) );
254 
255  builder_link->link_to(*builder);
256 
257  cf3_assert ( is_not_null(builder_link->follow()) );
258  cf3_assert ( is_not_null(Handle< BuilderT<BASE,CONCRETE> >(builder_link->follow())) );
259  cf3_assert ( is_not_null(Handle<Builder>(builder_link->follow())) );
260 
261  }
