62  void operator()(const UT& u, const NUT& nu_t, const DT& wall_dist, const Real& nu_lam)
63  {
64  typedef mesh::Integrators::GaussMappedCoords<1, UT::SupportT::EtypeT::shape> GaussT;
65  u.support().compute_shape_functions(GaussT::instance().coords.col(0));
66  u.support().compute_jacobian(GaussT::instance().coords.col(0));
67  u.compute_values(GaussT::instance().coords.col(0));
68  nu_t.compute_values(GaussT::instance().coords.col(0));
69 
70 
71  // nu_t.value() is a column vector with the nodal values of the viscosity for the element.
72  // mean comes from the Eigen library
73  Real nu_t_cell = nu_t.value().mean();
74  if(nu_t_cell < 0.)
75  {
76  nu_t_cell = 0.;
77  }
78  const Real chi = nu_t_cell / nu_lam;
79 
80  ft2 = ct3 * ::exp(-ct4*chi*chi);
81 
82  // Computing S needs the gradient, which is calculated at a mapped coordinate.
83  // We take the first gauss point here, i.e. we approximate the gradient by the value at the cell center.
84  // nabla is the gradient matrix of the shape function of u
85  Eigen::Matrix<Real, UT::dimension, UT::dimension> nabla_u = u.nabla() * u.value(); // The gradient of the velocity is the shape function gradient matrix multiplied with the nodal values
86 // std::cout << "u: " << u.value().transpose() << ", nabla_u:\n" << nabla_u << std::endl;
87  // wall distance
88  const Real d = wall_dist.value().mean(); // Mean cell wall distance
89  const Real omega = sqrt(0.5)*(nabla_u - nabla_u.transpose()).norm();
90 
91  const Real fv2 = 1. - chi/(1. + chi*fv1(chi, cv1));
92  const Real Sbar = nu_t_cell / (kappa*kappa*d*d)*fv2;
93  const Real c2 = 0.7;
94  const Real c3 = 0.9;
95  //std::cout << "omega: " << omega << std::endl;
96  if(Sbar >= -c2*omega)
97  Stilde = omega + Sbar;
98  else
99  Stilde = omega + omega*(c2*c2*omega + c3*Sbar) / ((c3 - 2*c2)*omega - Sbar);
100 
101  Real r = nu_t_cell/(Stilde*kappa*kappa*d*d);
102  if (!std::isfinite(r) || r > 10)
103  r = 10;
104  const Real g = r+cw2*(r*r*r*r*r*r-r);
105  const Real g6 = g*g*g*g*g*g;
106  const Real cw36 = cw3*cw3*cw3*cw3*cw3*cw3;
107  fw = g*::pow((1.+cw36)/(g6 + cw36),1./6.);
108 
109  //std::cout << "sa_params: " << ft2 << ", " << Stilde << ", " << fw << ", " << diag_diff << std::endl;
