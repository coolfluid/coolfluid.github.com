46  virtual void execute()
47  {
48  if(is_null(m_lss))
49  throw common::SetupError(FromHere(), "LSS not set for " + uri().path());
50 
51  math::LSS::Matrix& lss_mat = *m_lss->matrix();
52  math::LSS::Vector& lss_rhs = *m_lss->rhs();
53 
54  typedef mesh::LagrangeP1::Triag2D ElementT;
55 
56  BOOST_FOREACH(const Handle<mesh::Region>& region, m_loop_regions)
57  {
58  BOOST_FOREACH(const mesh::Elements& elements, common::find_components_recursively_with_filter<mesh::Elements>(*region, mesh::IsElementType<mesh::LagrangeP1::Triag2D>()))
59  {
60  const Uint nb_elems = elements.size();
61  const mesh::Connectivity& connectivity = elements.geometry_space().connectivity();
62  const mesh::Field& coordinates = elements.geometry_fields().coordinates();
63  Eigen::Matrix<Real, 3, 2> nodes, normals, velocity;
64  math::LSS::BlockAccumulator acc;
65  acc.resize(3, 3);
66 
67  Eigen::Matrix<Real, 3, 1> nu_eff;
68 
69  const mesh::Mesh& mesh = common::find_parent_component<mesh::Mesh>(elements);
70  const mesh::Field& solution_field = common::find_component_recursively_with_tag<mesh::Field>(mesh, "navier_stokes_solution");
71  const mesh::Field& nu_field = common::find_component_recursively_with_tag<mesh::Field>(mesh, "navier_stokes_viscosity");
72 
73  Eigen::Matrix<Real, 9, 9> A, T;
74  Eigen::Matrix<Real, 9, 1> x;
75 
76  for(Uint elem = 0; elem != nb_elems; ++elem)
77  {
78  const mesh::Connectivity::ConstRow conn_row = connectivity[elem];
79  acc.neighbour_indices(conn_row);
80  for(Uint i = 0; i != 3; ++i)
81  {
82  const Uint node_idx = conn_row[i];
83  nodes(i, XX) = coordinates[node_idx][XX];
84  nodes(i, YY) = coordinates[node_idx][YY];
85  velocity(i, XX) = solution_field[node_idx][0];
86  velocity(i, YY) = solution_field[node_idx][1];
87  x[i*3] = velocity(i, XX);
88  x[i*3+1] = velocity(i, YY);
89  x[i*3+2] = solution_field[node_idx][2];
90  nu_eff[i] = nu_field[i][0];
91  }
92 
93  const RealVector2 u_avg = velocity.colwise().mean();
94  const Real volume = ElementT::volume(nodes);
95  const Real fc = 0.5;
96  const Real nu = fabs(nu_eff.mean());
97 
98  // Face normals
99  normals(0, XX) = nodes(1, YY) - nodes(2, YY);
100  normals(0, YY) = nodes(2, XX) - nodes(1, XX);
101  normals(1, XX) = nodes(2, YY) - nodes(0, YY);
102  normals(1, YY) = nodes(0, XX) - nodes(2, XX);
103  normals(2, XX) = nodes(0, YY) - nodes(1, YY);
104  normals(2, YY) = nodes(1, XX) - nodes(0, XX);
105 
106  const Real umag = u_avg.norm();
107 
108  // Get the minimal edge length
109  Real h_rgn = 1e10;
110  for(Uint i = 0; i != 3; ++i)
111  {
112  for(Uint j = 0; j != 3; ++j)
113  {
114  if(i != j)
115  {
116  h_rgn = std::min(h_rgn, (nodes.row(i) - nodes.row(j)).squaredNorm());
117  }
118  }
119  }
120  h_rgn = sqrt(h_rgn);
121 
122  const Real h_ugn = h_rgn;//fabs(2.*umag / (u.eval()*u.nabla()).sum());
123 
124  const Real tau_adv_inv = (2.*umag)/h_ugn;
125  const Real tau_time_inv = 2.*invdt;
126  const Real tau_diff_inv = (nu)/(h_rgn*h_rgn);
127 
128  const Real tau_su = 1./(tau_adv_inv + tau_time_inv + 4.*tau_diff_inv);
129  const Real tau_ps = tau_su;
130  const Real tau_bulk = tau_su*umag*umag;
131 
132  A.setZero();
133  T.setZero();
134 
135  for(Uint i=0; i<3; ++i)
136  {
137  const Uint Ui = 3*i;
138  const Uint Vi = Ui + 1;
139  const Uint Pi = Ui + 2;
140  const Real u_ni = u_avg[XX]*normals(i, XX)+u_avg[YY]*normals(i, YY);
141 
142  for(Uint j=0; j<3; ++j)
143  {
144  const Uint Uj = 3*j;
145  const Uint Vj = Uj + 1;
146  const Uint Pj = Uj + 2;
147 
148  const Real uk=u_avg[XX];
149  const Real vk=u_avg[YY];
150  const Real uknj=uk*normals(j, XX)+vk*normals(j, YY);
151 
152  // Convection (Standard + SUPG)
153  Real val = 1./6.*uknj;
154  val += tau_su/(4.*volume)*uknj*u_ni;
155  A(Ui,Uj) += val;
156  A(Vi,Vj) += val;
157 
158  // Convection (PSPG)
159  val = tau_ps/(4.*volume)*uknj;
160  A(Pi,Uj) += val*normals(i, XX);
161  A(Pi,Vj) += val*normals(i, YY);
162 
163  // Convection Skewsymm (Standard + SUPG)
164  val = fc/6.;
165  val += fc*tau_su/(4.*volume)*u_ni;
166  A(Ui,Uj) += val*uk*normals(j, XX);
167  A(Ui,Vj) += val*uk*normals(j, YY);
168  A(Vi,Uj) += val*vk*normals(j, XX);
169  A(Vi,Vj) += val*vk*normals(j, YY);
170 
171  // Convection Skewsymm (PSPG)
172  val = fc*tau_ps/(4.*volume);
173  A(Pi,Uj) += val*normals(i, XX)*uk*normals(j, XX);
174  A(Pi,Vj) += val*normals(i, XX)*uk*normals(j, YY);
175  A(Pi,Uj) += val*normals(i, YY)*vk*normals(j, XX);
176  A(Pi,Vj) += val*normals(i, YY)*vk*normals(j, YY);
177 
178  //difusion (Standard)
179  val = nu/(4.*volume);
180  A(Ui,Uj)+=val*(4./3.*normals(i, XX)*normals(j, XX)+ normals(i, YY)*normals(j, YY));
181  A(Ui,Vj)+=val* normals(i, XX)*normals(j, YY)/3.;
182  A(Vi,Uj)+=val* normals(i, YY)*normals(j, XX)/3.;
183  A(Vi,Vj)+=val*( normals(i, XX)*normals(j, XX)+4./3.*normals(i, YY)*normals(j, YY));
184 
185  // Pressure (Standard + SUPG)
186  val = 1./(6.*rho);
187  val += tau_su/(4.*rho*volume)*u_ni;
188  A(Ui,Pj) += normals(j, XX)*val;
189  A(Vi,Pj) += normals(j, YY)*val;
190 
191  // Pressure (PSPG)
192  const Real laplacian = 1./(4.*rho*volume)*(normals(i, XX)*normals(j, XX)+normals(i, YY)*normals(j, YY));
193  A(Pi,Pj) += tau_ps*laplacian;
194 
195  // Continuity (Standard)
196  val = 1./6.;
197  A(Pi,Uj) += val*normals(j, XX);
198  A(Pi,Vj) += val*normals(j, YY);
199 
200  // Bulk viscosity (Standard)
201  val = tau_bulk/(4.*volume);
202  A(Ui,Uj) += val*normals(i, XX)*normals(j, XX);
203  A(Ui,Vj) += val*normals(i, XX)*normals(j, YY);
204  A(Vi,Uj) += val*normals(i, YY)*normals(j, XX);
205  A(Vi,Vj) += val*normals(i, YY)*normals(j, YY);
206 
207  // Time (Standard + SUPG)
208  val = volume/12.*(1.+((i)==(j)?1.:0.));
209  val += tau_su/6.*u_ni;
210  T(Ui,Uj) += val;
211  T(Vi,Vj) += val;
212 
213  // Time (PSPG)
214  val = tau_ps/6.;
215  T(Pi,Uj) += val*normals(i, XX);
216  T(Pi,Vj) += val*normals(i, YY);
217  }
218  }
219  acc.rhs = -A*x;
220  A.row(0) /= theta;
221  A.row(3) /= theta;
222  A.row(6) /= theta;
223  acc.mat = invdt * T + theta*A;
224  lss_mat.add_values(acc);
225  lss_rhs.add_rhs_values(acc);
226  }
227  }
228  }
229  }

239 NavierStokesManual::NavierStokesManual ( const std::string& name ) : LSSActionUnsteady( name )
240 {
241  // Option for the theta scheme
242  options().add("theta", 1.)
243  .pretty_name("Theta")
244  .description("Theta coefficient for the theta-method.");
245 
246  // This determines the name of the field that will be used to store the solution
247  set_solution_tag("navier_stokes_solution");
248 
249  // Create action components that wil be executed in the order they are created here:
250  // 1. Set the linear system matrix and vectors to zero
251  create_component<math::LSS::ZeroLSS>("ZeroLSS");
252 
253  // 2. Assemble the system matrix and RHS using an action component
254  Handle<NavierStokesManualAssembly> assembly = create_component<NavierStokesManualAssembly>("Assembly");
255  options().option("lss").link_to(&assembly->m_lss);
256 
257  // 3. Apply bondary conditions
258  Handle<BoundaryConditions> bc = create_component<BoundaryConditions>("BoundaryConditions");
259  bc->mark_basic();
260  bc->set_solution_tag(solution_tag());
261 
262  // 4. Solve the linear system
263  create_component<math::LSS::SolveLSS>("SolveLSS");
264 
265  // 5. Update the solution
266  FieldVariable<0, VectorField> u("Velocity", solution_tag());
267  FieldVariable<1, ScalarField> p("Pressure", solution_tag());
268  create_component<ProtoAction>("Update")->set_expression(nodes_expression(group(p += solution(p), u += solution(u))));
269 }
