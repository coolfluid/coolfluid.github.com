41 inline const common::Table<Real>& extract_coordinates(const mesh::Region& region)
42 {
43  const common::Table<Real>* coordinates = nullptr;
44  coordinates = common::find_component_ptr_with_tag<common::Table<Real> >(region, mesh::Tags::coordinates()).get();
45  if(!coordinates)
46  {
47  BOOST_FOREACH(const mesh::Elements& elements, common::find_components_recursively<mesh::Elements>(region))
48  {
49  if(coordinates)
50  {
51  cf3_assert(coordinates == &elements.geometry_fields().coordinates());
52  continue;
53  }
54  coordinates = &elements.geometry_fields().coordinates();
55  }
56  }
57 
58  return *coordinates;
59 }
