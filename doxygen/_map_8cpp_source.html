78 template <typename TYPE>
79 void Map::split_string ( const std::string& str, const std::string& delimiter,
80  std::vector<TYPE>& result, int size )
81 {
82  const char delim = delimiter[0];
83 
84  if(size >= 0)
85  result.reserve( result.size() + size );
86 
87  Uint in_brackets(0);
88  std::string::const_iterator first = str.begin();
89  std::string::const_iterator it = first;
90  std::string elem;
91  for ( ; it!=str.end(); ++it)
92  {
93  if (*it == '(') // opening bracket
94  ++in_brackets;
95  else if (*it == ')') // closing bracket
96  --in_brackets;
97  else if (*it == delim && in_brackets == 0)
98  {
99  bool is_delim=true;
100 
101  for (Uint i=0; i<delimiter.size(); ++i)
102  {
103  if (it+i==str.end() || *(it+i)!=delimiter[i])
104  {
105  is_delim=false;
106  break;
107  }
108  }
109  if (is_delim)
110  {
111  elem = std::string(first,it);
112  boost::algorithm::trim(elem);
113  try
114  {
115  if( !elem.empty() )
116  result.push_back( from_str<TYPE>(elem) );
117  }
118  catch(boost::bad_lexical_cast e)
119  {
120  throw CastingFailed(FromHere(), "Unable to cast [" + elem + "] to " +
121  common::class_name<TYPE>() + ".");
122  }
123  first = it+delimiter.size();
124  }
125  }
126  }
127  elem = std::string(first,it);
128  boost::algorithm::trim(elem);
129  try
130  {
131  if( !elem.empty() )
132  result.push_back( from_str<TYPE>(elem) );
133  }
134  catch(boost::bad_lexical_cast e)
135  {
136  throw CastingFailed(FromHere(), "Unable to cast [" + elem + "] to " +
137  common::class_name<TYPE>() + ".");
138  }
139 }

401 std::vector<TYPE> Map::array_to_vector ( const XmlNode & array_node, std::string * delim ) const
402 {
403  cf3_assert( array_node.is_valid() );

405  rapidxml::xml_attribute<> * delim_attr = nullptr;
406

414  *delim = delim_attr->value();

417  if ( !size_attr )

419
